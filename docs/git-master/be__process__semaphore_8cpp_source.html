23BiometricEvaluation::Process::Semaphore::Semaphore(
24 const std::string &name,
25 const mode_t mode,
26 const int value,
27 bool force) :
28 _name(name)
29{
30 this->_creatorPID = getpid();
31 int oflag = O_CREAT | O_EXCL;
32 if (force) {
33 /*
34 * Try to remove an existing semaphore, and test whether
35 * it existed afterwards.
36 */
37 if (sem_unlink(name.c_str()) != 0) {
38 /*
39 * OS-X returns EINVAL when the semaphore doesn't
40 * exist, contrary to POSIX.
41 */
42 if ((errno != ENOENT) && (errno != EINVAL)) {
43 throw BE::Error::StrategyError(
44 "Could not remove semaphore: " +
45 BE::Error::errorStr());
46 }
47 }
48 }
49 sem_t *sem = sem_open(
50 name.c_str(),
51 oflag,
52 mode,
53 value);
54 if (sem == SEM_FAILED) {
55 if (errno == EEXIST) {
56 throw BE::Error::ObjectExists();
57 } else {
58 throw BE::Error::StrategyError(
59 "Could not create semaphore: " +
60 BE::Error::errorStr());
61 }
62 }
63 this->_semaphore = sem;
64}

66BiometricEvaluation::Process::Semaphore::Semaphore(
67 const std::string &name) :
68 _name(name)
69{
70 this->_creatorPID = getpid();
71 sem_t *sem = sem_open(name.c_str(), 0);
72 if (sem == SEM_FAILED) {
73 if (errno == ENOENT) {
74 throw BE::Error::ObjectDoesNotExist();
75 } else {
76 throw BE::Error::StrategyError(
77 "Could not open semaphore: " +
78 BE::Error::errorStr());
79 }
80 }
81 this->_semaphore = sem;
82}

84BiometricEvaluation::Process::Semaphore::~Semaphore()
85{
86 sem_close(this->_semaphore);
87
88 /*
89 * Prevent unecessary unlinks of the semaphore when
90 * inherited by children, although no harm is done
91 * if a semaphore is unlinked more than once.
92 */
93 if (this->_creatorPID == getpid()) {
94 sem_unlink(this->_name.c_str());
95 }
96}

98bool
99BiometricEvaluation::Process::Semaphore::wait(
100 const bool interruptible)
101{
102 while (true) {
103 int ret = sem_wait(this->_semaphore);
104 if (ret == 0)
105 return (true);
106 switch (errno) {
107 case EINTR: /* Signal interrupted */
108 if (interruptible)
109 return (false);
110 break;
111 case EINVAL:
112 throw BE::Error::ObjectDoesNotExist("Invalid semaphore");
113 break; /* Not reached */
114 case EDEADLK:
115 throw BE::Error::StrategyError("Deadlock detected");
116 break; /* Not reached */
117 }
118 }
119}

147bool
148BiometricEvaluation::Process::Semaphore::timedwait(
149 const uint64_t interval,
150 const bool interruptible)
151{
152#ifdef Darwin
153 throw Error::NotImplemented();
154#else
155 /*
156 * Add the interval to the current time, then convert that value
157 * to microseconds to seconds/nanoseconds,
158 */
159 struct timeval tv;
160 (void) gettimeofday(&tv, NULL);
161 tv.tv_sec += (time_t)(interval / BE::Time::MicrosecondsPerSecond);
162
163 /* Roll over any excess microseconds into seconds field */
164 uint64_t usec = tv.tv_usec +
165 (interval % BE::Time::MicrosecondsPerSecond);
166 tv.tv_sec += (time_t)(usec / BE::Time::MicrosecondsPerSecond);
167 tv.tv_usec = (suseconds_t)(usec % BE::Time::MicrosecondsPerSecond);
168
169 struct timespec ts;
170 ts.tv_sec = tv.tv_sec;
171 ts.tv_nsec = tv.tv_usec * BE::Time::NanosecondsPerMicrosecond;
172 while (true) {
173 int ret = sem_timedwait(this->_semaphore, &ts);
174 if (ret == 0)
175 return (true);
176 switch (errno) {
177 case EINTR: /* Signal interrupted */
178 if (interruptible)
179 return (false);
180 break;
181 case EAGAIN: /* Already locked */
182 return(false);
183 break; /* Not reached */
184 case EINVAL:
185 throw BE::Error::ObjectDoesNotExist("Invalid semaphore");
186 break; /* Not reached */
187 case ETIMEDOUT:
188 return(false);
189 break; /* Not reached */
190 case EDEADLK:
191 throw BE::Error::StrategyError("Deadlock detected");
192 break; /* Not reached */
193 }
194 }
195#endif
196}

200{

202 if (ret != 0) {
203 switch (errno) {
204 case EINVAL:
