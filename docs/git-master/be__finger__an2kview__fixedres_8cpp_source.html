50void
51BiometricEvaluation::Finger::AN2KViewFixedResolution::readImageRecord(
52 const RecordType typeID)
53{
54 switch (typeID) {
55 case RecordType::Type_3:
56 case RecordType::Type_4:
57 case RecordType::Type_5:
58 case RecordType::Type_6:
59 break;
60 default:
61 throw Error::ParameterError("Invalid Record Type ID");
62 }
63 /*
64 * Read the information needed from the Type-1 record.
65 */
66 FIELD *field;
67 int idx;
68 Memory::AutoBuffer<ANSI_NIST> an2k = AN2KView::getAN2K();
69 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, NSR_ID, an2k->records[0])
70 != TRUE)
71 throw Error::DataError("Field NSR not found");
72 double nsr =
73 strtod((char *)field->subfields[0]->items[0]->value, nullptr);
74
75 RECORD *record = AN2KView::getAN2KRecord();
76
77 /*
78 * Using the ISR field, set the X/Y resolutions based on the
79 * native resolution read from the Type-1 record, or the minimum
80 * resolution from AN2K.
81 */
82 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, ISR_ID, record) != TRUE)
83 throw Error::DataError("Field ISR not found");
84 int isr = atoi((char *)field->subfields[0]->items[0]->value);
85
86 Image::Resolution ir;
87 ir.units = Image::Resolution::Units::PPMM;
88 switch ((*record).type) {
89 case TYPE_3_ID:
90 case TYPE_5_ID:
91 if (isr == 0) {
92 ir.xRes = ir.yRes =
93 AN2KView::HalfMinimumScanResolutionPPMM;
94 } else {
95 ir.xRes = ir.yRes = 0.5 * nsr;
96 }
97 break;
98 case TYPE_4_ID:
99 case TYPE_6_ID:
100 if (isr == 0) {
101 ir.xRes = ir.yRes =
102 AN2KView::MinimumScanResolutionPPMM;
103 } else {
104 ir.xRes = ir.yRes = nsr;
105 }
106 break;
107 }
108 /*
109 * The image resolution is the same as the scan resolution for
110 * these AN2K record types.
111 */
112 AN2KView::setImageResolution(ir);
113 AN2KView::setScanResolution(ir);
114 AN2KView::setImageColorDepth(FixedResolutionBitDepth);
115
116 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, BIN_CA_ID, record) != TRUE)
117 throw Error::DataError("Field BIN_CA not found");
118 AN2KView::setCompressionAlgorithm(
119 AN2KView::convertCompressionAlgorithm((*record).type,
120 field->subfields[0]->items[0]->value));
121
122 /* Retrieve the image data */
123 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, BIN_IMAGE_ID, record) != TRUE)
124 throw Error::DataError("Field BIN_IMAGE not found");
125 Memory::AutoArray<uint8_t> data = Memory::AutoArray<uint8_t>(
126 field->subfields[0]->items[0]->num_bytes);
127 for (unsigned int i = 0; i < data.size(); i++)
128 data[i] = field->subfields[0]->items[0]->value[i];
129
130 AN2KView::setImageData(data);
131}
