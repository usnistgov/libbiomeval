143/*
144 * Convert a point represented by "x,y" into a Image::Coordinate.
145 */
146static BE::Image::Coordinate
147pointFromStr(const std::string &pointStr)
148{
149 BE::Image::Coordinate point{};
150 size_t pos = pointStr.find(cDelim);
151 point.x = std::atoi((char*)(pointStr.substr(0, pos).c_str()));
152 point.y = std::atoi((char*)(pointStr.substr(
153 pos + cDelim.length(),
154 pointStr.length()).c_str()));
155 return(point);
156}

158/*
159 * Convert string in the form of "x1,y1-x2,y2,..,xn,yn" to
160 * a path (a set of points).
161 */
162static void
163pathFromStr(const std::string &pathStr, BE::Image::CoordinateSet &path)
164{
165 size_t start = 0;
166 size_t end = pathStr.find(pDelim);
167 while (end != std::string::npos) {
168 auto pointStr = pathStr.substr(start, end - start);
169 path.push_back(pointFromStr(pointStr));
170 start = end + pDelim.length();
171 end = pathStr.find(pDelim, start);
172 }
173 path.push_back(pointFromStr(pathStr.substr(start, end)));
174}

182static bool
183checkThenReadItem(
184 FIELD *field, int sfNum, int itemNum, bool &has_item, int &item)
185{
186 if (field->subfields[sfNum]->num_items < itemNum) {
187 has_item = false;
188 return false;
189 }
190 char *ptr = (char *)field->subfields[sfNum]->items[itemNum - 1]->value;
191 if (std::strlen(ptr) != 0) {
192 item = std::atoi(ptr);
193 has_item = true;
194 return true;
195 } else {
196 return false;
197 }
198}

200/*
201 * Note on processing AN2k field/subfield items:
202 * According to ANSI/NIST-ITL-2001 UPDATE 2015, if data items
203 * are optional, and there is more than optional one item possbile,
204 * empty items must be present, using the unit separator with
205 * nothing between. Check the length of the item string, which
206 * should be 0 when an empty item is followed by a item separator.
207 */
208static
209void
210readROI(
211 const RECORD *type9,
212 BiometricEvaluation::Image::ROI &roi)
213{
214 FIELD *field;
215 int idx;
216
217 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, EFS_ROI_ID, type9) == FALSE)
218 throw BE::Error::DataError("Field ROI not found");
219 roi.size.xSize = std::atoi((char*)field->subfields[0]->items[0]->value);
220 roi.size.ySize = std::atoi((char*)field->subfields[0]->items[1]->value);
221
222 if (field->subfields[0]->num_items == 2) {
223 return;
224 }
225 /* Assume that if we have horz offset, we have vert offset */
226 roi.horzOffset = std::atoi((char*)field->subfields[0]->items[2]->value);
227 roi.vertOffset = std::atoi((char*)field->subfields[0]->items[3]->value);
228
229 if (field->subfields[0]->num_items == 4) {
230 return;
231 }
232 std::string polygonStr =
233 std::string((char*)field->subfields[0]->items[4]->value);
234 pathFromStr(polygonStr, roi.path);
235}

251static
252void
253readFPP(
254 const RECORD *type9,
255 BiometricEvaluation::Feature::AN2K11EFS::FPPPosition &fpp)
256{
257 FIELD *field;
258 int idx;
259
260 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, EFS_FPP_ID, type9) == FALSE)
261 throw BE::Error::DataError("Field FPP not found");
262
263 /*********************************************************************/
264 /* Required Fields. */
265 /*********************************************************************/
266 /* FGP */
267 int fgp = std::atoi((char*)field->subfields[0]->items[0]->value);
268 /*
269 * AN2k11 EFS standard allows only for a subset of finger positions,
270 * and all palm positions.
271 */
272 if ((fgp >= 0 && fgp <= 10) || (fgp >= 16 && fgp <= 18)) {
273 fpp.fgp.posType = BE::Feature::PositionType::Finger;
274 fpp.fgp.position.fingerPos = static_cast<BE::Finger::Position>(fgp);
275 } else if ((fgp >= 20 && fgp <= 38) || (fgp >= 81 && fgp <= 86)) {
276 fpp.fgp.posType = BE::Feature::PositionType::Palm;
277 fpp.fgp.position.palmPos = static_cast<BE::Palm::Position>(fgp);
278 } else {
279 throw (BE::Error::DataError("Invalid FGP"));
280 }
281 if (field->subfields[0]->num_items == 1) {
282 return;
283 }
284 /*********************************************************************/
285 /* Optional Fields. */
286 /*********************************************************************/
287 fpp.has_fsm = false;
288 fpp.has_ocf = false;
289 fpp.has_sgp = false;
290
291 /* FSM */
292 if (field->subfields[0]->items[1]->num_chars != 0) {
293 std::string str((char*)field->subfields[0]->items[1]->value);
294 try {
295 fpp.fsm = FSMMap.at(str);
296 fpp.has_fsm = true;
297 } catch (...) {
298 throw (BE::Error::DataError(
299 "Unknown image segment value"));
300 }
301 }
302 if (field->subfields[0]->num_items == 2) {
303 return;
304 }
305 /* OCF */
306 if (field->subfields[0]->items[2]->num_chars != 0) {
307 char c = *(char*)field->subfields[0]->items[2]->value;
308 try {
309 fpp.ocf = OCFMap.at(c);
310 fpp.has_ocf = true;
311 } catch (...) {
312 throw (BE::Error::DataError("Invalid OCF value"));
313 }
314 }
315 if (field->subfields[0]->num_items == 3) {
316 return;
317 }
318 if (field->subfields[0]->items[3]->num_chars != 0) {
319 std::string polygonStr =
320 std::string((char*)field->subfields[0]->items[3]->value);
321 pathFromStr(polygonStr, fpp.sgp);
322 fpp.has_sgp = true;
323 }
324}

332static void
333readImageInfo(
334 const RECORD *type9,
335 BiometricEvaluation::Feature::AN2K11EFS::ImageInfo &ii)
336{
337 /*********************************************************************/
338 /* Required Fields. */
339 /*********************************************************************/
340 readROI(type9, ii.roi);
341 readFPP(type9, ii.fpp);
342
343 /*********************************************************************/
344 /* Optional Fields. */
345 /*********************************************************************/
346 FIELD *field;
347 int idx;
348 /*
349 * Orientation
350 */
351 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, EFS_ORT_ID, type9) == TRUE) {
352 ii.ort.encodingMethod = BE::Feature::AN2K11EFS::Orientation::
353 EncodingMethod::UserDefined;
354 ii.ort.eod =
355 std::atoi((char*)field->subfields[0]->items[0]->value);
356
357 /* EUC is optional */
358 if (field->subfields[0]->num_items == 1) {
359 ii.ort.has_euc = false;
360 } else {
361 ii.ort.has_euc = true;
362 ii.ort.euc = std::atoi(
363 (char*)field->subfields[0]->items[1]->value);
364
365 if (ii.ort.euc == BE::Feature::AN2K11EFS::Orientation::
366 EUCIndeterminate)
367 ii.ort.encodingMethod = BE::Feature::AN2K11EFS::
368 Orientation::EncodingMethod::Indeterminate;
369 }
370 } else { /* Assign default values per AN2k standard. */
371 ii.ort.encodingMethod = BE::Feature::AN2K11EFS::Orientation::
372 EncodingMethod::Default;
373 ii.ort.eod = BE::Feature::AN2K11EFS::Orientation::EODDefault;
374 ii.ort.euc = BE::Feature::AN2K11EFS::Orientation::EUCDefault;
375 }
376 ii.has_trv = false;
377 ii.has_plr = false;
378 /*
379 * Tonal reversal
380 */
381 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, EFS_TRV_ID, type9) == TRUE) {
382 char c = *(char*)field->subfields[0]->items[0]->value;
383 try {
384 ii.trv = TRVMap.at(c);
385 ii.has_trv = true;
386 } catch (...) {
387 throw (BE::Error::DataError("Invalid TRV value"));
388 }
389 }
390 /*
391 * Lateral reversal
392 */
393 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, EFS_PLR_ID, type9) == TRUE) {
394 char c = *(char*)field->subfields[0]->items[0]->value;
395 switch (c) {
396 case 'L':
397 ii.plr = BE::Feature::AN2K11EFS::LateralReversal::L;
398 break;
399 case 'U':
400 ii.plr = BE::Feature::AN2K11EFS::LateralReversal::U;
401 break;
402 default:
403 throw BE::Error::DataError("Invalid PLR value");
404 break;
405 }
406 ii.has_plr = true;
407 }
408}

416static void
417readMPS(
418 const RECORD *type9,
419 BiometricEvaluation::Feature::AN2K11EFS::MinutiaPointSet &mps)
420{
421 FIELD *field;
422 int idx;
423
424 /* Minutiae data is optional. */
425 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, EFS_MIN_ID, type9) == FALSE)
426 return;
427
428 for (int i = 0; i < field->num_subfields; i++) {
429 if (field->subfields[i]->num_items < 4) {
430 throw BE::Error::DataError(
431 "Insufficient item count in EFS minutia point");
432 }
433 /*
434 * Read four mandatory data items.
435 */
436 BE::Feature::AN2K11EFS::MinutiaPoint mp{};
437 mp.has_quality = false;
438 /* index starts at 1 for other Type-9 minutia */
439 mp.index = i + 1;
440 mp.coordinate.x =
441 std::atoi((char*)field->subfields[i]->items[0]->value);
442 mp.coordinate.y =
443 std::atoi((char*)field->subfields[i]->items[1]->value);
444 mp.theta =
445 std::atoi((char*)field->subfields[i]->items[2]->value);
446 mp.has_type = false;
447 mp.has_mru = false;
448 mp.has_mdu = false;
449 char type = *field->subfields[i]->items[3]->value;
450 try {
451 mp.type = MinutiaeTypeMap.at(type);
452 mp.has_type = true;
453 } catch (...) {
454 throw (BE::Error::DataError(
455 "Unknown minutia type value"));
456 }
457 /*
458 * Read two optional data items.
459 */
460 if (!checkThenReadItem(field, i, 5, mp.has_mru, mp.mru)) {
461 mps.push_back(mp);
462 continue;
463 }
464 if (!checkThenReadItem(field, i, 6, mp.has_mdu, mp.mdu)) {
465 mps.push_back(mp);
466 continue;
467 }
468 mps.push_back(mp);
469 }
470}

472static void
473readCPS(
474 const RECORD *type9,
475 BiometricEvaluation::Feature::AN2K11EFS::CorePointSet &cps)
476{
477 FIELD *field;
478 int idx;
479
480 /* Core data is optional. */
481 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, EFS_COR_ID, type9) == FALSE)
482 return;
483
484 for (int i = 0; i < field->num_subfields; i++) {
485 if (field->subfields[i]->num_items < 2) {
486 throw BE::Error::DataError(
487 "Insufficient item count in EFS core point");
488 }
489 BE::Feature::AN2K11EFS::CorePoint cp{};
490 /*
491 * Read two mandatory data items.
492 */
493 cp.location.x =
494 std::atoi((char*)field->subfields[i]->items[0]->value);
495 cp.location.y =
496 std::atoi((char*)field->subfields[i]->items[1]->value);
497 /*
498 * Read up to three optional data items.
499 */
500 if (!checkThenReadItem(field, i, 3, cp.has_cdi, cp.cdi)) {
501 cps.push_back(cp);
502 continue;
503 }
504 if (!checkThenReadItem(field, i, 4, cp.has_rpu, cp.rpu)) {
505 cps.push_back(cp);
506 continue;
507 }
508 if (!checkThenReadItem(field, i, 5, cp.has_duy, cp.duy)) {
509 cps.push_back(cp);
510 continue;
511 }
512 cps.push_back(cp);
513 }

534static void
535readDPS(
536 const RECORD *type9,
537 BiometricEvaluation::Feature::AN2K11EFS::DeltaPointSet &dps)
538{
539 FIELD *field;
540 int idx;
541
542 /* Delta data is optional. */
543 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, EFS_DEL_ID, type9) == FALSE)
544 return;
545
546 for (int i = 0; i < field->num_subfields; i++) {
547 if (field->subfields[i]->num_items < 2) {
548 throw BE::Error::DataError(
549 "Insufficient item count in EFS delta point");
550 }
551 BE::Feature::AN2K11EFS::DeltaPoint dp{};
552 /*
553 * Read two mandatory data items.
554 */
555 dp.location.x =
556 std::atoi((char*)field->subfields[i]->items[0]->value);
557 dp.location.y =
558 std::atoi((char*)field->subfields[i]->items[1]->value);
559 /*
560 * Read up to eight optional data items.
561 */
562 if (!checkThenReadItem(field, i, 3, dp.has_dup, dp.dup)) {
563 dps.push_back(dp);
564 continue;
565 }
566 if (!checkThenReadItem(field, i, 4, dp.has_dlf, dp.dlf)) {
567 dps.push_back(dp);
568 continue;
569 }
570 if (!checkThenReadItem(field, i, 5, dp.has_drt, dp.drt)) {
571 dps.push_back(dp);
572 continue;
573 }
574 if (field->subfields[i]->num_items < 6) {
575 dps.push_back(dp);
576 continue;
577 }
578 char *type = (char *)field->subfields[i]->items[5]->value;
579 try {
580 dp.dtp = DeltaTypeMap.at(std::string(type));
581 } catch (...) {
582 throw (BE::Error::DataError(
583 "Unknown delta type value"));
584 }
585 if (!checkThenReadItem(field, i, 7, dp.has_rpu, dp.rpu)) {
586 dps.push_back(dp);
587 continue;
588 }
589 if (!checkThenReadItem(field, i, 8, dp.has_duu, dp.duu)) {
590 dps.push_back(dp);
591 continue;
592 }
593 if (!checkThenReadItem(field, i, 9, dp.has_dul, dp.dul)) {
594 dps.push_back(dp);
595 continue;
596 }
597 if (!checkThenReadItem(field, i, 10, dp.has_dur, dp.dur)) {
598 dps.push_back(dp);
599 continue;
600 }
601 dps.push_back(dp);
602 }

840 {"PA", BE::Feature::AN2K11EFS::Pattern::
841 ArchSubclassification::Plain},

848 {"PW", BE::Feature::AN2K11EFS::Pattern::

852 {"DL", BE::Feature::AN2K11EFS::Pattern::
853 WhorlSubclassification::DoubleLoop},

859 {"I", BE::Feature::AN2K11EFS::Pattern::
860 WhorlDeltaRelationship::Inner},
861 {"M", BE::Feature::AN2K11EFS::Pattern::

863 {"O", BE::Feature::AN2K11EFS::Pattern::

872 return;

881 if (field->subfields[i]->num_items < 1)

889 items[0]->value);

891 throw BE::Error::DataError{"Invalid GCF in PAT "
892 "subfield #" + std::to_string(i + 1) + ": " +

894 value)};

899 empty()) {

906 GeneralClassification::Arch)
907 pat.subclass.arch = archMap.at((char*)field->
908 subfields[i]->items[1]->value);

912 subfields[i]->items[1]->value);

917 throw BE::Error::DataError{"Invalid SUB in PAT "
918 "subfield #" + std::to_string(i + 1) + ": " +

921 }

928 }

932 GeneralClassification::Whorl)

941 "subfield #" + std::to_string(i + 1) + ": " +
942 std::string((char*)field->subfields[i]->items[2]->
943 value)};
944 }
945
946 pats.push_back(pat);

950static void
951readMRCI(
952 const RECORD *type9,
953 BiometricEvaluation::Feature::AN2K11EFS::MinutiaeRidgeCountInfo &mrci)
954{
955 FIELD *field;
956 int idx;
957 /*
958 * Minutiae ridge count algorithm
959 */
960 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, EFS_MRA_ID, type9) == TRUE) {
961 try {
962 mrci.mra =
963 BE::Framework::Enumeration::to_enum
964 <BE::Feature::AN2K11EFS::MRA>(
965 std::string((char *)
966 field->subfields[0]->items[0]->value));
967 mrci.has_mra = true;
968 } catch (const BE::Error::ObjectDoesNotExist&) {
969 throw BE::Error::DataError("Invalid MRA value");
970 }
971 }
972 /*
973 * Minutiae ridge counts
974 */
975 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, EFS_MRC_ID, type9) == TRUE) {
976 mrci.has_mrcs = true;
977 for (int i = 0; i < field->num_subfields; i++) {
978 BE::Feature::AN2K11EFS::MinutiaeRidgeCount mrc{};
979 /*
980 * Read three mandatory data items.
981 */
982 mrc.mia = std::atoi(
983 (char*)field->subfields[i]->items[0]->value);
984 mrc.mib = std::atoi(
985 (char*)field->subfields[i]->items[1]->value);
986 mrc.mir = std::atoi(
987 (char*)field->subfields[i]->items[2]->value);
988 /*
989 * Read two optional data items.
990 */
991 if (!checkThenReadItem(
992 field, i, 4, mrc.has_mrn, mrc.mrn)) {
993 mrci.mrcs.push_back(mrc);
994 continue;
995 }
996 if (!checkThenReadItem(
997 field, i, 5, mrc.has_mrs, mrc.mrs)) {
998 mrci.mrcs.push_back(mrc);
999 continue;
1000 }
1001 mrci.mrcs.push_back(mrc);
1002 }
1003 }
1004 /*
1005 * Ridge count confidence
1006 */
1007 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, EFS_RCC_ID, type9) == TRUE) {
1008 mrci.has_rccs = true;
1009 for (int i = 0; i < field->num_subfields; i++) {
1010 /*
1011 * All data items are mandatory
1012 */
1013 BE::Image::Coordinate pointA{}, pointB;{}
1014 pointA.x = std::atoi(
1015 (char*)field->subfields[i]->items[0]->value);
1016 pointA.y = std::atoi(
1017 (char*)field->subfields[i]->items[1]->value);
1018 pointB.x = std::atoi(
1019 (char*)field->subfields[i]->items[2]->value);
1020 pointB.y = std::atoi(
1021 (char*)field->subfields[i]->items[3]->value);
1022
1023 char c = (char)*field->subfields[i]->items[4]->value;
1024 BE::Feature::AN2K11EFS::MORC morc;
1025 switch (c) {
1026 case 'A':
1027 morc = BE::Feature::AN2K11EFS::MethodOfRidgeCounting::A;
1028 break;
1029 case 'T':
1030 morc = BE::Feature::AN2K11EFS::MethodOfRidgeCounting::T;
1031 break;
1032 case 'M':
1033 morc = BE::Feature::AN2K11EFS::MethodOfRidgeCounting::M;
1034 break;
1035 default:
1036 throw BE::Error::DataError("Invalid MORC value");
1037 }
1038 int mcv = std::atoi(
1039 (char*)field->subfields[i]->items[5]->value);
1040
1041 BE::Feature::AN2K11EFS::MRCC
1042 rcc{pointA, pointB, morc, mcv};
1043 mrci.rccs.push_back(rcc);
1044 }
1045 }

1049BiometricEvaluation::Feature::AN2K11EFS::ExtendedFeatureSet::Impl::readType9Record(
1050 Memory::uint8Array &buf,
1051 int recordNumber)
1052{
1053 Memory::AutoBuffer<ANSI_NIST> an2k =
1054 Memory::AutoBuffer<ANSI_NIST>(&biomeval_nbis_alloc_ANSI_NIST,
1055 &biomeval_nbis_free_ANSI_NIST, &biomeval_nbis_copy_ANSI_NIST);
1056
1057 AN2KBDB bdb;
1058 INIT_AN2KBDB(&bdb, buf, buf.size());
1059 if (biomeval_nbis_scan_ANSI_NIST(&bdb, an2k) != 0)
1060 throw BE::Error::DataError(
1061 "Could not read complete AN2K record");
1062
1063 /*
1064 * Find the requested Type-9 in the file, throwing an exception
1065 * if not present. The first record in an AN2K file is always
1066 * the Type-1, so skip that one.
1067 */
1068 RECORD *type9 = nullptr;
1069 for (int i = 1; i < an2k->num_records; i++) {
1070 if (an2k->records[i]->type == TYPE_9_ID) {
1071 if (i == recordNumber) {
1072 type9 = an2k->records[i];
1073 break;
1074 }
1075 }
1076 }
1077 if (type9 == nullptr)
1078 throw (BE::Error::DataError(
1079 "Could not find requested Type-9 in AN2K record"));
1080
1081 readImageInfo(type9, this->_ii);
1082
1083 /*********************************************************************/
1084 /* Optional Fields. */
1085 /*********************************************************************/
1086 readMPS(type9, this->_mps);
1087 readCPS(type9, this->_cps);
1088 readDPS(type9, this->_dps);
1089 readLPM(type9, this->_lpm);
1090 readNFP(type9, this->_nfp);
1091 readMRCI(type9, this->_mrci);
1092 readEAA(type9, this->_eaa);
1093 readLSB(type9, this->_lsb);
1094 readPAT(type9, this->_pat);
1095}
