61/*
62 * Add a string key to the given buffer, preceded by the length of the
63 * key. The key is written as characters, without the nul terminator.
64 * The index out parameter is updated to the location of where the
65 * next write can take place.
66 */
67static void
68fillBufferWithKeyAndValue(
69 BE::Memory::uint8Array &buf,
70 const std::string &key,
71 BE::Memory::uint8Array &value,
72 BE::Memory::uint8Array::size_type &index)
73{
74 uint32_t keyLength = key.length();
75 uint64_t valueSize = value.size();
76 uint64_t neededSpace = index /* buffer space in use */
77 + sizeof(uint32_t) + keyLength /* space for key and length */
78 + sizeof(uint64_t) + valueSize; /* for value and size */
79 buf.resize(neededSpace);
80
81 /* Write the key length, value size, key, value if non-zero size */
82 uint32_t *pInt32 = (uint32_t *)&buf[index];
83 *pInt32 = (uint32_t)keyLength;
84 index += sizeof(uint32_t);
85 uint64_t *pInt64 = (uint64_t *)&buf[index];
86 *pInt64 = (uint64_t)valueSize;
87 index += sizeof(uint64_t);
88 std::memcpy((char *)&buf[index], key.data(), keyLength);
89 index += keyLength;
90 if (valueSize != 0) {
91 std::memcpy((void *)&buf[index], &value[0], valueSize);
92 index += valueSize;
93 }
94}

96void
97BiometricEvaluation::MPI::RecordStoreDistributor::createWorkPackage(
98 MPI::WorkPackage &workPackage)
99{
100 std::shared_ptr<BE::IO::Logsheet> log = this->getLogsheet();
101
102 /* Create the package data buffer at a reasonable starting size */
103 BE::Memory::uint8Array packageData(16384);
104
105 /*
106 * If there are no more keys to be read from the record store,
107 * send an empty work package.
108 */
109 if (this->_recordsRemaining == 0) {
110 workPackage.setNumElements(0);
111 workPackage.setData(packageData);
112 return;
113 }
114
115 /*
116 * Distribute a work package based on the chunk size given
117 * in the resources object. If a failure occurs reading a key,
118 * continue onto the next key. It is possible to send an empty
119 * work package due to sequential failures.
120 */
121 uint64_t keyCount;
122 if (this->_recordsRemaining > this->_resources->getChunkSize()) {
123 keyCount = this->_resources->getChunkSize();
124 } else {
125 keyCount = this->_recordsRemaining;
126 }
127
128 this->_recordsRemaining -= keyCount;
129
130 /*
131 * The value array must be 0-sized to start, and will stay that way
132 * if values are not to be sent.
133 */
134 BE::IO::RecordStore::Record record;
135 BE::Memory::uint8Array::size_type index = 0;
136 uint64_t realKeyCount = 0;
137 std::shared_ptr<IO::RecordStore> recordStore =
138 this->_resources->getRecordStore();
139
140 /*
141 * Pull keys, and possibly values, from the RecordStore and
142 * combine a chunk of them into a single work package.
143 */
144 for (uint64_t n = 0; n < keyCount; n++) {
145 try {
146 if (this->_includeValues)
147 record = recordStore->sequence();
148 else
149 record.key = recordStore->sequenceKey();
150 /*
151 * Save the last key sent for checkpointing purposes.
152 */
153 this->_lastDistributedKey = record.key;
154
155 fillBufferWithKeyAndValue(packageData, record.key,
156 record.data, index);
157 } catch (Error::Exception &e) {
158 log->writeDebug("Caught " + e.whatString());
159 continue;
160 }
161 realKeyCount++;
162 }
163 /*
164 * NOTE: At this point it is possible to have no keys in the package.
165 */
166 workPackage.setNumElements(realKeyCount);
167 workPackage.setData(packageData);
168}

197{
198 try {
199 auto chkData = this->getCheckpointData();
200 auto lastKey = chkData->getProperty(
201 BE::MPI::RecordStoreDistributor::CHECKPOINTLASTKEY);
202 /*
203 * Sequence into the input record store to the key past
204 * the checkpoint key.
205 */
206 auto recordStore = this->_resources->getRecordStore();
207 recordStore->setCursorAtKey(lastKey);
208 (void)recordStore->sequence();
209
210 this->_recordsRemaining -= chkData->getPropertyAsInteger(
211 BE::MPI::RecordStoreDistributor::CHECKPOINTNUMKEYS);
212 this->getLogsheet()->writeDebug(
213 "Checkpoint restore: " + chkData->getProperty(
214 BE::MPI::Distributor::CHECKPOINTREASON));
215 } catch (Error::Exception &e) {
216 this->getLogsheet()->writeDebug(
