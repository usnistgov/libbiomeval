27BiometricEvaluation::IO::CompressedRecordStore::Impl::Impl(
28 const std::string &pathname,
29 const std::string &description,
30 const RecordStore::Kind &recordStoreType,
31 const std::string &compressorType) :
32 RecordStore::Impl(pathname, description, RecordStore::Kind::Compressed)
33{
34 std::string rsPath = pathname + '/' + BACKING_STORE;
35 this->_rs = IO::RecordStore::createRecordStore(rsPath, description,
36 recordStoreType);
37 rsPath = rsPath + METADATA_SUFFIX;
38 this->_mdrs = IO::RecordStore::createRecordStore(rsPath, description,
39 recordStoreType);
40 try {
41 this->_compressor =
42 IO::Compressor::createCompressor(
43 to_enum<IO::Compressor::Kind>(compressorType));
44 } catch (const Error::ObjectDoesNotExist&) {
45 throw Error::StrategyError(compressorType + " is not a valid "
46 "compressor type");
47 }
48
49 /* Store compressor type */
50 std::shared_ptr<IO::Properties> props = this->getProperties();
51 props->setProperty(COMPRESSOR_TYPE_KEY, compressorType);
52 this->setProperties(props);
53}

109void
110BiometricEvaluation::IO::CompressedRecordStore::Impl::insert(
111 const std::string &key,
112 const void *const data,
113 const uint64_t size)
114{
115 if (this->getMode() == Mode::ReadOnly)
116 throw Error::StrategyError(RSREADONLYERROR);
117
118 Memory::uint8Array compressedData = _compressor->compress(
119 static_cast<const uint8_t *const>(data), size);
120 _rs->insert(key, compressedData);
121
122 std::ostringstream sizeStr;
123 sizeStr << size;
124 Memory::uint8Array sizeBuf(sizeStr.str().size());
125 sizeBuf.copy((uint8_t *)sizeStr.str().data(), sizeStr.str().size());
126 _mdrs->insert(key, sizeBuf);
127
128 RecordStore::Impl::insert(key, data, size);
129}

154BiometricEvaluation::IO::CompressedRecordStore::Impl::i_sequence(
155 bool returnData,
156 int cursor)
157{
158 BE::IO::RecordStore::Record record;
159 /* Obtain the next key, but not data, since it is compressed */
