30BiometricEvaluation::MPI::CSVResources::CSVResources(
31 const std::string &propertiesFileName) :
32 BiometricEvaluation::MPI::Resources(propertiesFileName),
33 _useBuffer(false),
34 _randomizeLines(false),
35 _delimiter(",")
36{
37 std::unique_ptr<IO::Properties> props;
38 try {
39 props.reset(new BE::IO::PropertiesFile(propertiesFileName,
40 IO::Mode::ReadOnly, {
41 {BE::MPI::CSVResources::TRIMPROPERTY, "true"}
42 }));
43 } catch (BE::Error::Exception &e) {
44 throw BE::Error::FileError("Could not open properties: " +
45 e.whatString());
46 }
47
48 /* Required properties */
49 try {
50 this->_chunkSize = props->getPropertyAsInteger(
51 BE::MPI::CSVResources::CHUNKSIZEPROPERTY);
52 this->_csvPath = props->getProperty(
53 BE::MPI::CSVResources::INPUTCSVPROPERTY);
54 } catch (BE::Error::Exception &e) {
55 throw BE::Error::ObjectDoesNotExist("Could not read "
56 "properties: " + e.whatString());
57 }
58
59 /* Optional properties */
60 try {
61 this->_useBuffer = props->getPropertyAsBoolean(
62 BE::MPI::CSVResources::USEBUFFERPROPERTY);
63 } catch (const BE::Error::Exception&) {}
64 try {
65 this->_delimiter = props->getProperty(
66 BE::MPI::CSVResources::DELIMITERPROPERTY);
67 } catch (const BE::Error::Exception&) {}
68 try {
69 this->_randomizeLines = props->getPropertyAsBoolean(
70 BE::MPI::CSVResources::RANDOMIZEPROPERTY);
71 } catch (const BE::Error::Exception&) {}
72 if (this->_randomizeLines) {
73 if (!this->_useBuffer)
74 throw BE::Error::StrategyError("\"" +
75 BE::MPI::CSVResources::RANDOMIZEPROPERTY + "\" "
76 "was enabled, but it requires \"" +
77 BE::MPI::CSVResources::USEBUFFERPROPERTY + ",\" "
78 "which was not.");
79 try {
80 this->_rngSeed = props->getPropertyAsInteger(
81 BE::MPI::CSVResources::RANDOMSEEDPROPERTY);
82 } catch (const BE::Error::ObjectDoesNotExist&) {
83 this->_rngSeed = std::random_device()();
84 }
85 this->_rng = std::mt19937_64(this->_rngSeed);
86 }
87 this->_trimWhitespace = props->getPropertyAsBoolean(
88 BE::MPI::CSVResources::TRIMPROPERTY);
89
90 this->openCSV();
91}

122void
123BiometricEvaluation::MPI::CSVResources::openCSV()
124{
125 /* Check that file exists */
126 if (!BE::IO::Utility::fileExists(this->_csvPath))
127 throw BE::Error::ObjectDoesNotExist("File does not exist: " +
128 this->_csvPath);
129
130 /* Completely read in file, or open a stream */
131 if (this->_useBuffer) {
132 this->_csvBuffer = BE::IO::Utility::readFile(this->_csvPath,
133 std::ios_base::in);
134 this->_offset = 0;
135 this->_numLines = BE::IO::Utility::countLines(this->_csvBuffer);
136
137 /* To randomize, we need to have all lines in memory */
138 if (this->_randomizeLines) {
139 uint64_t bufferSize = this->_csvBuffer.size();
140 uint64_t offset = 0, endOffset = 0;
141
142 for (uint64_t lineNum = 0; lineNum < this->_numLines;
143 lineNum++) {
144 endOffset = offset;
145 while ((endOffset < bufferSize) &&
146 (this->_csvBuffer[endOffset] != '\n'))
147 endOffset++;
148 std::string line((char *)&this->_csvBuffer[
149 offset], endOffset - offset);
150 offset = endOffset + 1;
151 this->_randomizedLines.push_back(std::make_pair(
152 lineNum + 1, line));
153 }
154
155 std::shuffle(this->_randomizedLines.begin(),
156 this->_randomizedLines.end(), this->_rng);
157 }
158 } else {
159 this->_numLines = BE::IO::Utility::countLines(this->_csvPath);
160 this->_csvStream = std::make_shared<std::ifstream>(
161 this->_csvPath);
162 if (!this->_csvStream)
163 throw Error::FileError("Error while opening CSV");
164 }
165 this->_remainingLines = this->_numLines;
166}

220std::pair<uint64_t, std::string>
221BiometricEvaluation::MPI::CSVResources::readLine()
222{
223 if (this->_useBuffer) {
224 if (this->_randomizeLines) {
225 this->_remainingLines -= 1;
226 if (this->_trimWhitespace) {
227 const auto ret = this->_randomizedLines[
228 this->_numLines - this->_remainingLines -
229 1];
230 return {ret.first, BE::Text::trimWhitespace(
231 ret.second)};
232 } else
233 return (this->_randomizedLines[this->_numLines -
234 this->_remainingLines - 1]);
235 } else {
236 uint64_t bufferSize = this->_csvBuffer.size();
237 if (this->_offset >= bufferSize)
238 throw BE::Error::ObjectDoesNotExist(
239 "Buffer exhausted");
240
241 uint64_t endOffset = this->_offset;
242 while ((endOffset < bufferSize) &&
243 (this->_csvBuffer[endOffset] != '\n'))
244 endOffset++;
245
246 std::string line((char *)&this->_csvBuffer[
247 this->_offset], endOffset - this->_offset);
248 if (this->_trimWhitespace)
249 line = BE::Text::trimWhitespace(line);
250 this->_offset = endOffset + 1;
251
252 this->_remainingLines -= 1;
253 return (std::make_pair(this->_numLines -
254 this->_remainingLines, line));
255 }
256 } else {
257 if (!this->_csvStream)
258 throw BE::Error::StrategyError("Stream not open");
259
260 std::string line;
261 std::getline(*this->_csvStream, line);
262 if (this->_trimWhitespace)
263 line = BE::Text::trimWhitespace(line);
264 if (!this->_csvStream)
265 throw BE::Error::ObjectDoesNotExist("Stream exhausted");
266
267 this->_remainingLines -= 1;
268 return (std::make_pair(this->_numLines - this->_remainingLines,
269 line));
270 }
271}
