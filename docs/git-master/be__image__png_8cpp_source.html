86BiometricEvaluation::Image::PNG::PNG(
87 const uint8_t *data,
88 const uint64_t size,
89 const std::string &identifier,
90 const statusCallback_t &statusCallback) :
91 Image::Image(
92 data,
93 size,
94 CompressionAlgorithm::PNG,
95 identifier,
96 statusCallback)
97{
98 png_structp png_ptr = png_create_read_struct(PNG_LIBPNG_VER_STRING,
99 (void *)this, png_error_callback, png_warning_callback);
100 if (png_ptr == nullptr)
101 throw Error::StrategyError("Could not initialize reading");
102
103 /* Read encoded PNG data from an AutoArray using our extension */
104 png_buffer png_buf = { this->getDataPointer(), this->getDataSize(), 0 };
105 png_set_read_fn(png_ptr, &png_buf, png_read_mem_src);
106
107 /* Read the header information */
108 png_infop png_info_ptr = png_create_info_struct(png_ptr);
109 if (png_info_ptr == nullptr) {
110 png_destroy_read_struct(&png_ptr, nullptr, nullptr);
111 throw Error::StrategyError("Could not initialize container for "
112 "information");
113 }
114 png_read_info(png_ptr, png_info_ptr);
115
116 const auto pngBitDepth = png_get_bit_depth(png_ptr, png_info_ptr);
117 setColorDepth(pngBitDepth *
118 png_get_channels(png_ptr, png_info_ptr));
119 /* Possible this could be <8-bit palette color */
120 const auto colorType = png_get_color_type(png_ptr, png_info_ptr);
121 if ((this->getColorDepth() <= 8) &&
122 ((colorType & PNG_COLOR_MASK_PALETTE) == PNG_COLOR_MASK_PALETTE))
123 throw BE::Error::NotImplemented("Color palette PNG image, "
124 "bit depth = " + std::to_string(pngBitDepth) + ", color "
125 "type = " + std::to_string(colorType));
126
127 this->setBitDepth(pngBitDepth);
128 setDimensions(Size(png_get_image_width(png_ptr, png_info_ptr),
129 png_get_image_height(png_ptr, png_info_ptr)));
130
131 png_uint_32 xres, yres;
132 int32_t type;
133 if (png_get_pHYs(png_ptr, png_info_ptr, &xres, &yres, &type) ==
134 PNG_INFO_pHYs) {
135 switch (type) {
136 case PNG_RESOLUTION_METER:
137 setResolution(Resolution(xres / 100.0, yres / 100.0,
138 Resolution::Units::PPCM));
139 break;
140 case PNG_RESOLUTION_UNKNOWN:
141 /* Resolution based on aspect ratio */
142 case PNG_RESOLUTION_LAST:
143 /* FALLTHROUGH */
144 default:
145 /* FALLTHROUGH */
146 /*
147 * For our purposes, there really is no good way to
148 * unambiguously set a resolution.
149 */
150 setResolution(Resolution(0, 0,
151 Resolution::Units::PPCM));
152 break;
153 }
154 } else {
155 /*
156 * Assume resolution is 72 dpi on both axis if resolution
157 * is not set, which is often the case in order to reduce
158 * file size.
159 */
160 setResolution(Resolution(72, 72, Resolution::Units::PPI));
161 }
162 png_byte color_type{png_get_color_type(png_ptr, png_info_ptr)};
163 this->setHasAlphaChannel((color_type & PNG_COLOR_MASK_ALPHA) ==
164 PNG_COLOR_MASK_ALPHA);
165
166 png_destroy_read_struct(&png_ptr, &png_info_ptr, nullptr);
167}

182BiometricEvaluation::Memory::uint8Array
183BiometricEvaluation::Image::PNG::getRawData()
184 const
185{
186 png_structp png_ptr = png_create_read_struct(PNG_LIBPNG_VER_STRING,
187 (void *)this, png_error_callback, png_warning_callback);
188 if (png_ptr == nullptr)
189 throw Error::StrategyError("Could not initialize reading");
190
191 /* Read encoded PNG data from a buffer using our extension */
192 png_buffer png_buf = { this->getDataPointer(), this->getDataSize(), 0 };
193 png_set_read_fn(png_ptr, &png_buf, png_read_mem_src);
194
195 /* Read the header information */
196 png_infop png_info_ptr = png_create_info_struct(png_ptr);
197 if (png_info_ptr == nullptr) {
198 png_destroy_read_struct(&png_ptr, nullptr, nullptr);
199 throw Error::StrategyError("Could not initialize container for "
200 "information");
201 }
202 png_read_info(png_ptr, png_info_ptr);
203
204 /* PNG default storage is big-endian */
205 auto pngBitDepth = png_get_bit_depth(png_ptr, png_info_ptr);
206 if ((pngBitDepth > 8) && BiometricEvaluation::Memory::isLittleEndian())
207 png_set_swap(png_ptr);
208
209
210 /* Let libpng help us do transformations. */
211 bool didTransformations{false};
212 auto color_type = png_get_color_type(png_ptr, png_info_ptr);
213 if ((color_type == PNG_COLOR_TYPE_GRAY) && (pngBitDepth < 8)) {
214 png_set_expand_gray_1_2_4_to_8(png_ptr);
215 didTransformations = true;
216 }
217
218 /* De-paletteize */
219 if (color_type == PNG_COLOR_TYPE_PALETTE) {
220 png_set_palette_to_rgb(png_ptr);
221 didTransformations = true;
222 }
223
224 /* Interpret tRNS block into alpha channel */
225 if (png_get_valid(png_ptr, png_info_ptr, PNG_INFO_tRNS)) {
226 png_set_tRNS_to_alpha(png_ptr);
227 didTransformations = true;
228 }
229
230 /* Update the info_ptr. Can only be called once! */
231 if (didTransformations) {
232 png_read_update_info(png_ptr, png_info_ptr);
233 pngBitDepth = png_get_bit_depth(png_ptr, png_info_ptr);
234 color_type = png_get_color_type(png_ptr, png_info_ptr);
235 }
236
237 /* Determine size of decompressed data */
238 const png_uint_32 rowbytes = png_get_rowbytes(png_ptr, png_info_ptr);
239 const uint32_t height = this->getDimensions().ySize;
240 Memory::AutoArray<png_bytep> row_pointers(height);
241 Memory::uint8Array rawData(rowbytes * height);
242
243 /* Tell libpng to store decompressed PNG data directly into AutoArray */
244 for (uint32_t row = 0; row < height; row++)
245 row_pointers[row] = rawData + (row * rowbytes);
246 png_read_image(png_ptr, row_pointers);
247
248 png_destroy_read_struct(&png_ptr, &png_info_ptr, nullptr);
249 return (rawData);
250}

260bool
261BiometricEvaluation::Image::PNG::isPNG(
262 const uint8_t *data,
263 uint64_t size)
264{
265 static const png_size_t PNG_SIG_LENGTH = 8;
266 if (size <= PNG_SIG_LENGTH)
267 return (false);
268
269 png_byte header[PNG_SIG_LENGTH];
270 std::memcpy(header, data, PNG_SIG_LENGTH);
271 return (png_sig_cmp(header, 0, PNG_SIG_LENGTH) == 0);
272}

274void
275png_read_mem_src(
276 png_structp png_ptr,
277 png_bytep buffer,
278 png_size_t length)
279{
280 // XXX: It appears that libpng calls this function in 4-byte
281 // increments, which can't be good for performance. See if there is
282 // a way to increase the chunk size. Ideally, one would set the chunk
283 // size to the size of the buffer since it is known ahead of time.
284
285 if (png_get_io_ptr(png_ptr) == nullptr)
286 throw BE::Error::StrategyError("Lost current offset while "
287 "reading from memory");
288
289 png_buffer *input = (png_buffer *)png_get_io_ptr(png_ptr);
290 if (length > (input->size - input->offset))
291 throw BE::Error::StrategyError("Attempted to read more data "
292 "than available");
293
294 memcpy(buffer, input->data + input->offset, length);
295 input->offset += length;
296}

298void
299png_error_callback(
300 png_structp png_ptr,
301 png_const_charp msg)
302{
303 const void *userData = png_get_error_ptr(png_ptr);
304 if (userData != nullptr) {
305 const BE::Image::PNG *png = static_cast<const BE::Image::PNG*>(
306 userData);
307 png->getStatusCallback()({BE::Framework::Status::Type::Error,
308 msg, png->getIdentifier()});
309 }
310
311 /*
312 * libpng cannot continue after errors, so if statusCallback won't
313 * throw, we will.
314 */
315 throw BE::Error::StrategyError(msg);
316}

329 png->getStatusCallback()({BE::Framework::Status::Type::Error, msg,
330 png->getIdentifier()});
331}
