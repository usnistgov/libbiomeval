42BiometricEvaluation::MPI::Distributor::Distributor(
43 const std::string &propertiesFileName)
44{
45 this->_resources = BE::Memory::make_unique<Resources>(
46 propertiesFileName);
47
48 /*
49 * Only create the real Logsheet and Checkpoint on the actual
50 * Distributor. MPI runtime will create a process for each rank,
51 * and those processes rank >= 1 contain the Distributor that does
52 * no work.
53 */
54 if (::MPI::COMM_WORLD.Get_rank() == 0) {
55 this->_logsheet =
56 BE::MPI::openLogsheet(
57 this->_resources->getLogsheetURL(),
58 "MPI::Distributor");
59
60 std::string chkFileName =
61 this->_resources->getCheckpointPath() + '/' +
62 BE::MPI::Distributor::CHECKPOINTFILENAME;
63 /*
64 * When checkpointing, the checkpoint file is optional for
65 * restore.
66 * On restore or save, update the checkpoint with items that
67 * are the responsibility of this class.
68 */
69 if (BE::MPI::checkpointEnable) {
70 if (BE::IO::Utility::fileExists(chkFileName)) {
71 BE::MPI::doCheckpointRestore = true;
72 } else {
73 BE::MPI::doCheckpointRestore = false;
74 }
75 /*
76 * Open the existing, or create a new checkpoint file
77 * with common data set by this class.
78 */
79 this->_checkpointData =
80 std::make_shared<BE::IO::PropertiesFile>(
81 chkFileName, IO::Mode::ReadWrite);
82 this->_checkpointData->setPropertyFromInteger(
83 BE::MPI::Distributor::CHECKPOINTPID, getpid());
84 this->_checkpointData->sync();
85 }
86 }

108void
109BiometricEvaluation::MPI::Distributor::start()
110{
111 if (BE::MPI::doCheckpointRestore) {
112 this->checkpointRestore();
113 }
114
115 /* Release other tasks to start up */
116 ::MPI::COMM_WORLD.Barrier();
117
118 /*
119 * Tell each child task to start requesting by sending an OK.
120 */
121 //XXX This should be asynchronous because some tasks
122 //XXX may have long startup times.
123 BE::IO::Logsheet *log = this->_logsheet.get();
124 MPI::logMessage(*log, "Sending messages to Task-N processes");
125 MPI::taskstat_t taskStatus;
126 for (int task = 1; task < this->_resources->getNumTasks(); task++) {
127 *log << "Tell Task-" << task << " to startup";
128 MPI::logEntry(*log);
129
130 MPI::taskcmd_t taskCmd =
131 to_int_type(MPI::TaskCommand::Continue);
132 ::MPI::COMM_WORLD.Send((void *)&taskCmd, 1, MPI_INT32_T,
133 task, to_int_type(MPI::MessageTag::Control));
134 ::MPI::COMM_WORLD.Recv(&taskStatus, 1, MPI_INT32_T,
135 task, to_int_type(MPI::MessageTag::Control));
136 if (taskStatus == to_int_type(MPI::TaskStatus::OK))
137 this->_activeMpiTasks.insert(task);
138 }
139 MPI::logMessage(*log, "Done sending start messages");
140
141 if (this->_activeMpiTasks.empty())
142 MPI::logMessage(*log, "No receiver tasks available");
143 else
144 this->distributeWork();
145
146 this->shutdown();
147}

149void
150BiometricEvaluation::MPI::Distributor::sendWorkPackage(
151 BE::MPI::WorkPackage &workPackage, int MPITask)
152{
153 /*
154 * Send three pieces of information:
155 * The raw data and length, in the first message;
156 * The number of elements in the second message.
157 */
158 BE::Memory::uint8Array data(0);
159 workPackage.getData(data);
160 int size = static_cast<int>(data.size());
161 ::MPI::COMM_WORLD.Send(
162 (void *)data, size, MPI_CHAR, MPITask,
163 to_int_type(BE::MPI::MessageTag::Data));
164
165 uint64_t numElements = workPackage.getNumElements();
166 ::MPI::COMM_WORLD.Send(
167 (void *)&numElements, 1, MPI_UINT64_T,
168 MPITask, to_int_type(BE::MPI::MessageTag::Data));
169
170 BE::IO::Logsheet *log = this->_logsheet.get();
171 std::ostringstream sstr;
172 sstr << "Sent package of size " << size << " to Task-" << MPITask;
173 MPI::logMessage(*log, sstr.str());
174}

176void
177BiometricEvaluation::MPI::Distributor::distributeWork()
178{
179 MPI::WorkPackage workPackage;
180 int numTasks = this->_activeMpiTasks.size();
181 auto taskStatus = BE::Memory::make_unique<MPI::taskstat_t[]>(numTasks);
182 auto indices = BE::Memory::make_unique<int[]>(numTasks);
183 auto MPIstatus = BE::Memory::make_unique<::MPI::Status[]>(numTasks);
184 auto requests = BE::Memory::make_unique<::MPI::Request[]>(numTasks);
185 int numRequests;
186 BE::IO::Logsheet *log = this->_logsheet.get();
187
188 /*
189 * Perform a non-blocking receive from all child tasks.
190 * This loop creates the initial set of receive requests,
191 * and in the loop below those requests are reposted for
192 * each Task that is sent a message.
193 */
194 int t = 0;
195 for (const auto &task : this->_activeMpiTasks) {
196 requests[t] = ::MPI::COMM_WORLD.Irecv(
197 &taskStatus[t], 1, MPI_INT32_T, task,
198 to_int_type(MPI::MessageTag::Control));
199 t++;
200 }
201
202 /*
203 * While there is work to be distributed, check for Exit conditions,
204 * gather up all work package requests fairly, dispatch work, etc.
205 */
206 bool haveWork = true;
207 MPI::taskcmd_t taskCmd;
208 while (haveWork) {
209
210 /*
211 * Check for exit signal conditions. The action
212 * taken for each condition will be done outside
213 * of this distribution loop.
214 */
215 if (BiometricEvaluation::MPI::Exit ||
216 BiometricEvaluation::MPI::QuickExit ||
217 BiometricEvaluation::MPI::TermExit) {
218 break;
219 }
220
221 /*
222 * Implement a fair message processing scheme, where all
223 * posted messages are checked in a non-blocking fashion
224 * as a set. Each MPI::Send() done by a Task must be matched
225 * with an MPI::Recv() (or Irecv(), etc.), so once a
226 * message is processed, we must call Irecv() for the Task
227 * so future messages will be processed.
228 */
229 numRequests = ::MPI::Request::Testsome(
230 numTasks, requests.get(), indices.get(), MPIstatus.get());
231 for (int r = 0; r < numRequests; r++) {
232 int task = MPIstatus[r].Get_source();
233 /*
234 * If the task says that it is done,
235 * then take it out of the list of
236 * active tasks.
237 */
238 *log << "Received ";
239 const auto ts = to_enum<MPI::TaskStatus>(
240 taskStatus[indices[r]]);
241 if ((ts == MPI::TaskStatus::Exit) ||
242 (ts == MPI::TaskStatus::Failed)) {
243 *log << "Exit/Failure from Task-" << task;
244 MPI::logEntry(*log);
245 this->_activeMpiTasks.erase(task);
246 continue;
247 } else if (ts == MPI::TaskStatus::
248 RequestJobTermination) {
249 *log << "Job termination request from Task-" <<
250 task;
251 MPI::logEntry(*log);
252 this->_activeMpiTasks.erase(task);
253 BE::MPI::TermExit = true;
254 continue;
255 }
256 *log << "OK from Task-" << task;
257 MPI::logEntry(*log);
258
259 this->createWorkPackage(workPackage);
260
261 /*
262 * If we are out of work, or in a shutdown
263 * condition, tell the task to ignore the
264 * reply. We need to do this so the
265 * communication send/recv pairs stay in sync.
266 */
267 if ((workPackage.getNumElements() == 0) ||
268 (BiometricEvaluation::MPI::Exit ||
269 BiometricEvaluation::MPI::QuickExit ||
270 BiometricEvaluation::MPI::TermExit)) {
271 taskCmd = to_int_type(MPI::TaskCommand::Ignore);
272 ::MPI::COMM_WORLD.Send(
273 (void *)&taskCmd, 1, MPI_INT32_T, task,
274 to_int_type(MPI::MessageTag::Control));
275 haveWork = false;
276 continue;
277 }
278 /*
279 * Tell the task to continue with the
280 * data coming in the next messages.
281 */
282 taskCmd = to_int_type(MPI::TaskCommand::Continue);
283 ::MPI::COMM_WORLD.Send((void *)&taskCmd, 1, MPI_INT32_T,
284 task, to_int_type(MPI::MessageTag::Control));
285
286 sendWorkPackage(workPackage, task);
287
288 /*
289 * Repost the non-blocking receive
290 * for the task just given work.
291 */
292 requests[indices[r]] = ::MPI::COMM_WORLD.Irecv(
293 &taskStatus[indices[r]], 1, MPI_INT32_T,
294 task, to_int_type(MPI::MessageTag::Control));
295 }
296 if (this->_activeMpiTasks.empty())
297 break;
298 }
299
300 /*
301 * Save the checkpoint when desired, on the forced, clean shutdown.
302 */
303 if (BE::MPI::checkpointEnable) {
304 if (BE::MPI::Exit) {
305 this->checkpointSave("Exit signal");
306 }
307 }
308
309 /*
310 * Send the Exit condition as an out-of-band message to
311 * all Task-N that are still asking for work.
312 * It's OK if Task-N never attempts to receive this message
313 * as it is of a different tag than the normal control
314 * messages that are kept in sync of the Send()/Recv() pairs.
315 * It is important that the job shutdown soon after this point.
316 */
317 if (BiometricEvaluation::MPI::Exit) {
318 taskCmd = to_int_type(MPI::TaskCommand::Exit);
319 } else if (BiometricEvaluation::MPI::QuickExit) {
320 taskCmd = to_int_type(MPI::TaskCommand::QuickExit);
321 } else if (BiometricEvaluation::MPI::TermExit) {
322 taskCmd = to_int_type(MPI::TaskCommand::TermExit);
323 }
324 for (const auto &task : this->_activeMpiTasks) {
325 ::MPI::COMM_WORLD.Isend(
326 (void *)&taskCmd, 1, MPI_INT32_T, task,
327 to_int_type(MPI::MessageTag::OOB));
328 }
329
330 /*
331 * Once all the work as been distributed, we must tell the
332 * Task there is no more work (via an Ignore message, and
333 * not call Irecv() for that task. This approach keeps the
334 * work message processing and shutdown message processing
335 * independent.
336 */
337 numRequests = ::MPI::Request::Testsome(
338 numTasks, requests.get(), indices.get(), MPIstatus.get());
339
340 /*
341 * MPI runtime will indicate when there are no postponed
342 * message requests in the set of requests.
343 */
344 taskCmd = to_int_type(MPI::TaskCommand::Ignore);
345 while (numRequests != MPI_UNDEFINED) {
346 for (int r = 0; r < numRequests; r++) {
347 int task = MPIstatus[r].Get_source();
348 const auto ts = to_enum<TaskStatus>(
349 taskStatus[indices[r]]);
350 if ((ts == MPI::TaskStatus::Exit) ||
351 (ts == MPI::TaskStatus::Failed)) {
352 *log << "Received Exit/Failure from Task-"
353 << task;
354 MPI::logEntry(*log);
355 this->_activeMpiTasks.erase(task);
356 } else {
357 ::MPI::COMM_WORLD.Send(
358 (void *)&taskCmd, 1,
359 MPI_INT32_T, task,
360 to_int_type(MPI::MessageTag::Control));
361 }
362 }
363 numRequests = ::MPI::Request::Testsome(
364 numTasks, requests.get(), indices.get(), MPIstatus.get());
365 }
366}

368void
369BiometricEvaluation::MPI::Distributor::shutdown()
370{
371 BE::IO::Logsheet *log = this->_logsheet.get();
372
373 /*
374 * Check the exit signal conditions and take appropriate action.
375 * On Exit, we let the Task-N processes to finish the work package.
376 * On QuickExit, tell Task-N to stop be allow children to finish.
377 * On TermExit, tell Task-N to stop and kill off all children.
378 */
379 MPI::taskcmd_t taskCmd;
380 *log << "Distribution end: ";
381 if (BiometricEvaluation::MPI::Exit) {
382 *log << "Exit signal";
383 taskCmd = to_int_type(MPI::TaskCommand::Exit);
384 } else if (BiometricEvaluation::MPI::QuickExit) {
385 *log << "Quick Exit signal";
386 taskCmd = to_int_type(MPI::TaskCommand::QuickExit);
387 } else if (BiometricEvaluation::MPI::TermExit) {
388 *log << "Termination Exit signal";
389 taskCmd = to_int_type(MPI::TaskCommand::TermExit);
390 } else {
391 *log << "Work completed";
392 taskCmd = to_int_type(MPI::TaskCommand::Exit);
393 }
394 MPI::logEntry(*log);
395
396 /*
397 * Wait for each child task to ask for more work, then
398 * then tell them to exit.
399 */
400 MPI::taskstat_t taskStatus;
401 ::MPI::Status MPIstatus;
402 while(!this->_activeMpiTasks.empty()) {
403
404 /* Wait for the receive of the work request */
405 ::MPI::COMM_WORLD.Recv(&taskStatus, 1, MPI_INT32_T,
406 MPI_ANY_SOURCE, to_int_type(MPI::MessageTag::Control),
407 MPIstatus);
408
409 /* Tell the task to exit */
410 int task = MPIstatus.Get_source();
411 ::MPI::COMM_WORLD.Send(
412 (void *)&taskCmd, 1, MPI_INT32_T, task,
413 to_int_type(MPI::MessageTag::Control));
414
415 this->_activeMpiTasks.erase(task);
416
417 *log << "Sent exit command to Task-" << task;
418 MPI::logEntry(*log);
419 }
420
421 /* Wait for other tasks to start the shut down */
422 ::MPI::COMM_WORLD.Barrier();
423
424 /*
425 * Wait for all tasks to send a final message even if
426 * they've done no receiving of work.
427 */
428 ::MPI::Status mpiStatus;
429 for (int task = 1; task < this->_resources->getNumTasks(); task++) {
430 ::MPI::COMM_WORLD.Recv(&taskStatus, 1, MPI_INT32_T,
431 MPI_ANY_SOURCE, to_int_type(MPI::MessageTag::Control),
432 mpiStatus);
433 *log << "Received " << to_enum<TaskStatus>(taskStatus) << " " <<
434 "from Task-" << mpiStatus.Get_source();
435 MPI::logEntry(*log);
436 }
437 /*
438 * If all the work has been distributed, remove the checkpoint file.
439 */
440 if ((BE::MPI::checkpointEnable) &&
441 !BE::MPI::Exit) {
442 std::string chkFileName =
443 this->_resources->getCheckpointPath() + '/' +
444 BE::MPI::Distributor::CHECKPOINTFILENAME;
445 this->_checkpointData.reset();
446 unlink(chkFileName.c_str());
447 }
448}
