29void
30BiometricEvaluation::MPI::CSVProcessor::processWorkPackage(
31 BiometricEvaluation::MPI::WorkPackage &workPackage)
32{
33 /*
34 * Extract the key/value data from the work package
35 */
36 Memory::uint8Array packageData(0);
37 workPackage.getData(packageData);
38 uint64_t numElements = workPackage.getNumElements();
39
40 /*
41 * Call the implementation's record processor function
42 * for each key.
43 */
44 uint64_t index = 0;
45 Memory::uint8Array value(0);
46 for (uint64_t count = 0; count < numElements; count++) {
47
48 /*
49 * Read the lineNum, line length, value size, create a string
50 * from the characters of that length, then create the data
51 * array if size is non-zero.
52 */
53
54 uint64_t *pInt64 = (uint64_t *)&packageData[index];
55 uint64_t lineNum = *pInt64;
56 index += sizeof(uint64_t);
57
58 pInt64 = (uint64_t *)&packageData[index];
59 uint64_t lineLength = *pInt64;
60 index += sizeof(uint64_t);
61
62 std::string line;
63 if (lineLength > 0) {
64 line = std::string((char *)&packageData[index],
65 lineLength);
66 index += lineLength;
67 } else
68 line = "";
69
70 /*
71 * Stop processing only when a quick or immediate exit
72 * condition exists. On a normal exit, we are allowed to
73 * finish with the work package.
74 */
75 if (MPI::QuickExit || MPI::TermExit) {
76 IO::Logsheet *log = this->getLogsheet().get();
77 log->writeDebug("Early exit: End record processing");
78 break;
79 }
80 try {
81 if (lineLength > 0)
82 this->processLine(lineNum, line);
83 /*
84 * The record processor is asking for termination.
85 * Rethrow the exception so the framekwork will start
86 * the shutdown.
87 */
88 } catch (const Error::Exception&) {
89 throw;
90 }
91 }
92}
