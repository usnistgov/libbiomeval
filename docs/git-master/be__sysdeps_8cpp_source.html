31 char*
32 call_libgen_method(
33 const char *p,
34 const Method method)
35 {
36 static char drive[_MAX_DRIVE];
37 static char dir[_MAX_DIR];
38 static char fname[_MAX_FNAME];
39 static char ext[_MAX_EXT];
40 static char path[_MAX_PATH];
41
42 std::memset(drive, '\0', _MAX_DRIVE);
43 std::memset(dir, '\0', _MAX_DIR);
44 std::memset(fname, '\0', _MAX_FNAME);
45 std::memset(ext, '\0', _MAX_EXT);
46 std::memset(path, '\0', _MAX_PATH);
47
48 const errno_t err = _splitpath_s<_MAX_DRIVE, _MAX_DIR,
49 _MAX_FNAME, _MAX_EXT>(p, drive, dir, fname, ext);
50 if (err == 0) {
51 switch (method) {
52 case Method::basename:
53 /* If no basename, it's dirname */
54 if (fname[0] == '\0') {
55 if (dir[0] == '\0')
56 strncpy(dir, ".", 2);
57 strncpy(fname, dir, 2);
58 }
59
60 if (_makepath_s<_MAX_PATH>(path,
61 nullptr, nullptr, fname, ext) != 0)
62 return (nullptr);
63 break;
64 case Method::dirname:
65 if (dir[0] == '\0')
66 strncpy(dir, ".", 2);
67
68 if (_makepath_s<_MAX_PATH>(path, drive,
69 dir, nullptr, nullptr) != 0)
70 return (nullptr);
71 break;
72 }
73
74 /* We do not want trailing slashes */
75 for (int i = strnlen(path, _MAX_PATH) - 1;
76 i > 0; --i) {
77 if (path[i] == '\\' ||
78 path[i] == '/')
79 path[i] = '\0';
80 else
81 break;
82 }
83
84 return (path);
85 }
86
87 return (nullptr);
88 }
89 }
90}

100char*
101dirname(
102 const char* p)
103{
104 /*
105 * On Windows, a trailing slash changes dirname. We want to mimic
106 * Linux, so remove training slashes.
107 */
108 std::string noSlash{p};
109 while ((noSlash.length() > 1) &&
110 ((noSlash.back() == '\\') || (noSlash.back() == '/')))
111 noSlash.pop_back();
112
113 return (BiometricEvaluation::libgen::call_libgen_method(
114 noSlash.c_str(), BiometricEvaluation::libgen::Method::dirname));
115}

178int gettimeofday(struct timeval* tp, struct timezone*)
179{
180 static const uint64_t EPOCH = ((uint64_t)116444736000000000ULL);
181
182 SYSTEMTIME systemTime;
183 FILETIME fileTime;
184 uint64_t time;
185
186 GetSystemTime(&systemTime);
187 SystemTimeToFileTime(&systemTime, &fileTime);
188 time = ((uint64_t)fileTime.dwLowDateTime);
189 time += ((uint64_t)fileTime.dwHighDateTime) << 32;
190
191 tp->tv_sec = (long)((time - EPOCH) / 10000000L);
192 tp->tv_usec = (long)(systemTime.wMilliseconds * 1000);
193
194 return (0);
195}
