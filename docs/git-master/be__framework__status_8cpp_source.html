29std::string
30BiometricEvaluation::Framework::to_string(
31 const Status &status)
32{
33 std::string prefix{::to_string(status.getType())};
34 std::for_each(prefix.begin(), prefix.end(), [](char &c) {
35 c = std::toupper(c);
36 });
37 std::string s{'[' + prefix + ']'};
38
39 const auto message = status.getMessage();
40 if (!message.empty())
41 s += ' ' + message;
42
43 const auto identifier = status.getIdentifier();
44 if (!identifier.empty())
45 s += " (" + identifier + ')';
46
47 return (s);
48}
