44uint64_t
45BiometricEvaluation::Memory::IndexedBuffer::scan(
46 void *buf,
47 uint64_t len)
48{
49 if (_index + len > _size)
50 throw (Error::DataError("Can't read beyond end of buffer"));
51 if (buf != nullptr)
52 (void)memcpy(buf, _data + _index, len);
53 _index += len;
54 return (len);
55}
