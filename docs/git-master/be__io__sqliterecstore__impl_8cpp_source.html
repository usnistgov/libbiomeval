41/*
42 * The maximum record size supported by the underlying SQLite file is
43 * 2^30 by default, and never larger than 2^31. This class will break
44 * larger records up into multiple key/value pairs, creating the new
45 * keys using a reserved key character.
46 *
47 * NOTE:
48 * The maximum size of a record can change when SQLite is compiled, so
49 * the value below is set at the default. V2 allows one to lookup the maximum,
50 * but this will be different system to system. Examine the maximum for your
51 * system if SQLiteRecordStore appears to be truncating data read from
52 * large records created on another system.
53 * NOTE:
54 * replace() could perform an UPDATE, but with segmented records, it is much
55 * simpler to remove() all old bits and insert() new ones, so this class
56 * depends on the parent's implementation to replace().
57 */
58static const uint64_t MAX_REC_SIZE = (uint64_t)1000000000U;

140BiometricEvaluation::IO::SQLiteRecordStore::Impl::move(
141 const std::string &pathname)
142{
143 if (getMode() == Mode::ReadOnly)
144 throw Error::StrategyError("RecordStore was opened read-only");
145
146 this->cleanup();
147
148 std::string oldDBName, newDBName;
149 /*
150 * Preserve the old name of the DB files, which is based on the
151 * directory name.
152 */
153 oldDBName = BE::Text::basename(this->getDBFilename());
154 RecordStore::Impl::move(pathname);
155
156 /*
157 * The DB files are now in the new directory name.
158 */
159 oldDBName = pathname + '/' + oldDBName;
160 newDBName = pathname + '/' + BE::Text::basename(pathname);
161 if (rename(oldDBName.c_str(), newDBName.c_str()))
162 throw Error::StrategyError("sqlite3: Could not rename "
163 "database (" + Error::errorStr() + ")");
164
165 this->_dbname = this->getDBFilename();
166 if (!IO::Utility::fileExists(this->_dbname))
167 throw Error::StrategyError("sqlite3: Database " + _dbname +
168 "does not exist");
169
170#ifdef SQLITE_V2_SUPPORT
171 int32_t rv = sqlite3_open_v2(this->_dbname.c_str(), &this->_db,
172 SQLITE_OPEN_READWRITE | SQLITE_OPEN_FULLMUTEX, nullptr);
173#else
174 int32_t rv = sqlite3_open(this->_dbname.c_str(), &this->_db);
175#endif
176 if ((rv != SQLITE_OK) || (this->_db == nullptr))
177 sqliteError(rv);

192void
193BiometricEvaluation::IO::SQLiteRecordStore::Impl::insert(
194 const std::string &key,
195 const void *const data,
196 const uint64_t size)
197{
198 if (getMode() == Mode::ReadOnly)
199 throw Error::StrategyError("RecordStore was opened read-only");
200 if (!validateKeyString(key))
201 throw Error::StrategyError("Invalid key format");
202
203 /* Warn if key is already in database */
204 try {
205 this->length(key);
206 throw Error::ObjectExists(key);
207 } catch (const Error::ObjectDoesNotExist&) {}
208
209 sqlite3_stmt *statement = nullptr;
210 std::string activeTable = PRIMARY_KV_TABLE;
211 uint64_t segnum = 0;
212 uint64_t remSize = size, bindSize = 0;
213 uint8_t *bindData = (uint8_t *)data;
214 while ((remSize > 0) ||
215 ((remSize == 0) && (segnum < KEY_SEGMENT_START))) {
216 std::string sqlCommand = "INSERT INTO " + activeTable + " " +
217 "VALUES ('" + genKeySegName(key, segnum) +
218 "', $value)";
219
220 /* Prepare the statement */
221#ifdef SQLITE_V2_SUPPORT
222 int32_t rv = sqlite3_prepare_v2(_db, sqlCommand.c_str(),
223 sqlCommand.length(), &statement, nullptr);
224#else
225 int32_t rv = sqlite3_prepare(_db, sqlCommand.c_str(),
226 sqlCommand.length(), &statement, nullptr);
227#endif
228 if (rv != SQLITE_OK) {
229 sqlite3_finalize(statement);
230 sqliteError(rv);
231 }
232 if (statement == nullptr)
233 throw Error::StrategyError("SQLite: Could not allocate "
234 "statement");
235
236 /* Bind data to the statement, segmenting if necessary */
237 if (remSize < MAX_REC_SIZE) {
238 bindSize = remSize;
239 remSize = 0;
240 } else {
241 bindSize = MAX_REC_SIZE;
242 remSize -= MAX_REC_SIZE;
243 }
244 rv = sqlite3_bind_blob(statement,
245 sqlite3_bind_parameter_index(statement, "$value"),
246 bindData, bindSize, SQLITE_STATIC);
247 if (rv != SQLITE_OK) {
248 sqlite3_finalize(statement);
249 sqliteError(rv);
250 }
251
252 /* Execute the statement */
253 rv = sqlite3_step(statement);
254 if (rv != SQLITE_DONE) {
255 sqlite3_finalize(statement);
256 sqliteError(rv);
257 }
258
259 /* Free the statement */
260 rv = sqlite3_finalize(statement);
261 if (rv != SQLITE_OK)
262 sqliteError(rv);
263
264 /* Increment data position and segment */
265 bindData += bindSize;
266 switch (segnum) {
267 case 0:
268 segnum = KEY_SEGMENT_START;
269 activeTable = SUBORDINATE_KV_TABLE;
270 break;
271 default:
272 segnum++;
273 break;
274 }
275 }
276
277 /* Propagate to parent class */
278 RecordStore::Impl::insert(key, data, size);
279}

281void
282BiometricEvaluation::IO::SQLiteRecordStore::Impl::remove(
283 const std::string &key)
284{
285 if (getMode() == Mode::ReadOnly)
286 throw Error::StrategyError("RecordStore was opened read-only");
287 if (!validateKeyString(key))
288 throw Error::StrategyError("Invalid key format");
289
290 sqlite3_stmt *statement;
291 std::string activeTable = PRIMARY_KV_TABLE;
292 int64_t segnum = 0;
293 bool moreSegments = true;
294 while (moreSegments) {
295 std::string sqlCommand = "DELETE FROM " + activeTable + " " +
296 "WHERE " + KEY_COL + " = '" + genKeySegName(key, segnum) +
297 "'";
298
299 /* Prepare the statement */
300#ifdef SQLITE_V2_SUPPORT
301 int32_t rv = sqlite3_prepare_v2(_db, sqlCommand.c_str(),
302 sqlCommand.length(), &statement, nullptr);
303#else
304 int32_t rv = sqlite3_prepare(_db, sqlCommand.c_str(),
305 sqlCommand.length(), &statement, nullptr);
306#endif
307 if (rv != SQLITE_OK) {
308 sqlite3_finalize(statement);
309 sqliteError(rv);
310 }
311 if (statement == nullptr)
312 throw Error::StrategyError("SQLite: Could not allocate "
313 "statement");
314
315 /* Execute the statement */
316 rv = sqlite3_step(statement);
317 if (rv != SQLITE_DONE) {
318 sqlite3_finalize(statement);
319 sqliteError(rv);
320 }
321
322 /* Free the statement */
323 rv = sqlite3_finalize(statement);
324 if (rv != SQLITE_OK)
325 sqliteError(rv);
326
327 /* Increment segment number */
328 switch (segnum) {
329 case 0:
330 /* Check if any rows were actually deleted */
331 if (sqlite3_changes(_db) == 0)
332 throw Error::ObjectDoesNotExist(key);
333
334 segnum = KEY_SEGMENT_START;
335 activeTable = SUBORDINATE_KV_TABLE;
336 break;
337 default:
338 /* Check if there could be more segments */
339 if (sqlite3_changes(_db) == 0)
340 moreSegments = false;
341 else {
342 moreSegments = true;
343 segnum++;
344 }
345
346 break;
347 }
348 }
349
350 /* Propagate changes to parent */
351 RecordStore::Impl::remove(key);
352}

373uint64_t
374BiometricEvaluation::IO::SQLiteRecordStore::Impl::readSegments(
375 const std::string &key,
376 void * const data)
377 const
378{
379 if (!validateKeyString(key))
380 throw Error::StrategyError("Invalid key format");
381
382 sqlite3_stmt *statement;
383 uint64_t segnum = 0;
384 uint64_t totalBytes = 0, segBytes;
385 std::string activeTable = PRIMARY_KV_TABLE;
386 uint8_t *dataPtr = (uint8_t *)data;
387 bool moreSegments = true;
388 while (moreSegments) {
389 std::string sqlCommand = "SELECT " + VALUE_COL + " FROM " +
390 activeTable + " WHERE " + KEY_COL + " = '" +
391 genKeySegName(key, segnum) + "' LIMIT 1";
392
393 /* Prepare the statement */
394#ifdef SQLITE_V2_SUPPORT
395 int32_t rv = sqlite3_prepare_v2(_db, sqlCommand.c_str(),
396 sqlCommand.length(), &statement, nullptr);
397#else
398 int32_t rv = sqlite3_prepare(_db, sqlCommand.c_str(),
399 sqlCommand.length(), &statement, nullptr);
400#endif
401 if (rv != SQLITE_OK) {
402 sqlite3_finalize(statement);
403 sqliteError(rv);
404 }
405 if (statement == nullptr)
406 throw Error::StrategyError("SQLite: Could not allocate "
407 "statement");
408
409 /* Execute the statement */
410 rv = sqlite3_step(statement);
411 switch (segnum) {
412 case 0:
413 if (rv != SQLITE_ROW) {
414 sqlite3_finalize(statement);
415 throw Error::ObjectDoesNotExist(key);
416 }
417 /* FALLTHROUGH */
418 default:
419 segBytes = sqlite3_column_bytes(statement, 0);
420 totalBytes += segBytes;
421
422 if (data != nullptr) {
423 memcpy(dataPtr,
424 sqlite3_column_blob(statement, 0),
425 segBytes);
426 dataPtr += segBytes;
427 break;
428 }
429 }
430
431 /* Free the statement */
432 rv = sqlite3_finalize(statement);
433 if (rv != SQLITE_OK)
434 sqliteError(rv);
435
436 /* Increment segment number if there's more data */
437 if (segBytes == MAX_REC_SIZE) {
438 switch (segnum) {
439 case 0:
440 segnum = KEY_SEGMENT_START;
441 activeTable = SUBORDINATE_KV_TABLE;
442 break;
443 default:
444 segnum++;
445 break;
446 }
447 moreSegments = true;
448 } else
449 moreSegments = false;
450 }
451
452 return (totalBytes);
453}

470BiometricEvaluation::IO::RecordStore::Record
471BiometricEvaluation::IO::SQLiteRecordStore::Impl::i_sequence(
472 bool returnData,
473 int cursor)
474{
475 if ((cursor != BE_RECSTORE_SEQ_START) &&
476 (cursor != BE_RECSTORE_SEQ_NEXT))
477 throw Error::StrategyError("Invalid cursor position as "
478 "argument");
479
480 int32_t rv;
481 if ((cursor == BE_RECSTORE_SEQ_START) || (_sequencer == nullptr)) {
482 std::string sqlCommand = "SELECT *,ROWID FROM " +
483 PRIMARY_KV_TABLE + " " + "ORDER BY ROWID";
484
485 /* Finalize previous statement */
486 rv = sqlite3_finalize(_sequencer);
487 if (rv != SQLITE_OK)
488 sqliteError(rv);
489
490 /* Prepare the statement */
491#ifdef SQLITE_V2_SUPPORT
492 rv = sqlite3_prepare_v2(_db, sqlCommand.c_str(),
493 sqlCommand.length(), &_sequencer, nullptr);
494#else
495 rv = sqlite3_prepare(_db, sqlCommand.c_str(),
496 sqlCommand.length(), &_sequencer, nullptr);
497#endif
498 if ((rv != SQLITE_OK) || (_sequencer == nullptr))
499 sqliteError(rv);
500 _sequenceEnd = false;
501 }
502
503 /* Must reset sequencer first if we've exhausted the list */
504 if (_sequenceEnd == true)
505 throw Error::ObjectDoesNotExist();
506
507 /*
508 * Must reselect cursor at sequence to protect against other methods
509 * modifying the database between setCursorAtKey() and sequence().
510 */
511 if (_cursorRow != 0) {
512 /* Finalize previous sequence() SELECT */
513 rv = sqlite3_finalize(_sequencer);
514 if (rv != SQLITE_OK)
515 sqliteError(rv);
516
517 std::stringstream sqlCommand;
518 sqlCommand << "SELECT *,ROWID FROM " <<
519 PRIMARY_KV_TABLE << " " << "WHERE ROWID >= " <<
520 _cursorRow << " ORDER BY ROWID";
521 _cursorRow = 0;
522
523 /* Prepare the statement */
524#ifdef SQLITE_V2_SUPPORT
525 rv = sqlite3_prepare_v2(_db, sqlCommand.str().c_str(),
526 sqlCommand.str().length(), &_sequencer, nullptr);
527#else
528 rv = sqlite3_prepare(_db, sqlCommand.str().c_str(),
529 sqlCommand.str().length(), &_sequencer, nullptr);
530#endif
531 if ((rv != SQLITE_OK) || (_sequencer == nullptr))
532 sqliteError(rv);
533 }
534
535 /* Execute the statement */
536 rv = sqlite3_step(_sequencer);
537
538 /* End of entries */
539 BiometricEvaluation::IO::RecordStore::Record record;
540 switch (rv) {
541 case SQLITE_ROW: {
542 record.key.assign(
543 (const char *)sqlite3_column_text(_sequencer, 0));
544 uint64_t bytes = sqlite3_column_bytes(_sequencer, 1);
545 if (returnData) {
546 record.data.resize(bytes);
547 record.data.copy(
548 (uint8_t *)sqlite3_column_blob(_sequencer, 1),
549 bytes);
550 }
551 break;
552 } case SQLITE_DONE:
553 _sequenceEnd = true;
554 throw Error::ObjectDoesNotExist();
555
556 /* Not reached */
557 break;
558 default:
559 /* Free the statement */
560 rv = sqlite3_finalize(_sequencer);
561 if (rv != SQLITE_OK)
562 sqliteError(rv);
563
564 /* Not reached */
565 break;
566 }
567 return(record);
568}

586void
587BiometricEvaluation::IO::SQLiteRecordStore::Impl::setCursorAtKey(
588 const std::string &key)
589{
590 if (!validateKeyString(key))
591 throw Error::StrategyError("Invalid key format");
592
593 int32_t rv;
594
595 sqlite3_stmt *statement;
596 std::string sqlCommand = "SELECT ROWID FROM " + PRIMARY_KV_TABLE + " " +
597 "WHERE " + KEY_COL + " = '" + key + "'";
598
599 /* Prepare the statement */
600#ifdef SQLITE_V2_SUPPORT
601 rv = sqlite3_prepare_v2(_db, sqlCommand.c_str(), sqlCommand.length(),
602 &statement, nullptr);
603#else
604 rv = sqlite3_prepare(_db, sqlCommand.c_str(), sqlCommand.length(),
605 &statement, nullptr);
606#endif
607 if ((rv != SQLITE_OK) || (statement == nullptr))
608 sqliteError(rv);
609
610 /* Execute the statement */
611 rv = sqlite3_step(statement);
612
613 /* End of entries */
614 switch (rv) {
615 case SQLITE_ROW: {
616 _cursorRow = (uint64_t)strtoll(
617 (const char *)sqlite3_column_text(statement, 0), nullptr,
618 10);
619
620 rv = sqlite3_finalize(statement);
621 if (rv != SQLITE_OK)
622 sqliteError(rv);
623 break;
624 } case SQLITE_DONE:
625 rv = sqlite3_finalize(statement);
626 if (rv != SQLITE_OK)
627 sqliteError(rv);
628 throw Error::ObjectDoesNotExist();
629
630 /* Not reached */
631 break;
632 default:
633 rv = sqlite3_finalize(statement);
634 if (rv != SQLITE_OK)
635 sqliteError(rv);
636 throw Error::StrategyError();
637
638 /* Not reached */
639 break;
640 }
641
642 /* Reset cursor */
643 _sequenceEnd = false;
644}

731bool
732BiometricEvaluation::IO::SQLiteRecordStore::Impl::validateKeyValueTable(
733 const std::string &table)
734{
735 sqlite3_stmt *statement;
736
737 /* Schema is good if there exists key and value columns */
738 std::string sqlCommand = "SELECT " + KEY_COL + "," + VALUE_COL +
739 " FROM " + table + " LIMIT 1";
740#ifdef SQLITE_V2_SUPPORT
741 int32_t rv = sqlite3_prepare_v2(_db, sqlCommand.c_str(),
742 sqlCommand.length(), &statement, nullptr);
743#else
744 int32_t rv = sqlite3_prepare(_db, sqlCommand.c_str(),
745 sqlCommand.length(), &statement, nullptr);
746#endif
747 if ((rv != SQLITE_OK) || (statement == nullptr))
748 sqliteError(rv);
749
750 /* Execute the statement */
751 bool valid = false;
752 rv = sqlite3_step(statement);
753 if (rv == SQLITE_DONE || rv == SQLITE_ROW)
754 valid = true;
755
756 /* Free the statement */
757 rv = sqlite3_finalize(statement);
758 if (rv != SQLITE_OK)
759 sqliteError(rv);
760
761 return (valid);
762}
