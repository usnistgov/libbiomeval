47BiometricEvaluation::MPI::Resources::Resources(
48 const std::string &propertiesFileName)
49{
50 this->_propertiesFileName = propertiesFileName;
51 this->_rank = ::MPI::COMM_WORLD.Get_rank();
52 this->_numTasks = ::MPI::COMM_WORLD.Get_size();
53
54 /* Read the properties file */
55 std::unique_ptr<IO::PropertiesFile> props;
56 try {
57 props.reset(new IO::PropertiesFile(propertiesFileName,
58 IO::Mode::ReadOnly));
59 } catch (Error::Exception &e) {
60 throw Error::FileError("Could not open properties: " +
61 e.whatString());
62 }
63 /*
64 * Required properties.
65 */
66 try {
67 auto wpn = props->getProperty(
68 MPI::Resources::WORKERSPERNODEPROPERTY);
69 if (BE::Text::caseInsensitiveCompare(wpn, NUMCPUS)) {
70 this->_workersPerNode = BE::System::getCPUCount();
71 } else if (BE::Text::caseInsensitiveCompare(wpn, NUMCORES)) {
72 this->_workersPerNode = BE::System::getCPUCoreCount();
73 } else if (BE::Text::caseInsensitiveCompare(wpn, NUMSOCKETS)) {
74 this->_workersPerNode = BE::System::getCPUSocketCount();
75 } else {
76 this->_workersPerNode = props->getPropertyAsInteger(
77 MPI::Resources::WORKERSPERNODEPROPERTY);
78 }
79 } catch (Error::Exception &e) {
80 throw Error::ObjectDoesNotExist("Could not read properties: " +
81 e.whatString());
82 }
83
84 /*
85 * Optional properties.
86 */
87 try {
88 this->_logsheetURL =
89 props->getProperty(MPI::Resources::LOGSHEETURLPROPERTY);
90 } catch (Error::Exception &e) {
91 this->_logsheetURL = "";
92 }
93 /*
94 * Checkpoint path is only required when checkpointing
95 * is done.
96 */
97 try {
98 this->_checkpointPath =
99 props->getProperty(MPI::Resources::CHECKPOINTPATHPROPERTY);
100 } catch (Error::Exception &e) {
101 if (MPI::checkpointEnable) {
102 throw Error::ObjectDoesNotExist(
103 "Could not read " +
104 MPI::Resources::CHECKPOINTPATHPROPERTY);
105 } else {
106 this->_checkpointPath = "";
107 }
108 }
109}
