19/*
20 * Setup access to the container stream using FFMPEG libraries.
21 */
22void
23BiometricEvaluation::Video::Container::Impl::openContainer()
24{
25 this->_fmtCtx = avformat_alloc_context();
26 if (this->_fmtCtx == nullptr)
27 throw BE::Error::MemoryError("Could not allocate format context");
28 /* fill opaque structure used by the AVIOContext read callback */
29 this->_IOCtxBufferData.ptr = *this->_containerBuf.get();
30 this->_IOCtxBufferData.size = this->_containerBuf->size();
31 this->_IOCtxBufferData.pos = 0;
32
33 uint8_t *ctxBuf = nullptr;
34 ctxBuf = (uint8_t *)av_malloc(BE::Video::AVIOCTXBUFFERSIZE);
35 if (ctxBuf == nullptr)
36 throw BE::Error::MemoryError(
37 "Could not allocate IO context buffer");
38
39 this->_avioCtx = avio_alloc_context(
40 ctxBuf, AVIOCTXBUFFERSIZE,
41 0, &this->_IOCtxBufferData, &read_packet, NULL, &seek);
42 if (this->_avioCtx == nullptr) {
43 av_free(ctxBuf);
44 throw BE::Error::MemoryError(
45 "Could not allocate IO context");
46 }
47
48 this->_fmtCtx->pb = this->_avioCtx;
49
50 int ret = avformat_open_input(&this->_fmtCtx, NULL, NULL, NULL);
51 if (ret < 0)
52 throw (Error::StrategyError("Could not read container"));
53
54 ret = avformat_find_stream_info(this->_fmtCtx, NULL);
55 if (ret < 0)
56 throw (Error::StrategyError(
57 "Could not find stream information"));
58}

60/*
61 * Construct this object by opening the container then counting audio/video
62 * streams.
63 */
64void
65BiometricEvaluation::Video::Container::Impl::construct()
66{
67 av_register_all();
68 this->openContainer();
69 this->_audioCount = 0;
70 this->_videoCount = 0;
71 for(unsigned int i = 0; i < this->_fmtCtx->nb_streams; i++) {
72 if (this->_fmtCtx->streams[i]->codec->codec_type ==
73 AVMEDIA_TYPE_VIDEO) {
74 this->_videoCount++;
75 }
76 if (this->_fmtCtx->streams[i]->codec->codec_type ==
77 AVMEDIA_TYPE_AUDIO) {
78 this->_audioCount++;
79 }
80 }
81}

83/*
84 * Tear down an open stream be releasing FFMPEG libary objects.
85 */
86void
87BiometricEvaluation::Video::Container::Impl::closeContainer()
88{
89 if (this->_fmtCtx != nullptr) {
90 avformat_close_input(&this->_fmtCtx);
91 avformat_free_context(this->_fmtCtx);
92 }
93 /*
94 * NOTE: The internal buffer could have changed, and not be
95 * the buffer we allocated. We don't need to free that buffer
96 * here.
97 */
98 if (this->_avioCtx != nullptr) {
99 av_freep(&this->_avioCtx->buffer);
100 av_freep(&this->_avioCtx);
101 }
102}

104/*
105 * Find a video stream by comparing each stream's type and index number.
106 * Caller must check that requested stream number doesn't exceed total
107 * number of video streams.
108 * Returns the stream index within the container.
109 */
110static uint32_t
111findVideoStream(
112 AVFormatContext *fmtCtx,
113 uint32_t videoNum)
114{
115 uint32_t videoCount = 0;
116 unsigned int stream;
117 for(stream = 0; stream < fmtCtx->nb_streams; stream++) {
118 if(fmtCtx->streams[stream]->codec->codec_type ==
119 AVMEDIA_TYPE_VIDEO) {
120 videoCount++;
121 if (videoCount == videoNum) {
122 break;
123 }
124 }
125 }
126 return (stream);
127}

164std::unique_ptr<BiometricEvaluation::Video::Stream>
165BiometricEvaluation::Video::Container::Impl::getVideoStream(
166 uint32_t videoNum)
167{
168 if ((videoNum == 0) || (videoNum > this->_videoCount))
169 throw Error::ParameterError("Requested stream not present");
170 uint32_t streamIndex = findVideoStream(this->_fmtCtx, videoNum);
171 std::unique_ptr<BiometricEvaluation::Video::Stream> ptr;
172 ptr.reset(new BE::Video::StreamImpl(streamIndex, this->_containerBuf));
