87bool
88BiometricEvaluation::Process::Worker::waitForMessage(
89 int numSeconds)
90 const
91{
92 bool result = false;
93
94 struct timeval timeout;
95 bool userTimeout;
96 if (numSeconds >= 0) {
97 timeout.tv_sec = numSeconds;
98 timeout.tv_usec = 0;
99 userTimeout = true;
100 } else {
101 timeout.tv_sec = 3;
102 timeout.tv_usec = 0;
103 userTimeout = false;
104 }
105
106 /*
107 * We need to handle the case where the signal that terminated
108 * the child did not interrupt the select call, thereby creating
109 * a race condition when the caller specified no timeout value
110 * and we sit in select forever.
111 * First, at the top of the loop, check whether we were
112 * requested to stop;
113 * Second, if there is no user timeout, set our own so the
114 * stop reqested flag can be checked, closing a small race window.
115 * In that case we don't exit the loop because the user wants to
116 * wait forever for a message, forever meaning until this process
117 * is told to stop asynchronously.
118 */
119 bool finished = false;
120 fd_set set;
121 FD_ZERO(&set);
122 FD_SET(_pipeToChild[0], &set);
123 struct timeval l_timeout = timeout;
124 while (!finished && !_stopRequested) {
125 int ret = select(_pipeToChild[0] + 1, &set,
126 nullptr, nullptr, &l_timeout);
127 if (ret == 0) {
128 /* Nothing available */
129 if (userTimeout) {
130 result = false;
131 finished = true;
132 } else {
133 FD_SET(_pipeToChild[0], &set);
134 /* Reset timeout as some systems change it */
135 l_timeout = timeout;
136 }
137 } else if (ret < 0) {
138 /* Could have been interrupted while blocking */
139 if (errno == EINTR) {
140 /* Give up if we need to exit anyway */
141 if (_stopRequested)
142 return (false);
143 } else {
144 finished = true;
145 }
146 } else {
147 /* Something available -- check what */
148 if (FD_ISSET(_pipeToChild[0], &set)) {
149 result = true;
150 finished = true;
151 } else {
152 result = false;
153 }
154 }
155 }
156 return (result);
157}

159void
160BiometricEvaluation::Process::Worker::sendMessageToManager(
161 const Memory::uint8Array &message)
162{
163 if (_communicationEnabled == false)
164 throw Error::StrategyError("Communication is not enabled");
165
166 /*
167 * Send the message length, then the message contents.
168 * All exceptions float out.
169 */
170 uint64_t length = message.size();
171 IO::Utility::writePipe(&length, sizeof(length), _pipeFromChild[1]);
172 IO::Utility::writePipe(message, _pipeFromChild[1]);
173}

184 message.resize(length);
185 IO::Utility::readPipe(message, _pipeToChild[0]);

212void
213BiometricEvaluation::Process::Worker::_initCommunication()
214{
215 if (_communicationEnabled == false) {
216 if (pipe(_pipeToChild) != 0)
217 throw Error::StrategyError("Could not create send "
218 "pipe ( " + Error::errorStr() + ")");
219 if (pipe(_pipeFromChild) != 0) {
220 close(_pipeToChild[0]);
221 close(_pipeToChild[1]);
222
223 throw Error::StrategyError("Could not create receive "
224 "pipe ( " + Error::errorStr() + ")");
225 }
226
227 _communicationEnabled = true;
228 }
229}
