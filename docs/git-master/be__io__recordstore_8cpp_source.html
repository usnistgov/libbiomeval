222void
223BiometricEvaluation::IO::RecordStoreIterator::setBegin()
224{
225 try {
226 std::string key = this->_recordStore->sequenceKey(
227 RecordStore::BE_RECSTORE_SEQ_START);
228 this->_recordStore->setCursorAtKey(key);
229 } catch (const Error::ObjectDoesNotExist&) {
230 this->setEnd();
231 }
232
233 this->step(1);
234}
235
236void
237BiometricEvaluation::IO::RecordStoreIterator::step(
238 BiometricEvaluation::IO::RecordStoreIterator::difference_type numSteps)
239{
240 /* Backwards */
241 if (numSteps <= 0)
242 return;
243
244 /* Forward one step */
245 if (numSteps == 1) {
246 try {
247 this->_currentRecord = this->_recordStore->sequence();
248 } catch (const Error::ObjectDoesNotExist&) {
249 this->setEnd();
250 }
251 return;
252 }
253
254 /* Forward 2..n steps */
255 std::string key;
256 for (difference_type i = 0; i < numSteps; i++) {
257 try {
258 key = this->_recordStore->sequenceKey();
259 } catch (const Error::ObjectDoesNotExist&) {
260 this->setEnd();
261 return;
262 }
263 }
264
265 Memory::uint8Array data;
266 data = this->_recordStore->read(key);
267 this->_currentRecord = RecordStore::Record(key, data);
268}
269
270void
271BiometricEvaluation::IO::RecordStoreIterator::setEnd()
272{
273 this->_atEnd = true;
274 this->_currentRecord = RecordStore::Record();
275}
