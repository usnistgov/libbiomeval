41void
42BiometricEvaluation::Time::Timer::stop()
43{
44 /* Get the time immediately */
45 const auto stopTime = BE_CLOCK_TYPE::now();
46
47 if (!this->_inProgress) {
48 throw Error::StrategyError("Timing not in progress");
49 }
50 this->_finish = stopTime;

54uint64_t
55BiometricEvaluation::Time::Timer::elapsed(
56 bool nano)
57 const
58{
59 if (this->_inProgress) {
60 throw Error::StrategyError("Timing in progress");
61 }
62
63 /*
64 * On some systems with some clocks, we may be losing precision by
65 * returning microseconds. Therefore, we must use a duration_cast
66 * instead of simply instantiating a microseconds object.
67 */
68 if (nano) {
69 return (std::chrono::duration_cast<std::chrono::nanoseconds>(
70 this->_finish - this->_start).count());
71 } else {
72 return (std::chrono::duration_cast<std::chrono::microseconds>(
73 this->_finish - this->_start).count());
74 }
75}
