21BiometricEvaluation::Image::WSQ::WSQ(
22 const uint8_t *data,
23 const uint64_t size,
24 const std::string &identifier,
25 const statusCallback_t &statusCallback) :
26 Image::Image(
27 data,
28 size,
29 CompressionAlgorithm::WSQ20,
30 identifier,
31 statusCallback)
32{
33 uint8_t *marker_buf = (uint8_t *)this->getDataPointer();
34 uint8_t *wsq_buf = marker_buf;
35
36 /* Read to the "start of image" marker */
37 uint16_t marker, tbl_size;
38 uint32_t rv = 0;
39 if ((rv = biomeval_nbis_getc_marker_wsq(&marker, SOI_WSQ, &marker_buf,
40 wsq_buf + size)))
41 throw Error::StrategyError("Could not read to SOI_WSQ");
42
43 /* Step through any tables up to the "start of frame" marker */
44 for (;;) {
45 if ((rv = biomeval_nbis_getc_marker_wsq(&marker, TBLS_N_SOF, &marker_buf,
46 wsq_buf + size)))
47 throw Error::StrategyError("Could not read to "
48 "TBLS_N_SOF");
49
50 if (marker == SOF_WSQ)
51 break;
52
53 if ((rv = biomeval_nbis_getc_ushort(&tbl_size, &marker_buf, wsq_buf + size)))
54 throw Error::StrategyError("Could not read size "
55 "of table");
56 /* Table size includes size of field but not the marker */
57 marker_buf += tbl_size - sizeof(tbl_size);
58 }
59
60 /* Read the frame header */
61 FRM_HEADER_WSQ wsq_header;
62 if ((rv = biomeval_nbis_getc_frame_header_wsq(&wsq_header, &marker_buf,
63 wsq_buf + size)))
64 throw Error::DataError("Could not read frame header");
65 setDimensions(Size(wsq_header.width, wsq_header.height));
66
67 /* Read PPI from NISTCOM, if present */
68 int ppi{-1};
69 if (biomeval_nbis_getc_ppi_wsq(&ppi, wsq_buf, size) == 0) {
70 /* Resolution does not have to be defined */
71 if (ppi == -1)
72 /* WSQ is a 500 ppi specification */
73 setResolution(Resolution(500, 500,
74 Resolution::Units::PPI));
75 else
76 setResolution(Resolution(ppi, ppi,
77 Resolution::Units::PPI));
78 } else {
79 /* WSQ is a 500 ppi specification */
80 setResolution(Resolution(500, 500, Resolution::Units::PPI));
81 }
82
83 /*
84 * "Source fingerprint images shall be captured with 8 bits of
85 * precision per pixel."
86 */
87 setColorDepth(8);
88 this->setBitDepth(8);
89
90 this->setHasAlphaChannel(false);
91}

106BiometricEvaluation::Memory::uint8Array
107BiometricEvaluation::Image::WSQ::getRawData()
108 const
109{
110 uint8_t *rawbuf = nullptr;
111 int32_t depth, height, lossy, ppi, rv, width;
112 if ((rv = biomeval_nbis_wsq_decode_mem(&rawbuf, &width, &height, &depth, &ppi,
113 &lossy, (unsigned char *)this->getDataPointer(),
114 this->getDataSize())))
115 throw Error::DataError("Could not convert WSQ to raw.");
116
117 /* rawbuf allocated within libwsq. Copy to manage with AutoArray. */
118 /* TODO: AutoBuffer-wrapped AutoArray */
119 Memory::uint8Array rawData(width * height * (depth / 8));
120 rawData.copy(rawbuf);
121 free(rawbuf);
122
123 return (rawData);
124}

134bool
135BiometricEvaluation::Image::WSQ::isWSQ(
136 const uint8_t *data,
137 uint64_t size)
138{
139 static const uint8_t WSQ_SOI[2] = {0xFF, 0xA0};
140 if (size < (sizeof(WSQ_SOI) / sizeof(uint8_t)))
141 return (false);
142
143 return (memcmp(data, WSQ_SOI, 2) == 0);
