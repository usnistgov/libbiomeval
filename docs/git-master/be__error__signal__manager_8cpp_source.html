91void
92BiometricEvaluation::Error::SignalManager::start()
93{
94 struct sigaction sa{};
95 sigemptyset(&sa.sa_mask);
96 sa.sa_flags = SA_SIGINFO;
97 sa.sa_sigaction = SignalManagerSighandler;
98 for (int sig = SIGHUP; sig <= SIGUSR2; sig++) {
99 if ((sig == SIGKILL) || (sig == SIGSTOP)) {
100 continue;
101 }
102 if (sigismember(&_signalSet, sig)) {
103 if (sigaction(sig, &sa, nullptr) == -1) {
104 throw (Error::StrategyError(
105 "Registering signal handler failed"));
106 }
107 }
108 }
109 _canSigJump = true;
110}

112void
113BiometricEvaluation::Error::SignalManager::stop()
114{
115 struct sigaction sa{};
116 sigemptyset(&sa.sa_mask);
117 sa.sa_flags = 0;
118 sa.sa_handler = SIG_DFL;
119 for (int sig = SIGHUP; sig <= SIGUSR2; sig++) {
120 if ((sig == SIGKILL) || (sig == SIGSTOP)) {
121 continue;
122 }
123 if (sigismember(&_signalSet, sig)) {
124 if (sigaction(sig, &sa, nullptr) == -1) {
125 throw (Error::StrategyError(
126 "Setting default signal handler failed"));
127 }
128 }
129 }
130 _canSigJump = false;
