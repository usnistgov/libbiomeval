89void
90BiometricEvaluation::MPI::CSVDistributor::createWorkPackage(
91 BiometricEvaluation::MPI::WorkPackage &workPackage)
92{
93 std::shared_ptr<BE::IO::Logsheet> log = this->getLogsheet();
94
95 /* Create the package data buffer at a reasonable starting size */
96 BE::Memory::uint8Array packageData(16384);
97
98 /*
99 * If there are no more keys to be read from the record store,
100 * send an empty work package.
101 */
102 if (this->_resources->getNumRemainingLines() == 0) {
103 workPackage.setNumElements(0);
104 workPackage.setData(packageData);
105 return;
106 }
107
108 /*
109 * Distribute a work package based on the chunk size given
110 * in the resources object. If a failure occurs reading a key,
111 * continue onto the next key. It is possible to send an empty
112 * work package due to sequential failures.
113 */
114 uint64_t lineCount;
115 if (this->_resources->getNumRemainingLines() >
116 this->_resources->getChunkSize())
117 lineCount = this->_resources->getChunkSize();
118 else
119 lineCount = this->_resources->getNumRemainingLines();
120
121 /*
122 * The value array must be 0-sized to start, and will stay that way
123 * if values are not to be sent.
124 */
125 BE::Memory::uint8Array::size_type index = 0;
126 uint64_t realLineCount = 0;
127
128 /*
129 * Pull lines from the file and combine a chunk of them into a
130 * single work package.
131 */
132 std::pair<uint64_t, std::string> lineData;
133 for (uint64_t n = 0; n < lineCount; n++) {
134 try {
135 lineData = this->_resources->readLine();
136 fillBufferWithTokens(packageData, lineData.first,
137 lineData.second, index);
138 } catch (BE::Error::Exception &e) {
139 log->writeDebug("Caught " + e.whatString());
140 continue;
141 }
142 realLineCount++;
143 }
144
145 /*
146 * NOTE: At this point it is possible to have no keys in the package.
147 */
148 this->_distributedLineCount += realLineCount;
149 workPackage.setNumElements(realLineCount);
150 workPackage.setData(packageData);
151}

153void
154BiometricEvaluation::MPI::CSVDistributor::checkpointSave(
155 const std::string &reason)
156{
157 try {
158 auto chkData = this->getCheckpointData();
159 chkData->setProperty(
160 BE::MPI::Distributor::CHECKPOINTREASON, reason);
161 chkData->setPropertyFromInteger(
162 BE::MPI::CSVDistributor::CHECKPOINTLINECOUNT,
163 this->_distributedLineCount);
164 /* Save the randomizer seed when present */
165 try {
166 auto seed = this->_resources->getRandomSeed();
167 chkData->setPropertyFromInteger(
168 BE::MPI::CSVDistributor::CHECKPOINTRANDOMSEED,
169 seed);
170 } catch (...) {
171 }
172 chkData->sync();
173 this->getLogsheet()->writeDebug("Checkpoint saved: " + reason);
174 } catch (Error::Exception &e) {
175 this->getLogsheet()->writeDebug(
176 "Checkpoint save: Caught " + e.whatString());
177 }

180void
181BiometricEvaluation::MPI::CSVDistributor::checkpointRestore()
182{
183 try {
184 auto chkData = this->getCheckpointData();
185 this->_distributedLineCount =
186 chkData->getPropertyAsInteger(
187 BE::MPI::CSVDistributor::CHECKPOINTLINECOUNT);
188
189 /*
190 * Check the randomizer seed against what has been
191 * checkpointed. Presence of the seed in the resources
192 * and checkpoint and the value (if present) must match.
193 */
194 uint64_t seed, chkSeed;
195 bool haveSeed{false}, haveChkSeed{false};
196 try {
197 seed = this->_resources->getRandomSeed();
198 haveSeed = true;
199 } catch (...) {
200 }
201 try {
202 chkSeed = chkData->getPropertyAsInteger(
203 BE::MPI::CSVDistributor::CHECKPOINTRANDOMSEED);
204 haveChkSeed = true;
205 } catch (...) {
206 }
207 if ((haveSeed && !haveChkSeed) || (haveChkSeed && !haveSeed)) {
208 throw Error::ObjectDoesNotExist(
209 "Missing required RNG seed in resources "
210 "or checkpoint file");
211 }
212 if ((haveSeed && haveChkSeed) && (seed != chkSeed)) {
213 throw Error::DataError(
214 "RNG seed in resources does not match checkpoint");
215 }
216 /*
217 * Skip over the lines used during the checkpointed run.
218 */
219 for (uint64_t n = 0; n < this->_distributedLineCount; n++) {
220 (void)this->_resources->readLine();
221 }
222 this->getLogsheet()->writeDebug(
223 "Checkpoint restore: " + chkData->getProperty(
224 BE::MPI::Distributor::CHECKPOINTREASON));
225 } catch (Error::Exception &e) {
226 this->getLogsheet()->writeDebug(
227 "Checkpoint restore: Caught " + e.whatString());
228 throw;
229 }
230}
