27/*
28 * The base constructor for all INCITS and ISO iris
29 * image data records opens the file and reads the contents
30 * into the state object.
31 */
32BE::Iris::INCITSView::INCITSView(
33 const std::string &filename,
34 const uint32_t viewNumber)
35{
36 FILE *fp;
37 if (!BE::IO::Utility::fileExists(filename)) {
38 throw (BE::Error::FileError("File not found."));
39 }
40 fp = std::fopen(filename.c_str(), "rb");
41 if (fp == NULL) {
42 throw (BE::Error::FileError("Could not open file."));
43 }
44 uint64_t size = IO::Utility::getFileSize(filename);
45 this->_iir.resize(size);
46 if (fread(this->_iir, 1, size, fp) != size){
47 fclose(fp);
48 throw (BE::Error::FileError("Could not read file"));
49 }
50 fclose(fp);
51}

167void
168BiometricEvaluation::Iris::INCITSView::readHeader(
169 Memory::IndexedBuffer &buf,
170 const uint32_t formatStandard)
171{
172 if (formatStandard != BE::Iris::INCITSView::ISO2011_STANDARD)
173 throw (Error::ParameterError("Invalid standard"));
174
175 (void)buf.scanBeU32Val(); /* record length */
176 (void)buf.scanBeU16Val(); /* number of iris */
177 this->_certFlag = buf.scanU8Val(); /* certification flag */
178 (void)buf.scanU8Val(); /* number of eyes */
179}

181void
182BiometricEvaluation::Iris::INCITSView::readIrisView(
183 BiometricEvaluation::Memory::IndexedBuffer &buf)
184{
185 static const uint8_t IMAGE_FORMAT_MONO_RAW = 0x02;
186 static const uint8_t IMAGE_FORMAT_JPEG2000 = 0x0A;
187 static const uint8_t IMAGE_FORMAT_MONO_PNG = 0x0E;
188
189 uint8_t uval8;
190 uint32_t uval32;
191
192 uval32 = buf.scanBeU32Val(); /* Representation length */
193 buf.scan(this->_captureDate, CAPTURE_DATE_LENGTH);
194 /*
195 * Convert the data in the date array to YYYY-MM-DD HH:MM:SS
196 * format.
197 */
198 BE::Memory::IndexedBuffer dateBuf(
199 const_cast<uint8_t *>(this->_captureDate), CAPTURE_DATE_LENGTH);
200
201 uint16_t year = dateBuf.scanBeU16Val();
202 uint8_t month = dateBuf.scanU8Val();
203 uint8_t day = dateBuf.scanU8Val();
204 uint8_t hour = dateBuf.scanU8Val();
205 uint8_t minute = dateBuf.scanU8Val();
206 uint8_t seconds = dateBuf.scanU8Val();
207
208 std::stringstream sstr;
209 sstr << (int)year << "-" << (int)month << "-" << (int)day << " ";
210 sstr << (int)hour << ":" << (int)minute << ":" << (int)seconds;
211 this->_captureDateString = sstr.str();
212
213 uval8 = buf.scanU8Val();
214 this->_captureDeviceTechnology =
215 to_enum<BE::Iris::CaptureDeviceTechnology>(uval8);
216
217 this->_captureDeviceVendor = buf.scanBeU16Val();
218 this->_captureDeviceType = buf.scanBeU16Val();
219
220 /*
221 * Quality blocks: Length field (number of blocks) and blocks
222 */
223 uval8 = buf.scanU8Val();
224 BE::Iris::INCITSView::QualitySubBlock qsb;
225 for (uint8_t count = 0; count < uval8; count++) {
226 qsb.score = buf.scanU8Val();
227 qsb.vendorID = buf.scanBeU16Val();
228 qsb.algorithmID = buf.scanBeU16Val();
229 this->_qualitySet.push_back(qsb);
230 }
231
232 (void)buf.scanBeU16Val(); /* number of representations */
233
234 uval8 = buf.scanU8Val();
235 this->_eyeLabel = to_enum<BE::Iris::EyeLabel>(uval8);
236
237 uval8 = buf.scanU8Val();
238 this->_imageType = to_enum<BE::Iris::ImageType>(uval8);
239
240 uval8 = buf.scanU8Val(); /* Image format */
241 switch (uval8) {
242 case IMAGE_FORMAT_MONO_RAW:
243 this->setCompressionAlgorithm(
244 BE::Image::CompressionAlgorithm::None);
245 break;
246 case IMAGE_FORMAT_JPEG2000:
247 this->setCompressionAlgorithm(
248 BE::Image::CompressionAlgorithm::JP2);
249 break;
250 case IMAGE_FORMAT_MONO_PNG:
251 this->setCompressionAlgorithm(
252 BE::Image::CompressionAlgorithm::PNG);
253 break;
254 default:
255 //XXX throw?
256 break;
257 }
258 uval8 = buf.scanU8Val(); /* Image properties */
259 this->_horizontalOrientation =
260 to_enum<BE::Iris::Orientation>(uval8 & 0x03);
261 this->_verticalOrientation =
262 to_enum<BE::Iris::Orientation>((uval8 & 0x0C) >> 2);
263 this->_compressionHistory =
264 to_enum<BE::Iris::ImageCompression>((uval8 & 0xC0) >> 6);
265
266 uint16_t width = buf.scanBeU16Val(); /* Image width */
267 uint16_t height = buf.scanBeU16Val(); /* Image height */
268 this->setImageSize(BE::Image::Size(width, height));
269 uval8 = buf.scanU8Val(); /* Image bit depth */
270 this->setImageColorDepth((uint32_t)uval8);
271
272 this->_cameraRange = buf.scanBeU16Val();
273 this->_rollAngle = buf.scanBeU16Val();
274 this->_rollAngleUncertainty = buf.scanBeU16Val();
275
276 this->_irisCenterSmallestX = buf.scanBeU16Val();
277 this->_irisCenterLargestX = buf.scanBeU16Val();
278 this->_irisCenterSmallestY = buf.scanBeU16Val();
279 this->_irisCenterLargestY = buf.scanBeU16Val();
280 this->_irisDiameterSmallest = buf.scanBeU16Val();
281 this->_irisDiameterLargest = buf.scanBeU16Val();
282
283 uval32 = buf.scanBeU32Val(); /* image length */
284 BE::Memory::uint8Array imageData(uval32);
285 buf.scan(&imageData[0], uval32);
286 this->setImageData(imageData);
287}
