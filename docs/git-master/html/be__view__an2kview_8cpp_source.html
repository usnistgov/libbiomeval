69BiometricEvaluation::View::AN2KView::AN2KView(
70 const std::string filename,
71 const RecordType typeID,
72 const uint32_t recordNumber) :
73 _an2kRecord(nullptr)
74{
75 if (!IO::Utility::fileExists(filename))
76 throw (Error::FileError("File not found."));
77
78 FILE *fp = std::fopen(filename.c_str(), "rb");
79 if (fp == nullptr)
80 throw (Error::FileError("Could not open file."));
81
82 _an2k = Memory::AutoBuffer<ANSI_NIST>(&biomeval_nbis_alloc_ANSI_NIST,
83 &biomeval_nbis_free_ANSI_NIST, &biomeval_nbis_copy_ANSI_NIST);
84 if (biomeval_nbis_read_ANSI_NIST(fp, _an2k) != 0) {
85 fclose(fp);
86 throw Error::FileError("Could not read AN2K file");
87 }
88 fclose(fp);
89
90 readImageCommon(_an2k, typeID, recordNumber);
91 associateMinutiaeData(filename);
92}

119BiometricEvaluation::Image::CompressionAlgorithm
120BiometricEvaluation::View::AN2KView::convertCompressionAlgorithm(
121 const uint16_t recordType,
122 const unsigned char *an2kValue)
123{
124 switch (recordType) {
125 case static_cast<std::underlying_type<RecordType>::type>(RecordType::Type_15):
126 case static_cast<std::underlying_type<RecordType>::type>(RecordType::Type_14):
127 case static_cast<std::underlying_type<RecordType>::type>(RecordType::Type_13):
128 if (!strcmp((const char*)an2kValue, "NONE"))
129 return (BE::Image::CompressionAlgorithm::None);
130 else if (!strcmp((const char*)an2kValue, "WSQ20"))
131 return (BE::Image::CompressionAlgorithm::WSQ20);
132 else if (!strcmp((const char*)an2kValue, "JPEGB"))
133 return (BE::Image::CompressionAlgorithm::JPEGB);
134 else if (!strcmp((const char*)an2kValue, "JPEGL"))
135 return (BE::Image::CompressionAlgorithm::JPEGL);
136 else if (!strcmp((const char*)an2kValue, "JP2"))
137 return (BE::Image::CompressionAlgorithm::JP2);
138 else if (!strcmp((const char*)an2kValue, "JP2L"))
139 return (BE::Image::CompressionAlgorithm::JP2L);
140 else if (!strcmp((const char*)an2kValue, "PNG"))
141 return (BE::Image::CompressionAlgorithm::PNG);
142 else
143 throw Error::DataError("Invalid compression algorithm");
144 break;
145
146 case static_cast<std::underlying_type<RecordType>::type>(RecordType::Type_3):
147 case static_cast<std::underlying_type<RecordType>::type>(RecordType::Type_4):
148 /* ANSI/NIST-ITL 1-2007, Page 7 */
149 switch (atoi((const char*)an2kValue)) {
150 case 0: return (BE::Image::CompressionAlgorithm::None);
151 case 1: return (BE::Image::CompressionAlgorithm::WSQ20);
152 case 2: return (BE::Image::CompressionAlgorithm::JPEGB);
153 case 3: return (BE::Image::CompressionAlgorithm::JPEGL);
154 case 4: return (BE::Image::CompressionAlgorithm::JP2);
155 case 5: return (BE::Image::CompressionAlgorithm::JP2L);
156 case 6: return (BE::Image::CompressionAlgorithm::PNG);
157 default:
158 throw Error::DataError("Invalid compression algorithm");
159 }
160 break;
161
162 case static_cast<std::underlying_type<RecordType>::type>(RecordType::Type_5):
163 case static_cast<std::underlying_type<RecordType>::type>(RecordType::Type_6):
164 /* ANSI/NIST-ITL 1-2007, Page 9 */
165 switch (atoi((const char*)an2kValue)) {
166 case 0: return (BE::Image::CompressionAlgorithm::None);
167 case 1: return (BE::Image::CompressionAlgorithm::Facsimile);
168 default:
169 throw Error::DataError("Invalid compression algorithm");
170 }
171 break;
172
173 default:
174 throw Error::ParameterError("Invalid record type to decode compression algorithm");
175 break;
176 }

234/*
235 * Read those fields that are common across all image record types:
236 * IDC, HLL, VLL, Image data.
237 */
238void
239BiometricEvaluation::View::AN2KView::readImageCommon(
240 const ANSI_NIST *an2k,
241 const RecordType typeID,
242 const uint32_t recordNumber)
243{
244 if (an2k == nullptr)
245 throw Error::ParameterError("Null pointer passed in");
246
247 switch (typeID) {
248 case RecordType::Type_3:
249 case RecordType::Type_4:
250 case RecordType::Type_5:
251 case RecordType::Type_6:
252 case RecordType::Type_13:
253 case RecordType::Type_14:
254 case RecordType::Type_15:
255 break;
256 default:
257 throw Error::ParameterError("Invalid Record Type ID");
258 }
259 _recordType = typeID;
260
261 /*
262 * Find the nth record of the requested type in the file, throwing
263 * an exception if not present. The 0th record in an AN2K
264 * file is always the Type-1, so skip that one.
265 * The pointer is set to an object inside the complete ANSI-NIST
266 * record, and that object is free'd in the destructor by the
267 * AutoBuffer that wraps it. Therefore the single RECORD object
268 * is not explicitly destroyed.
269 */
270 uint32_t count = 1;
271 for (int i = 1; i < an2k->num_records; i++) {
272 if (an2k->records[i]->type == static_cast<
273 std::underlying_type<
274 BiometricEvaluation::View::AN2KView::RecordType>::type>(
275 typeID)) {
276 if (count == recordNumber) {
277 _an2kRecord = _an2k->records[i];
278 break;
279 }
280 count++;
281 }
282 }
283
284 if (_an2kRecord == nullptr)
285 throw (Error::DataError("Could not find image record in AN2K"));
286
287 FIELD *field;
288 int idx;
289
290 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, IDC_ID, _an2kRecord) != TRUE)
291 throw Error::DataError("Field IDC not found");
292 _idc = atoi((char *)field->subfields[0]->items[0]->value);
293
294 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, HLL_ID, _an2kRecord) != TRUE)
295 throw Error::DataError("Field HLL not found");
296
297 Image::Size imageSize;
298 imageSize.xSize = atoi((char *)field->subfields[0]->items[0]->value);
299
300 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, VLL_ID, _an2kRecord) != TRUE)
301 throw Error::DataError("Field VLL not found");
302 imageSize.ySize = atoi((char *)field->subfields[0]->items[0]->value);
303 this->setImageSize(imageSize);
304
305 switch (typeID) {
306 case RecordType::Type_3:
307 case RecordType::Type_4:
308 case RecordType::Type_5:
309 case RecordType::Type_6:
310 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, BIN_CA_ID,
311 _an2kRecord) != TRUE)
312 throw Error::DataError("Field CA not found");
313 break;
314 case RecordType::Type_13:
315 case RecordType::Type_14:
316 case RecordType::Type_15:
317 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, TAG_CA_ID,
318 _an2kRecord) != TRUE)
319 throw Error::DataError("Field CA not found");
320 break;
321 default:
322 /* Not reached */
323 throw Error::ParameterError("Invalid Record Type ID");
324 }
325 BE::Image::CompressionAlgorithm ca =
326 convertCompressionAlgorithm(to_int_type(typeID),
327 field->subfields[0]->items[0]->value);
328 this->setCompressionAlgorithm(ca);
329
330 switch (typeID) {
331 case RecordType::Type_3:
332 case RecordType::Type_4:
333 case RecordType::Type_5:
334 case RecordType::Type_6:
335 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, BIN_IMAGE_ID,
336 _an2kRecord) != TRUE)
337 throw Error::DataError("Field DATA not found");
338 break;
339 case RecordType::Type_13:
340 case RecordType::Type_14:
341 case RecordType::Type_15:
342 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, DAT2_ID,
343 _an2kRecord) != TRUE)
344 throw Error::DataError("Field DATA not found");
345 break;
346 default:
347 /* Not reached */
348 throw Error::ParameterError("Invalid Record Type ID");
349 }
350 BE::Memory::uint8Array imageData;
351 imageData.copy(field->subfields[0]->items[0]->value,
352 field->subfields[0]->items[0]->num_bytes);
353 this->setImageData(imageData);
354}

356void
357BiometricEvaluation::View::AN2KView::associateMinutiaeData(
358 Memory::uint8Array &buf)
359{
360 FIELD *field;
361 int idx;
362 std::set<int> type9Recs =
363 DataInterchange::AN2KRecord::recordLocations(buf, RecordType::Type_9);
364 for (std::set<int>::const_iterator it = type9Recs.begin();
365 it != type9Recs.end(); it++) {
366 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, IDC_ID,
367 _an2k->records[*it]) == TRUE) {
368 if (_idc == atoi((char *)field->subfields[0]->
369 items[0]->value)) {
370 Finger::AN2KMinutiaeDataRecord amdr(buf, *it);
371 addMinutiaeDataRecord(amdr);
372 }
373 }
374 }
375}

377void
378BiometricEvaluation::View::AN2KView::associateMinutiaeData(
379 const std::string &filename)
380{
381 if (!IO::Utility::fileExists(filename))
382 throw (Error::FileError("File not found."));
383
384 FILE *fp = std::fopen(filename.c_str(), "rb");
385 if (fp == nullptr)
386 throw (Error::FileError("Could not open file."));
387
388 uint64_t sz = IO::Utility::getFileSize(filename);
389 Memory::uint8Array buf(sz);
390 if (fread(buf, 1, sz, fp) != sz) {
391 fclose(fp);
392 throw Error::FileError("Could not read AN2K file");
393 }
394 fclose(fp);
395
396 associateMinutiaeData(buf);
397}
