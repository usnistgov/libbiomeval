81int32_t
82BiometricEvaluation::Process::WorkerController::getExitStatus()
83 const
84{
85 if (!this->everWorked())
86 throw Error::StrategyError("Worker was never started");
87 else if (this->isWorking())
88 throw Error::StrategyError("Worker is still working");
89 else if (!this->_rvSet)
90 throw Error::ObjectDoesNotExist();
91
92 return (this->_rv);
93}

99void
100BiometricEvaluation::Process::WorkerController::sendMessageToWorker(
101 const Memory::uint8Array &message)
102{
103 uint64_t length = message.size();
104 int pipeFD = getWorker()->getSendingPipe();
105
106 /*
107 * Send the message length, then the message contents.
108 * All exceptions float out.
109 */
110 IO::Utility::writePipe(&length, sizeof(length), pipeFD);
111 IO::Utility::writePipe(message, pipeFD);
112}
