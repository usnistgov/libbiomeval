53BiometricEvaluation::DataInterchange::ANSI2004Record::ANSI2004Record(
54 const std::initializer_list<BE::Finger::ANSI2004View> &views)
55{
56 BE::Image::Size size(0, 0);
57 for (const auto &view : views) {
58 /* Ensure all view image sizes are identical */
59 if ((size.xSize == 0) && (size.ySize == 0))
60 size = view.getImageSize();
61 else if (view.getImageSize() != size)
62 throw BE::Error::StrategyError("Not all view image "
63 "sizes are identical.");
64
65 /* Add to collection */
66 this->_views.push_back(view);
67 }

87uint64_t
88BiometricEvaluation::DataInterchange::ANSI2004Record::getFMRLength()
89 const
90{
91 /* Minimum header size */
92 uint64_t size = 26;
93
94 for (const auto &view : this->_views) {
95 /* Finger view header size */
96 size += 4;
97
98 /* Minutia */
99 size += (6 * view.getMinutiaeData().getMinutiaPoints().size());
100
101 /* EDB */
102 size += this->getEDBLength();
103 }
104
105 /* Large records use 6 bytes instead of 2 for length in header */
106 if (size > (UINT16_MAX - 4))
107 size += 4;
108
109 return (size);
110}

112uint64_t
113BiometricEvaluation::DataInterchange::ANSI2004Record::getEDBLength()
114 const
115{
116 /* Minimum length of EDB (EDB Length field) */
117 uint64_t size = 2;
118
119 for (const auto &view : this->_views) {
120 auto minutiaData = view.getMinutiaeData();
121
122 /* Cores */
123 const auto cores = minutiaData.getCores();
124 for (const auto &core : cores)
125 size += core.has_angle ? 6 : 5;
126
127 /* Deltas */
128 const auto deltas = minutiaData.getDeltas();
129 for (const auto &delta : deltas)
130 size += delta.has_angle ? 8 : 5;
131
132 /* Core and delta header */
133 if ((cores.size() != 0) || (deltas.size() != 0))
134 size += 4;
135
136 /* If core w/o delta (or v.v.), must indicate 0 of the other */
137 if ((cores.size() == 0) && (deltas.size() != 0))
138 ++size;
139 if ((deltas.size() == 0) && (cores.size() != 0))
140 ++size;
141
142 /* Ridge counts (+ 1 for extraction method) */
143 const auto ridgeCounts = minutiaData.getRidgeCountItems();
144 size += (ridgeCounts.size() * 3);
145
146 /* Ridge count header + extraction method */
147 if (ridgeCounts.size() != 0)
148 size += 5;
149 }
150
151 return (size);
152}

154BiometricEvaluation::Memory::uint8Array
155BiometricEvaluation::DataInterchange::ANSI2004Record::getFMR()
156 const
157{
158 BE::Memory::uint8Array tmpl(this->getFMRLength());
159 BE::Memory::MutableIndexedBuffer buf(tmpl);
160
161 /* Format identifier */
162// buf.pushBeU32Val(BE::Finger::INCITSView::FMR_BASE_FORMAT_ID);
163 buf.pushBeU32Val(0x464D5200);
164 /* Version number */
165// buf.pushBeU32Val(BE::Finger::ANSI2004View::BASE_SPEC_VERSION);
166 buf.pushBeU32Val(0x20323000);
167 /* Length of record */
168 buf.pushBeU16Val(tmpl.size());
169 /* CBEFF product identifier and type */
170 buf.pushBeU32Val(0);
171 /* Capture equipment compliance and equipment ID */
172 buf.pushBeU16Val(0);
173 /* Width */
174 auto dimensions = this->_views[0].getImageSize();
175 buf.pushBeU16Val(dimensions.xSize);
176 /* Height */
177 buf.pushBeU16Val(dimensions.ySize);
178 /* X resolution */
179 buf.pushBeU16Val(197);
180 /* Y resolution */
181 buf.pushBeU16Val(197);
182 /* Number of views */
183 buf.pushU8Val(this->_views.size());
184 /* Reserved */
185 buf.pushU8Val(0);
186
187 uint8_t viewNumber = 0;
188 uint16_t u16Val = 0;
189 for (const auto &view : this->_views) {
190 /* Finger position */
191 buf.pushU8Val(to_int_type(view.getPosition()));
192 /* View number and impression type */
193 uint8_t viewAndImpression = ((viewNumber++ << 4) & 0xF0);
194 viewAndImpression |= to_int_type(view.getImpressionType());
195 buf.pushU8Val(viewAndImpression);
196 /* Quality */
197 buf.pushU8Val(view.getQuality());
198
199 /* Number of minutia */
200 const auto minutiaeData = view.getMinutiaeData();
201 const auto minutiaPoints = minutiaeData.getMinutiaPoints();
202 buf.pushU8Val(minutiaPoints.size());
203
204 /* Minutia points */
205 for (const auto &minutiae : minutiaPoints) {
206 /* Minutiae type and X coordinate */
207 switch (minutiae.type) {
208 case Feature::MinutiaeType::RidgeEnding:
209 u16Val = (1 << 14) & 0xC000;
210 break;
211 case Feature::MinutiaeType::Bifurcation:
212 u16Val = (2 << 14) & 0xC000;
213 break;
214 case Feature::MinutiaeType::Other:
215 u16Val = 0;
216 break;
217 default:
218 throw BE::Error::StrategyError("Unsupported "
219 "minutiae type (" +
220 to_string(minutiae.type) + ")");
221 break;
222
223 }
224 u16Val |= minutiae.coordinate.x;
225 buf.pushBeU16Val(u16Val);
226
227 /* Y coordinate. Upper 2 bits are reserved */
228 u16Val = minutiae.coordinate.y;
229 buf.pushBeU16Val(u16Val);
230
231 /* Angle */
232 buf.pushU8Val(minutiae.theta);
233 /* Quality */
234 buf.pushU8Val(minutiae.has_quality ?
235 minutiae.quality : 0);
236 }
237
238 /* Extended data size */
239 buf.pushBeU16Val(this->getEDBLength() - sizeof(uint16_t));
240
241 /* EDB: Ridge counts */
242 auto ridgeCountData = minutiaeData.getRidgeCountItems();
243 if (ridgeCountData.size() != 0) {
244 /* Extended data type for ridge counts */
245 buf.pushBeU16Val(1);
246 /* Length = Num RC + extraction method + header */
247 buf.pushBeU16Val((ridgeCountData.size() * 3) + 1 + 4);
248 buf.pushU8Val(to_int_type(
249 ridgeCountData[0].extraction_method));
250 for (const auto &rc : ridgeCountData) {
251 /* Index 1 */
252 buf.pushU8Val(rc.index_one);
253 /* Index 2 */
254 buf.pushU8Val(rc.index_two);
255 /* Count */
256 buf.pushU8Val(rc.count);
257 }
258 }
259
260 /* EDB: Cores and deltas */
261 auto cores = minutiaeData.getCores();
262 auto deltas = minutiaeData.getDeltas();
263
264 uint8_t u8Val = 0;
265 if ((cores.size() != 0) || (deltas.size() != 0)) {
266 /* Extended data type for cores and deltas */
267 buf.pushBeU16Val(2);
268
269 /* Calculate size of core and delta section */
270 uint16_t cdSize = 4;
271 if (cores.size() == 0)
272 ++cdSize;
273 else
274 cdSize += (cores.size() *
275 (cores[0].has_angle ? 6 : 5));
276 if (deltas.size() == 0)
277 ++cdSize;
278 else
279 cdSize += (deltas.size() *
280 (deltas[0].has_angle ? 8 : 5));
281 buf.pushBeU16Val(cdSize);
282
283 /* Cores */
284 if (cores.size() == 0) {
285 buf.pushU8Val(0);
286 } else {
287 /* Use first core as truth */
288 bool coresHaveAngle = cores[0].has_angle;
289 /* 2 bits angle boolean, 2 bits set to 0 */
290 u8Val = ((coresHaveAngle ? 1 : 0) << 6) & 0xC0;
291 /* Lower 4 bits contain number of cores */
292 u8Val |= cores.size();
293 buf.pushU8Val(u8Val);
294 for (const auto &core : cores) {
295 /* X coordinate */
296 buf.pushBeU16Val(core.coordinate.x);
297 /* Y coordinate */
298 buf.pushBeU16Val(core.coordinate.y);
299 /* Angle, if applicable */
300 if (coresHaveAngle)
301 buf.pushU8Val(core.angle);
302 }
303 }
304
305 /* Deltas */
306 if (deltas.size() == 0) {
307 buf.pushU8Val(0);
308 } else {
309 /* Use first delta as truth */
310 bool deltasHaveAngles = deltas[0].has_angle;
311 /* Highest 2 bits have angle boolean */
312 u8Val = ((deltasHaveAngles ? 1 : 0) << 6) &
313 0xC0;
314 /* Lower 6 bits have number of deltas */
315 u8Val |= deltas.size();
316 buf.pushU8Val(u8Val);
317 for (const auto &delta : deltas) {
318 /* X coordinate */
319 buf.pushBeU16Val(delta.coordinate.x);
320 /* Y coordinate */
321 buf.pushBeU16Val(delta.coordinate.y);
322 /* Angles, if applicable */
323 if (deltasHaveAngles) {
324 buf.pushU8Val(delta.angle1);
325 buf.pushU8Val(delta.angle2);
326 buf.pushU8Val(delta.angle3);
327 }
328 }
329 }
330 }
331
332 /* EDB: Proprietary data */
333 /* TODO: Not parsed by ANSI2004View */
334 }
335
336 return (tmpl);
337}
