17bool
18BiometricEvaluation::Feature::Sort::XY::operator()(
19 const BiometricEvaluation::Feature::MinutiaPoint &lhs,
20 const BiometricEvaluation::Feature::MinutiaPoint &rhs)
21 const
22{
23 /* Ascending X */
24 if (lhs.coordinate.x > rhs.coordinate.x)
25 return (false);
26 if (lhs.coordinate.x < rhs.coordinate.x)
27 return (true);
28
29 /* Ascending Y */
30 if (lhs.coordinate.y > rhs.coordinate.y)
31 return (false);
32 if (lhs.coordinate.y < rhs.coordinate.y)
33 return (true);
34
35 /* Equivalent */
36 return (false);
37}

86uint64_t
87BiometricEvaluation::Feature::Sort::Polar::distanceFromCenter(
88 const BiometricEvaluation::Image::Coordinate &coordinate)
89 const
90{
91 /*
92 * We don't care about the actual distance value, so don't use
93 * BiometricEvaluation::Image::distance() (to avoid floating point,
94 * for speed)).
95 */
96 const int64_t xDelta = static_cast<int64_t>(coordinate.x) -
97 static_cast<int64_t>(this->_center.x);
98 const int64_t yDelta = static_cast<int64_t>(coordinate.y) -
99 static_cast<int64_t>(this->_center.y);
100 return ((xDelta * xDelta) + (yDelta * yDelta));
101}

103BiometricEvaluation::Image::Coordinate
104BiometricEvaluation::Feature::Sort::Polar::centerOfMinutiaeMass(
105 const BiometricEvaluation::Feature::MinutiaPointSet &mps)
106{
107 const uint64_t count = mps.size();
108 if (count == 0)
109 throw BE::Error::StrategyError("No minutia");
110
111 uint64_t sumX = 0, sumY = 0;
112 for (const auto &minutiae : mps) {
113 sumX += minutiae.coordinate.x;
114 sumY += minutiae.coordinate.y;
115 }
116
117 /* Guaranteed to be 32-bit unsigned, so downcast is safe */
118 return {static_cast<uint32_t>(sumX / count),
119 static_cast<uint32_t>(sumY / count)};
120}

129bool
130BiometricEvaluation::Feature::Sort::Polar::operator()(
131 const BiometricEvaluation::Feature::MinutiaPoint &lhs,
132 const BiometricEvaluation::Feature::MinutiaPoint &rhs)
133 const
134{
135 const uint64_t lhsDistance = Polar::distanceFromCenter(lhs.coordinate);
136 const uint64_t rhsDistance = Polar::distanceFromCenter(rhs.coordinate);
137
138 /* Increasing distance from center */
139 if (lhsDistance > rhsDistance)
140 return (false);
141 if (lhsDistance < rhsDistance)
142 return (true);
143
144 /* Increasing theta */
145 if (lhs.theta > rhs.theta)
146 return (false);
147 if (lhs.theta < rhs.theta)
148 return (true);
149
150 return (false);
151}

191std::vector<BiometricEvaluation::Feature::MinutiaPoint>
192BiometricEvaluation::Feature::Sort::sort(
193 std::vector<BiometricEvaluation::Feature::MinutiaPoint> &minutia,
194 const BiometricEvaluation::Feature::Sort::Kind &sortOrder)
195{
196 switch (sortOrder) {
197 case BE::Feature::Sort::Kind::XYAscending:
198 std::sort(minutia.begin(), minutia.end(),
199 BE::Feature::Sort::XY());
200 break;
201 case BE::Feature::Sort::Kind::XYDescending:
202 std::sort(minutia.rbegin(), minutia.rend(),
203 BE::Feature::Sort::XY());
204 break;
205 case BE::Feature::Sort::Kind::YXAscending:
206 std::sort(minutia.begin(), minutia.end(),
207 BE::Feature::Sort::YX());
208 break;
209 case BE::Feature::Sort::Kind::YXDescending:
210 std::sort(minutia.rbegin(), minutia.rend(),
211 BE::Feature::Sort::YX());
212 break;
213 case BE::Feature::Sort::Kind::QualityAscending:
214 std::sort(minutia.begin(), minutia.end(),
215 BE::Feature::Sort::Quality());
216 break;
217 case BE::Feature::Sort::Kind::QualityDescending:
218 std::sort(minutia.rbegin(), minutia.rend(),
219 BE::Feature::Sort::Quality());
220 break;
221 case BE::Feature::Sort::Kind::AngleAscending:
222 std::sort(minutia.begin(), minutia.end(),
223 BE::Feature::Sort::Angle());
224 break;
225 case BE::Feature::Sort::Kind::AngleDescending:
226 std::sort(minutia.rbegin(), minutia.rend(),
227 BE::Feature::Sort::Angle());
228 break;
229 case BE::Feature::Sort::Kind::PolarCOMAscending:
230 try {
231 std::sort(minutia.begin(), minutia.end(),
232 BE::Feature::Sort::Polar(BE::Feature::Sort::Polar::
233 centerOfMinutiaeMass(minutia)));
234 } catch (const BE::Error::StrategyError&) {
235 /* No minutia, sorting not important */
236 }
237 break;
238 case BE::Feature::Sort::Kind::PolarCOMDescending:
239 try {
240 std::sort(minutia.rbegin(), minutia.rend(),
241 BE::Feature::Sort::Polar(BE::Feature::Sort::Polar::
242 centerOfMinutiaeMass(minutia)));
243 } catch (const BE::Error::StrategyError&) {
244 /* No minutia, sorting not important */
245 }
246 break;
247 default:
248 throw BE::Error::NotImplemented(to_string(sortOrder));
249 }
250
251 return (minutia);
252}
