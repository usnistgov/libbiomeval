37int32_t
38BiometricEvaluation::Process::MessageCenterReceiver::workerMain()
39{
40 this->parseArgs();
41
42 bool stop = false;
43 Memory::uint8Array message;
44 while (!stop) {
45 /*
46 * Check and forward message from client to listener.
47 */
48 try {
49 if (MessageCenterUtility::dataAvailable(
50 this->_clientSocket,
51 MessageCenter::DEFAULT_TIMEOUT)) {
52 message = this->receive();
53 this->sendMessageToManager(message);
54 }
55 } catch (const Error::Exception&) {
56 /* Most likely a connection failure, close connection */
57 stop = true;
58 continue;
59 }
60
61 /*
62 * Check and forward message from Listener to client.
63 */
64 try {
65 if (this->waitForMessage(
66 MessageCenter::DEFAULT_TIMEOUT)) {
67 this->receiveMessageFromManager(message);
68 this->send(message);
69 }
70 } catch (const Error::Exception&) {
71 /* Most likely a server issue, close connection */
72 stop = true;
73 continue;
74 }
75
76 if (this->stopRequested())
77 stop = true;
78 }
79
80 ::close(this->_clientSocket);
81 return (EXIT_SUCCESS);
82}

84BiometricEvaluation::Memory::uint8Array
85BiometricEvaluation::Process::MessageCenterReceiver::receive()
86 const
87{
88 int rv;
89 Memory::uint8Array buffer(MessageCenter::MAX_MESSAGE_LENGTH);
90
91 /* Read message */
92 for (;;) {
93 rv = ::recv(this->_clientSocket, buffer, buffer.size(), 0);
94 if (rv < 0) {
95 if (errno != EINTR)
96 throw BE::Error::StrategyError(
97 BE::Error::errorStr());
98 } else if (rv == 0) {
99 /* Client-side closed connection */
100 break;
101 } else {
102 /* Data received */
103 buffer.resize(rv - 1);
104 buffer[buffer.size() - 1] = '\0';
105 break;
106 }
107 }
108
109 if (rv == 0)
110 throw Error::ObjectDoesNotExist();
111
112 return (buffer);
113}
