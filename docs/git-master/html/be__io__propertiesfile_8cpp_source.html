37void
38BiometricEvaluation::IO::PropertiesFile::initPropertiesFile(
39 const std::map<std::string, std::string> &defaults)
40{
41 try {
42 this->initWithBuffer(IO::Utility::readFile(this->_pathname),
43 defaults);
44 } catch (const Error::ObjectDoesNotExist&) {
45 /* Create a new file if one does not exist */
46 if (this->getMode() == Mode::ReadOnly)
47 throw Error::StrategyError("Properties file does not "
48 "exist and mode is read-only");
49
50 FILE *fp = fopen(this->_pathname.c_str(), "w");
51 if (fp == nullptr)
52 throw Error::FileError("Could not create new "
53 "properties file");
54 fclose(fp);
55 }
56}

58void
59BiometricEvaluation::IO::PropertiesFile::sync()
60{
61 if (this->getMode() == Mode::ReadOnly)
62 throw Error::StrategyError(RO_ERR_MSG);
63
64 std::ofstream ofs(_pathname.c_str());
65 if (!ofs)
66 throw Error::FileError("Could not write properties file");
67
68 std::vector<std::string> keys = this->getPropertyKeys();
69 for (auto k = keys.begin(); k != keys.end(); ++k) {
70 ofs << *k << " = " << this->getProperty(*k) << '\n';
71 }
72 ofs.close();
73}

75void
76BiometricEvaluation::IO::PropertiesFile::changeName(
77 const std::string &pathname)
78{
79 if (this->getMode() == Mode::ReadOnly)
80 throw Error::StrategyError(RO_ERR_MSG);
81
82 if (IO::Utility::fileExists(pathname))
83 throw Error::ObjectExists(pathname);
84
85 if (::rename(this->_pathname.c_str(), pathname.c_str()) != 0)
86 throw Error::StrategyError("Could not move \"" +
87 this->_pathname + "\" to \"" + pathname + "\" (" +
88 Error::errorStr() + ")");
89
90 this->_pathname = pathname;
91}
