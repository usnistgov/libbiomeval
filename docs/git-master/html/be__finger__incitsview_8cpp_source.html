28/*
29 * The base constructor for all INCITS and ISO finger
30 * minutiae records opens the file and reads the contents
31 * into the object state _fmr and _fir. The parsing of the
32 * data is done by the child classes with help from this
33 * class.
34 */
35BiometricEvaluation::Finger::INCITSView::INCITSView(
36 const std::string &fmrFilename,
37 const std::string &firFilename,
38 const uint32_t viewNumber)
39{
40 if (fmrFilename != "") {
41 try {
42 this->_fmr = BE::IO::Utility::readFile(fmrFilename);
43 } catch (const BE::Error::Exception&) {
44 throw (BE::Error::FileError(
45 "FMR file could not be opened."));
46 }
47 }
48 if (firFilename != "") {
49 try {
50 this->_fir = BE::IO::Utility::readFile(firFilename);
51 } catch (const BE::Error::Exception&) {
52 throw (BE::Error::FileError(
53 "FIR file could not be opened."));
54 }
55 }
56}

150static std::tuple<BiometricEvaluation::Feature::MinutiaPoint, uint8_t>
151scanFMD(BiometricEvaluation::Memory::IndexedBuffer &buf)
152{
153 uint16_t sval;
154 uint8_t cval;
155 BE::Feature::MinutiaPoint m;
156
157 sval = buf.scanBeU16Val();
158
159 m.has_type = true;
160 uint8_t nativeType = ((sval &
161 BE::Feature::INCITSMinutiae::FMD_MINUTIA_TYPE_MASK) >>
162 BE::Feature::INCITSMinutiae::FMD_MINUTIA_TYPE_SHIFT);
163
164 switch (nativeType) {
165 case BE::Feature::INCITSMinutiae::FMD_MINUTIA_TYPE_OTHER:
166 m.type = BE::Feature::MinutiaeType::Other;
167 break;
168 case BE::Feature::INCITSMinutiae::FMD_MINUTIA_TYPE_RIDGE_ENDING:
169 m.type = BE::Feature::MinutiaeType::RidgeEnding;
170 break;
171 case BE::Feature::INCITSMinutiae::FMD_MINUTIA_TYPE_BIFURCATION:
172 m.type = BE::Feature::MinutiaeType::Bifurcation;
173 break;
174 default:
175 throw (BE::Error::DataError("Invalid minutiae type."));
176 break;
177 }
178 m.coordinate.x = sval & BE::Feature::INCITSMinutiae::FMD_X_COORD_MASK;
179 sval = buf.scanBeU16Val();
180 m.coordinate.y = sval & BE::Feature::INCITSMinutiae::FMD_Y_COORD_MASK;
181 uint8_t reservedValue = sval &
182 BE::Feature::INCITSMinutiae::FMD_RESERVED_MASK;
183
184 /* Angle and quality */
185 cval = buf.scanU8Val();
186 m.theta = cval;
187 cval = buf.scanU8Val();
188 m.has_quality = true;
189 m.quality = cval;
190
191 return (std::make_tuple(m, reservedValue));
192}

194static
195BiometricEvaluation::Feature::RidgeCountItem scanRCD(
196 BiometricEvaluation::Memory::IndexedBuffer &buf,
197 uint8_t nativeExtrMethod)
198{
199 uint8_t idx1 = buf.scanU8Val();
200 uint8_t idx2 = buf.scanU8Val();
201 uint8_t count = buf.scanU8Val();
202 BE::Feature::RidgeCountExtractionMethod extrMethod;
203 switch (nativeExtrMethod) {
204 case BE::Feature::INCITSMinutiae::RCE_NONSPECIFIC:
205 extrMethod = BE::Feature::RidgeCountExtractionMethod::NonSpecific;
206 break;
207 case BE::Feature::INCITSMinutiae::RCE_FOUR_NEIGHBOR:
208 extrMethod = BE::Feature::RidgeCountExtractionMethod::FourNeighbor;
209 break;
210 case BE::Feature::INCITSMinutiae::RCE_EIGHT_NEIGHBOR:
211 extrMethod = BE::Feature::RidgeCountExtractionMethod::EightNeighbor;
212 break;
213 default:
214 throw (BE::Error::DataError(
215 "Invalid ridge count extraction method"));
216 break;
217 }
218 BE::Feature::RidgeCountItem rcd(extrMethod, idx1, idx2, count);
219 return (rcd);
220}

333void
334BiometricEvaluation::Finger::INCITSView::readFMRHeader(
335 BiometricEvaluation::Memory::IndexedBuffer &buf,
336 const uint32_t formatStandard)
337{
338 static const uint16_t HDR_SCANNER_ID_MASK = 0x0FFF;
339 static const uint16_t HDR_COMPLIANCE_MASK = 0xF000;
340 static const uint8_t HDR_COMPLIANCE_SHIFT = 12;
341
342 if ((formatStandard != Finger::INCITSView::ANSI2004_STANDARD) &&
343 (formatStandard != Finger::INCITSView::ISO2005_STANDARD))
344 throw (Error::ParameterError("Invalid standard parameter"));
345
346 /* Record length, 2/4/6 bytes */
347 if (formatStandard == Finger::INCITSView::ANSI2004_STANDARD) {
348 this->_recordLength = buf.scanBeU16Val();
349 if (this->_recordLength == 0) {
350 this->_recordLength = buf.scanBeU32Val();
351 }
352 } else {
353 this->_recordLength = buf.scanBeU32Val();
354 }
355
356 /* CBEFF Product ID */
357 if (formatStandard == Finger::INCITSView::ANSI2004_STANDARD) {
358 _productIDOwner = buf.scanBeU16Val();
359 _productIDType = buf.scanBeU16Val();
360 }
361
362 /* Capture equipment compliance/scanner ID */
363 uint16_t sval = buf.scanBeU16Val();
364 _captureEquipmentID = sval & HDR_SCANNER_ID_MASK;
365 _appendixFCompliance =
366 (sval & HDR_COMPLIANCE_MASK) >> HDR_COMPLIANCE_SHIFT;
367
368 /* Image size and resolution */
369 uint16_t xval, yval;
370 xval = buf.scanBeU16Val();
371 yval = buf.scanBeU16Val();
372 this->setImageSize(Image::Size(xval, yval));
373 xval = buf.scanBeU16Val();
374 yval = buf.scanBeU16Val();
375 auto resolution = BE::Image::Resolution(xval, yval,
376 BE::Image::Resolution::Units::PPCM);
377 this->setImageResolution(resolution);
378 setScanResolution(resolution);
379
380 /* Number of views and reserved field */
381 this->_numFingerViews = buf.scanU8Val();
382 this->_fmrReservedByte = buf.scanU8Val();
383}

385void
386BiometricEvaluation::Finger::INCITSView::readFVMR(
387 BiometricEvaluation::Memory::IndexedBuffer &buf)
388{
389 static const uint8_t FVMR_VIEW_NUMBER_MASK = 0xF0;
390 static const uint8_t FVMR_VIEW_NUMBER_SHIFT = 4;
391 static const uint8_t FVMR_IMPRESSION_MASK = 0x0F;
392
393 uint8_t cval = buf.scanU8Val();
394 _position = BE::Finger::INCITSView::convertPosition(cval);
395
396 cval = buf.scanU8Val();
397 _viewNumber = (cval & FVMR_VIEW_NUMBER_MASK) >> FVMR_VIEW_NUMBER_SHIFT;
398 _impression = BE::Finger::INCITSView::convertImpression(
399 cval & FVMR_IMPRESSION_MASK);
400
401 _quality = (uint32_t)buf.scanU8Val();
402
403 /* Read the minutiae data items. */
404 cval = buf.scanU8Val(); /* Number of minutiae */
405 auto minutiaData = this->readMinutiaeDataPoints(buf, cval);
406 this->_minutiae.setMinutiaPoints(std::get<0>(minutiaData));
407 this->_fmdReserved = std::get<1>(minutiaData);
408 this->readExtendedDataBlock(buf);
409}

411std::tuple<BiometricEvaluation::Feature::MinutiaPointSet, std::vector<uint8_t>>
412BiometricEvaluation::Finger::INCITSView::readMinutiaeDataPoints(
413 BiometricEvaluation::Memory::IndexedBuffer &buf,
414 uint32_t count)
415{
416 BE::Feature::MinutiaPointSet mps(count);
417 std::vector<uint8_t> reserved(count);
418 for (uint32_t i = 0; i < count; i++) {
419 std::tie(mps[i], reserved[i]) = scanFMD(buf);
420 mps[i].index = i;
421 }
422 return (std::make_tuple(mps, reserved));
423}

425BiometricEvaluation::Feature::RidgeCountItemSet
426BiometricEvaluation::Finger::INCITSView::readRidgeCountData(
427 BiometricEvaluation::Memory::IndexedBuffer &buf,
428 uint32_t dataLength)
429{
430 uint8_t nativeExtrMethod = buf.scanU8Val();
431 int32_t remLength = dataLength -
432 BE::Feature::INCITSMinutiae::FED_HEADER_LENGTH - 1;
433 if ((remLength % BE::Feature::INCITSMinutiae::FED_RCD_ITEM_LENGTH) != 0)
434 throw (Error::DataError(
435 "Ridge count data block has bad length"));
436 BE::Feature::RidgeCountItemSet rcis;
437 while (remLength > 0) {
438 rcis.push_back(scanRCD(buf, nativeExtrMethod));
439 remLength -= BE::Feature::INCITSMinutiae::FED_RCD_ITEM_LENGTH;
440 }
441 return (rcis);
442}

444void
445BiometricEvaluation::Finger::INCITSView::readExtendedDataBlock(
446 BiometricEvaluation::Memory::IndexedBuffer &buf)
447{
448 /* Extended data block length */
449 this->_edbLength = buf.scanBeU16Val();
450 if (this->_edbLength == 0)
451 return;
452
453 int32_t blockLength = static_cast<int32_t>(this->_edbLength);
454 while (blockLength > 0) {
455 uint16_t typeID = buf.scanBeU16Val();
456 uint16_t dataLength = buf.scanBeU16Val();
457 if (dataLength == 0)
458 throw (Error::DataError("Extended data length is 0"));
459 if (dataLength > blockLength) {
460 std::ostringstream sstr;
461 sstr << "Extended data length " << dataLength <<
462 " is greater than remaining block length of " <<
463 blockLength;
464 throw (Error::DataError(sstr.str()));
465 }
466 switch (typeID) {
467 case BE::Feature::INCITSMinutiae::FED_RIDGE_COUNT:
468 {
469 BE::Feature::RidgeCountItemSet rcis =
470 this->readRidgeCountData(buf, dataLength);
471 _minutiae.setRidgeCountItems(rcis);
472 }
473 break;
474
475 case BE::Feature::INCITSMinutiae::FED_CORE_AND_DELTA:
476 {
477 BE::Feature::CorePointSet cps;
478 BE::Feature::DeltaPointSet dps;
479 this->readCoreDeltaData(buf, dataLength, cps, dps);
480 _minutiae.setCorePointSet(cps);
481 _minutiae.setDeltaPointSet(dps);
482 }
483 break;
484
485 default:
486 //XXX read an opaque object
487 break;
488 }
489 blockLength -= dataLength;
490 }
491}
