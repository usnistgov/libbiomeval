21BiometricEvaluation::Image::JPEGL::JPEGL(
22 const uint8_t *data,
23 const uint64_t size,
24 const std::string &identifier,
25 const statusCallback_t &statusCallback) :
26 Image::Image(
27 data,
28 size,
29 CompressionAlgorithm::JPEGL,
30 identifier,
31 statusCallback)
32{
33 uint8_t *markerBuf = (uint8_t *)this->getDataPointer();
34 uint8_t *endPtr = (uint8_t *)this->getDataPointer() +
35 this->getDataSize();
36
37 uint16_t marker;
38 if (biomeval_nbis_getc_marker_jpegl(&marker, SOI, &markerBuf, endPtr))
39 throw Error::DataError("No SOI marker");
40 if (biomeval_nbis_getc_marker_jpegl(&marker, APP0, &markerBuf, endPtr))
41 throw Error::DataError("No APP0 marker");
42
43 /* Parse JFIF header for resolution information */
44 JFIF_HEADER *JFIFHeader;
45 if (biomeval_nbis_getc_jfif_header(&JFIFHeader, &markerBuf, endPtr))
46 throw Error::DataError("Could not read JFIF header");
47
48 switch (JFIFHeader->units) {
49 case 1: /* PPI */
50 setResolution(Resolution(JFIFHeader->dx, JFIFHeader->dy,
51 Resolution::Units::PPI));
52 break;
53 case 2: /* PPCM */
54 setResolution(Resolution(JFIFHeader->dx, JFIFHeader->dy,
55 Resolution::Units::PPCM));
56 break;
57 case 0: /* Resolution undefined */
58 /* FALLTHROUGH */
59 default:
60 setResolution(Resolution(0,0));
61 break;
62 }
63 free(JFIFHeader);
64
65 /* Step through any tables up to the "start of frame" marker */
66 uint16_t tableSize;
67 for (;;) {
68 if (biomeval_nbis_getc_marker_jpegl(&marker, TBLS_N_SOF, &markerBuf, endPtr))
69 throw Error::DataError("Could not read to TBLS_N_SOF");
70
71 if (marker == SOF3)
72 break;
73
74 if (biomeval_nbis_getc_ushort(&tableSize, &markerBuf, endPtr))
75 throw Error::DataError("Could not read size of table");
76 /* Table size includes size of field but not the marker */
77 markerBuf += tableSize - sizeof(tableSize);
78 }
79
80 /* Parse frame header for depth and dimension information */
81 FRM_HEADER_JPEGL *frameHeader;
82 if (biomeval_nbis_getc_frame_header_jpegl(&frameHeader, &markerBuf, endPtr))
83 throw Error::DataError("Could not read frame header");
84 setColorDepth((uint16_t)frameHeader->Nf * 8);
85 this->setBitDepth(8);
86 this->setHasAlphaChannel(false);
87 setDimensions(Size(frameHeader->x, frameHeader->y));
88 free(frameHeader);
89}

138bool
139BiometricEvaluation::Image::JPEGL::isJPEGL(
140 const uint8_t *data,
141 uint64_t size)
142{
143 uint8_t *markerBuf = (uint8_t *)data;
144 uint8_t *endPtr = (uint8_t *)data + size;
145
146 /*
147 * JPEG markers (ISO/IEC 10918-1:1993)
148 */
149 static const uint16_t startOfScan = 0xFFDA;
150 static const uint16_t startOfImage = 0xFFD8;
151
152 /* Start of frame, non-differential, Huffman coding */
153 static const uint16_t SOFBaselineDCT = 0xFFC0;
154 static const uint16_t SOFExtendedSequentialDCT = 0xFFC1;
155 static const uint16_t SOFProgressiveDCT = 0xFFC2;
156 static const uint16_t SOFLosslessSequential = 0xFFC3;
157 /* Start of frame, differential, Huffman coding */
158 static const uint16_t SOFDifferentialSequentialDCT = 0xFFC5;
159 static const uint16_t SOFDifferentialProgressiveDCT = 0xFFC6;
160 static const uint16_t SOFDifferentialLossless = 0xFFC7;
161 /* Start of frame, non-differential, arithmetic coding */
162 static const uint16_t SOFExtendedSequentialDCTArith = 0xFFC9;
163 static const uint16_t SOFProgressiveDCTArith = 0xFFCA;
164 static const uint16_t SOFLosslessArith = 0xFFCB;
165 /* Start of frame, differential, arithmetic coding */
166 static const uint16_t SOFDifferentialSequentialDCTArith = 0xFFCD;
167 static const uint16_t SOFDifferentialProgressiveDCTArith = 0xFFCE;
168 static const uint16_t SOFDifferentialLosslessArith = 0xFFCF;
169
170 /* First marker should be start of image */
171 uint16_t marker;
172 if (biomeval_nbis_getc_ushort(&marker, &markerBuf, endPtr) != 0)
173 return (false);
174 if (marker != startOfImage)
175 return (false);
176
177 /* Read markers until end of buffer or an identifying marker is found */
178 for (;;) {
179 /* Get next 16 bits */
180 if (biomeval_nbis_getc_ushort(&marker, &markerBuf, endPtr) != 0)
181 return (false);
182
183 /* 16-bit markers start with 0xFF but aren't 0xFF00 or 0xFFFF */
184 while (((marker >> 8) != 0xFF) &&
185 ((marker == 0xFF00) || (marker == 0xFFFF)))
186 if (biomeval_nbis_getc_ushort(&marker, &markerBuf, endPtr) != 0)
187 return (false);
188
189 switch (marker) {
190 /* Lossless start of frame markers */
191 case SOFLosslessSequential:
192 /* FALLTHROUGH */
193 case SOFDifferentialLossless:
194 /* FALLTHROUGH */
195 case SOFLosslessArith:
196 /* FALLTHROUGH */
197 case SOFDifferentialLosslessArith:
198 return (true);
199
200 /* Lossy start of frame markers */
201 case SOFBaselineDCT:
202 /* FALLTHROUGH */
203 case SOFExtendedSequentialDCT:
204 /* FALLTHROUGH */
205 case SOFProgressiveDCT:
206 /* FALLTHROUGH */
207 case SOFDifferentialSequentialDCT:
208 /* FALLTHROUGH */
209 case SOFDifferentialProgressiveDCT:
210 /* FALLTHROUGH */
211 case SOFExtendedSequentialDCTArith:
212 /* FALLTHROUGH */
213 case SOFProgressiveDCTArith:
214 /* FALLTHROUGH */
215 case SOFDifferentialSequentialDCTArith:
216 /* FALLTHROUGH */
217 case SOFDifferentialProgressiveDCTArith:
218 /* FALLTHROUGH */
219
220 /* Start of scan found before a start of frame */
221 case startOfScan:
222 return (false);
223 }
224
225 /* Reposition marker pointer after current marker segment */
226 if (JPEG::getc_skip_marker_segment(marker, &markerBuf, endPtr))
227 return (false);
228 }
229
230 return (false);
231}
