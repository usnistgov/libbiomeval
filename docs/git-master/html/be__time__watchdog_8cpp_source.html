81void
82BiometricEvaluation::Time::Watchdog::start()
83{
84 if (_interval == 0) {
85 return;
86 }
87
88 struct sigaction sa{};
89 int signo;
90 int which;
91 internalMapWatchdogType(&signo, &which);
92 sigemptyset(&sa.sa_mask);
93 sa.sa_flags = SA_SIGINFO;
94 sa.sa_sigaction = WatchdogSignalHandler;
95 if (sigaction(signo, &sa, nullptr) != 0) {
96 throw (Error::StrategyError("Registering signal handler failed"));
97 }
98 time_t sec, usec;
99 struct itimerval timerval{};
100 /*
101 * Convert _interval to sec/usec from usec
102 */
103 sec = static_cast<time_t>(_interval / Time::MicrosecondsPerSecond);
104 usec = static_cast<time_t>(_interval % Time::MicrosecondsPerSecond);
105 timerclear(&timerval.it_interval);
106 timerval.it_value.tv_sec = sec;
107 timerval.it_value.tv_usec = usec;
108 if (setitimer(which, &timerval, nullptr) != 0) {
109 throw (Error::StrategyError("Registering system timer failed"));
110 }
111}

113void
114BiometricEvaluation::Time::Watchdog::stop()
115{
116 struct sigaction sa{};
117 int signo;
118 int which;
119 internalMapWatchdogType(&signo, &which);
120 sigemptyset(&sa.sa_mask);
121 sa.sa_flags = 0;
122 sa.sa_handler = SIG_DFL;
123 if (sigaction(signo, &sa, nullptr) == -1) {
124 throw (Error::StrategyError("Clearing signal handler failed"));
125 }
126
127 /*
128 * Cancel the timer.
129 */
130 struct itimerval timerval{};
131 timerclear(&timerval.it_interval);
132 timerclear(&timerval.it_value);
133 if (setitimer(which, &timerval, nullptr) != 0) {
134 throw (Error::StrategyError("Clearing system timer failed"));
135 }
136}
