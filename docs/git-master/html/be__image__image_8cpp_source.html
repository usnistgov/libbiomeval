111BiometricEvaluation::Memory::uint8Array
112BiometricEvaluation::Image::Image::getRawData(
113 const bool removeAlphaChannelIfPresent)
114 const
115{
116 if (!removeAlphaChannelIfPresent || !this->hasAlphaChannel())
117 return (this->getRawData());
118
119 /* Set the last channel to be removed */
120 std::vector<bool> components(this->getColorDepth() /
121 this->getBitDepth(), false);
122 *(std::prev(components.end(), 1)) = true;
123
124 return (BiometricEvaluation::Image::removeComponents(this->getRawData(),
125 this->getBitDepth(), components));
126}

128BiometricEvaluation::Memory::uint8Array
129BiometricEvaluation::Image::Image::getRawGrayscaleData(
130 uint8_t depth)
131 const
132{
133 if (depth != 16 && depth != 8 && depth != 1)
134 throw Error::ParameterError("Invalid value for bit depth");
135
136 /* Return no-effort conversion */
137 if (this->getColorDepth() == depth)
138 return (this->getRawData());
139
140 const uint8_t bpcIn = static_cast<uint8_t>(
141 std::ceil(this->getColorDepth() / 8.0));
142 const Memory::uint8Array rawColor{this->getRawData()};
143 Memory::IndexedBuffer inBuffer{rawColor};
144
145 const uint8_t bpcOut = static_cast<uint8_t>(std::ceil(depth / 8.0));
146 Memory::uint8Array rawGray(
147 bpcOut * this->getDimensions().xSize * this->getDimensions().ySize);
148 Memory::MutableIndexedBuffer outBuffer(rawGray);
149
150 /* Constants from ITU-R BT.601 */
151 static const float redFactor = 0.299;
152 static const float greenFactor = 0.587;
153 static const float blueFactor = 0.114;
154
155 uint16_t rValue, bValue, gValue;
156
157 /*
158 * Convert to 16-bit or 8-bit. 1,2,4-bit conversions will be quantized
159 * after converting to 8-bit.
160 */
161 for (uint32_t i = 0; i < rawColor.size(); i += bpcIn) {
162 switch (this->getColorDepth()) {
163 case 1:
164 /* Images are upped to 8-bit in getRawData() */
165 /* FALLTHROUGH */
166 case 2:
167 /* Images are upped to 8-bit in getRawData() */
168 /* FALLTHROUGH */
169 case 4:
170 /* Images are upped to 8-bit in getRawData() */
171 /* FALLTHROUGH */
172 case 8: /* 8-bit single-channel (grayscale) */
173 if (depth == 8) {
174 outBuffer.pushU8Val(inBuffer.scanU8Val());
175 } else if (depth == 16) {
176 /* Get 16-bit colorspace interpolated value */
177 outBuffer.pushU16Val(valueInColorspace(
178 inBuffer.scanU8Val(), UINT8_MAX, 16));
179 }
180
181 break;
182 case 16: /* 16-bit single-channel (grayscale) */
183 if (depth == 8) {
184 /* Interpolate value in 8-bit colorspace */
185 outBuffer.pushU8Val(valueInColorspace(
186 inBuffer.scanU16Val(), UINT16_MAX, 8));
187 }
188
189 break;
190 case 32: /* 8-bit RGBA */
191 /* FALLTHROUGH (ignoring alpha channel) */
192 case 24: /* 8-bit RGB */
193 if (depth == 16) {
194 /* Get 16-bit colorspace interpolated value */
195 rValue = static_cast<uint16_t>(
196 valueInColorspace(inBuffer.scanU8Val(),
197 UINT8_MAX, 16));
198 gValue = static_cast<uint16_t>(
199 valueInColorspace(inBuffer.scanU8Val(),
200 UINT8_MAX, 16));
201 bValue = static_cast<uint16_t>(
202 valueInColorspace(inBuffer.scanU8Val(),
203 UINT8_MAX, 16));
204
205 /* Pull Y' component from Y'CbCr */
206 outBuffer.pushU16Val((rValue * redFactor) +
207 (gValue * greenFactor) +
208 (bValue * blueFactor));
209 } else {
210 /* Pull Y' component from Y'CbCr */
211 outBuffer.pushU8Val(static_cast<uint8_t>(
212 (inBuffer.scanU8Val() * redFactor) +
213 (inBuffer.scanU8Val() * greenFactor) +
214 (inBuffer.scanU8Val() * blueFactor)));
215 }
216
217 /* Increment over the alpha channel */
218 if (this->getColorDepth() == 32)
219 inBuffer.scanU8Val();
220
221 break;
222 case 64: /* 16-bit RGBA */
223 /* FALLTHROUGH (ignoring alpha channel) */
224 case 48: /* 16-bit RGB */
225 /* Extract 16-bit values */
226 rValue = inBuffer.scanU16Val();
227 gValue = inBuffer.scanU16Val();
228 bValue = inBuffer.scanU16Val();
229
230 if (depth == 16) {
231 /* Y' component from Y'CbCr */
232 outBuffer.pushU16Val((rValue * redFactor) +
233 (gValue * greenFactor) +
234 (bValue * blueFactor));
235 } else {
236 /* Interpolate colors in 8-bit colorspace */
237 rValue = static_cast<uint8_t>(
238 valueInColorspace(rValue, UINT16_MAX, 8));
239 gValue = static_cast<uint8_t>(
240 valueInColorspace(gValue, UINT16_MAX, 8));
241 bValue = static_cast<uint8_t>(
242 valueInColorspace(bValue, UINT16_MAX, 8));
243
244 /* Pull Y' component from Y'CbCr */
245 outBuffer.pushU8Val((rValue * redFactor) +
246 (gValue * greenFactor) +
247 (bValue * blueFactor));
248 }
249
250 /* Increment over the alpha channel */
251 if (this->getColorDepth() == 64)
252 inBuffer.scanU16Val();
253
254 break;
255 default:
256 throw BE::Error::NotImplemented("Grayscale conversion "
257 "for " + std::to_string(this->getColorDepth()) + "-bit "
258 "depth imagery");
259 }
260 }
261
262 /* Quantize down to black and white */
263 if (depth == 1)
264 std::transform(rawGray.begin(), rawGray.end(), rawGray.begin(),
265 [](const uint8_t &i) { return (i <= 127 ? 0x00 : 0xFF); });
266
267 return (rawGray);
268}

324uint64_t
325BiometricEvaluation::Image::Image::valueInColorspace(
326 uint64_t color,
327 uint64_t maxColorValue,
328 uint8_t depth)
329{
330 /*
331 * Solve for X in:
332 *
333 * color X
334 * ------------- = -------------
335 * maxColorValue 2^(depth) - 1
336 */
337
338 return ((((uint64_t)pow(2.0, depth) - 1) * color) / maxColorValue);
339}

341std::shared_ptr<BiometricEvaluation::Image::Image>
342BiometricEvaluation::Image::Image::openImage(
343 const uint8_t *data,
344 const uint64_t size,
345 const std::string &identifier,
346 const statusCallback_t &statusCallback)
347{
348 switch (Image::getCompressionAlgorithm(data, size)) {
349 case CompressionAlgorithm::JPEGB:
350 return (std::shared_ptr<Image>(new JPEG(data, size,
351 identifier, statusCallback)));
352 case CompressionAlgorithm::JPEGL:
353 return (std::shared_ptr<Image>(new JPEGL(data, size,
354 identifier, statusCallback)));
355 case CompressionAlgorithm::JP2:
356 /* FALLTHROUGH */
357 case CompressionAlgorithm::JP2L:
358 return (std::shared_ptr<Image>(new JPEG2000(data, size,
359 identifier, statusCallback)));
360 case CompressionAlgorithm::PNG:
361 return (std::shared_ptr<Image>(new PNG(data, size,
362 identifier, statusCallback)));
363 case CompressionAlgorithm::NetPBM:
364 return (std::shared_ptr<Image>(new NetPBM(data, size,
365 identifier, statusCallback)));
366 case CompressionAlgorithm::WSQ20:
367 return (std::shared_ptr<Image>(new WSQ(data, size,
368 identifier, statusCallback)));
369 case CompressionAlgorithm::BMP:
370 return (std::shared_ptr<Image>(new BMP(data, size,
371 identifier, statusCallback)));
372 case CompressionAlgorithm::TIFF:
373 return (std::shared_ptr<Image>(new TIFF(data, size,
374 identifier, statusCallback)));
375 default:
376 throw Error::StrategyError("Could not determine compression "
377 "algorithm");
378 }
379}

424BiometricEvaluation::Image::CompressionAlgorithm
425BiometricEvaluation::Image::Image::getCompressionAlgorithm(
426 const uint8_t *data,
427 const uint64_t size)
428{
429 if (NetPBM::isNetPBM(data, size))
430 return (CompressionAlgorithm::NetPBM);
431 else if (JPEG2000::isJPEG2000(data, size))
432 return (CompressionAlgorithm::JP2);
433 else if (JPEG::isJPEG(data, size))
434 return (CompressionAlgorithm::JPEGB);
435 else if (JPEGL::isJPEGL(data, size))
436 return (CompressionAlgorithm::JPEGL);
437 else if (PNG::isPNG(data, size))
438 return (CompressionAlgorithm::PNG);
439 else if (BMP::isBMP(data, size))
440 return (CompressionAlgorithm::BMP);
441 else if (WSQ::isWSQ(data, size))
442 return (CompressionAlgorithm::WSQ20);
443 else if (TIFF::isTIFF(data, size))
444 return (CompressionAlgorithm::TIFF);
445
446 return (CompressionAlgorithm::None);
447}
