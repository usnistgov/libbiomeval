36int32_t
37BiometricEvaluation::Process::MessageCenterListener::workerMain()
38{
39 this->parseArgs();
40
41 try {
42 this->setupSocket();
43 this->listen();
44 } catch (Error::Exception &e) {
45 return (EXIT_FAILURE);
46 }
47
48 int client;
49 std::shared_ptr<BE::Process::WorkerController> wc;
50 BE::Memory::uint8Array message(0);
51 while (!this->stopRequested()) {
52 /*
53 * Check for new connections.
54 */
55 try {
56 if (MessageCenterUtility::dataAvailable(
57 this->_socket, MessageCenter::DEFAULT_TIMEOUT))
58 this->spawnReceiver(this->accept());
59 } catch (Error::Exception &e) {
60 /* TODO */
61 }
62
63 /*
64 * Read and forward messages from client to manager.
65 */
66 try {
67 if (this->_manager->getNextMessage(wc, message,
68 MessageCenter::DEFAULT_TIMEOUT)) {
69 client = -1;
70 for (auto it = this->_clientMap.cbegin();
71 it != this->_clientMap.cend(); it++)
72 if (it->second == wc)
73 client = it->first;
74 if (client != -1) {
75 /* Prepend client ID on message */
76 message = MessageCenterUtility::
77 setClientID(client, message);
78
79 /* Forward the message onward */
80 this->sendMessageToManager(message);
81 }
82 }
83 } catch (Error::Exception &e) {
84 /* TODO */
85 }
86
87 /*
88 * Read and forward messages from manager to client.
89 */
90 try {
91 if (this->waitForMessage(
92 MessageCenter::DEFAULT_TIMEOUT)) {
93 this->receiveMessageFromManager(message);
94
95 /* Decode client ID from message */
96 client = MessageCenterUtility::
97 getClientID(message);
98
99 /*
100 * Check for known messages.
101 */
102
103 /* Disconnect. */
104 if (to_string(MessageCenterUtility::getMessage(
105 message)) == MessageCenterReceiver::
106 MSG_DISCONNECT)
107 this->_manager->stopWorker(
108 this->_clientMap[client]);
109
110 /* Not a known message. */
111 else
112 this->_clientMap[client]->
113 sendMessageToWorker(message);
114 }
115 } catch (BE::Error::Exception &e) {
116 /* TODO */
117 }
118
119 }
120
121 this->tearDown();
122 return (EXIT_SUCCESS);
123}

137void
138BiometricEvaluation::Process::MessageCenterListener::spawnReceiver(
139 int clientSocket)
140{
141 std::shared_ptr<WorkerController> wc = this->_manager->addWorker(
142 std::shared_ptr<Worker>(new MessageCenterReceiver()));
143 wc->setParameterFromInteger(
144 MessageCenterReceiver::PARAM_CLIENT_SOCKET, clientSocket);
145
146 /* TODO: Lock this ID, once the semaphore code is in refactoring. */
147 static uint32_t CLIENT_ID = 0;
148 wc->setParameterFromInteger(
149 MessageCenterReceiver::PARAM_CLIENT_ID, ++CLIENT_ID);
150 this->_manager->startWorker(wc, false, true);
151
152 this->_clientMap[CLIENT_ID] = wc;
153
154 /* We don't need the client socket */
155 ::close(clientSocket);
156}

170void
171BiometricEvaluation::Process::MessageCenterListener::setupSocket()
172{
173 struct addrinfo hints;
174 std::memset(&hints, 0, sizeof(hints));
175 hints.ai_family = AF_UNSPEC;
176 hints.ai_socktype = SOCK_STREAM;
177 hints.ai_flags = AI_PASSIVE;
178
179 struct addrinfo *addrs;
180 int rv = ::getaddrinfo(NULL, std::to_string(this->_port).c_str(),
181 &hints, &addrs);
182 if (rv != 0) {
183 std::string errorMessage = gai_strerror(rv);
184 throw BE::Error::StrategyError("getaddrinfo() -- " +
185 errorMessage);
186 }
187
188 /* Bind to the first available address */
189 int reuse = 1;
190 struct addrinfo *addr;
191 for (addr = addrs; addr != NULL; addr = addr->ai_next) {
192 this->_socket = ::socket(addr->ai_family, addr->ai_socktype,
193 addr->ai_protocol);
194 if (this->_socket != -1) {
195 ::setsockopt(this->_socket, SOL_SOCKET, SO_REUSEADDR,
196 &reuse, sizeof(uint32_t));
197 if (::bind(this->_socket, addr->ai_addr,
198 addr->ai_addrlen) == -1)
199 close(this->_socket);
200 else
201 break;
202 }
203 }
204 if (this->_socket == -1)
205 throw BE::Error::StrategyError("Failed to bind socket");
206
207 this->_addr = addr;
208}

210int
211BiometricEvaluation::Process::MessageCenterListener::accept()
212{
213 struct sockaddr_storage clientAddr;
214 socklen_t clientAddrSize;
215
216 int clientSocket;
217 bool stop = false;
218 while (!stop) {
219 errno = 0;
220 clientSocket = ::accept(this->_socket,
221 (struct sockaddr *)&clientAddr, &clientAddrSize);
222 if (clientSocket == -1) {
223 switch (errno) {
224 case EINTR:
225 continue;
226 default:
227 stop = true;
228 throw BE::Error::StrategyError("accept() -- " +
229 BE::Error::errorStr());
230 }
231 } else
232 stop = true;
233 }
234
235 return (clientSocket);
236}
