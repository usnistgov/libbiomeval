18/*
19 * Setup access to the container stream using FFMPEG libraries.
20 */
21void
22BiometricEvaluation::Video::StreamImpl::openContainer()
23{
24 this->_currentFrameNum = 0;
25 this->_currentFrameTS = 0;
26
27 this->_fmtCtx = avformat_alloc_context();
28 if (this->_fmtCtx == nullptr)
29 throw BE::Error::MemoryError("Could not allocate format context");
30 /* fill opaque structure used by the AVIOContext read callback */
31 this->_IOCtxBufferData.ptr = *this->_containerBuf.get();
32 this->_IOCtxBufferData.size = this->_containerBuf->size();
33 this->_IOCtxBufferData.pos = 0;
34
35 uint8_t *ctxBuf = nullptr;
36 ctxBuf = (uint8_t *)av_malloc(AVIOCTXBUFFERSIZE);
37 if (ctxBuf == nullptr)
38 throw BE::Error::MemoryError(
39 "Could not allocate IO context buffer");
40
41 this->_avioCtx = avio_alloc_context(
42 ctxBuf, AVIOCTXBUFFERSIZE,
43 0, &this->_IOCtxBufferData,
44 &BE::Video::read_packet, NULL, &BE::Video::seek);
45 if (this->_avioCtx == nullptr) {
46 av_free(ctxBuf);
47 throw BE::Error::MemoryError(
48 "Could not allocate IO context");
49 }
50
51 this->_fmtCtx->pb = this->_avioCtx;
52
53 int ret = avformat_open_input(&this->_fmtCtx, NULL, NULL, NULL);
54 if (ret < 0)
55 throw (Error::StrategyError("Could not read container"));
56
57 ret = avformat_find_stream_info(this->_fmtCtx, NULL);
58 if (ret < 0)
59 throw (Error::StrategyError(
60 "Could not find stream information"));
61
62 /*
63 * Create a codec context for the stream using the codec that
64 * was used for the stream. This context will be closed whenever
65 * the container is closed.
66 */
67 AVCodec *codec = avcodec_find_decoder(
68 this->_fmtCtx->streams[this->_streamIndex]->codec->codec_id);
69 if (codec == nullptr)
70 throw (Error::StrategyError("Unsupported codec"));
71
72 this->_codecCtx = avcodec_alloc_context3(nullptr);
73 if (this->_codecCtx == nullptr)
74 throw (Error::MemoryError(
75 "Could not allocate codec context"));
76 /*
77 * Copy all the settings from the codec allocated by the
78 * library into our codec. This is necessary for certain
79 * stream types, H264 at least.
80 */
81 avcodec_copy_context(this->_codecCtx,
82 this->_fmtCtx->streams[this->_streamIndex]->codec);
83
84 AVDictionary *opts = NULL;
85 av_dict_set(&opts, "refcounted_frames", "1", 0);
86 if (avcodec_open2(this->_codecCtx, codec, &opts) < 0 )
87 throw (Error::StrategyError("Could not open codec context"));
88 av_dict_free(&opts);
89
90 this->_swsCtx = nullptr;
91}

109/*
110 * Tear down an open stream be releasing FFMPEG library objects.
111 */
112void
113BiometricEvaluation::Video::StreamImpl::closeContainer()
114{
115 /*
116 * NOTE: The internal buffer could have changed, and not be
117 * the buffer we allocated. We don't need to free that buffer
118 * here.
119 */
120 if (this->_avioCtx != nullptr) {
121 av_freep(&this->_avioCtx->buffer);
122 av_freep(&this->_avioCtx);
123 }
124 if (this->_fmtCtx != nullptr) {
125 avformat_close_input(&this->_fmtCtx);
126 avformat_free_context(this->_fmtCtx);
127 }
128 if (this->_codecCtx != nullptr) {
129 avcodec_free_context(&this->_codecCtx);
130 }
131 if (this->_swsCtx != nullptr) {
132 sws_freeContext(this->_swsCtx);
133 }
134}

165uptrAVFrame
166BiometricEvaluation::Video::StreamImpl::getNextAVFrame()
167{
168 /* Allocate video frame for the FFMPEG internal encoded data */
169 AVFrame *frameNative;
170 frameNative = av_frame_alloc();
171 if (frameNative == nullptr)
172 throw (BE::Error::StrategyError("Could not allocate frame"));
173 uptrAVFrame pFrame(frameNative, freeAVFrame);
174
175 /*
176 * Read the stream to get the next frame. Note that the stream position
177 * within the context depends on previous calls to this function.
178 */
179 int gotFrame = 0;
180 AVPacket packet;
181 av_init_packet(&packet);
182 packet.size = 0;
183 packet.data = nullptr;
184
185 /*
186 * Grab the next frame from the video stream.
187 */
188 while(av_read_frame(this->_fmtCtx, &packet) >= 0) {
189 if(packet.stream_index == (int)this->_streamIndex) {
190 avcodec_decode_video2(
191 this->_codecCtx, frameNative, &gotFrame, &packet);
192 av_packet_unref(&packet);
193 if (gotFrame != 0) {
194 break;
195 } else {
196 av_frame_unref(frameNative);
197 }
198 } else {
199 av_packet_unref(&packet);
200 }
201 }
202 /*
203 * We need to flush any cached frames.
204 * See http://ffmpeg.org/doxygen/trunk/decoding_encoding_8c-example.html
205 */
206 //XXX check packet.stream_index?
207 if (gotFrame == 0) {
208 packet.size = 0;
209 packet.data = nullptr;
210 avcodec_decode_video2(
211 this->_codecCtx, frameNative, &gotFrame, &packet);
212 av_packet_unref(&packet);
213 }
214 if (gotFrame == 0) {
215 throw (BE::Error::ParameterError("Frame could not be found"));
216 }
217 this->_currentFrameNum++;
218 this->_currentFrameTS = av_frame_get_best_effort_timestamp(frameNative);
219 return (pFrame);
220}

222/*
223 * This function uses the scaling context from FFMPEG, part of this
224 * object's state data, and that context is essentially managed by the
225 * FFMPEG library. Therefore, this is a member function so the context
226 * pointer can be updated.
227 */
228BiometricEvaluation::Video::Frame
229BiometricEvaluation::Video::StreamImpl::convertAVFrame(
230 AVFrame *frameNative)
231{
232 BE::Video::Frame staticFrame;
233 staticFrame.size.xSize = this->_codecCtx->width * this->_xScale;
234 staticFrame.size.ySize = this->_codecCtx->height * this->_yScale;
235 staticFrame.timestamp = av_frame_get_best_effort_timestamp(frameNative);
236
237 /* Calculate the size of the decoded frame */
238 int frameSize = av_image_get_buffer_size(
239 this->_avPixelFormat,
240 staticFrame.size.xSize, staticFrame.size.ySize, 1);
241
242 /*
243 * Reuse the scaling context, if possible. If there is more than
244 * one video stream, with different codec parameters (width, etc.)
245 * then a new scaling context will be allocated, the old one
246 * being free'd.
247 */
248 this->_swsCtx = sws_getCachedContext(
249 this->_swsCtx,
250 this->_codecCtx->width, this->_codecCtx->height,
251 this->_codecCtx->pix_fmt,
252 staticFrame.size.xSize, staticFrame.size.ySize,
253 this->_avPixelFormat,
254 SWS_ACCURATE_RND, nullptr, nullptr, nullptr);
255
256 //XXX We can either allocate a buffer for the output frame by using
257 //XXX the ffmpeg library function, or we can create an AutoArray
258 //XXX and associate that with the frame. The library function is
259 //XXX supposed to take care of alignment.
260/*
261 uint8_t *buffer = (uint8_t *)av_malloc(frameSize * sizeof(uint8_t));
262 std::unique_ptr<uint8_t, decltype(freeAVBuffer)>
263 pBuffer(buffer, freeAVBuffer);
264 if (buffer == nullptr)
265 throw (Error::StrategyError("Could not allocate frame buffer"));
266*/
267 /* Allocate video frame for the for the output frame data */
268 AVFrame *frameOut;
269 frameOut = av_frame_alloc();
270 if (frameOut == nullptr)
271 throw (BE::Error::StrategyError("Could not allocate frame"));
272 uptrAVFrame pFrameOut(frameOut, freeAVFrame);
273
274 staticFrame.data.resize(frameSize);
275 av_image_fill_arrays(frameOut->data, frameOut->linesize,
276 &staticFrame.data[0], this->_avPixelFormat, staticFrame.size.xSize,
277 staticFrame.size.ySize, 1);
278
279 sws_scale(
280 this->_swsCtx, frameNative->data, frameNative->linesize,
281 0, this->_codecCtx->height,
282 frameOut->data, frameOut->linesize);
283
284// XXX Copy is only necessary when the av_malloc'd buffer is used.
285// std::memcpy(frameBuffer, frameOut->data[0], frameSize);
286
287 return (staticFrame);
288}

290BiometricEvaluation::Video::Frame
291BiometricEvaluation::Video::StreamImpl::getFrame(
292 uint32_t frameNum)
293{
294 /*
295 * If the last frame read from the stream is after the requested
296 * frame, reset close and open the container stream and start
297 * reading from the beginning.
298 * XXX The av_seek* functions do not seem to work; try again some time.
299 */
300 if (frameNum <= this->_currentFrameNum) {
301 this->closeContainer();
302 this->openContainer();
303 }
304 /*
305 * Let exceptions float out from here.
306 */
307 while(true) {
308 auto uptrFrame = getNextAVFrame();
309 if (frameNum == this->_currentFrameNum) {
310 AVFrame *frameNative = uptrFrame.get();
311 return (convertAVFrame(frameNative));
312 }
313 }
314}

316std::vector<BiometricEvaluation::Video::Frame>
317BiometricEvaluation::Video::StreamImpl::getFrameSequence(
318 int64_t startTime,
319 int64_t endTime)
320{
321 uint32_t streamIdx = this->_streamIndex;
322 int64_t startTS = av_rescale(
323 startTime,
324 this->_fmtCtx->streams[streamIdx]->time_base.den,
325 this->_fmtCtx->streams[streamIdx]->time_base.num);
326 startTS /= BE::Time::MillisecondsPerSecond;
327 int64_t endTS = av_rescale(
328 endTime,
329 this->_fmtCtx->streams[streamIdx]->time_base.den,
330 this->_fmtCtx->streams[streamIdx]->time_base.num);
331 endTS /= BE::Time::MillisecondsPerSecond;
332
333 /*
334 * If the last scanned frame has a time stamp later than
335 * the time of the requested start of sequence, then
336 * we need to start at the beginning of the container so
337 * we can grab frames at any point. This is the equivalent
338 * of seeking to the beginning.
339 */
340 if (this->_currentFrameTS >= startTS) {
341 this->closeContainer();
342 this->openContainer();
343 }
344
345 std::vector<BE::Video::Frame> frames;
346 while (true) {
347 try {
348 auto uptrFrame = getNextAVFrame();
349 AVFrame *frameNative = uptrFrame.get();
350 if (this->_currentFrameTS > endTS) {
351 break; /* past the point of caring */
352 }
353 if ((this->_currentFrameTS >= startTS)
354 && (this->_currentFrameTS <= endTS)) {
355 auto frame = convertAVFrame(frameNative);
356 frames.push_back(std::move(frame));
357 }
358 } catch (const Error::ParameterError&) {
359 break; /* Ran out of frames */
360 }
361 }
362 return (frames);
363}
