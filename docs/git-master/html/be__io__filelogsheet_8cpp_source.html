24/*
25 * Helper function to parse a string and pull the pathname from it,
26 * handling 'real' URL with 'file://' and 'fake' URLs that are just
27 * a pathname.
28 */
29static bool
30parseURL(const std::string &url, std::string &pathname)
31{
32 /*
33 * Check for a file URL and parse that if found.
34 */
35 try {
36 BE::IO::Logsheet::Kind lsType =
37 BE::IO::Logsheet::getTypeFromURL(url);
38 if (lsType != BE::IO::Logsheet::Kind::File)
39 return (false);
40 } catch (const BE::Error::Exception&) {
41 /* Assume we have no scheme, so return the URL as pathname */
42 pathname = url;
43 return (true);
44 }
45 /*
46 * Return the path name as everything after the scheme.
47 */
48 std::string::size_type start = url.find("://");
49 pathname = url.substr(start + 3, std::string::npos);
50 return (true);
51}

53BiometricEvaluation::IO::FileLogsheet::FileLogsheet(
54 const std::string &url,
55 const std::string &description) :
56 Logsheet(),
57 _cursor(0)
58{
59 std::string pathname;
60 if (parseURL(url, pathname) != true)
61 throw Error::ParameterError("Malformed URL");
62 if (IO::Utility::fileExists(pathname))
63 throw Error::ObjectExists();
64
65 /* Open the log sheet file as a file output stream */
66 std::fstream *ofs = new std::fstream(pathname.c_str(),
67 std::ios_base::out);
68 if (!ofs || ofs->fail())
69 throw Error::StrategyError("Could not open FileLogsheet file");
70
71 _theLogFile.reset(ofs);
72 *_theLogFile << BE::IO::Logsheet::DescriptionTag << " " << description
73 << std::endl;
74 if (_theLogFile->fail())
75 throw Error::StrategyError("Could not write description to "
76 "log file");
77
78 _sequenceFile.reset(new std::fstream(pathname.c_str(), in));
79 if (_sequenceFile->fail())
80 throw Error::StrategyError("Could not open FileLogsheet sequencer");
81}

83BiometricEvaluation::IO::FileLogsheet::FileLogsheet(
84 const std::string &url) :
85 Logsheet(),
86 _cursor(0)
87{
88 std::string pathname;
89 if (parseURL(url, pathname) != true)
90 throw Error::ParameterError("Malformed URL");
91 if (!IO::Utility::fileExists(pathname))
92 throw Error::ObjectDoesNotExist();
93
94 /* Open the log sheet file as a file input stream so we can
95 * obtain the last entry number. */
96 std::ifstream ifs(pathname.c_str(), std::ios_base::in);
97 if (ifs.fail())
98 throw Error::StrategyError("Could not open FileLogsheet file");
99
100 /*
101 * Determine the current entry number by counting lines that
102 * begin with the entry delimiter, etc.
103 */
104 std::string oneline;
105 while (!ifs.eof()) {
106 std::getline(ifs, oneline);
107 if (BE::IO::Logsheet::lineIsEntry(oneline))
108 this->incrementEntryNumber();
109 }
110
111 /* Open the log sheet file as a file output stream */
112 std::fstream *ofs = new std::fstream(pathname.c_str(), app | out);
113 if (!ofs || ofs->fail())
114 throw Error::StrategyError("Could not open FileLogsheet file");
115
116 _theLogFile.reset(ofs);
117
118 _sequenceFile.reset(new std::fstream(pathname.c_str(), in));
119 if (_sequenceFile->fail())
120 throw Error::StrategyError("Could not open FileLogsheet sequencer");
121}

125{
126 if (this->getCommit() == false)

135 << " to log file";

137 }

179std::string
180BiometricEvaluation::IO::FileLogsheet::sequence(
181 bool allEntries,
182 bool trim,
183 int32_t cursor)
184{
185 if ((cursor != BE_FILELOGSHEET_SEQ_START) &&
186 (cursor != BE_FILELOGSHEET_SEQ_NEXT))
187 throw Error::StrategyError("Invalid cursor position as "
188 "argument");
189
190 /* Sync to make sure that fstream knows about recent writes */
191 _sequenceFile->sync();
192 /* Reset EOF */
193 _sequenceFile->clear();
194
195 /* Reset stream and cursor when starting over */
196 if ((cursor == BE_FILELOGSHEET_SEQ_START) ||
197 ((_cursor == 0) && (cursor == BE_FILELOGSHEET_SEQ_NEXT))) {
198 _sequenceFile->seekg(0, std::ios::beg);
199 this->updateCursor();
200 } else {
201 _sequenceFile->seekg(_cursor, std::ios::beg);
202 _sequenceFile->peek();
203 }
204
205 /* Check that new cursor position is not the EOF */
206 if (_sequenceFile->eof())
207 throw Error::ObjectDoesNotExist();
208
209 /* Get next entry, skipping any comments, debug, or descriptions */
210 std::string entry = "", line = "";
211 while (true) {
212 std::getline(*_sequenceFile, entry);
213 if (_sequenceFile->eof())
214 throw Error::ObjectDoesNotExist();
215 if (_sequenceFile->fail())
216 throw Error::StrategyError("Priming sequence "
217 "read failed");
218
219 /* Skip description, comments, and debug */
220 if ((allEntries == false) &&
221 (BE::IO::Logsheet::lineIsEntry(entry) == true))
222 break;
223
224 /* or just skip description */
225 if ((allEntries == true) &&
226 ((BE::IO::Logsheet::lineIsEntry(entry) == true) ||
227 (BE::IO::Logsheet::lineIsComment(entry) == true) ||
228 (BE::IO::Logsheet::lineIsDebug(entry) == true)))
229 break;
230 }
231
232 /* Check for multiline entry */
233 this->updateCursor();
234 std::getline(*_sequenceFile, line);
235 while ((_sequenceFile->fail() == false) &&
236 (BE::IO::Logsheet::lineIsEntry(line) == false) &&
237 (BE::IO::Logsheet::lineIsComment(line) == false) &&
238 (BE::IO::Logsheet::lineIsDebug(line) == false)) {
239 entry += '\n' + line;
240 this->updateCursor();
241 std::getline(*_sequenceFile, line);
242 }
243 if (_sequenceFile->eof())
244 return (trim ? BE::IO::Logsheet::trim(entry) : entry);
245 if (_sequenceFile->fail())
246 throw Error::StrategyError("Failed sequencing multiline entry");
247
248 return (trim ? BE::IO::Logsheet::trim(entry) : entry);
249}

251void
252BiometricEvaluation::IO::FileLogsheet::mergeLogsheets(
253 std::vector<std::shared_ptr<FileLogsheet>> &logSheets)
254{
255 if (logSheets.size() == 1)
256 return;
257
258 std::shared_ptr<FileLogsheet> master;
259 try {
260 master = logSheets.at(0);
261 } catch (const std::out_of_range&) {
262 throw Error::StrategyError("out_of_range 0");
263 }
264
265 bool firstEntry = true;
266 std::vector<std::shared_ptr<FileLogsheet>>::const_iterator it;
267 std::string entry;
268 for (it = logSheets.begin() + 1; it != logSheets.end(); it++) {
269 for (;;) {
270 try {
271 if (firstEntry)
272 entry = (*it)->sequence(true, false,
273 BE_FILELOGSHEET_SEQ_START);
274 else
275 entry = (*it)->sequence(true, false,
276 BE_FILELOGSHEET_SEQ_NEXT);
277 } catch (const Error::ObjectDoesNotExist&) {
278 break;
279 }
280
281 if (BE::IO::Logsheet::lineIsComment(entry)) {
282 master->writeComment(
283 BE::IO::Logsheet::trim(entry));
284 } else {
285 if (BE::IO::Logsheet::lineIsDebug(entry)) {
286 master->writeComment(
287 BE::IO::Logsheet::trim(entry));
288 } else {
289 master->write(
290 BE::IO::Logsheet::trim(entry));
291 }
292 }
293 firstEntry = false;
294 }
295
296 master->sync();
297 }
