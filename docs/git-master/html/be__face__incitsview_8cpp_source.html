26/*
27 * The base constructor for all INCITS and ISO face
28 * image data records opens the file and reads the contents
29 * into the state object.
30 */
31BiometricEvaluation::Face::INCITSView::INCITSView(
32 const std::string &filename,
33 const uint32_t viewNumber)
34{
35 FILE *fp;
36 if (!BE::IO::Utility::fileExists(filename)) {
37 throw (BE::Error::FileError("File not found."));
38 }
39 fp = std::fopen(filename.c_str(), "rb");
40 if (fp == NULL) {
41 throw (BE::Error::FileError("Could not open file."));
42 }
43 uint64_t size = IO::Utility::getFileSize(filename);
44 this->_fid.resize(size);
45 if (fread(this->_fid, 1, size, fp) != size){
46 fclose(fp);
47 throw (BE::Error::FileError("Could not read file"));
48 }
49 fclose(fp);
50}

169void
170BiometricEvaluation::Face::INCITSView::readFaceView(
171 BiometricEvaluation::Memory::IndexedBuffer &buf)
172{
173 uint8_t uval8;
174
175 /*
176 * Facial Information block.
177 */
178 uint32_t remainLen = buf.scanBeU32Val();/* Facial Record Data length */
179 uint16_t numFeaturePoints = buf.scanBeU16Val();
180
181 this->_gender = to_enum<BE::Face::Gender>(buf.scanU8Val());
182 this->_eyeColor = to_enum<BE::Face::EyeColor>(buf.scanU8Val());
183 this->_hairColor = to_enum<BE::Face::HairColor>(buf.scanU8Val());
184
185 uint32_t propMask = 0;
186 uval8 = buf.scanU8Val();
187 propMask = (uint32_t)uval8;
188 propMask = propMask << 8;
189 uval8 = buf.scanU8Val();
190 propMask += uval8;
191 propMask = propMask << 8;
192 uval8 = buf.scanU8Val();
193 propMask += uval8;
194
195 if (propMask & 0x01)
196 this->_propertiesConsidered = true;
197 else
198 this->_propertiesConsidered = false;
199
200 if (propMask & 0x00000002)
201 this->_propertySet.push_back(BE::Face::Property::Glasses);
202 if (propMask & 0x00000004)
203 this->_propertySet.push_back(BE::Face::Property::Moustache);
204 if (propMask & 0x00000008)
205 this->_propertySet.push_back(BE::Face::Property::Beard);
206 if (propMask & 0x00000010)
207 this->_propertySet.push_back(BE::Face::Property::Teeth);
208 if (propMask & 0x00000020)
209 this->_propertySet.push_back(BE::Face::Property::Blink);
210 if (propMask & 0x00000040)
211 this->_propertySet.push_back(BE::Face::Property::MouthOpen);
212 if (propMask & 0x00000080)
213 this->_propertySet.push_back(BE::Face::Property::LeftEyePatch);
214 if (propMask & 0x00000100)
215 this->_propertySet.push_back(BE::Face::Property::RightEyePatch);
216 if (propMask & 0x00000200)
217 this->_propertySet.push_back(BE::Face::Property::DarkGlasses);
218 if (propMask & 0x00000400)
219 this->_propertySet.push_back(
220 BE::Face::Property::MedicalCondition);
221
222 this->_expression = to_enum<BE::Face::Expression>(buf.scanBeU16Val());
223
224 BE::Face::PoseAngle pa;
225 pa.yaw = buf.scanU8Val();
226 pa.pitch = buf.scanU8Val();
227 pa.roll = buf.scanU8Val();
228 pa.yawUncertainty = buf.scanU8Val();
229 pa.pitchUncertainty = buf.scanU8Val();
230 pa.rollUncertainty = buf.scanU8Val();
231 this->_poseAngle = pa;
232
233 remainLen = remainLen - 20; /* Sum of above data item lengths */
234
235 /*
236 * Feature point blocks
237 */
238 BE::Feature::MPEGFacePoint fp;
239 for (uint16_t count = 0; count < numFeaturePoints; count++) {
240 fp.type = buf.scanU8Val();
241 uval8 = buf.scanU8Val();
242 fp.major = (uval8 & 0xF0) >> 4;
243 fp.minor = uval8 & 0x0F;
244 uint16_t hp = buf.scanBeU16Val();
245 uint16_t vp = buf.scanBeU16Val();
246 fp.coordinate = BE::Image::Coordinate(hp, vp);
247 (void)buf.scanBeU16Val(); /* reserved field */
248 this->_featurePointSet.push_back(fp);
249 remainLen = remainLen - 8; /* Sum of above data item lengths */
250 }
251
252 /*
253 * Image Information
254 */
255 this->_imageType = to_enum<BE::Face::ImageType>(buf.scanU8Val());
256 this->_imageDataType = to_enum<BE::Face::ImageDataType>( buf.scanU8Val());
257 uint16_t width = buf.scanBeU16Val();
258 uint16_t height = buf.scanBeU16Val();
259 this->setImageSize(BE::Image::Size(width, height));
260
261 switch (this->_imageDataType) {
262 case BE::Face::ImageDataType::JPEG:
263 this->setCompressionAlgorithm(
264 BE::Image::CompressionAlgorithm::JPEGB);
265 break;
266 case BE::Face::ImageDataType::JPEG2000:
267 this->setCompressionAlgorithm(
268 BE::Image::CompressionAlgorithm::JP2);
269 break;
270 default:
271 //XXX throw?
272 break;
273 }
274 this->_colorSpace = to_enum<BE::Face::ColorSpace>(buf.scanU8Val());
275 this->_sourceType = to_enum<BE::Face::SourceType>(buf.scanU8Val());
276 this->_deviceType = buf.scanBeU16Val();
277 this->_quality = buf.scanBeU16Val();
278
279 remainLen = remainLen - 12; /* Sum of above data item lengths */
280
281 /*
282 * Image data
283 */
284 this->setImageColorDepth(0);
285 this->setImageResolution(
286 Image::Resolution(0, 0, BE::Image::Resolution::Units::NA));
287 this->setScanResolution(
288 Image::Resolution(0, 0, BE::Image::Resolution::Units::NA));
289 BE::Memory::uint8Array imageData(remainLen);
290 buf.scan(&imageData[0], remainLen);
291 this->setImageData(imageData);
292}
