71BiometricEvaluation::View::AN2KViewVariableResolution::QualityMetricSet
72BiometricEvaluation::View::AN2KViewVariableResolution::extractQuality(
73 FIELD *field,
74 Feature::PositionType type)
75{
76 QualityMetricSet qms;
77
78 for (int i = 0; i < field->num_subfields; i++) {
79 if (field->subfields[i]->num_items != 4)
80 throw BE::Error::StrategyError("Missing item fields "
81 "in FQM.");
82
83 AN2KQualityMetric qm;
84 qm.fgp.posType = type;
85 int pos = atoi((char *)field->subfields[i]->items[0]->value);
86 if (type == Feature::PositionType::Palm) {
87 qm.fgp.position.palmPos =
88 to_enum<BE::Palm::Position>(pos);
89 } else {
90 qm.fgp.position.fingerPos =
91 Finger::AN2KView::convertPosition(pos);
92 }
93 qm.score = atoi((char *)field->subfields[i]->items[1]->value);
94 qm.vendorID = atoi((char*)field->subfields[i]->items[2]->value);
95 qm.productCode = atoi((char *)field->subfields[i]->items[3]->value);
96 qms.push_back(qm);
97 }
98 return (qms);
99}

101BiometricEvaluation::Memory::uint8Array
102BiometricEvaluation::View::AN2KViewVariableResolution::getUserDefinedField(
103 const uint16_t field)
104 const
105{
106 /* Check for cached version of this field */
107 const auto it = this->_udf.find(field);
108 if (it != _udf.end())
109 return (it->second);
110
111 /* Insert pair and return the data */
112 const auto ret = this->_udf.emplace(field,
113 parseUserDefinedField(AN2KView::getAN2KRecord(), field));
114 if (ret.second)
115 return (ret.first->second);
116 throw Error::StrategyError("Field " + std::to_string(field) +
117 " could not be cached");
118}

158static
159BiometricEvaluation::Finger::PositionDescriptors
160parsePositionDescriptors(
161 const BE::View::AN2KView::RecordType typeID,
162 const RECORD *record)
163{
164 unsigned int field_num;
165 switch (typeID) {
166 case BE::View::AN2KView::RecordType::Type_13: field_num = SPD_ID; break;
167 case BE::View::AN2KView::RecordType::Type_14: field_num = PPD_ID; break;
168 default: throw BE::Error::DataError("Invalid type -- no position "
169 "descriptor field");
170 }
171
172 int idx;
173 FIELD *field;
174 BE::Finger::PositionDescriptors pd;
175 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, field_num, record) != TRUE)
176 throw BE::Error::DataError(
177 "Position descriptor field not found");
178 for (int i = 0; i < field->num_subfields; i++) {
179 if (field->subfields[i]->num_items != 2)
180 throw BE::Error::DataError(
181 "Not enough position descriptor fields.");
182 pd[BE::Finger::AN2KView::convertPosition(atoi((char *)field->
183 subfields[i]->items[0]->value))] =
184 BE::Finger::AN2KView::convertFingerImageCode((char *)field->
185 subfields[i]->items[1]->value);
186 }
187
188 return (pd);
189}

270static BE::Feature::FGPSet
271populateFGP(FIELD *field)
272{
273 BE::Feature::FGP fgp;
274 BE::Feature::FGPSet posSet;
275 int an2kFGP;
276 for (int i = 0; i < field->num_subfields; i++) {
277 an2kFGP = atoi((char *)field->subfields[i]->items[0]->value);
278 try {
279 fgp.position.fingerPos =
280 to_enum<BE::Finger::Position>(an2kFGP);
281 fgp.posType = BE::Feature::PositionType::Finger;
282 } catch (...) {
283 try {
284 fgp.position.palmPos =
285 to_enum<BE::Palm::Position>(an2kFGP);
286 fgp.posType = BE::Feature::PositionType::Palm;
287 } catch (...) {
288 fgp.position.plantarPos =
289 to_enum<BE::Plantar::Position>(an2kFGP);
290 fgp.posType = BE::Feature::PositionType::Plantar;
291 } /* exceptions float out */
292 }
293 posSet.push_back(fgp);
294 }
295 return (posSet);
296}

298void
299BiometricEvaluation::View::AN2KViewVariableResolution::readImageRecord(
300 const RecordType typeID)
301{
302 switch (typeID) {
303 case RecordType::Type_13:
304 case RecordType::Type_14:
305 case RecordType::Type_15:
306 break;
307 default:
308 throw Error::ParameterError("Invalid Record Type ID");
309 }
310
311 RECORD* record = AN2KView::getAN2KRecord();
312 FIELD *field;
313 int idx;
314
315 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, FGP3_ID, record) != TRUE)
316 throw Error::DataError("Field FGP not found");
317 _positions = populateFGP(field);
318
319 /*********************************************************************/
320 /* Optional Fields. */
321 /*********************************************************************/
322 for (auto fp: _positions) {
323 if ((fp.posType == Feature::PositionType::Finger) &&
324 (fp.position.fingerPos == BE::Finger::Position::EJI)) {
325 /* Print Position Descriptors */
326 int pd_id;
327 switch (typeID) {
328 case RecordType::Type_13:
329 pd_id = SPD_ID;
330 break;
331 case RecordType::Type_14:
332 default:
333 pd_id = PPD_ID;
334 break;
335 }
336 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, pd_id, record)
337 == TRUE) {
338 _pd = parsePositionDescriptors(typeID, record);
339 }
340 /* Print Position Coordinates */
341 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, PPC_ID, record)
342 == TRUE) {
343 for (int i = 0; i < field->num_subfields; i++) {
344 _ppcs.push_back(
345 convertPrintPositionCoordinate(
346 field->subfields[i]));
347 }
348 }
349 }
350 }
351
352 /*********************************************************************/
353 /* Required Fields. */
354 /*********************************************************************/
355 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, IMP_ID, record) != TRUE)
356 throw Error::DataError("Field IMP not found");
357 _imp = BE::Finger::AN2KView::convertImpression(
358 field->subfields[0]->items[0]->value);
359
360 /* For some required fields, where no other information is dependent
361 * on them, we'll allow them to be missing.
362 */
363 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, SRC_ID, record) == TRUE)
364 _sourceAgency = (char *)field->subfields[0]->items[0]->value;
365
366 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, CD_ID, record) == TRUE)
367 _captureDate = (char *)field->subfields[0]->items[0]->value;
368
369 /*
370 * Convert the Horizontal/Vertical pixel scale to the image
371 * resolution. Field SLC has the scale units indicator.
372 */
373 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, SLC_ID, record) != TRUE)
374 throw Error::DataError("Field SLC not found");
375 int slc = atoi((char *)field->subfields[0]->items[0]->value);
376
377 BE::Image::Resolution ir;
378 switch (slc) {
379 case 0: ir.units = BE::Image::Resolution::Units::NA; break;
380 case 1: ir.units = BE::Image::Resolution::Units::PPI; break;
381 case 2: ir.units = BE::Image::Resolution::Units::PPCM; break;
382 }
383 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, HPS_ID, record) != TRUE)
384 throw Error::DataError("Field HPS not found");
385 ir.xRes = atoi((char *)field->subfields[0]->items[0]->value);
386 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, VPS_ID, record) != TRUE)
387 throw Error::DataError("Field VPS not found");
388 ir.yRes = atoi((char *)field->subfields[0]->items[0]->value);
389 AN2KView::AN2KView::setImageResolution(ir);
390
391 /* Compression Algorithm, ASCII version */
392 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, TAG_CA_ID, record) != TRUE)
393 throw Error::DataError("Field TAG_CA not found");
394 AN2KView::setCompressionAlgorithm(
395 AN2KView::convertCompressionAlgorithm(
396 (*record).type, field->subfields[0]->items[0]->value));
397
398 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, BPX_ID, record) != TRUE)
399 throw Error::DataError("Field BPX not found");
400 AN2KView::setImageColorDepth(
401 atoi((char *)field->subfields[0]->items[0]->value));
402
403 /* Read the image data */
404 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, DAT2_ID, record) != TRUE)
405 throw Error::DataError("Field DAT2 not found");
406 Memory::AutoArray<uint8_t>imageData = Memory::AutoArray<uint8_t>(
407 field->subfields[0]->items[0]->num_bytes);
408 for (unsigned int i = 0; i < imageData.size(); i++)
409 imageData[i] = field->subfields[0]->items[0]->value[i];
410 AN2KView::setImageData(imageData);
411
412 /*********************************************************************/
413 /* Optional Fields. */
414 /*********************************************************************/
415
416 /* Reuse units from SLC */
417 ir.xRes = ir.yRes = 0.0;
418 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, SHPS_ID, record) == TRUE)
419 ir.xRes = atoi((char *)field->subfields[0]->items[0]->value);
420 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, SVPS_ID, record) == TRUE)
421 ir.yRes = atoi((char *)field->subfields[0]->items[0]->value);
422 AN2KView::setScanResolution(ir);
423
424 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, COM_ID, record) == TRUE)
425 _comment = (char *)field->subfields[0]->items[0]->value;
426
427 /* Latent/finger/palm quality metric */
428 int quality_id;
429 Feature::PositionType type;
430 switch (typeID) {
431 case RecordType::Type_13:
432 quality_id = LQM_ID;
433 type = Feature::PositionType::Finger;
434 break;
435 case RecordType::Type_15:
436 quality_id = PQM_ID;
437 type = Feature::PositionType::Palm;
438 break;
439 case RecordType::Type_14:
440 default:
441 quality_id = FQM_ID;
442 type = Feature::PositionType::Finger;
443 break;
444 }
445 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, quality_id, record) == TRUE)
446 _qms = extractQuality(field, type);
447
448 /* FrictionRidgeCaptureTechnology (FRCT) */
449 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, 901, record) == TRUE)
450 this->_frct = convertCaptureTechnology(
451 (char *)field->subfields[0]->items[0]->value);
452}

454BiometricEvaluation::Memory::uint8Array
455BiometricEvaluation::View::AN2KViewVariableResolution::parseUserDefinedField(
456 const RECORD* const record,
457 int fieldID)
458{
459
460 if (fieldID < 200 || fieldID > 998)
461 throw Error::DataError("Invalid user-defined field number");
462
463 FIELD *field;
464 int idx;
465 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, fieldID, record) == FALSE)
466 throw Error::ObjectDoesNotExist("Field " +
467 std::to_string(fieldID) + " does not exist");
468
469 /* Byte-for-byte copy of the field as would appear in an AN2K file */
470 size_t offset = 0;
471 Memory::uint8Array buf(field->num_bytes);
472 for (int sf = 0; sf < field->num_subfields; sf++) {
473 for (int item = 0; item < field->subfields[sf]->num_items;
474 item++) {
475 for (int item_offset = 0;
476 item_offset < field->subfields[sf]->items[item]->
477 num_bytes; item_offset++) {
478 buf[offset++] = field->subfields[sf]->
479 items[item]->value[item_offset];
480 }
481 buf[offset++] = field->subfields[sf]->
482 items[item]->us_char;
483 }
484 buf[offset++] = field->subfields[sf]->rs_char;
485 }
486 buf[offset++] = field->gs_char;
487
488 return (buf);
489}
