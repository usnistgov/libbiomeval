79uint64_t
80BiometricEvaluation::System::getRealMemorySize()
81{
82#if defined Linux
83 long numpages = sysconf(_SC_PHYS_PAGES);
84 if (numpages <= 0)
85 throw (Error::NotImplemented());
86
87 long pagesize = sysconf(_SC_PAGE_SIZE);
88 if (pagesize <= 0)
89 throw (Error::NotImplemented());
90
91 /*
92 * POSIX states that this value can overflow; check if we've
93 * gone negative.
94 */
95 long total = numpages * pagesize;
96 if (total <= 0)
97 throw (Error::NotImplemented());
98 return ((uint64_t)(total / 1024));
99
100#elif defined Darwin
101 int mib[2];
102 uint64_t total;
103 size_t len = sizeof(total);
104 mib[0] = CTL_HW;
105 mib[1] = HW_MEMSIZE;
106 int retval = sysctl(mib, 2, &total, &len, nullptr, 0);
107 if (retval != 0)
108 throw (Error::NotImplemented());
109 total = total / 1024;
110 return (total);
111#else
112 throw (Error::NotImplemented());
113#endif
114}
