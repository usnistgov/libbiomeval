69/*
70 * Check the encoding flag bit in the given 8-bit value for BER-TLV primitive.
71 */
72static bool
73primitiveFlag(const uint8_t cval)
74{
75 uint8_t data_encoding = (cval & BERTLV_TAG_DATA_ENCODING_MASK) >>
76 BERTLV_TAG_DATA_ENCODING_SHIFT;
77 if (data_encoding == BERTLV_TAG_DATA_ENCODING_PRIMITIVE)
78 return(true);
79 else
80 return(false);
81}

114void
115BiometricEvaluation::Device::TLV::Impl::setTag(const Memory::uint8Array &tag)
116{
117 if (primitiveFlag(tag[0]) == true) {
118 if (this->_children.size() != 0) {
119 throw (Error::DataError(
120 "Primitive flag set in tag, but TLV has children"));
121 }
122 } else { /* primitive flag is off */
123 if (this->_primitive.size() != 0) {
124 throw (Error::DataError(
125 "Primitive flag not set in tag, but TLV has data"));
126 }
127 }
128
129 if (tag.size() > BERTLV_MAX_TAG_LENGTH)
130 throw (Error::ParameterError("Tag length too large"));
131 BE::Memory::IndexedBuffer ibuf(tag);
132 parseTag(ibuf);
133 this->_tag_field = tag;
134}

204BiometricEvaluation::Memory::uint8Array
205BiometricEvaluation::Device::TLV::Impl::getRawTLV() const
206{
207 BE::Memory::uint8Array buf(
208 this->_tag_field.size() +
209 this->_length_field.size() +
210 this->_length);
211 BE::Memory::MutableIndexedBuffer ibuf(buf);
212
213 /* Push the the encoded tag field into the buffer */
214 for (auto n : this->_tag_field)
215 ibuf.pushU8Val(n);
216
217 /* Push the length, encoded, into the buffer */
218 for (auto n : this->_length_field)
219 ibuf.pushU8Val(n);
220
221 /* Push primitive data */
222 if (this->_isPrimitive == true) {
223 ibuf.push(this->_primitive, this->_primitive.size());
224 } else {
225 /* Push any children */
226 for (auto c : this->_children) {
227 auto buf = c.getRawTLV();
228 ibuf.push(buf, buf.size());
229 }
230 }
231
232 return (buf);
233}

235void
236BiometricEvaluation::Device::TLV::Impl::parseTag(
237 BE::Memory::IndexedBuffer &ibuf)
238{
239 uint8_t cval;
240
241 /* Read first byte, determine if single/multi-byte tag */
242 cval = ibuf.scanU8Val();
243
244 /* The tag field is always has the capacity to hold the largest tag */
245 this->_tag_field[0] = cval;
246 this->_tagclass = (cval & BERTLV_TAG_CLASS_MASK) >>
247 BERTLV_TAG_CLASS_SHIFT;
248 this->_isPrimitive = primitiveFlag(cval);
249
250 /*
251 * If all bits of the rest of the tag are 1, the tag value is in
252 * subsequent bytes. Otherwise, the tag value is in this byte.
253 */
254 int tag_field_length = 1;
255 if ((cval & BERTLV_SB_MB_TAGNUM_MASK) == BERTLV_SB_MB_TAGNUM_MASK) {
256 /* ISO 7816-4 says tag value is in next 1 or 2 bytes */
257 cval = ibuf.scanU8Val();
258 this->_tag_field[1] = cval;
259 this->_tagnum = cval & BERTLV_MB_TAGNUM_MASK;
260 tag_field_length++;
261 if (cval & BERTLV_MB_TAGNUM_TERMINATOR_MASK) {
262 cval = ibuf.scanU8Val();
263 this->_tag_field[2] = cval;
264 this->_tagnum = (this->_tagnum << 8) |
265 (cval & BERTLV_MB_TAGNUM_MASK);
266 tag_field_length++;
267 }
268 } else {
269 this->_tagnum = cval & BERTLV_MB_TAGNUM_MASK;
270 }
271 this->_tag_field.resize(tag_field_length);
272}

274void
275BiometricEvaluation::Device::TLV::Impl::updateLengthField()
276{
277 if (this->_length <= BERTLV_SB_MAX_VALUE) {
278 this->_length_field.resize(1);
279 this->_length_field[0] = this->_length;
280 return;
281 }
282 if (this->_length <= BERTLV_MB_2_MAX_VALUE) {
283 this->_length_field.resize(2);
284 uint8_t cval = this->_length;
285 this->_length_field[0] = BERTLV_SB_MB_LENGTH_MB_2;
286 this->_length_field[1] = cval;
287 return;
288 }
289 if (this->_length <= BERTLV_MB_3_MAX_VALUE) {
290 this->_length_field.resize(3);
291 uint16_t sval = htons(this->_length);
292 this->_length_field[0] = BERTLV_SB_MB_LENGTH_MB_3;
293 this->_length_field[1] = ((uint8_t *)&sval)[0];
294 this->_length_field[2] = ((uint8_t *)&sval)[1];
295 return;
296 }
297 if (this->_length <= BERTLV_MB_4_MAX_VALUE) {
298 this->_length_field.resize(4);
299 uint32_t lval = htonl(this->_length);
300 this->_length_field[0] = BERTLV_SB_MB_LENGTH_MB_4;
301 this->_length_field[1] = ((uint8_t *)&lval)[1];
302 this->_length_field[2] = ((uint8_t *)&lval)[2];
303 this->_length_field[3] = ((uint8_t *)&lval)[3];
304 return;
305 }
306 if (this->_length <= BERTLV_MB_5_MAX_VALUE) {
307 this->_length_field.resize(5);
308 uint32_t lval = htonl(this->_length);
309 this->_length_field[0] = BERTLV_SB_MB_LENGTH_MB_5;
310 this->_length_field[1] = ((uint8_t *)&lval)[0];
311 this->_length_field[2] = ((uint8_t *)&lval)[1];
312 this->_length_field[3] = ((uint8_t *)&lval)[2];
313 this->_length_field[4] = ((uint8_t *)&lval)[3];
314 return;
315 }

318void
319BiometricEvaluation::Device::TLV::Impl::readTLV(BE::Memory::IndexedBuffer &ibuf)
320{
321 uint8_t cval;
322 uint16_t sval;
323
324 /* First, parse the tag and set decoded field values */
325 parseTag(ibuf);
326
327 /* Read first length byte, determine if single/multi-byte length */
328 cval = ibuf.scanU8Val();
329 if (cval <= BERTLV_SB_MAX_VALUE) {
330 this->_length = cval;
331 } else {
332 switch (cval) {
333 case BERTLV_SB_MB_LENGTH_MB_2 :
334 this->_length = ibuf.scanU8Val();
335 break;
336 case BERTLV_SB_MB_LENGTH_MB_3 :
337 this->_length = ibuf.scanBeU16Val();
338 break;
339 case BERTLV_SB_MB_LENGTH_MB_4 :
340 sval = ibuf.scanBeU16Val();
341 this->_length = sval;
342 cval = ibuf.scanU8Val();
343 this->_length = (this->_length << 8) + cval;
344 break;
345 case BERTLV_SB_MB_LENGTH_MB_5 :
346 this->_length = ibuf.scanBeU32Val();
347 break;
348 default :
349 throw (Error::DataError());
350 break; // not reached
351 }
352 }
353 updateLengthField();
354
355 if (this->_length == 0)
356 return;
357
358 int64_t length = this->_length;
359 /*
360 * Read the value field if primitive type; otherwise, read the child
361 * TLV and add it to this TLV's set of children. Set the primitive
362 * indicator appropriately.
363 */
364 while (length > 0) {
365 if (this->_isPrimitive == true) {
366 this->_primitive.resize(this->_length);
367 ibuf.scan(this->_primitive, this->_length);
368 length -= this->_length;
369 } else {
370 /* Add a new TLV into the vector of children */
371 Device::TLV child(ibuf);
372 length -= (child.pimpl->_length +
373 child.pimpl->_tag_field.size() +
374 child.pimpl->_length_field.size());
375 this->_children.push_back(child);
376 }
377 }
