19BiometricEvaluation::Finger::AN2KMinutiaeDataRecord::AN2KMinutiaeDataRecord(
20 const std::string &filename,
21 int recordNumber)
22{
23 if (IO::Utility::fileExists(filename) == false)
24 throw (Error::FileError("File not found."));
25
26 FILE *fp = std::fopen(filename.c_str(), "rb");
27 if (fp == nullptr)
28 throw (Error::FileError("Could not open file."));
29
30 uint64_t sz = IO::Utility::getFileSize(filename);
31 Memory::uint8Array buf(sz);
32 if (fread(buf, 1, sz, fp) != sz) {
33 fclose(fp);
34 throw Error::FileError("Could not read AN2K file");
35 }
36 fclose(fp);
37
38 readType9Record(buf, recordNumber);
39}

98void
99BiometricEvaluation::Finger::AN2KMinutiaeDataRecord::readRegisteredVendorBlock(
100 RECORD *type9,
101 BiometricEvaluation::Feature::MinutiaeFormat vendor)
102{
103 /* First field of IAFIS format features */
104 static const uint16_t IAFISFieldStart = 13;
105 /* Last field of IAFIS format features */
106 static const uint16_t IAFISFieldEnd = 30;
107 /* First field of Cogent Systems format features */
108 static const uint16_t CogentFieldStart = 31;
109 /* Last field of Cogent Systems format features */
110 static const uint16_t CogentFieldEnd = 55;
111 /* First field of Motorola format features */
112 static const uint16_t MotorolaFieldStart = 56;
113 /* Last field of Motorola format features */
114 static const uint16_t MotorolaFieldEnd = 70;
115 /* First field of Sagem Morpho format features */
116 static const uint16_t SagemFieldStart = 71;
117 /* Last field of Sagem Morpho format features */
118 static const uint16_t SagemFieldEnd = 99;
119 /* First field of NEC format features */
120 static const uint16_t NECFieldStart = 100;
121 /* Last field of NEC format features */
122 static const uint16_t NECFieldEnd = 125;
123 /* First field of M1-378 format features */
124 static const uint16_t M1FieldStart = 126;
125 /* Last field of M1-378 format features */
126 static const uint16_t M1FieldEnd = 150;
127 /* First field of Identix format features */
128 static const uint16_t IdentixFieldStart = 151;
129 /* Last field of Identix format features */
130 static const uint16_t IdentixFieldEnd = 175;
131
132 uint16_t startField, endField;
133 std::map<uint16_t, Memory::uint8Array> *features;
134 switch (vendor) {
135 case Feature::MinutiaeFormat::IAFIS:
136 startField = IAFISFieldStart;
137 endField = IAFISFieldEnd;
138 features = &_IAFISFeatures;
139 break;
140 case Feature::MinutiaeFormat::Cogent:
141 startField = CogentFieldStart;
142 endField = CogentFieldEnd;
143 features = &_cogentFeatures;
144 break;
145 case Feature::MinutiaeFormat::Motorola:
146 startField = MotorolaFieldStart;
147 endField = MotorolaFieldEnd;
148 features = &_motorolaFeatures;
149 break;
150 case Feature::MinutiaeFormat::Sagem:
151 startField = SagemFieldStart;
152 endField = SagemFieldEnd;
153 features = &_sagemFeatures;
154 break;
155 case Feature::MinutiaeFormat::NEC:
156 startField = NECFieldStart;
157 endField = NECFieldEnd;
158 features = &_NECFeatures;
159 break;
160 case Feature::MinutiaeFormat::Identix:
161 startField = IdentixFieldStart;
162 endField = IdentixFieldEnd;
163 features = &_identixFeatures;
164 break;
165 case Feature::MinutiaeFormat::M1:
166 startField = M1FieldStart;
167 endField = M1FieldEnd;
168 features = &_M1Features;
169 break;
170 case Feature::MinutiaeFormat::AN2K7:
171 /* FALLTHROUGH */
172 default:
173 throw (Error::NotImplemented());
174 }
175
176 FIELD *field;
177 int idx;
178 uint64_t offset = 0;
179
180 /*
181 * Recreate the bit stream from the original AN2K file that was
182 * parsed to create the RECORD struct.
183 */
184 for (uint16_t i = startField; i < endField; i++) {
185 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, i, type9) == FALSE)
186 continue;
187
188 /*
189 * XXX: num_bytes is not the size of a FIELD, though num_bytes +
190 * num_subfields is over the size by a few bytes.
191 */
192 features->insert(std::pair<uint16_t, Memory::uint8Array>(i,
193 Memory::uint8Array(field->num_bytes +
194 field->num_subfields)));
195 for (int sf = 0; sf < field->num_subfields; sf++, offset = 0) {
196 for (int item = 0; item <
197 field->subfields[sf]->num_items; item++) {
198 memcpy(features->find(i)->second,
199 field->subfields[sf]->items[item]->value,
200 field->subfields[sf]->items[item]->
201 num_bytes);
202 /* num_bytes *should* include separator */
203 offset += field->subfields[sf]->items[item]->
204 num_bytes;
205 }
206 features->find(i)->second[offset++] =
207 field->subfields[sf]->rs_char;
208 }
209 features->find(i)->second[offset++] = field->gs_char;
210
211 /* Always seems to be slightly larger than it needs to be */
212 features->find(i)->second.resize(offset);
213 }
214}

216void
217BiometricEvaluation::Finger::AN2KMinutiaeDataRecord::readType9Record(
218 Memory::uint8Array &buf,
219 int recordNumber)
220{
221 Memory::AutoBuffer<ANSI_NIST> an2k =
222 Memory::AutoBuffer<ANSI_NIST>(&biomeval_nbis_alloc_ANSI_NIST,
223 &biomeval_nbis_free_ANSI_NIST, &biomeval_nbis_copy_ANSI_NIST);
224
225 AN2KBDB bdb;
226 INIT_AN2KBDB(&bdb, buf, buf.size());
227 if (biomeval_nbis_scan_ANSI_NIST(&bdb, an2k) != 0)
228 throw Error::DataError("Could not read complete AN2K record");
229
230 /*
231 * Find the requested Type-9 in the file, throwing an exception
232 * if not present. The first record in an AN2K file is always
233 * the Type-1, so skip that one.
234 */
235 RECORD *type9 = nullptr;
236 for (int i = 1; i < an2k->num_records; i++) {
237 if (an2k->records[i]->type == TYPE_9_ID) {
238 if (i == recordNumber) {
239 type9 = an2k->records[i];
240 break;
241 }
242 }
243 }
244 if (type9 == nullptr)
245 throw (Error::DataError("Could not find requested Type-9 in "
246 "AN2K record"));
247
248 FIELD *field;
249 int idx;
250
251 /* Impression type (IMP) -- Field 9.003 (mandatory) */
252 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, IMP_ID, type9) == FALSE)
253 throw Error::DataError("Field IMP not found");
254 _imp = Finger::AN2KView::convertImpression(field->subfields[0]->
255 items[0]->value);
256
257 /* Try to read AN2K7 feature data, although it may not be present */
258 try {
259 _AN2K7Features.reset(
260 new Feature::AN2K7Minutiae(buf, recordNumber));
261 } catch (const Error::Exception&) {}
262
263 readRegisteredVendorBlock(type9, Feature::MinutiaeFormat::IAFIS);
264 readRegisteredVendorBlock(type9, Feature::MinutiaeFormat::Cogent);
265 readRegisteredVendorBlock(type9, Feature::MinutiaeFormat::Motorola);
266 readRegisteredVendorBlock(type9, Feature::MinutiaeFormat::Sagem);
267 readRegisteredVendorBlock(type9, Feature::MinutiaeFormat::NEC);
268 readRegisteredVendorBlock(type9, Feature::MinutiaeFormat::M1);
269 readRegisteredVendorBlock(type9, Feature::MinutiaeFormat::Identix);
270
271 /*
272 * Try to read AN2K11 extended feature set data, although it may not
273 * be present.
274 */
275 try {
276 _AN2K11EFS.reset(
277 new Feature::AN2K11EFS::ExtendedFeatureSet(buf,
278 recordNumber));
279 } catch (const Error::Exception&) {}
280
281}
