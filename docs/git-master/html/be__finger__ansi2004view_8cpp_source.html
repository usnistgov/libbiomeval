41void
42BiometricEvaluation::Finger::ANSI2004View::init(
43 const Memory::uint8Array &fmrBuffer,
44 const Memory::uint8Array &firBuffer,
45 const uint32_t viewNumber)
46{
47 if (fmrBuffer.size() != 0) {
48 Memory::IndexedBuffer iBuf(fmrBuffer, fmrBuffer.size());
49 this->readFMRHeader(iBuf);
50 try {
51 for (uint32_t i = 0; i < viewNumber; i++) {
52 this->readFVMR(iBuf);
53 }
54 } catch (const BE::Error::DataError&) {
55 throw BE::Error::ObjectDoesNotExist("Error reading "
56 "view number = " + std::to_string(viewNumber));
57 }
58 }

62void
63BiometricEvaluation::Finger::ANSI2004View::readFMRHeader(
64 Memory::IndexedBuffer &buf)
65{
66 uint32_t uval32;
67
68 uval32 = buf.scanBeU32Val(); /* Format ID */
69 if (uval32 != BE::Finger::INCITSView::FMR_BASE_FORMAT_ID)
70 throw (BE::Error::DataError("Invalid Format ID in data"));
71
72 uval32 = buf.scanBeU32Val(); /* Spec Version */
73 if (uval32 != BE::Finger::ANSI2004View::BASE_SPEC_VERSION)
74 throw (Error::DataError("Invalid Spec Version in data"));
75
76 BE::Finger::INCITSView::readFMRHeader(
77 buf, BE::Finger::INCITSView::ANSI2004_STANDARD);
78}

80void
81BiometricEvaluation::Finger::ANSI2004View::readCoreDeltaData(
82 Memory::IndexedBuffer &buf,
83 uint32_t dataLength,
84 Feature::CorePointSet &cores,
85 Feature::DeltaPointSet &deltas)
86{
87 static const uint16_t CORE_TYPE_MASK = 0xC0;
88 static const uint16_t CORE_TYPE_SHIFT = 6;
89 static const uint16_t CORE_NUM_CORES_MASK = 0x0F;
90 static const uint16_t CORE_X_COORD_MASK = 0x3FFF;
91 static const uint16_t CORE_Y_COORD_MASK = 0x3FFF;
92
93 static const uint16_t DELTA_TYPE_MASK = 0xC0;
94 static const uint16_t DELTA_TYPE_SHIFT = 6;
95 static const uint16_t DELTA_NUM_DELTAS_MASK = 0x3F;
96 static const uint16_t DELTA_X_COORD_MASK = 0x3FFF;
97 static const uint16_t DELTA_Y_COORD_MASK = 0x3FFF;
98
99 /* Read the core info */
100 uint8_t cval = buf.scanU8Val();
101 uint8_t type = (cval & CORE_TYPE_MASK) >> CORE_TYPE_SHIFT;
102 uint8_t count = cval & CORE_NUM_CORES_MASK;
103 bool hasAngle = false;
104 if (type == Feature::INCITSMinutiae::CORE_TYPE_ANGULAR)
105 hasAngle = true;
106 for (int i = 0; i < count; i++) {
107 uint16_t x = buf.scanBeU16Val() & CORE_X_COORD_MASK;
108 uint16_t y = buf.scanBeU16Val() & CORE_Y_COORD_MASK;
109 uint8_t angle = 0;
110 if (hasAngle)
111 angle = buf.scanU8Val();
112 Image::Coordinate c(x, y);
113 cores.push_back(Feature::CorePoint(c, hasAngle, angle));
114 }
115
116 /* Read the delta info */
117 cval = buf.scanU8Val();
118 type = (cval & DELTA_TYPE_MASK) >> DELTA_TYPE_SHIFT;
119 count = cval & DELTA_NUM_DELTAS_MASK;
120 hasAngle = false;
121 if (type == Feature::INCITSMinutiae::DELTA_TYPE_ANGULAR)
122 hasAngle = true;
123 for (int i = 0; i < count; i++) {
124 uint16_t x = buf.scanBeU16Val() & DELTA_X_COORD_MASK;
125 uint16_t y = buf.scanBeU16Val() & DELTA_Y_COORD_MASK;
126 uint8_t angle1 = 0, angle2 = 0, angle3 = 0;
127 if (hasAngle) {
128 angle1 = buf.scanU8Val();
129 angle2 = buf.scanU8Val();
130 angle3 = buf.scanU8Val();
131 }
132 Image::Coordinate c(x, y);
133 deltas.push_back(
134 Feature::DeltaPoint(c, hasAngle, angle1, angle2, angle3));
135 }
136}
