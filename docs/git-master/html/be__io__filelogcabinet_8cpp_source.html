35BiometricEvaluation::IO::FileLogCabinet::FileLogCabinet(
36 const std::string &pathname,
37 const std::string &description)
38{
39 if (IO::Utility::fileExists(pathname))
40 throw Error::ObjectExists();
41
42 if (mkdir(pathname.c_str(), S_IRWXU) != 0)
43 throw Error::StrategyError("Could not create directory");
44
45 this->_count = 0;
46 this->_pathname = pathname;
47 this->_description = description;
48 try {
49 (void)writeControlFile();
50 } catch (Error::StrategyError& e) {
51 throw;
52 }
53}

132void
133BiometricEvaluation::IO::FileLogCabinet::readControlFile()
134{
135 std::string str;
136
137 /* Read the cabinet name and description from the control file.
138 * _pathname must be set before calling this method.
139 */
140 std::ifstream ifs(canonicalName(controlFileName).c_str());
141 if (!ifs)
142 throw Error::StrategyError("Could not open control file");
143
144 std::getline(ifs, this->_description);
145 if (ifs.eof())
146 throw Error::StrategyError("Premature EOF on control file");
147
148 ifs >> this->_count;
149 if (!ifs.good())
150 throw Error::StrategyError("Could not read count from control "
151 "file");
152
153 ifs.close();
154}

156void
157BiometricEvaluation::IO::FileLogCabinet::writeControlFile()
158{
159 std::ofstream ofs(canonicalName(controlFileName).c_str());
160 if (!ofs)
161 throw Error::StrategyError("Could not write control file");
162
163 ofs << this->_description << std::endl;
164 ofs << this->_count << std::endl;
165 if (!ofs.good())
166 throw Error::StrategyError("Could not write count to control "
167 "file");
168 ofs.close();
169}
