32static const mode_t DBRS_MODE_RW =
33 S_IRUSR | S_IWUSR | S_IRGRP | S_IWGRP | S_IROTH | S_IWOTH;
34static const mode_t DBRS_MODE_R = S_IRUSR | S_IRGRP | S_IROTH;
35static const std::string DBFILENAME = "rsdb";
36static const std::string SUBORDINATE_DBEXT = ".subordinate";

38/*
39 * The maximum record size supported by the underlying Berkeley DB is
40 * 2^32. This class will break larger records up into multiple key/value
41 * pairs, inserting the 2..n segments into a separate DB file with
42 * uniquely generated key names.
43 */
44static const uint64_t MAX_REC_SIZE = (uint64_t)4294967295U;
45
46static void setBtreeInfo(std::shared_ptr<Db> db)
47{
48 db->set_lorder(4321); /* Big-endian */
49}

51/*
52 * The name property in the control file has been removed, but we
53 * check for it to determine whether this is an old-style DBRecordStore
54 * or the new style.
55 */
56static const std::string NAMEPROPERTY("Name");
57
58/*
59 * Obtain the name of the underlying Berkeley DB files:
60 * In the old format, they are named after the record store name, and
61 * that property will exists (as "non-core") in the properties file, so
62 * return it. Otherwise, assume we have a new format record store with
63 * no name, and use the new namining scheme.
64 */
65std::string
66BiometricEvaluation::IO::DBRecordStore::Impl::getDBFilePathname() const
67{
68 std::string filename;
69 std::shared_ptr<IO::Properties> props = this->getProperties();
70 try {
71 filename = props->getProperty(NAMEPROPERTY);
72 } catch (const Error::ObjectDoesNotExist&) {
73 filename = DBFILENAME;
74 }
75 return (this->getPathname() + '/' + filename);
76}

78void
79BiometricEvaluation::IO::DBRecordStore::Impl::i_setup(
80 const std::string &pathname,
81 int dbFlags,
82 IO::Mode mode)
83{
84 /*
85 * This function assumes that files and paths either exist when
86 * needed or don't exist when not needed.
87 */
88 this->_dbnameP = this->getDBFilePathname();
89
90 int dbMode;
91 switch (mode) {
92 case IO::Mode::ReadOnly:
93 dbMode = DBRS_MODE_R;
94 break;
95 case IO::Mode::ReadWrite:
96 dbMode = DBRS_MODE_RW;
97 break;
98 }
99 /* Open the primary DB file */
100 this->_dbP = std::make_shared<Db>(nullptr, 0);
101 setBtreeInfo(this->_dbP);
102 try {
103 this->_dbP->open(nullptr, this->_dbnameP.c_str(), nullptr,
104 DB_BTREE, dbFlags, dbMode);
105 } catch (const DbException &e) {
106 throw Error::StrategyError("Could not open primary DB (DB "
107 "error = " + std::to_string(e.get_errno()) + " -- " +
108 e.what() + ")");
109 } catch (const std::exception &e) {
110 throw Error::StrategyError("Could not open primary DB (" +
111 Error::errorStr() + ")");
112 }
113 /* Create the cursor */
114 try {
115 Dbc *dbC{nullptr};
116 this->_dbP->cursor(nullptr, &dbC, 0);
117 /*
118 * DB Cursors cannot be destructed, so set the deleter to
119 * do nothing.
120 */
121 this->_dbC = std::shared_ptr<Dbc>(dbC, [](Dbc *c) {});
122 } catch (const DbException &e) {
123 throw BE::Error::StrategyError("Could not create DB cursor "
124 "(DB error = " + std::to_string(
125 e.get_errno()) + " -- " + e.what() + ")");
126 } catch (const std::exception &e) {
127 throw Error::StrategyError("Could not create DB cursor (" +
128 Error::errorStr() + ")");
129 }
130
131 /*
132 * Initialize the cursor to DB_FIRST, check for errors (empty database).
133 * and set the indicator of cursor state. If the cursor cannot be
134 * initialized here, the first insert will do it. Set the Dbt handle to
135 * indicate that no data need be read.
136 */
137 Dbt dbtkey;
138 Dbt dbtdata;
139 dbtdata.set_dlen(0);
140 dbtdata.set_flags(DB_DBT_PARTIAL);
141 auto rv = this->_dbC->get(&dbtkey, &dbtdata, DB_FIRST);
142 switch (rv) {
143 case 0:
144 this->_cursorIsInit = true;
145 break;
146 case DB_NOTFOUND:
147 this->_cursorIsInit = false;
148 break;
149 default:
150 this->_cursorIsInit = false;
151 break;
152 }
153
154 /* Open the subordinate DB file */
155 this->_dbnameS = this->_dbnameP + SUBORDINATE_DBEXT;
156 this->_dbS = std::make_shared<Db>(nullptr, 0);
157 setBtreeInfo(this->_dbS);
158 try {
159 this->_dbS->open(nullptr, this->_dbnameS.c_str(), nullptr,
160 DB_BTREE, dbFlags, dbMode);
161 } catch (const DbException &e) {
162 throw Error::StrategyError("Could not open subordinate DB "
163 "(DB error = " + std::to_string(e.get_errno()) + " -- " +
164 e.what() + ")");
165 } catch (const std::exception &e) {
166 throw Error::StrategyError("Could not open subordinate DB "
167 "(" + Error::errorStr() + ")");
168 }
169}

174 RecordStore::Impl(pathname, description, RecordStore::Kind::BerkeleyDB)
175{
176 /*
177 * The BDB files previously were named after the RecordStore
178 * name, but name has been removed as concept. However, the

190BiometricEvaluation::IO::DBRecordStore::Impl::Impl(
191 const std::string &pathname,
192 IO::Mode mode) :
193 RecordStore::Impl(pathname, mode)
194{
195 this->_dbnameP = this->getDBFilePathname();
196 if (!IO::Utility::fileExists(this->_dbnameP)) {
197 throw Error::ObjectDoesNotExist("Database does not exist");
198 }
199 /*
200 * Create the subordinate DB file if necessary in order
201 * to migrate older DBRecordStores. If we can't open the file,
202 * but the mode is ReadWrite, throw an exception; otherwise we'll
203 * just not use it later with the assumption there are no large
204 * records in the existing record store.
205 */
206 std::string tmpDBName = this->_dbnameP + SUBORDINATE_DBEXT;
207 if (!IO::Utility::fileExists(tmpDBName)) {
208 std::shared_ptr<Db> tmpDB = std::make_shared<Db>(nullptr, 0);
209 setBtreeInfo(tmpDB);
210 try {
211 tmpDB->open(nullptr, tmpDBName.c_str(),
212 nullptr, DB_BTREE, DB_CREATE | DB_EXCL,
213 DBRS_MODE_RW);
214 } catch (const DbException &e) {
215 if (mode == Mode::ReadWrite) {
216 throw Error::StrategyError(
217 "Could not upgrade database.");
218 } else {
219 return;
220 }
221 } catch (const std::exception &e) {
222 if (mode == Mode::ReadWrite) {
223 throw Error::StrategyError(
224 "Could not upgrade database.");
225 } else {
226 return;
227 }
228 }
229 tmpDB->close(0);
230 tmpDB = nullptr;
231 }
232
233 switch (mode) {
234 case IO::Mode::ReadOnly:
235 i_setup(pathname, DB_RDONLY, mode);
236 break;
237 case IO::Mode::ReadWrite:
238 i_setup(pathname, 0, mode);
239 break;
240 }

254void
255BiometricEvaluation::IO::DBRecordStore::Impl::move(const std::string &pathname)
256{
257 if (getMode() == Mode::ReadOnly)
258 throw Error::StrategyError("RecordStore was opened read-only");
259
260 if (this->_dbC != nullptr)
261 this->_dbC->close();
262 if (this->_dbP != nullptr)
263 this->_dbP->close(0);
264 if (this->_dbS != nullptr)
265 this->_dbS->close(0);
266 this->_dbC = nullptr;
267 this->_dbP = nullptr;
268 this->_dbS = nullptr;
269
270 std::string oldDBName, newDBName;
271 /*
272 * Preserve the old name of the DB files.
273 */
274 oldDBName = BE::Text::basename(this->getDBFilePathname());
275 RecordStore::Impl::move(pathname);
276
277 /*
278 * Remove the name property if it exists and have the parent class
279 * store the new Properties object.
280 */
281 std::shared_ptr<IO::Properties> props = this->getProperties();
282 try {
283 props->removeProperty(NAMEPROPERTY);
284 this->setProperties(props);
285 } catch (const BE::Error::ObjectDoesNotExist&) {
286 }
287
288 /*
289 * The DB files are now in the new directory and will be
290 * always named in the new manner.
291 */
292 oldDBName = pathname + '/' + oldDBName;
293 newDBName = pathname + '/' + DBFILENAME;
294
295 if (rename(oldDBName.c_str(), newDBName.c_str()))
296 throw Error::StrategyError(
297 "Could not rename primary DB (" +
298 Error::errorStr() + ")");
299
300 oldDBName = oldDBName + SUBORDINATE_DBEXT;
301 newDBName = newDBName + SUBORDINATE_DBEXT;
302 if (rename(oldDBName.c_str(), newDBName.c_str()))
303 throw Error::StrategyError(
304 "Could not rename subordinate DB (" +
305 Error::errorStr() + ")");
306
307 this->_dbnameP = this->getDBFilePathname();
308 if (!IO::Utility::fileExists(this->_dbnameP))
309 throw Error::StrategyError("Database " + this->_dbnameP +
310 " does not exist");
311 this->_dbnameS = this->_dbnameP + SUBORDINATE_DBEXT;
312 if (!IO::Utility::fileExists(this->_dbnameP))
313 throw Error::StrategyError("Database " + this->_dbnameS +
314 "does not exist");
315
316 i_setup(pathname, 0, IO::Mode::ReadWrite);
317}

367void
368BiometricEvaluation::IO::DBRecordStore::Impl::insert(
369 const std::string &key,
370 const void *const data,
371 const uint64_t size)
372{
373 if (getMode() == Mode::ReadOnly)
374 throw Error::StrategyError("RecordStore was opened read-only");
375 if (!validateKeyString(key))
376 throw Error::StrategyError("Invalid key format");
377
378 insertRecordSegments(key, data, size);
379 if (!this->_cursorIsInit) {
380 Dbt dbtkey;
381 Dbt dbtdata;
382 /* Do not read any data as we are just moving the cursor */
383 dbtdata.set_dlen(0);
384 dbtdata.set_flags(DB_DBT_PARTIAL);
385 auto rv = this->_dbC->get(&dbtkey, &dbtdata, DB_FIRST);
386 if (rv == 0) {
387 this->_cursorIsInit = true;
388 } else {
389 throw Error::StrategyError(
390 "Could not move cursor during insert");
391 }
392 }
393 /*
394 * If we were at the end, the insert may have added beyond the cursor,
395 * so try to move the cursor.
396 */
397 if (this->_atEnd) {
398 Dbt dbtkey;
399 Dbt dbtdata;
400 /* Do not read any data as we are just moving the cursor */
401 dbtdata.set_dlen(0);
402 dbtdata.set_flags(DB_DBT_PARTIAL);
403 auto rv = this->_dbC->get(&dbtkey, &dbtdata, DB_NEXT);
404 if (rv == 0) {
405 this->_atEnd = false;
406 }
407 }
408 RecordStore::Impl::insert(key, data, size);
409}

411void
412BiometricEvaluation::IO::DBRecordStore::Impl::remove(
413 const std::string &key)
414{
415 if (getMode() == Mode::ReadOnly)
416 throw Error::StrategyError("RecordStore was opened read-only");
417 if (!validateKeyString(key))
418 throw Error::StrategyError("Invalid key format");
419
420 /* Allow exceptions to float out of this function. */
421 removeRecordSegments(key);
422
423 /*
424 * Move the cursor if it was pointing to the deleted key; set _atEnd
425 * if deleted the last record.
426 */
427 Dbt dbtkey;
428 Dbt dbtdata;
429 /* Do not read any data as we are just moving the cursor */
430 dbtdata.set_dlen(0);
431 dbtdata.set_flags(DB_DBT_PARTIAL);
432 auto rv = this->_dbC->get(&dbtkey, &dbtdata, DB_CURRENT);
433 if (rv == DB_KEYEMPTY) {
434 rv = this->_dbC->get(&dbtkey, &dbtdata, DB_NEXT);
435 if (rv == DB_NOTFOUND) {
436 this->_atEnd = true;
437 this->_cursorIsInit = false;
438 }
439 }
440
441 RecordStore::Impl::remove(key);
442}

478{
479 if (getMode() == Mode::ReadOnly)
480 throw Error::StrategyError("RecordStore was opened read-only");
481 if (!validateKeyString(key))

484 /*
485 * Because we sync the entire database, we really don't care
486 * whether the key exists, but check anyway with a read,

501BiometricEvaluation::IO::RecordStore::Record
502BiometricEvaluation::IO::DBRecordStore::Impl::i_sequence(
503 bool returnData,
504 int cursor)
505{
506 if ((cursor != IO::RecordStore::BE_RECSTORE_SEQ_START) &&
507 (cursor != IO::RecordStore::BE_RECSTORE_SEQ_NEXT)) {
508 throw Error::StrategyError("Invalid cursor position as "
509 "argument");
510 }
511 /* If the current cursor position is START, then it doesn't matter
512 * what the client requests; we start at the first record.
513 */
514 u_int32_t pos;
515 if ((getCursor() == IO::RecordStore::BE_RECSTORE_SEQ_START) ||
516 (cursor == IO::RecordStore::BE_RECSTORE_SEQ_START)) {
517 pos = DB_FIRST;
518 } else {
519 if (this->_atEnd) {
520 throw BE::Error::ObjectDoesNotExist();
521 }
522 pos = DB_CURRENT;
523 }
524 /*
525 * Check the Berkeley DB cursor; do not read any data as we are just
526 * moving the cursor in the case where the record under the cursor has
527 * been removed. Read of data is done in another function.
528 */
529 Dbt dbtkey;
530 Dbt dbtdata;
531 dbtdata.set_dlen(0);
532 dbtdata.set_flags(DB_DBT_PARTIAL);
533 auto rv = this->_dbC->get(&dbtkey, &dbtdata, pos);
534 switch (rv) {
535 case DB_NOTFOUND:
536 throw BE::Error::ObjectDoesNotExist();
537 break;
538 /*
539 * The record at the current cursor position has been deleted.
540 * If the cursor cannot retrieve data, we are at the end of
541 * the database.
542 */
543 case DB_KEYEMPTY:
544 rv = this->_dbC->get(&dbtkey, &dbtdata, DB_NEXT);
545 if (rv == DB_NOTFOUND) {
546 throw BE::Error::ObjectDoesNotExist();
547 }
548 break;
549 default:
550 break;
551 }
552
553 BE::IO::RecordStore::Record record;
554 record.key.assign((const char *)dbtkey.get_data(), dbtkey.get_size());
555 if (returnData == true) {
556 /* Don'just copy dbtdata, this may span into subordinate */
557 record.data = this->read(record.key);
558 }
559 /*
560 * Move the BDB cursor to the next record because the cursor points
561 * to either the first record, or last record returned. If there is no
562 * next record, the next call to this function will handle the error.
563 */
564 rv = this->_dbC->get(&dbtkey, &dbtdata, DB_NEXT);
565 if (rv == DB_NOTFOUND) {
566 this->_atEnd = true;
567 } else {
568 this->_atEnd = false;
569 }
570 setCursor(IO::RecordStore::BE_RECSTORE_SEQ_NEXT);
571
572 return (record);
573}

590void
591BiometricEvaluation::IO::DBRecordStore::Impl::setCursorAtKey(
592 const std::string &key)
593{
594 if (!validateKeyString(key))
595 throw Error::StrategyError("Invalid key format");
596 Dbt dbtkey, dbtdata;
597
598 dbtkey.set_data((void *)key.data());
599 dbtkey.set_size(key.length());
600 /*
601 * There is no need to be concerned about subordinate record
602 * segments here because the sequence is maintained entirely
603 * within the primary DB file.
604 */
605 /* Do not read any data as we are just moving the cursor */
606 dbtdata.set_dlen(0);
607 dbtdata.set_flags(DB_DBT_PARTIAL);
608 try {
609 if (this->_dbC->get(&dbtkey, &dbtdata, DB_SET) == DB_NOTFOUND)
610 throw BE::Error::ObjectDoesNotExist(key);
611 } catch (const DbException &e) {
612 throw BE::Error::StrategyError("Could not set Dbc (DB "
613 "error = " + std::to_string(e.get_errno()) + " -- " +
614 e.what() + ")");
615 } catch (const std::exception &e) {
616 throw Error::StrategyError("Could not set Dbc (" +
617 Error::errorStr() + ")");
618 }
619 this->_atEnd = false;
620 setCursor(BE_RECSTORE_SEQ_NEXT);
621}

626/*
627 * Functions to insert all components of a record to the database.
628 */
629static void
630insertIntoDB(
631 std::shared_ptr<Db> db,
632 Dbt &dbtkey,
633 Dbt &dbtdata)
634{
635 try {
636 if (db->put(nullptr, &dbtkey, &dbtdata, DB_NOOVERWRITE) == DB_KEYEXIST)
637 throw BiometricEvaluation::Error::ObjectExists(
638 std::string(static_cast<char*>(dbtkey.get_data()),
639 dbtkey.get_size()));
640 } catch (const DbDeadlockException &e) {
641 throw BiometricEvaluation::Error::StrategyError(
642 "Could not insert to database. Deadlock. (" +
643 std::to_string(e.get_errno()) + ": " + e.what() + ")");
644 } catch (const DbLockNotGrantedException &e) {
645 throw BiometricEvaluation::Error::StrategyError(
646 "Could not insert to database. Lock not granted (" +
647 std::to_string(e.get_errno()) + ": " + e.what() + ")");
648 } catch (const DbRepHandleDeadException &e) {
649 throw BiometricEvaluation::Error::StrategyError(
650 "Could not insert to database. Handle dead. (" +
651 std::to_string(e.get_errno()) + ": " + e.what() + ")");
652 } catch (const DbException &e) {
653 switch (e.get_errno()) {
654 case EACCES:
655 throw BiometricEvaluation::Error::StrategyError(
656 "Could not insert to database. Opened read-only.");
657 default:
658 throw BiometricEvaluation::Error::StrategyError(
659 "Could not insert to database (" +
660 std::to_string(e.get_errno()) + ": " + e.what() +
661 ")");
662 }
663 } catch (const std::exception &e) {
664 throw BiometricEvaluation::Error::StrategyError(
665 "Could not insert to database (" + std::string(e.what()) +
666 ")");
667 }
668}

670void
671BiometricEvaluation::IO::DBRecordStore::Impl::insertRecordSegments(
672 const std::string &key,
673 const void *data,
674 const uint64_t size)
675{
676 /*
677 * Insert all segments.
678 * There is no danger of not having the subordinate DB here
679 * because when the store is opened ReadWrite, the upgrade
680 * to create the subordinate DB is done.
681 */
682 Dbt dbtkey, dbtdata;
683
684 /* Handle the case of a zero-length record */
685 if ((size == 0) || (data == nullptr)) {
686 dbtkey.set_data((void *)key.data());
687 dbtkey.set_size(key.size());
688 dbtdata.set_data(nullptr);
689 dbtdata.set_size(0);
690
691 insertIntoDB(this->_dbP, dbtkey, dbtdata);
692 } else {
693 int segnum = KEY_SEGMENT_START;
694 uint64_t remsize = size;
695 std::string keyseg = key;
696 uint8_t *ptr = (uint8_t *)data;
697
698 /* First segment key same as input key */
699 std::shared_ptr<Db> DBin = this->_dbP; /* Start with primary DB file */
700 while (remsize > 0) {
701 dbtkey.set_data((void *)keyseg.data());
702 dbtkey.set_size(keyseg.size());
703
704 if (remsize < MAX_REC_SIZE) {
705 dbtdata.set_size(remsize);
706 remsize = 0;
707 } else {
708 dbtdata.set_size(MAX_REC_SIZE);
709 remsize = remsize - MAX_REC_SIZE;
710 }
711 dbtdata.set_data(ptr);
712 ptr += dbtdata.get_size();
713 try {
714 insertIntoDB(DBin, dbtkey, dbtdata);
715 } catch (Error::ObjectExists &e) {
716 throw;
717 } catch (Error::StrategyError &e) {
718 throw;
719 }
720 keyseg = genKeySegName(key, segnum);
721 segnum++;
722 DBin = this->_dbS; /* Switch to subordinate DB */
723 }
724 }
725}

727/*
728 * Function to read all components of a record from the database.
729 */
730uint64_t
731BiometricEvaluation::IO::DBRecordStore::Impl::readRecordSegments(
732 const std::string &key,
733 void *const data)
734 const
735{
736 if (!validateKeyString(key))
737 throw Error::StrategyError("Invalid key format");
738
739 /*
740 * Read all segments.
