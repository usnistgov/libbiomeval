55void
56BiometricEvaluation::Image::NetPBM::parseHeader()
57{
58 const uint8_t *data = this->getDataPointer();
59 const uint64_t dataSize = this->getDataSize();
60
61 size_t offset = 0;
62 skipComment(data, this->getDataSize(), offset);
63 if (data[offset++] != 'P')
64 throw Error::DataError("Not a valid NetPBM file");
65
66 /* Integer at second byte indiciates the format of the image data */
67 switch (data[offset++]) {
68 case '1':
69 _kind = Kind::ASCIIPortableBitmap;
70 break;
71 case '2':
72 _kind = Kind::ASCIIPortableGraymap;
73 break;
74 case '3':
75 _kind = Kind::ASCIIPortablePixmap;
76 break;
77 case '4':
78 _kind = Kind::BinaryPortableBitmap;
79 break;
80 case '5':
81 _kind = Kind::BinaryPortableGraymap;
82 break;
83 case '6':
84 _kind = Kind::BinaryPortablePixmap;
85 break;
86 default:
87 throw Error::DataError("Not a valid NetPBM magic number");
88 }
89
90 /* Space separated width and height immediately follow magic number */
91 uint32_t width = atoi((char *)getNextValue(data, dataSize, offset).c_str());
92 uint32_t height = atoi((char *)getNextValue(data, dataSize, offset).c_str());
93 setDimensions(Size(width, height));
94
95 /* Maximum color value follow dimensions on non-bitmap formats */
96 switch (_kind) {
97 case Kind::ASCIIPortableGraymap:
98 /* FALLTHROUGH */
99 case Kind::BinaryPortableGraymap:
100 /* FALLTHROUGH */
101 case Kind::ASCIIPortablePixmap:
102 /* FALLTHROUGH */
103 case Kind::BinaryPortablePixmap:
104 _maxColorValue =
105 atoi((char *)getNextValue(data, dataSize, offset).c_str());
106 break;
107 default:
108 break;
109 }
110
111 /* Set depth (based on max color value) */
112 switch (_kind) {
113 case Kind::ASCIIPortableBitmap:
114 /* FALLTHROUGH */
115 case Kind::BinaryPortableBitmap:
116 /* Bitmaps are 1-bit depth by definition */
117 setColorDepth(1);
118 this->setBitDepth(1);
119 this->setHasAlphaChannel(false);
120 break;
121 case Kind::ASCIIPortableGraymap:
122 /* FALLTHROUGH */
123 case Kind::BinaryPortableGraymap:
124 /* Graymaps can provide gray levels in the 1 - 65535 range */
125 if (this->_maxColorValue < 256) {
126 this->setColorDepth(8);
127 this->setBitDepth(8);
128 } else {
129 this->setColorDepth(16);
130 this->setBitDepth(16);
131 }
132 this->setHasAlphaChannel(false);
133 break;
134 case Kind::ASCIIPortablePixmap:
135 /* FALLTHROUGH */
136 case Kind::BinaryPortablePixmap:
137 /* Pixmaps can provide R, G, B values in the 1 - 65535 range */
138 if (this->_maxColorValue < 256) {
139 this->setColorDepth(24);
140 this->setBitDepth(8);
141 } else {
142 this->setColorDepth(48);
143 this->setBitDepth(16);
144 }
145 this->setHasAlphaChannel(false);
146 break;
147 default:
148 break;
149 }
150
151 /* Resolution is unspecified */
152 setResolution(Resolution(72, 72, Resolution::Units::PPI));
153
154 /* Payload comes a minimum of one whitespace after last header item */
155 _headerLength = offset + 1;
156}

171std::string
172BiometricEvaluation::Image::NetPBM::getNextValue(
173 const uint8_t *data,
174 size_t dataSize,
175 size_t &offset,
176 size_t sizeOfValue)
177{
178 bool nonSpaceEncountered = false;
179 std::stringstream value;
180 char c;
181
182 for ( ; offset < dataSize; offset++) {
183 c = data[offset];
184
185 /*
186 * An arbitrary amount of whitespace may be between information
187 * elements and the presence of whitespace or the beginning of
188 * a comment indicates the end of an information element.
189 */
190 if (isspace(c)) {
191 if (nonSpaceEncountered)
192 /* End of information item */
193 break;
194 else
195 /* Awaiting start of information item */
196 continue;
197 } else if (c == '#') {
198 /* Once a comment is encountered, skip to EOL */
199 skipLine(data, dataSize, offset);
200
201 /* Could have a comment begin after a value */
202 if (nonSpaceEncountered)
203 break;
204 else
205 continue;
206 }
207
208 nonSpaceEncountered = true;
209 value << c;
210
211 /* Break if we have obtained an expected value size */
212 if (value.str().size() == sizeOfValue) {
213 offset++;
214 break;
215 }
216 }
217
218 return (value.str());
219}

241BiometricEvaluation::Memory::uint8Array
242BiometricEvaluation::Image::NetPBM::getRawData()
243 const
244{
245 const uint8_t *data = this->getDataPointer() + this->_headerLength;
246 const uint64_t dataSize = this->getDataSize() - this->_headerLength;
247
248 switch (_kind) {
249 case Kind::ASCIIPortableBitmap:
250 return (ASCIIBitmapTo8Bit(data, dataSize, getDimensions().xSize,
251 getDimensions().ySize));
252 case Kind::BinaryPortableBitmap:
253 return (BinaryBitmapTo8Bit(data, dataSize,
254 getDimensions().xSize, getDimensions().ySize));
255 case Kind::ASCIIPortableGraymap:
256 /* FALLTHROUGH */
257 case Kind::ASCIIPortablePixmap:
258 return (ASCIIPixmapToBinaryPixmap(data, dataSize,
259 getDimensions().xSize, getDimensions().ySize,
260 getColorDepth(), this->_maxColorValue));
261 case Kind::BinaryPortableGraymap:
262 /* FALLTHROUGH */
263 case Kind::BinaryPortablePixmap: {
264 Memory::uint8Array rawData(dataSize);
265 rawData.copy(data);
266
267 /* NetPBM stores data big-endian */
268 if ((this->getColorDepth() == 16 ||
269 this->getColorDepth() == 48) && Memory::isLittleEndian())
270 for (uint64_t i = 0; i < (rawData.size() - 1); i += 2)
271 std::swap(rawData[i], rawData[i + 1]);
272
273 return (rawData);
274 }
275 default:
276 throw Error::NotImplemented(
277 std::to_string(std::underlying_type<Kind>::type(_kind)));
278 }
279}

281BiometricEvaluation::Memory::uint8Array
282BiometricEvaluation::Image::NetPBM::ASCIIBitmapTo8Bit(
283 const uint8_t *bitmap,
284 uint64_t bitmapSize,
285 uint32_t width,
286 uint32_t height)
287{
288 Memory::uint8Array eightBitData(width * height);
289 Memory::MutableIndexedBuffer buffer(eightBitData);
290
291 std::string nextValue;
292 uint8_t byte;
293 size_t bitmapOffset = 0;
294 while (bitmapOffset < bitmapSize) {
295 /* Get next one-byte non-space value */
296 nextValue = getNextValue(bitmap, bitmapSize, bitmapOffset, 1);
297
298 /* Extraneous spaces/newline at end of file */
299 if (nextValue.empty())
300 break;
301
302 byte = nextValue[0];
303
304 /* 0 is white, 1 is black */
305 buffer.pushU8Val((byte == '0') ? 0xFF : 0x00);
306 }
307
308 return (eightBitData);
309}

311BiometricEvaluation::Memory::uint8Array
312BiometricEvaluation::Image::NetPBM::ASCIIPixmapToBinaryPixmap(
313 const uint8_t *ASCIIBuf,
314 uint64_t ASCIIBufSize,
315 uint32_t width,
316 uint32_t height,
317 uint8_t depth,
318 uint32_t maxColor)
319{
320 /* Ensure valid bit depth */
321 if (((depth % 8) != 0) || (depth > 48))
322 throw Error::ParameterError("Invalid depth");
323 const uint8_t bytesPerPixel = (depth / 8);
324
325 Memory::uint8Array binaryBuf(width * height * bytesPerPixel);
326 Memory::MutableIndexedBuffer buffer(binaryBuf);
327
328 size_t ASCIIOffset = 0;
329 std::string nextValue;
330 uint32_t decimal;
331 while (ASCIIOffset < ASCIIBufSize) {
332 /* Read space separated ASCII integer */
333 nextValue = getNextValue(ASCIIBuf, ASCIIBufSize, ASCIIOffset);
334
335 /* Extraneous spaces/newline at end of file */
336 if (nextValue.empty())
337 break;
338
339 /* Scale to colorspace */
340 decimal = valueInColorspace(std::stol(nextValue),
341 maxColor, depth);
342
343 if (maxColor <= 255) {
344 /* One byte per component */
345 buffer.pushU8Val(decimal);
346 } else {
347 /* Two bytes per component (max color limited) */
348 buffer.pushU16Val(decimal);
349 }
350 }
351
352 return (binaryBuf);
353}

355BiometricEvaluation::Memory::uint8Array
356BiometricEvaluation::Image::NetPBM::BinaryBitmapTo8Bit(
357 const uint8_t *bitmap,
358 uint64_t bitmapSize,
359 uint32_t width,
360 uint32_t height)
361{
362 Memory::uint8Array eightBitData(width * height);
363 Memory::MutableIndexedBuffer buffer(eightBitData);
364
365 uint8_t byte, mask;
366 for (size_t i = 0, offset = 0; i < bitmapSize; i++) {
367 byte = bitmap[i];
368
369 mask = 0x80; /* 0b10000000 */
370 for (int j = 0; j < 8; j++, mask >>= 1) {
371 /* 0 is white, 1 is black */
372 buffer.pushU8Val(((byte & mask) == 0) ? 0xFF : 0x00);
373
374 /* Skip filler bits when width not a multiple of 8 */
375 if ((++offset % width) == 0)
376 break;
377 }
378 }
379
380 return (eightBitData);
381}

391bool
392BiometricEvaluation::Image::NetPBM::isNetPBM(
393 const uint8_t *data,
394 uint64_t size)
395{
396 /* Skip any comments that exist before the magic bits */
397 size_t offset = 0;
398 while (data[offset] == '#') {
399 while (offset < size && data[offset] != '\n')
400 offset++;
401 if (offset + 1 < size)
402 offset++;
403 }
404
405 /* Ran off the edge of the image without encountering magic bits */
406 if (offset + 1 >= size)
407 return (false);
408
409 if ((data[offset] == 'P') &&
410 (data[offset + 1] >= '1' && data[offset + 1] <= '6'))
411 return (true);
412
413 return (false);
414}
