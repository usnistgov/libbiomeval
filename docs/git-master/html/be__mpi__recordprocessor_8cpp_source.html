39BiometricEvaluation::MPI::RecordProcessor::processWorkPackage(
40 MPI::WorkPackage &workPackage)
41{
42 /*
43 * Extract the key/value data from the work package
44 */
45 Memory::uint8Array packageData(0);
46 workPackage.getData(packageData);
47 uint64_t numElements = workPackage.getNumElements();
48
49 /*
50 * Call the implementation's record processor function
51 * for each key.
52 */
53 uint64_t index = 0;
54 Memory::uint8Array value(0);
55 for (uint64_t count = 0; count < numElements; count++) {
56
57 /*
58 * Read the key length, value size, create a std::string
59 * from the characters of that length, then create the data
60 * array if size is non-zero.
61 */
62 uint32_t *pInt32 = (uint32_t *)&packageData[index];
63 uint32_t keyLength = *pInt32;
64 index += sizeof(uint32_t);
65 uint64_t *pInt64 = (uint64_t *)&packageData[index];
66 uint64_t valueSize = *pInt64;
67 index += sizeof(uint64_t);
68 std::string key((char *)&packageData[index], keyLength);
69 index += keyLength;
70 if (valueSize > 0) {
71 value.resize(valueSize);
72 std::memcpy((void *)&value[0],
73 (void *)&packageData[index], valueSize);
74 index += valueSize;
75 }
76
77 /*
78 * Stop processing only when a quick or immediate exit
79 * condition exists. On a normal exit, we are allowed to
80 * finish with the work package.
81 */
82 if (MPI::QuickExit || MPI::TermExit) {
83 IO::Logsheet *log = this->getLogsheet().get();
84 log->writeDebug("Early exit: End record processing");
85 break;
86 }
87 try {
88 if (valueSize > 0) {
89 this->processRecord(key, value);
90 } else {
91 this->processRecord(key);
92 }
93 /*
94 * The record processor is asking for termination.
95 * Rethrow the exception so the framekwork will start
96 * the shutdown.
97 */
98 } catch (const Error::Exception&) {
99 throw;
100 }
101 }
102}
