66std::map<const std::string,
67const std::shared_ptr<BiometricEvaluation::IO::RecordStore>>
68BiometricEvaluation::IO::RecordStoreUnion::Impl::initRecordStoreMap(
69 const std::map<const std::string, const std::string> &input)
70 const
71{
72 std::map<const std::string, const std::shared_ptr<
73 BiometricEvaluation::IO::RecordStore>> recordStores;
74 for (const auto &rsInfo : input)
75 recordStores.emplace(std::make_pair(rsInfo.first,
76 BE::IO::RecordStore::openRecordStore(rsInfo.second,
77 BE::IO::Mode::ReadOnly)));
78 return (recordStores);
79}

81void
82BiometricEvaluation::IO::RecordStoreUnion::Impl::verifyRecordStoreNames(
83 const std::map<const std::string,
84 BiometricEvaluation::Memory::uint8Array> &data)
85 const
86{
87 if (data.size() < this->_recordStores.size())
88 throw BE::Error::ParameterError("Missing RecordStore names");
89
90 /* Check that all keys are valid (no duplicates possible) */
91 for (const auto &dataPair : data)
92 if (this->_recordStores.find(dataPair.first) ==
93 this->_recordStores.cend())
94 throw BE::Error::ObjectDoesNotExist(dataPair.first);
95}

123std::map<const std::string, BiometricEvaluation::Memory::uint8Array>
124BiometricEvaluation::IO::RecordStoreUnion::Impl::read(
125 const std::string &key)
126 const
127{
128 std::string exceptions;
129 std::map<const std::string,
130 BiometricEvaluation::Memory::uint8Array> ret;
131
132 for (const auto &rsPair : this->_recordStores) {
133 try {
134 ret.emplace(std::make_pair(rsPair.first,
135 rsPair.second->read(key)));
136 } catch (const BE::Error::ObjectDoesNotExist&) {
137 /* Swallow */
138 } catch (BE::Error::Exception &e) {
139 if (!exceptions.empty())
140 exceptions += '\n';
141 exceptions += e.whatString() + " (" + rsPair.first +
142 ')';
143 }
144 }
145
146 if (!exceptions.empty())
147 throw BE::Error::StrategyError(exceptions);
148 if (ret.size() == 0)
149 throw BE::Error::ObjectDoesNotExist(key);
150
151 return (ret);
152}
