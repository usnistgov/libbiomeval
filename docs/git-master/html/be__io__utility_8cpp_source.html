59void
60BiometricEvaluation::IO::Utility::copyDirectoryContents(
61 const std::string &sourcepath,
62 const std::string &targetpath,
63 const bool removesource)
64{
65 struct stat sb;
66 struct dirent *entry;
67 DIR *dir = nullptr;
68
69 if (!IO::Utility::pathIsDirectory(sourcepath))
70 throw Error::ObjectDoesNotExist(sourcepath + " is not a path");
71 if (IO::Utility::fileExists(targetpath)) {
72 if (!IO::Utility::pathIsDirectory(targetpath))
73 throw Error::ObjectDoesNotExist(targetpath +
74 " is not a path");
75 } else {
76 /* Create the target dir with same perms as source */
77 stat(sourcepath.c_str(), &sb);
78 if(IO::Utility::makePath(targetpath, sb.st_mode) != 0)
79 throw Error::StrategyError(targetpath +
80 " could not be created");
81 }
82
83 dir = opendir(sourcepath.c_str());
84 if (dir == nullptr)
85 throw Error::StrategyError(sourcepath + " could not be opened");
86
87 while ((entry = readdir(dir)) != nullptr) {
88#ifndef _WIN32
89 if (entry->d_ino == 0)
90 continue;
91#endif
92 if ((strcmp(entry->d_name, ".") == 0) ||
93 (strcmp(entry->d_name, "..") == 0))
94 continue;
95
96 std::string sourcefile = sourcepath + "/" + entry->d_name;
97 std::string targetfile = targetpath + "/" + entry->d_name;
98
99 /* Recursively copy subdirectories and files */
100 try {
101 if (IO::Utility::pathIsDirectory(sourcefile)) {
102 copyDirectoryContents(sourcefile, targetfile);
103 } else {
104 /* copy the file */
105 IO::Utility::writeFile(
106 IO::Utility::readFile(sourcefile),
107 targetfile);
108 }
109 } catch (Error::Exception &e) {
110 closedir(dir);
111 throw;
112 }
113 }
114 closedir(dir);
115 if (removesource)
116 IO::Utility::removeDirectory(sourcepath);
117}

119void
120BiometricEvaluation::IO::Utility::removeDirectory(
121 const std::string &directory,
122 const std::string &prefix)
123{
124 struct stat sb;
125 struct dirent *entry;
126 DIR *dir = nullptr;
127 std::string dirpath, filename;
128
129 dirpath = prefix + "/" + directory;
130 if (stat(dirpath.c_str(), &sb) != 0)
131 throw Error::ObjectDoesNotExist(dirpath + " does not exist");
132 dir = opendir(dirpath.c_str());
133 if (dir == nullptr)
134 throw Error::StrategyError(dirpath + " could not be opened");
135
136 while ((entry = readdir(dir)) != nullptr) {
137#ifndef _WIN32
138 if (entry->d_ino == 0)
139 continue;
140#endif
141 if ((strcmp(entry->d_name, ".") == 0) ||
142 (strcmp(entry->d_name, "..") == 0))
143 continue;
144
145 filename = dirpath + "/" + entry->d_name;
146 if (stat(filename.c_str(), &sb) != 0) {
147 if (dir != nullptr) {
148 if (closedir(dir)) {
149 throw Error::StrategyError("Could not "
150 "close " + dirpath + " (" +
151 Error::errorStr() + ")");
152 }
153 }
154 throw Error::StrategyError("Could not stat " +
155 filename);
156 }
157
158 /* Recursively remove subdirectories and files */
159 if (S_ISDIR(sb.st_mode)) {
160 removeDirectory(entry->d_name, dirpath);
161 } else {
162 if (std::remove(filename.c_str())) {
163 const std::string rmErr{Error::errorStr()};
164 if (dir != nullptr) {
165 if (closedir(dir)) {
166 throw Error::StrategyError(
167 "Could not close " +
168 dirpath + " (" +
169 Error::errorStr() + ") or "
170 "remove " + filename + " "
171 "(" + rmErr + ")");
172 }
173 }
174 throw Error::StrategyError(filename + " "
175 "could not be removed (" + rmErr + ")");
176 }
177 }
178 }
179
180 if (dir != nullptr) {
181 if (closedir(dir)) {
182 throw Error::StrategyError("Could not close " +
183 dirpath + " (" + Error::errorStr() + ")");
184 }
185 }
186
187 /* Remove parent directory, now that children have been removed */
188 if (rmdir(dirpath.c_str()))
189 throw Error::StrategyError(dirpath + " could not be removed (" +
190 Error::errorStr() + ")");
191}

202void
203BiometricEvaluation::IO::Utility::setAsideName(
204 const std::string &path)
205{
206 if (!fileExists(path))
207 throw Error::ObjectDoesNotExist(path + " does not exist");
208
209 /* Find the next index available for the set aside name */
210 uint32_t idx;
211 std::ostringstream sstr;
212 for (idx = 1; idx <= UINT16_MAX; idx++) {
213 sstr << path << "." << idx;
214 if (!fileExists(sstr.str()))
215 break;
216 sstr.str("");
217 }
218 if (idx > UINT16_MAX)
219 throw (Error::StrategyError("All possible names in use"));
220
221 int rc = rename(path.c_str(), sstr.str().c_str());
222 if (rc != 0)
223 throw (Error::StrategyError("Could not rename " + path + ": " +
224 Error::errorStr()));
225}

243uint64_t
244BiometricEvaluation::IO::Utility::sumDirectoryUsage(const std::string &pathname)
245{
246 uint64_t total = 0;
247 struct stat sb;
248 struct dirent *entry;
249 DIR *dir = NULL;
250 std::string filename;
251
252 /*
253 * A list to contain directory entry inode numbers that have
254 * more than one hard link associated with it.
255 */
256 std::list<ino_t> multiLinked;
257 std::list<ino_t>::iterator it;
258
259 if (stat(pathname.c_str(), &sb) != 0)
260 throw Error::ObjectDoesNotExist(pathname + " does not exist");
261
262 /* Add the parent directory to the total */
263 total += sb.st_size;
264
265 dir = opendir(pathname.c_str());
266 if (dir == NULL)
267 throw Error::StrategyError("Could not open " + pathname + " ("
268 + Error::errorStr() + ")");
269
270 while ((entry = readdir(dir)) != NULL) {
271#ifndef _WIN32
272 if (entry->d_ino == 0)
273 continue;
274#endif
275 if ((strcmp(entry->d_name, ".") == 0) ||
276 (strcmp(entry->d_name, "..") == 0))
277 continue;
278
279 filename = pathname + "/" + entry->d_name;
280
281 if (lstat(filename.c_str(), &sb) != 0) {
282 if (dir != NULL) {
283 if (closedir(dir)) {
284 throw Error::StrategyError("Could not "
285 "close " + pathname + " (" +
286 Error::errorStr() + ")");
287 }
288 }
289 throw Error::StrategyError("Could not stat " +
290 filename);
291 }
292
293 /*
294 * If there is more than one hard link to the file, we
295 * only count its size once.
296 */
297 if (sb.st_nlink > 1) {
298 it = std::find(
299 multiLinked.begin(), multiLinked.end(),
300 sb.st_ino);
301 if (it == multiLinked.end())
302 multiLinked.push_back(sb.st_ino);
303 else
304 continue;
305 }
306
307 if (S_ISDIR(sb.st_mode)) {
308 /* Recursively sum subdirectories and files */
309 total += sumDirectoryUsage(filename);
310 } else {
311 total += sb.st_size;
312 }
313 }
314
315 if (dir != NULL) {
316 if (closedir(dir)) {
317 throw Error::StrategyError("Could not close " +
318 pathname + " (" + Error::errorStr() + ")");
319 }
320 }
321 return (total);
322}

397void
398BiometricEvaluation::IO::Utility::readPipe(
399 void *data,
400 size_t size,
401 int pipeFD)
402{
403#ifdef _WIN32
404 throw BE::Error::NotImplemented("IO::Utility::readPipe()");
405#else
406 size_t remaining = size;
407 uint8_t *ptr = (uint8_t *)data;;
408 while (true) {
409 ssize_t sz = read(pipeFD, ptr, remaining);
410 if (sz == -1) {
411 throw (Error::StrategyError("Could not read pipe: "
412 + Error::errorStr()));
413 }
414 remaining -= sz;
415 ptr += sz;
416 if (remaining == 0) {
417 break;
418 }
419 /*
420 * Reading from a pipe that is closed on the write end
421 * results in no data being returned; no signal.
422 */
423 if (sz == 0) {
424 throw Error::ObjectDoesNotExist("Widowed pipe");
425 }
426 }
427#endif
428}

438void
439BiometricEvaluation::IO::Utility::writePipe(
440 const void *data,
441 size_t size,
442 int pipeFD)
443{
444#ifdef _WIN32
445 throw BE::Error::NotImplemented("IO::Utility::writePipe()");
446#else
447 sigset_t sigset;
448 sigemptyset(&sigset);
449 sigaddset(&sigset, SIGPIPE);
450 Error::SignalManager signalManager(sigset);
451
452 size_t remaining = size;
453 uint8_t *ptr = (uint8_t *)data;;
454 ssize_t sz = 0;
455 while (true) {
456 BEGIN_SIGNAL_BLOCK(&signalManager, pipe_write_length_block);
457 sz = write(pipeFD, data, remaining);
458 END_SIGNAL_BLOCK(&signalManager, pipe_write_length_block);
459 if (sz == -1) {
460 throw (Error::StrategyError("Could not write pipe: "
461 + Error::errorStr()));
462 }
463 remaining -= sz;
464 ptr += sz;
465 /*
466 * Writing to a pipe that is closed on the read end
467 * results in a signal.
468 */
469 if (signalManager.sigHandled())
470 throw Error::ObjectDoesNotExist("Widowed pipe");
471 if (remaining == 0) {
472 break;
473 }
474 }
475#endif
476}

512FILE*
513BiometricEvaluation::IO::Utility::createTemporaryFile(
514 std::string &path,
515 const std::string &prefix,
516 const std::string &parentDir)
517{
518 path.clear();
519
520#ifdef _WIN32
521 /* Default argument is "/tmp" but that won't exist on Windows */
522 if (parentDir == "/tmp") {
523 char winDefaultParentDir[MAX_PATH];
524 const auto len = GetTempPath(MAX_PATH, winDefaultParentDir);
525 if ((len > MAX_PATH) || (len == 0))
526 throw BE::Error::StrategyError("Invalid return "
527 "from GetTempPath (" + BE::Error::errorStr() +
528 ')');
529 /* GetTempPath appends trailing slash */
530 path = winDefaultParentDir;
531 } else
532#endif /* _WIN32 */
533 if (!parentDir.empty() && parentDir != ".")
534 path = parentDir + '/';
535 std::string tmpl = "-XXXXXX";
536 if (prefix.empty() )
537 tmpl = path + "libbiomeval-XXXXXX";
538 else
539 tmpl = path + prefix + "-XXXXXX";
540
541 char *name = strdup(tmpl.c_str());
542 if (name == nullptr)
543 throw Error::MemoryError();
544
545 int fildes = mkstemp(name);
546 if (fildes == -1) {
547 free(name);
548 throw Error::FileError("Could not create temporary file (" +
549 Error::errorStr() + ')');
550 }
551
552 path = name;
553 free(name);
554
555 FILE *fp = fdopen(fildes, "w+x");
556 if (fp == nullptr)
557 throw Error::FileError("Could not wrap with stream (" +
558 Error::errorStr() + ')');
559
560 return (fp);
561}
