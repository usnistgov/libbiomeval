83int32_t
84BiometricEvaluation::MPI::Receiver::PackageWorker::workerMain()
85{
86 /*
87 * Open a Logsheet, and if that fails, indicate that
88 * we are finished.
89 */
90 try {
91 this->_logsheet =
92 BE::MPI::openLogsheet(
93 this->_resources->getLogsheetURL(),
94 "MPI::Worker");
95 } catch (const Error::Exception &e) {
96 MPI::printStatus("Worker failed to open log sheet (" +
97 e.whatString() + ')');
98 return(-1);
99 }
100 BE::IO::Logsheet *log = this->_logsheet.get();
101
102 /*
103 * At this point, we are in a child process with its
104 * own copy of the package processor object.
105 */
106 BiometricEvaluation::MPI::WorkPackage workPackage;
107 BE::Memory::uint8Array message;
108 MPI::TaskStatus taskStatus = MPI::TaskStatus::OK;
109 MPI::TaskCommand taskCommand;
110
111 /*
112 * The child process needs its own copy of the package
113 * processor so that it can have a unique copy of all
114 * file references and resources.
115 */
116 try {
117 this->_workPackageProcessor =
118 this->_workPackageProcessor->newProcessor(this->_logsheet);
119 } catch (BE::Error::Exception &e) {
120 std::string error{"Worker failed to create a child package "
121 "processor (" + e.whatString() + ")"};
122 MPI::printStatus(error);
123 MPI::logMessage(*log, error);
124 return (-1);
125 } catch (...) {
126 std::string error{"Worker failed to create a child package "
127 "processor (caught unknown exception)"};
128 MPI::printStatus(error);
129 MPI::logMessage(*log, error);
130 return (-1);
131 }
132
133 if (this->_workPackageProcessor == nullptr) {
134 std::string error{"Worker failed to create a child package "
135 "processor (package processor was NULL)"};
136 MPI::printStatus(error);
137 MPI::logMessage(*log, error);
138 return (-1);
139 }
140
141 /*
142 * The processing of a work package loop. We only break out
143 * of this loop if there's an inability to communicate, which
144 * can happen if the parent closes its pipe, or waiting for
145 * message returns false, which means that this process most
146 * likely has been requested to stop.
147 */
148 while (this->stopRequested() == false) {
149
150 /*
151 * Stop asking for work packages if any exit condition
152 * exists.
153 */
154 if (MPI::Exit || MPI::QuickExit || MPI::TermExit) {
155 MPI::logMessage(*log,
156 "Early Exit: End package requests");
157 taskStatus = MPI::TaskStatus::Exit;
158 }
159
160 /*
161 * Send a status message to report status, asking for more
162 * work unless we are in a bad state; then exit.
163 */
164 statusToMessage(taskStatus, message);
165 try {
166 this->sendMessageToManager(message);
167 if (taskStatus != MPI::TaskStatus::OK) {
168 break;
169 }
170 } catch (Error::Exception &e) {
171 MPI::logMessage(*log, "Worker send message failure: "
172 + e.whatString());
173 break;
174 }
175 /*
176 * This call will prevent hangs on job end, (because it
177 * always times out and checks stopRequested) but there
178 * still is a race condition between here and the
179 * receiveMessage() call, but that call should fail then.
180 */
181 try {
182 if (this->waitForMessage() == false)
183 break;
184 this->receiveMessageFromManager(message);
185 } catch (Error::Exception &e) {
186 MPI::logMessage(*log, "Worker receive message failure: "
187 + e.whatString());
188 taskStatus = MPI::TaskStatus::Failed;
189 continue; /* Attempt to send one final status */
190 }
191
192 /*
193 * XXX Check for checkpoint messages.
194 * Note that we don't check for Exit command because the
195 * process management framework controls normal exit.
196 */
197 taskCommand = messageToCommand(message);
198 if (taskCommand == MPI::TaskCommand::Ignore) {
199 continue;
200 }
201 /*
202 * Receieve the work package and hand it off to the
203 * package processor.
204 */
205 try {
206 this->waitForMessage();
207 this->receiveMessageFromManager(message);
208 uint64_t wpCount;
209 std::memcpy(&wpCount, &message[0], sizeof(wpCount));
210 this->waitForMessage();
211 this->receiveMessageFromManager(message);
212 workPackage = MPI::WorkPackage(message);
213 workPackage.setNumElements(wpCount);
214 } catch (Error::Exception &e) {
215 MPI::logMessage(*log, "Failed to receive work package: "
216 + e.whatString());
217 taskStatus = MPI::TaskStatus::Failed;
218 continue; /* Attempt to send one final status */
219 }
220 try {
221 this->_workPackageProcessor->processWorkPackage(
222 workPackage);
223 } catch (MPI::TerminateJob &e) {
224 MPI::logMessage(*log,
225 "Package processor wants complete job termination: "
226 + e.whatString());
227 taskStatus = MPI::TaskStatus::RequestJobTermination;
228 continue; /* Attempt to send one final status */
229 } catch (Error::Exception &e) {
230 MPI::logMessage(*log,
231 "Package processor wants shutdown: "
232 + e.whatString());
233 taskStatus = MPI::TaskStatus::Failed;
234 continue; /* Attempt to send one final status */
235 }
236 }
237
238 this->_workPackageProcessor.reset();
239 MPI::logMessage(*log, "Worker process exiting");
240 return(0);
241}

263void
264BiometricEvaluation::MPI::Receiver::sendWorkPackage(
265 MPI::WorkPackage &workPackage)
266{
267 /*
268 * While there is some worker available, send the work package
269 * to the first worker from which we receive a request. If that
270 * worker wants to go on furlough, we move on to the next worker.
271 * If there are no more workers to process the package, then
272 * we log the fact that the package is lost and return.
273 */
274 std::shared_ptr<Process::WorkerController> worker;
275 BE::Memory::uint8Array message;
276 BE::Memory::uint8Array wpData;
277 MPI::TaskStatus taskStatus;
278 BE::IO::Logsheet *log = this->_logsheet.get();
279
280 /*
281 * Wait for a request from a worker. A request starts
282 * with a status message. Handle the case where a worker
283 * is exiting, and when there may be no more workers.
284 */
285 while (true) {
286 if (this->_processManager.getNumActiveWorkers() == 0)
287 throw (Error::StrategyError("No workers"));
288
289 /*
290 * Check for an out-of-band message indicating whether
291 * Task-0 has an Exit condition requring that we stop
292 * disitributing work packages. We must check here
293 * because we can be waiting a long time for a worker
294 * to request a work package.
295 */
296 bool oobmsg = ::MPI::COMM_WORLD.Iprobe(0,
297 to_int_type(MPI::MessageTag::OOB));
298 if (oobmsg) {
299 MPI::taskcmd_t oobCmd;
300 ::MPI::COMM_WORLD.Recv((void *)&oobCmd, 1, MPI_INT32_T,
301 0, to_int_type(MPI::MessageTag::OOB));
302 const auto oobCmdE = to_enum<TaskCommand>(oobCmd);
303 if (oobCmdE == MPI::TaskCommand::QuickExit) {
304 MPI::logMessage(*log, "OOB Quick Exit received");
305 MPI::QuickExit = true;
306 }
307 if (oobCmdE == MPI::TaskCommand::TermExit) {
308 MPI::logMessage(*log, "OOB Term Exit received");
309 MPI::TermExit = true;
310 }
311 }
312
313 /*
314 * If Quick or Term Exit, do not send out the work package.
315 * Normal Exit, send it out. Actual shutting down of the
316 * Receiver is not done here.
317 */
318 if (MPI::QuickExit || MPI::TermExit) {
319 return;
320 }
321
322 /*
323 * If no worker is ready, pause for a bit, then go back
324 * to the top of the loop and start over.
325 */
326 bool msgAvail =
327 this->_processManager.getNextMessage(worker, message, 0);
328 if (!msgAvail) {
329 struct timespec ts;
330 ts.tv_sec = 0;
331 ts.tv_nsec = 100000000L; /* 100 milliseconds */
332 nanosleep(&ts, NULL);
333 continue;
334 }
335
336 /*
337 * Once a worker is ready, we're dedicated to sending off
338 * the work package, so no checks for Exit conditions here.
339 */
340 taskStatus = messageToStatus(message);
341
342 /*
343 * When a worker gets into trouble, have it stop processing.
344 */
345 if (taskStatus == MPI::TaskStatus::RequestJobTermination)
346 throw MPI::TerminateJob();
347 if (taskStatus != MPI::TaskStatus::OK) {
348 try {
349 this->_processManager.stopWorker(worker);
350 } catch (Error::Exception &e) {
351 MPI::logMessage(*log,
352 "Task-N stopping worker: Caught: "
353 + e.whatString());
354 }
355 } else {
356 break;
357 }
358 }
359
360 /*
361 * Tell the worker to continue on.
362 */
363 commandToMessage(MPI::TaskCommand::Continue, message);
364 worker->sendMessageToWorker(message);
365
366 /*
367 * A work package is sent in two parts:
368 * The number of elements, and the raw data.
369 */
370 uint64_t wpCount = workPackage.getNumElements();
371 message.resize(sizeof(wpCount));
372 std::memcpy(&message[0], &wpCount, sizeof(wpCount));
373 worker->sendMessageToWorker(message);
374 workPackage.getData(wpData);
375 worker->sendMessageToWorker(wpData);
376 *log << "Sent work package of size " << wpData.size() << " to worker";
377 MPI::logEntry(*log);
378}

380BiometricEvaluation::MPI::TaskStatus
381BiometricEvaluation::MPI::Receiver::requestWorkPackages()
382{
383 BE::Memory::uint8Array workPackageRaw(0);
384
385 ::MPI::Status MPIstatus;
386 BE::MPI::taskstat_t taskStatus;
387 BE::MPI::taskcmd_t taskCommand;
388 MPI::TaskStatus status = MPI::TaskStatus::OK;
389 BE::IO::Logsheet *log = this->_logsheet.get();
390
391 while (true) {
392
393 /*
394 * Check local exit conditions.
395 * Tell workers to exit when an immediate exit condition
396 * exists. In all cases, tell Task-0 we are done.
397 */
398 if (MPI::Exit) {
399 MPI::logMessage(*log, "Exit signal");
400 taskStatus = to_int_type(MPI::TaskStatus::Exit);
401 ::MPI::COMM_WORLD.Send(
402 (void *)&taskStatus, 1, MPI_INT32_T,
403 0, to_int_type(MPI::MessageTag::Control));
404 status = MPI::TaskStatus::Exit;
405 break;
406 }
407 if (MPI::QuickExit) {
408 MPI::logMessage(*log, "Quick Exit signal");
409 this->_processManager.broadcastSignal(SIGINT);
410 taskStatus = to_int_type(MPI::TaskStatus::Exit);
411 ::MPI::COMM_WORLD.Send(
412 (void *)&taskStatus, 1, MPI_INT32_T,
413 0, to_int_type(MPI::MessageTag::Control));
414 status = MPI::TaskStatus::Exit;
415 break;
416 }
417 if (MPI::TermExit) {
418 MPI::logMessage(*log, "Termination Exit signal");
419 this->_processManager.broadcastSignal(SIGKILL);
420 taskStatus = to_int_type(MPI::TaskStatus::Exit);
421 ::MPI::COMM_WORLD.Send(
422 (void *)&taskStatus, 1, MPI_INT32_T,
423 0, to_int_type(MPI::MessageTag::Control));
424 status = MPI::TaskStatus::Exit;
425 break;
426 }
427
428 MPI::logMessage(*log, "Asking for work package");
429 taskStatus = to_int_type(MPI::TaskStatus::OK);
430 ::MPI::COMM_WORLD.Sendrecv(
431 (void *)&taskStatus, 1, MPI_INT32_T, 0,
432 to_int_type(MPI::MessageTag::Control), &taskCommand, 1,
433 MPI_INT32_T, 0, to_int_type(MPI::MessageTag::Control));
434
435 const BE::MPI::TaskCommand taskCommandE =
436 to_enum<TaskCommand>(taskCommand);
437 MPI::logMessage(*log, to_string(taskCommandE) + " command");
438 if (taskCommandE == MPI::TaskCommand::Ignore) {
439 continue;
440 }
441 if (taskCommandE == MPI::TaskCommand::Exit) {
442 break;
443 }
444 if (taskCommandE == MPI::TaskCommand::QuickExit) {
445 this->_processManager.broadcastSignal(SIGINT);
446 break;
447 }
448 if (taskCommandE == MPI::TaskCommand::TermExit) {
449 this->_processManager.broadcastSignal(SIGKILL);
450 break;
451 }
452 /*
453 * Receive three pieces of information:
454 * The raw data and length in the first message;
455 * The number of elements in the second message.
456 */
457 ::MPI::COMM_WORLD.Probe(0, to_int_type(MPI::MessageTag::Data),
458 MPIstatus);
459 uint64_t length = MPIstatus.Get_count(MPI_CHAR);
460 workPackageRaw.resize(length);
461 ::MPI::COMM_WORLD.Recv(
462 (void *)&workPackageRaw[0], length, MPI_CHAR, 0,
463 to_int_type(MPI::MessageTag::Data));
464 workPackageRaw.resize(length);
465
466 uint64_t numElements;
467 ::MPI::COMM_WORLD.Recv(
468 (void *)&numElements, 1, MPI_UINT64_T, 0,
469 to_int_type(MPI::MessageTag::Data));
470 try {
471 MPI::WorkPackage workPackage(workPackageRaw);
472 workPackage.setNumElements(numElements);
473 this->sendWorkPackage(workPackage);
474 } catch (MPI::TerminateJob &e) {
475 MPI::logMessage(*log,
476 "Package processor requested job termination " +
477 e.whatString());
478 taskStatus = to_int_type(
479 MPI::TaskStatus::RequestJobTermination);
480 ::MPI::COMM_WORLD.Send(
481 (void *)&taskStatus, 1, MPI_INT32_T, 0,
482 to_int_type(MPI::MessageTag::Control));
483 status = MPI::TaskStatus::RequestJobTermination;
484 } catch (Error::Exception &e) {
485 MPI::logMessage(*log,
486 "Failure to process work package: "
487 + e.whatString());
488 taskStatus = to_int_type(MPI::TaskStatus::Failed);
489 ::MPI::COMM_WORLD.Send(
490 (void *)&taskStatus, 1, MPI_INT32_T, 0,
491 to_int_type(MPI::MessageTag::Control));
492 status = MPI::TaskStatus::Failed;
493 break;
494 }
495 }
496 return (status);
497}

519void
520BiometricEvaluation::MPI::Receiver::start()
521{
522 /* Release other tasks to start up */
523 ::MPI::COMM_WORLD.Barrier();
524
525 BE::MPI::taskstat_t taskStatus;
526 try {
527 this->_logsheet =
528 BE::MPI::openLogsheet(
529 this->_resources->getLogsheetURL(),
530 "MPI::Receiver");
531 } catch (const Error::Exception&) {
532 taskStatus = to_int_type(MPI::TaskStatus::Failed);
533 ::MPI::COMM_WORLD.Send((void *)&taskStatus, 1, MPI_INT32_T,
534 0, to_int_type(MPI::MessageTag::Control));
535 this->shutdown(MPI::TaskStatus::Failed,
536 "Failed opening Logsheet()");
537 return;
538 }
539 BE::IO::Logsheet *log = this->_logsheet.get();
540 MPI::logMessage(*log, "Wait for startup message");
541 BE::MPI::taskstat_t flag;
542 ::MPI::COMM_WORLD.Recv(&flag, 1, MPI_INT32_T, 0,
543 to_int_type(MPI::MessageTag::Control));
544
545 /* Shutdown Task-N if Task-0 says not OK */
546 taskStatus = to_int_type(MPI::TaskStatus::OK);
547 if (flag == to_int_type(MPI::TaskStatus::Failed)) {
548 ::MPI::COMM_WORLD.Send((void *)&taskStatus, 1, MPI_INT32_T,
549 0, to_int_type(MPI::MessageTag::Control));
550 this->shutdown(MPI::TaskStatus::OK, "Distributor says abort");
551 return;
552 }
553
554 /*
555 * Call pre-fork initialization function in the work package
556 * processor, bailing out if that fails.
557 */
558 try {
559 this->_workPackageProcessor.get()->performInitialization(
560 this->_logsheet);
561 } catch (Error::Exception &e) {
562 MPI::logMessage(*log, "Could not initialize package processor: "
563 + e.whatString());
564 taskStatus = to_int_type(MPI::TaskStatus::Failed);
565 ::MPI::COMM_WORLD.Send((void *)&taskStatus, 1, MPI_INT32_T,
566 0, to_int_type(MPI::MessageTag::Control));
567 this->shutdown(MPI::TaskStatus::Failed,
568 "Failed performInitalization()");
569 return;
570 }
571 this->startWorkers();
572
573 //XXX Open log sheet
574 if (this->_processManager.getNumActiveWorkers() == 0) {
575 taskStatus = to_int_type(MPI::TaskStatus::Failed);
576 ::MPI::COMM_WORLD.Send((void *)&taskStatus, 1, MPI_INT32_T,
577 0, to_int_type(MPI::MessageTag::Control));
578 this->shutdown(MPI::TaskStatus::Failed, "No workers");
579 return;
580 }
581
582 ::MPI::COMM_WORLD.Send((void *)&taskStatus, 1, MPI_INT32_T,
583 0, to_int_type(MPI::MessageTag::Control));
584
585 MPI::TaskStatus status = this->requestWorkPackages();
586 std::string str;
587 if (status == MPI::TaskStatus::OK) {
588 str = "Normal end";
589 } else if (status == MPI::TaskStatus::Exit) {
590 str = "Early exit";
591 } else if (status == MPI::TaskStatus::RequestJobTermination) {
592 str = "Early exit (job termination requested)";
593 } else {
594 str = "Failed";
595 }
596 this->shutdown(status, str);
597}

599void
600BiometricEvaluation::MPI::Receiver::shutdown(
601 const MPI::TaskStatus &taskStatus,
602 const std::string &reason)
603{
604 BE::IO::Logsheet *log = this->_logsheet.get();
605 MPI::logMessage(*log, "Shutting down: " + reason);
606
607 /*
608 * Tell all workers to shut down.
609 */
610 uint32_t workerCount = this->_processManager.getNumActiveWorkers();
611
612 /*
613 * If TermExit occurred, the workers were forcibly killed
614 * so don't try to communicate with them.
615 */
616 //
617 // XXX Need to check whether TermExit command was received
618 // XXX because we then killed all the workers in the same
619 // XXX manner as when TermExit signal was received.
620 //
621 if ((MPI::TermExit == false) && (workerCount > 0)) {
622 MPI::logMessage(*log, "Stopping workers");
623 BE::Memory::uint8Array
624 inMessage(sizeof(BE::MPI::TaskStatus));
625 std::shared_ptr<Process::WorkerController> worker;
626
627 bool msgAvail;
628 for (uint32_t i = 0; i < workerCount; i++) {
629 try {
630 msgAvail = this->_processManager.getNextMessage(
631 worker, inMessage);
632 } catch (Error::Exception &e) {
633 MPI::logMessage(*log, "Task-N receiving message: "
634 "Caught: " + e.whatString());
635 /*
636 * It is debatable whether to break here as
637 * there may not be enough stop commands going
638 * out then, but if we don't break every call
639 * above will most likely fail anyway.
640 */
641 break;
642 }
643 if (!msgAvail)
644 break;
645 try {
646 this->_processManager.stopWorker(worker);
647 } catch (Error::Exception &e) {
648 MPI::logMessage(*log, "Task-N stopping worker: "
649 "Caught: " + e.whatString());
650 }
651 }
652 }
653 /*
654 * Call shutdown function in the work package processor. If that
655 * fails, continue with the shutdown.
656 */
657 try {
658 this->_workPackageProcessor.get()->performShutdown();
659 } catch (Error::Exception &e) {
660 MPI::logMessage(*log, "Could not shutdown package processor: "
661 + e.whatString());
662 }
663
664 /*
665 * We must synchronize here so these messages don't end up in
666 * the queue for a receive operation done when the Task-0 is
667 * still sending out data.
668 */
669 ::MPI::COMM_WORLD.Barrier();
670 MPI::logMessage(*log, "Sending final message");
671 const BE::MPI::taskstat_t rawTaskStatus = to_int_type(taskStatus);
672 ::MPI::COMM_WORLD.Send((void *)&rawTaskStatus, 1, MPI_INT32_T,
673 0, to_int_type(MPI::MessageTag::Control));
674}
