32/*
33 * There is no standard method to obtain process statistics from the OS.
34 * So we'll define static-local functions here for each OS that is supported.
35 * Unfortunately, these functions may be dependent not only on the OS, but
36 * a specfic version of the OS, but we'll try to avoid that.
37 * The alternative is to either link with an OS library that provides what
38 * we need, or import the source code; either way adds its own set of
39 * complications.
40 */

73static std::string
74internalGetProcName(pid_t pid)
75{
76#if defined Linux
77 std::ostringstream tp;
78 tp << "/proc/" << pid << "/cmdline";
79 if (!BE::IO::Utility::fileExists(tp.str()))
80 throw BE::Error::StrategyError(
81 "Could not find " + tp.str() + ".");
82
83 /* Read the process cmdline into a string so we can parse it. */
84 std::ifstream ifs(tp.str().c_str());
85 if (ifs.fail())
86 throw BE::Error::StrategyError(
87 "Could not open " + tp.str() + ".");
88
89 std::string line;
90 ifs >> line;
91 ifs.close();
92
93 /*
94 * /proc/<pid>cmdline represents the command line used to execute the
95 * program, with the arguments separated by the nul character.
96 * Therefore, the cmdline from the start until the first nul is the
97 * command name and we'll call that the process name.
98 */
99 std::size_t loc = line.find_first_of('\0');
100 if (loc != std::string::npos)
101 line.erase(loc, loc);
102
103 return (BE::Text::basename(line));
104#else
105 throw BE::Error::NotImplemented();
106#endif
107}

109static PSTATS
110internalGetPstats(pid_t pid)
111#if defined Linux
112{
113 std::ostringstream tp;
114 tp << "/proc/" << pid << "/status";
115 if (!BE::IO::Utility::fileExists(tp.str()))
116 throw BE::Error::StrategyError(
117 "Could not find " + tp.str() + ".");
118
119 /* Read the process status into a string so we can parse it. */
120 std::ifstream ifs(tp.str().c_str());
121 if (ifs.fail())
122 throw BE::Error::StrategyError(
123 "Could not open " + tp.str() + ".");
124
125 /*
126 * The status info for a process is composed on n lines in this form:
127 * key: value <units>
128 * so, for example:
129 * VmSize: 2164 kB
130 */
131 std::string oneline, key, value;
132 std::string::size_type idx;
133 PSTATS stats;
134 while (!ifs.eof()) {
135 std::getline(ifs, oneline);
136 idx = oneline.find(":");
137 value = oneline.substr(idx + 1, oneline.length());
138 key = BE::Text::trimWhitespace(oneline.substr(0, idx));
139
140 //XXX May want to remove non-digits in value string
141
142 /*
143 * The call to strtoll() works for all the stat values here
144 * because the entry in /proc/<pid>/status has the number
145 * separated from the units by a whitespace.
146 */
147 if (key == VmRSSProp) {
148 stats.vmrss = (uint64_t)std::strtoll(value.c_str(),
149 nullptr, 10);
150 continue;
151 }
152 if (key == VmSizeProp) {
153 stats.vmsize = (uint64_t)std::strtoll(value.c_str(),
154 nullptr, 10);
155 continue;
156 }
157 if (key == VmPeakProp) {
158 stats.vmpeak = (uint64_t)std::strtoll(value.c_str(),
159 nullptr, 10);
160 continue;
161 }
162 if (key == VmDataProp) {
163 stats.vmdata = (uint64_t)std::strtoll(value.c_str(),
164 nullptr, 10);
165 continue;
166 }
167 if (key == VmStackProp) {
168 stats.vmstack = (uint64_t)std::strtoll(value.c_str(),
169 nullptr, 10);
170 continue;
171 }
172 if (key == ThreadsProp) {
173 stats.threads = (uint64_t)std::strtoll(value.c_str(),
174 nullptr, 10);
175 continue;
176 }
177 }
178 ifs.close();
179 return (stats);
180}

201static void internalGetCPUTimes(
202 uint64_t *usertime,
203 uint64_t *systemtime)
204{
205 struct rusage ru;
206 int ret = getrusage(RUSAGE_SELF, &ru);
207 if (ret != 0)
208 throw BE::Error::StrategyError("OS call failed: " +
209 BE::Error::errorStr());
210
211 *usertime = (uint64_t)(ru.ru_utime.tv_sec *
212 BE::Time::MicrosecondsPerSecond + ru.ru_utime.tv_usec);
213 *systemtime = (uint64_t)(ru.ru_stime.tv_sec *
214 BE::Time::MicrosecondsPerSecond + ru.ru_stime.tv_usec);
215}

301void
302BiometricEvaluation::Process::Statistics::logStats()
303{
304 if (!_logging)
305 throw BE::Error::ObjectDoesNotExist();
306
307 PSTATS ps;
308 uint64_t usertime, systemtime;
309 pthread_mutex_lock(&this->_logMutex);
310 try {
311 ps = internalGetPstats(_pid);
312 internalGetCPUTimes(&usertime, &systemtime);
313 } catch (BE::Error::Exception &e) {
314 pthread_mutex_unlock(&this->_logMutex);
315 throw;
316 }
317 *_logSheet << usertime << " " << systemtime << " ";
318 *_logSheet << ps.vmrss << " " << ps.vmsize << " " << ps.vmpeak << " ";
319 *_logSheet << ps.vmdata << " " << ps.vmstack << " " << ps.threads;
320 _logSheet->newEntry();
321
322 pthread_mutex_unlock(&this->_logMutex);
323}

336 pthread_cond_t logCond;
337};

339extern "C" void *
340autoLogger(void *ptr)
341{
342 int type;
343
344 /*
345 * We need some control over when this thread will be cancelled,
346 * so defer cancellation, but we'll test for the cancel event and
347 * give up control.
348 */
349 pthread_setcanceltype(PTHREAD_CANCEL_DEFERRED, &type);
350 pthread_setcancelstate(PTHREAD_CANCEL_ENABLE, &type);
351
352 /*
353 * We need to copy data out of the logging package in a manner
354 * that is synchronized with the owner of the package.
355 */
356 struct loggerPackage *lp = (struct loggerPackage *)(ptr);
357 pthread_mutex_lock(&lp->logMutex);
358 BE::Process::Statistics *stat = lp->stat;
359 /*
360 * Convert _interval to sec/nsec from usec
361 */
362 time_t sec = (time_t)(lp->interval / BE::Time::MicrosecondsPerSecond);
363 long nsec = (long)((lp->interval % BE::Time::MicrosecondsPerSecond) * 1000);
364 lp->flag = 1;
365 pthread_cond_signal(&lp->logCond);
366 pthread_mutex_unlock(&lp->logMutex);
367
368 struct timespec req, rem;
369
370 while (true) {
371
372 /*
373 * Test for a cancel request. Note that we could create one
374 * more log entry AFTER a request comes in.
375 */
376 pthread_testcancel();
377
378 /* We want the logging operation to complete, so disable
379 * cancellation while that is going on.
380 */
381 pthread_setcancelstate(PTHREAD_CANCEL_DISABLE, &type);
382 stat->callStatistics_logStats();
383 pthread_setcancelstate(PTHREAD_CANCEL_ENABLE, &type);
384 req.tv_sec = sec;
385 req.tv_nsec = nsec;
386
387 /* We use nanosleep(2) to avoid causing signals sometimes
388 * used by sleep(3).
389 */
390 int retval = nanosleep(&req, &rem);
391
392 /* If a signal occurs, there will be remaining time on
393 * the sleep interval, so use it up.
394 */
395 if (retval == -1) {
396 while (rem.tv_sec > 0) {
397 req = rem;
398 nanosleep(&req, &rem);
399 }
400 }
401 }
402
403 return (nullptr);
404}

406void
407BiometricEvaluation::Process::Statistics::startAutoLogging(
408 uint64_t interval)
409{
410 if (!_logging)
411 throw BE::Error::ObjectDoesNotExist();
412 if (_autoLogging)
413 throw BE::Error::ObjectExists();
414 if (interval == 0)
415 return;
416
417 struct loggerPackage *lp;
418 lp = (struct loggerPackage *)malloc(sizeof(struct loggerPackage));
419 if (lp == nullptr)
420 throw BE::Error::StrategyError("Memory allocation error");
421
422 lp->interval = interval;
423 lp->stat = this;
424 lp->flag = 0;
425 pthread_mutex_init(&lp->logMutex, nullptr);
426 pthread_cond_init(&lp->logCond, nullptr);
427
428 int retval = pthread_create(&_loggingThread, nullptr, autoLogger,
429 (void *)lp);
430 if (retval != 0) {
431 free (lp);
432 throw BE::Error::StrategyError("Creating thread failed: " +
433 BE::Error::errorStr());
434 }
435
436 std::ostringstream comment;
437 comment << StartAutologComment << lp->interval << " microseconds.";
438 _logSheet->writeComment(comment.str());
439
440 /*
441 * Synchronize with the logging thread so it can copy the info
442 * out of the logging package before it is freed.
443 */
444 pthread_mutex_lock(&lp->logMutex);
445 while (lp->flag != 1) {
446 pthread_cond_wait(&lp->logCond, &lp->logMutex);
447 }
448 pthread_mutex_unlock(&lp->logMutex);
449 pthread_cond_destroy(&lp->logCond);
450 pthread_mutex_destroy(&lp->logMutex);
451 free(lp);
452 _autoLogging = true;
453}

455void
456BiometricEvaluation::Process::Statistics::stopAutoLogging()
457{
458 if (!_autoLogging)
459 throw BE::Error::ObjectDoesNotExist();
460 _autoLogging = false;
461 int retval = pthread_cancel(_loggingThread);
462 if (retval != 0)
463 throw BE::Error::StrategyError(
464 "Cancel of logging thread failed: " +
465 BE::Error::errorStr());
466
467 /* Wait for the logging thread to exit */
468 pthread_join(_loggingThread, nullptr);
469 std::ostringstream comment;
470 comment << StopAutologComment;
471 _logSheet->writeComment(comment.str());
472}
