40std::string
41BiometricEvaluation::Time::getCurrentCalendarInformation(
42 const std::string &formatString)
43{
44 auto theTime = std::chrono::system_clock::to_time_t(
45 std::chrono::system_clock::now());
46
47 /* "For portable code tzset() should be called before localtime_r()." */
48 ::tzset();
49 std::tm now{};
50 ::localtime_r(&theTime, &now);
51
52 return (Time::put_time(&now, formatString.c_str()));
53}

56BiometricEvaluation::Time::put_time(

59{
60 Memory::AutoArray<char> buffer;
61 size_t size;
62
63 do {
64 buffer.resize(buffer.size() + 255);
65 size = ::strftime(buffer, buffer.size(), fmt, tmb);
66 } while (size == 0);
67 buffer.resize(size + 1);
68
69 return (to_string(buffer));
