219float
220BiometricEvaluation::Image::distance(
221 const Image::Coordinate &p1,
222 const Image::Coordinate &p2)
223{
224 const uint32_t xDif = p2.x >= p1.x ? p2.x - p1.x : p1.x - p2.x;
225 const uint32_t yDif = p2.y >= p1.y ? p2.y - p1.y : p1.y - p2.y;
226
227 return (sqrtf(powf(xDif, 2.0) + powf(yDif, 2.0)));
228}

231BiometricEvaluation::Image::Resolution
232BiometricEvaluation::Image::Resolution::toUnits(
233 const BiometricEvaluation::Image::Resolution::Units &units)
234 const
235{
236 switch (units) {
237 case Image::Resolution::Units::NA:
238 throw BE::Error::StrategyError("Cannot convert to an "
239 "unknown unit");
240 case Image::Resolution::Units::PPI:
241 switch (this->units) {
242 case Image::Resolution::Units::NA:
243 throw BE::Error::StrategyError("Can't convert because "
244 "original units are not known");
245 case Image::Resolution::Units::PPI:
246 return {this->xRes, this->yRes, units};
247 case Image::Resolution::Units::PPCM:
248 return {this->xRes * CentimetersPerInch,
249 this->yRes * CentimetersPerInch, units};
250 case Image::Resolution::Units::PPMM:
251 return {this->xRes * MillimetersPerInch,
252 this->yRes * MillimetersPerInch, units};
253 }
254 case Image::Resolution::Units::PPCM:
255 switch (this->units) {
256 case Image::Resolution::Units::NA:
257 throw BE::Error::StrategyError("Can't convert because "
258 "original units are not known");
259 case Image::Resolution::Units::PPCM:
260 return {this->xRes, this->yRes, units};
261 case Image::Resolution::Units::PPMM:
262 return {this->xRes * 10, this->yRes * 10, units};
263 case Image::Resolution::Units::PPI:
264 return {this->xRes / CentimetersPerInch,
265 this->yRes / CentimetersPerInch, units};
266 }
267 case Image::Resolution::Units::PPMM:
268 switch (this->units) {
269 case Image::Resolution::Units::NA:
270 throw BE::Error::StrategyError("Can't convert because "
271 "original units are not known");
272 case Image::Resolution::Units::PPMM:
273 return {this->xRes, this->yRes, units};
274 case Image::Resolution::Units::PPCM:
275 return {this->xRes / 10, this->yRes / 10, units};
276 case Image::Resolution::Units::PPI:
277 return {this->xRes / MillimetersPerInch,
278 this->yRes / MillimetersPerInch, units};
279 }
280 }
281
282 throw BE::Error::StrategyError("Can't convert because original units "
283 "are not known");
284}

286BiometricEvaluation::Memory::uint8Array
287BiometricEvaluation::Image::removeComponents(
288 const BiometricEvaluation::Memory::uint8Array &rawData,
289 const uint8_t bitDepth,
290 const std::vector<bool> &components)
291{
292 /* Useful derived definitions */
293 uint8_t numComponentsToRemove{0};
294 std::for_each(components.begin(), components.end(), [&](const bool &c) {
295 if (c) { ++numComponentsToRemove; }
296 });
297 const uint8_t numComponents = components.size();
298 const uint8_t pixelStride = ((numComponents * bitDepth) / 8);
299 const uint8_t componentStride = (bitDepth / 8);
300
301 /* If we aren't removing anything */
302 if (numComponentsToRemove == 0)
303 return {rawData};
304
305 /* If we're removing everything */
306 if (numComponentsToRemove == components.size())
307 return (BE::Memory::uint8Array());
308
309 /* Only support 8-bit and 16-bit images */
310 if ((bitDepth != 8) && (bitDepth != 16))
311 throw BE::Error::ParameterError("Unsupported bit depth (" +
312 std::to_string(bitDepth) + ")");
313
314 /* Check that raw data appears to be large enough */
315 if ((rawData.size() % pixelStride) != 0)
316 throw BE::Error::StrategyError("Raw data is sized incorrectly "
317 "for " + std::to_string(numComponents) + ' ' +
318 std::to_string(bitDepth) + "-bit components");
319
320 BE::Memory::uint8Array out((rawData.size() / pixelStride) *
321 (numComponents - numComponentsToRemove) * componentStride);
322 BE::Memory::MutableIndexedBuffer outBuf(out);
323
324 /* Naively loop over image, skipping over specified components */
325 for (uint64_t px = 0; px < rawData.size(); px += pixelStride) {
326 for (uint8_t comp = 0; comp < numComponents; ++comp) {
327 if (!components[comp]) {
328 switch (bitDepth) {
329 case 8:
330 outBuf.pushU8Val(rawData[px +
331 (comp * componentStride)]);
332 break;
333 case 16:
334 outBuf.pushBeU16Val(rawData[px +
335 (comp * componentStride)]);
336 break;
337 }
338 }
339 }
340 }
341
342 return (out);
343}
