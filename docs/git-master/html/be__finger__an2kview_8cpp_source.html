93BiometricEvaluation::Finger::FingerImageCode
94BiometricEvaluation::Finger::AN2KView::convertFingerImageCode(
95 const char *str)
96{
97 if (strncmp(str, "EJI", 3) == 0)
98 return (BE::Finger::FingerImageCode::EJI);
99 else if (strncmp(str, "TIP", 3) == 0)
100 return (BE::Finger::FingerImageCode::RolledTip);
101 else if (strncmp(str, "FV1", 3) == 0)
102 return (BE::Finger::FingerImageCode::FullFingerRolled);
103 else if (strncmp(str, "FV2", 3) == 0)
104 return (BE::Finger::FingerImageCode::FullFingerPlainLeft);
105 else if (strncmp(str, "FV3", 3) == 0)
106 return (BE::Finger::FingerImageCode::FullFingerPlainCenter);
107 else if (strncmp(str, "FV4", 3) == 0)
108 return (BE::Finger::FingerImageCode::FullFingerPlainRight);
109 else if (strncmp(str, "PRX", 3) == 0)
110 return (BE::Finger::FingerImageCode::ProximalSegment);
111 else if (strncmp(str, "DST", 3) == 0)
112 return (BE::Finger::FingerImageCode::DistalSegment);
113 else if (strncmp(str, "MED", 3) == 0)
114 return (BE::Finger::FingerImageCode::MedialSegment);
115 else if (strncmp(str, "NA", 2) == 0)
116 return (BE::Finger::FingerImageCode::NA);
117 else
118 throw BE::Error::DataError("Invalid finger image code value");
119}

140void
141BiometricEvaluation::Finger::AN2KView::readImageRecord(
142 const RecordType typeID,
143 const uint32_t recordNumber)
144{
145 switch (typeID) {
146 case RecordType::Type_3:
147 case RecordType::Type_4:
148 case RecordType::Type_5:
149 case RecordType::Type_6:
150 break;
151 default:
152 throw Error::ParameterError("Invalid Record Type ID");
153 }
154 RECORD *record = AN2KView::getAN2KRecord();
155
156 FIELD *field;
157 int idx;
158
159 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, IMP_ID, record) != TRUE)
160 throw Error::DataError("Field IMP not found");
161 _imp = convertImpression(field->subfields[0]->items[0]->value);
162
163 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, FGP_ID, record) != TRUE)
164 throw Error::DataError("Field FGP not found");
165 _positions = populateFGP(field);
166}
