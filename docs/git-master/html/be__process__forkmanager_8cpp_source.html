112void
113BiometricEvaluation::Process::ForkManager::startWorkers(
114 bool wait,
115 bool communicate)
116{
117 /* Ensure all Workers have finished their previous assignments */
118 if (this->getNumActiveWorkers() != 0)
119 throw Error::ObjectExists();
120 this->reset();
121
122 for (uint32_t i = 0; i < getTotalWorkers(); i++) {
123 std::shared_ptr<ForkWorkerController> fwc =
124 std::static_pointer_cast<ForkWorkerController>(_workers[i]);
125 fwc->start(communicate);
126 _wcStatus[fwc].pid = fwc->getPID();
127 _wcStatus[fwc].isWorking = true;
128 }
129
130 /* In the child case, start() will eventually exit the child */
131 _parent = true;
132
133 /* Optionally wait for all processes to exit. */
134 if (wait)
135 _wait();
136 /*
137 * Otherwise, setup a SIGCHLD handler to reap children on exit,
138 * assuming the parent will at least stick around to perform some
139 * sort of control.
140 */
141 else {
142 struct sigaction reapSignal;
143 memset(&reapSignal, 0, sizeof(reapSignal));
144 reapSignal.sa_handler = ForkManager::reap;
145 sigaction(SIGCHLD, &reapSignal, nullptr);
146 }
147}

189void
190BiometricEvaluation::Process::ForkWorkerController::start(
191 bool communicate)
192{
193 if (this->isWorking())
194 throw Error::ObjectExists();
195 this->reset();
196
197 if (communicate)
198 getWorker()->_initCommunication();
199 int32_t pid = fork();
200
201 switch (pid) {
202 case 0: { /* Child */
203 /* Update self references */
204 _pid = getpid();
205 BiometricEvaluation::Process::ForkManager::FORKMANAGERS.erase(
206 BiometricEvaluation::Process::ForkManager::
207 FORKMANAGERS.begin(),
208 BiometricEvaluation::Process::ForkManager::
209 FORKMANAGERS.end());
210
211 /* Copy to a static var only for this process's instance */
212 _staticWorker = getWorker();
213 if (communicate)
214 _staticWorker->closeManagerPipeEnds();
215
216 /* Catch SIGUSR1 to quit child on demand */
217 struct sigaction stopSignal;
218 memset(&stopSignal, 0, sizeof(stopSignal));
219 stopSignal.sa_handler = ForkWorkerController::_stop;
220 sigaction(SIGUSR1, &stopSignal, nullptr);
221
222 /* Run workerMain() -- required method */
223 int32_t rv;
224 try {
225 rv = getWorker()->workerMain();
226 } catch (...) {
227 rv = EXIT_FAILURE;
228 }
229 std::exit(rv);
230
231 /* Not reached */
232 break;
233 } case -1: /* Error */
234 throw Error::StrategyError("Error during fork(): " +
235 Error::errorStr());
236 break;
237 default: /* Parent */
238 _pid = pid;
239 if (communicate)
240 getWorker()->closeWorkerPipeEnds();
241 break;
242 }
243}

303void
304BiometricEvaluation::Process::ForkManager::_wait()
305{
306 /* No need for children to wait */
307 if (_parent == false)
308 return;
309
310 bool stop = false;
311 pid_t process = -1;
312 int status;
313 while (getNumActiveWorkers() > 0) {
314 stop = false;
315 while (!stop) {
316 process = ::wait(&status);
317 switch (process) {
318 case -1:
319 switch (errno) {
320 case ECHILD: /* No child processes */
321 if (_exitCallback != NULL) {
322 _exitCallback(
323 std::shared_ptr<
324 ForkWorkerController>(),
325 0);
326 }
327 stop = true;
328 break;
329 case EINTR: /* Interrupted */
330 /*
331 * Try to reap again.
332 *
333 * If SIGCHLD had the SA_RESTART flag
334 * set, this should set errno to ECHILD
335 * on the next iteration.
336 */
337 continue;
338 default:
339 throw Error::StrategyError(
340 Error::errorStr());
341 }
342 break;
343 case 0: /* Child exists but hasn't changed state */
344 break;
345 default:
346 try {
347 this->setNotWorking(process);
348 } catch (const Error::ObjectDoesNotExist&) {
349 throw Error::StrategyError(
350 Error::errorStr());
351 }
352
353 this->setExitStatus(process, status);
354
355 if (_exitCallback != NULL) {
356 _exitCallback(
357 getProcessWithPID(process), status);
358 }
359 stop = true;
360 break;
361 }
362 }
363 }

374void
375BiometricEvaluation::Process::ForkManager::reap(
376 int signal)
377{
378 /* This handler is for SIGCHLD only */
379 if (signal != SIGCHLD)
380 return;
381
382 int32_t status;
383 bool stop = false;
384 pid_t pid;
385
386 /* Try to reap until there are no more available processes to reap */
387 while (!stop) {
388 /* Reap the first available child without waiting */
389 pid = ::waitpid(-1, &status, WNOHANG);
390
391 switch (pid) {
392 case 0: /* No child ready to be reaped */
393 stop = true;
394 break;
395 case -1: /* Error */
396 switch (errno) {
397 case ECHILD: /* No child processes */
398 stop = true;
399 break;
400 case EINTR: /* Interrupted */
401 /*
402 * Try to reap again.
403 *
404 * If SIGCHLD had the SA_RESTART flag
405 * set, this should set errno to ECHILD
406 * on the next iteration.
407 */
408 break;
409 default:
410 throw Error::StrategyError(
411 Error::errorStr());
412 }
413 break;
414 default: /* Reap successful */
415 /* Update the Status list */
416 for (auto &it : FORKMANAGERS) {
417 if (it->responsibleFor(pid)) {
418 it->setExitStatus(pid, status);
419 it->setNotWorking(pid);
420 }
421 }
422 break;
423 }
424 }

436void
437BiometricEvaluation::Process::ForkManager::defaultExitCallback(
438 std::shared_ptr<ForkWorkerController> child,
439 int status)
440{
441 if (child == nullptr) {
442 std::cerr << "Unknown child exited with unknown status." <<
443 std::endl;
444 return;
445 }
446
447 std::cerr << "PID " << child->getPID() << ": ";
448 if (WIFEXITED(status))
449 std::cerr << "Exited with status " << WEXITSTATUS(status);
450 else if (WIFSIGNALED(status)) {
451 std::cerr << "Exited due to signal " << WTERMSIG(status) <<
452 " (" << strsignal(WTERMSIG(status)) << ')';
453 if ((WCOREDUMP(status)))
454 std::cerr << " and core dumped";
455 } else
456 std::cerr << "Exited with unknown status";
457 std::cerr << '.' << std::endl;
