85static void writeToLogsheet(
86 BE::IO::Logsheet &logsheet,
87 std::string message)
88{
89 static bool displayedMessage = false;
90
91 /*
92 * If logging should fail at some point, errors are
93 * not propagated, but a message is displayed.
94 */
95 try {
96 logsheet.writeDebug(message);
97 } catch (BE::Error::Exception &e) {
98 if (!displayedMessage) {
99 BE::MPI::printStatus("Log failure: " + e.whatString()
100 + "; logging disabled");
101 displayedMessage = true;
102 logsheet.setCommit(false);
103 logsheet.setDebugCommit(false);
104 logsheet.setCommentCommit(false);
105 }
106 }
107}

125std::shared_ptr<BiometricEvaluation::IO::Logsheet>
126BiometricEvaluation::MPI::openLogsheet(
127 const std::string &url,
128 const std::string &description)
129{
130 std::shared_ptr<BE::IO::Logsheet> logsheet;
131 if (url == "") {
132 logsheet.reset(new BE::IO::Logsheet());
133 return (logsheet);
134 }
135 BE::IO::Logsheet::Kind lsKind = BE::IO::Logsheet::getTypeFromURL(url);
136 switch (lsKind) {
137 case BE::IO::Logsheet::Kind::File: {
138 std::string locURL = url + "-"
139 + BE::MPI::generateUniqueID();
140 try {
141 logsheet.reset(new BE::IO::FileLogsheet(
142 locURL,
143 description));
144 } catch (const BE::Error::ObjectExists&) {
145 try {
146 logsheet.reset(
147 new BE::IO::FileLogsheet(locURL));
148 } catch (BE::Error::Exception &e) {
149 throw BE::Error::Exception(
150 "Could not open FileLogsheet: "
151 + e.whatString());
152 }
153 }
154 break;
155 }
156 case BE::IO::Logsheet::Kind::Syslog: {
157 /* Use the MPI rank as the application name */
158 try {
159 logsheet.reset(new BE::IO::SysLogsheet(
160 url,
161 description,
162 std::to_string(::MPI::COMM_WORLD.Get_rank()),
163 true, true));
164 } catch (BE::Error::Exception &e) {
165 throw BE::Error::Exception(
166 "Could not open SysLogsheet: "
167 + e.whatString());
168 }
169 break;
170 }
171 default:
172 throw (
173 BE::Error::ParameterError("Invalid Logsheet URL"));
174 }
175 return (logsheet);
176}
