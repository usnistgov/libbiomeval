47void
48BiometricEvaluation::Iris::ISO2011View::readISOHeader(
49 Memory::IndexedBuffer &buf)
50{
51 uint32_t uval32;
52
53 uval32 = buf.scanBeU32Val(); /* Format ID */
54 if (uval32 != BE::Iris::INCITSView::BASE_FORMAT_ID)
55 throw (BE::Error::DataError("Invalid Format ID in data"));
56
57 uval32 = buf.scanBeU32Val(); /* Spec Version */
58 if (uval32 != BE::Iris::ISO2011View::BASE_SPEC_VERSION)
59 throw (Error::DataError("Invalid Spec Version in data"));
60
61 BE::Iris::INCITSView::readHeader(
62 buf, Iris::INCITSView::ISO2011_STANDARD);
63}
