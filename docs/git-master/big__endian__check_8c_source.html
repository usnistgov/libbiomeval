44/*
45 * Check endianess of the system, printing 1 when big-endian, 0 otherwise.
46 */
47int main()
48{
49 /*
50 * On a little-endian machine, x is 0x01000000, on big-endian
51 * it is 0x00000001.
52 */
53 int x = 1;
54 char *y = (char*)&x; /* points to first octet of x */
55 if (*y == 0)
56 printf("1");
57 else
58 printf("0");
59 return (0);
60}
