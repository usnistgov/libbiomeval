35void
36BiometricEvaluation::Process::POSIXThreadManager::startWorkers(
37 bool wait,
38 bool communicate)
39{
40 /* Ensure all Workers have finished their previous assignments */
41 if (this->getNumActiveWorkers() != 0)
42 throw Error::ObjectExists();
43 this->reset();
44
45 std::vector<std::shared_ptr<WorkerController>>::const_iterator it;
46 for (it = _workers.begin(); it != _workers.end(); it++)
47 std::static_pointer_cast<POSIXThreadWorkerController>(*it)->
48 start(communicate);
49
50 if (wait)
51 _wait();
52}

136void *
137BiometricEvaluation::Process::POSIXThreadWorkerController::workerMainWrapper(
138 void *_this)
139{
140 ((POSIXThreadWorkerController *)_this)->_hasWorked = true;
141 ((POSIXThreadWorkerController *)_this)->_working = true;
142 ((POSIXThreadWorkerController *)_this)->_rvSet = false;
143 try {
144 ((POSIXThreadWorkerController *)_this)->_rv =
145 ((POSIXThreadWorkerController *)_this)->getWorker()->
146 workerMain();
147 } catch (...) {
148 ((POSIXThreadWorkerController *)_this)->_rv = EXIT_FAILURE;
149 }
150 ((POSIXThreadWorkerController *)_this)->_rvSet = true;
151 ((POSIXThreadWorkerController *)_this)->_working = false;
152
