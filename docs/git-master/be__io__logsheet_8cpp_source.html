26BiometricEvaluation::IO::Logsheet::Kind
27BiometricEvaluation::IO::Logsheet::getTypeFromURL(
28 const std::string &url)
29{
30 /*
31 * Search for the URL scheme, then compare in a
32 * case-insensitive manner to known schemes,
33 * removing whitespace before-hand.
34 */
35 std::string::size_type start = url.find("://");
36 if (start == std::string::npos)
37 throw (Error::ParameterError("Missing URL scheme"));
38
39 std::string prefix{BE::Text::trimWhitespace(url.substr(0, start))};
40 if (BE::Text::caseInsensitiveCompare(
41 prefix, BE::IO::Logsheet::SYSLOGURLSCHEME) == true) {
42 return (BE::IO::Logsheet::Kind::Syslog);
43 }
44 if (BE::Text::caseInsensitiveCompare(
45 prefix, BE::IO::Logsheet::FILEURLSCHEME) == true) {
46 return (BE::IO::Logsheet::Kind::File);
47 }
48 throw (Error::ParameterError("Invalid URL scheme"));
49}

75std::string
76BiometricEvaluation::IO::Logsheet::trim(
77 const std::string &entry)
78{
79 /* Length of a Logsheet entry prefix */
80 static const uint8_t entryDelimiterLength = 12;
81 /* Length of a Logsheet prefix */
82 static const uint8_t commentDelimiterLength = 2;
83
84 if (entry.length() <= 0)
85 return (entry);
86
87 switch (entry[0]) {
88 case CommentDelimiter:
89 return (entry.substr(commentDelimiterLength));
90 case EntryDelimiter:
91 return (entry.substr(entryDelimiterLength));
92 default:
93 return (entry);
94 }
95}
