63uint64_t
64BiometricEvaluation::IO::FileRecordStore::Impl::getSpaceUsed()
65 const
66{
67 this->sync();
68
69 DIR *dir;
70 dir = opendir(this->_theFilesDir.c_str());
71 if (dir == nullptr)
72 throw Error::StrategyError("Cannot open store directory");
73
74 uint64_t total = RecordStore::Impl::getSpaceUsed();
75 struct dirent *entry;
76 struct stat sb;
77 std::string cname;
78 while ((entry = readdir(dir)) != nullptr) {
79#ifndef _WIN32
80 if (entry->d_ino == 0)
81 continue;
82#endif
83 cname = entry->d_name;
84 cname = FileRecordStore::Impl::canonicalName(cname);
85 if (stat(cname.c_str(), &sb) != 0) {
86 const std::string errorStr{"Cannot stat store file (" +
87 Error::errorStr() + ")"};
88
89 if (dir != nullptr) {
90 if (closedir(dir)) {
91 throw Error::StrategyError("Could not "
92 "close " + this->_theFilesDir + " "
93 "(" + Error::errorStr() + ") "
94 "while exiting with error " +
95 errorStr);
96 }
97 }
98
99 throw Error::StrategyError{errorStr};
100 }
101 if ((S_IFMT & sb.st_mode) == S_IFDIR) /* skip '.' and '..' */
102 continue;
103 total += sb.st_size;
104 }
105
106 if (dir != nullptr) {
107 if (closedir(dir)) {
108 throw Error::StrategyError("Could not close " +
109 this->_theFilesDir + "(" + Error::errorStr() + ")");
110 }
111 }
112
113 return (total);
114}

158BiometricEvaluation::Memory::uint8Array
159BiometricEvaluation::IO::FileRecordStore::Impl::read(
160 const std::string &key)
161 const
162{
163 if (!validateKeyString(key))
164 throw Error::StrategyError("Invalid key format");
165 std::string pathname = FileRecordStore::Impl::canonicalName(key);
166 if (!IO::Utility::fileExists(pathname))
167 throw Error::ObjectDoesNotExist();
168
169 /* Allow exceptions to propagate out of here */
170 uint64_t size = IO::Utility::getFileSize(pathname);
171 std::FILE *fp = std::fopen(pathname.c_str(), "rb");
172 if (fp == nullptr)
173 throw Error::StrategyError("Could not open " + pathname +
174 " (" + Error::errorStr() + ")");
175
176 Memory::uint8Array data(size);
177 std::size_t sz = fread(data, 1, size, fp);
178 std::fclose(fp);
179 if (sz != size)
180 throw Error::StrategyError("Could not write " + pathname +
181 " (" + Error::errorStr() + ")");
182 return(data);
183}

241BiometricEvaluation::IO::RecordStore::Record
242BiometricEvaluation::IO::FileRecordStore::Impl::i_sequence(
243 bool returnData,
244 int cursor)
245{
246 if ((cursor != BE_RECSTORE_SEQ_START) &&
247 (cursor != BE_RECSTORE_SEQ_NEXT))
248 throw Error::StrategyError("Invalid cursor position as "
249 "argument");
250
251 DIR *dir;
252 dir = opendir(_theFilesDir.c_str());
253 if (dir == nullptr)
254 throw Error::StrategyError("Cannot open store directory");
255
256 /* If the current cursor position is START, then it doesn't matter
257 * what the client requests; we start at the first record.
258 */
259 if ((getCursor() == BE_RECSTORE_SEQ_START) ||
260 (cursor == BE_RECSTORE_SEQ_START))
261 _cursorPos = 1;
262
263 if (_cursorPos > getCount()) { /* Client needs to start over */
264 const std::string errorStr{"No record at position"};
265 if (dir != nullptr) {
266 if (closedir(dir)) {
267 throw Error::StrategyError("Could not close " +
268 this->_theFilesDir + "(" +
269 Error::errorStr() + ") while exiting with "
270 "error " + errorStr);
271 }
272 }
273
274 throw Error::ObjectDoesNotExist(errorStr);
275 }
276
277 struct dirent *entry;
278 struct stat sb;
279 uint64_t i = 1;
280 std::string cname;
281 while ((entry = readdir(dir)) != nullptr) {
282#ifndef _WIN32
283 if (entry->d_ino == 0)
284 continue;
285#endif
286 cname = _theFilesDir + "/" + entry->d_name;
287 if (stat(cname.c_str(), &sb) != 0) {
288 const std::string errorStr{"Cannot stat store file (" +
289 Error::errorStr() + ")"};
290 if (dir != nullptr) {
291 if (closedir(dir)) {
292 throw Error::StrategyError("Could not "
293 "close " + this->_theFilesDir + " "
294 "(" + Error::errorStr() + ") "
295 "while exiting with error " +
296 errorStr);
297 }
298 }
299
300 throw BE::Error::StrategyError{errorStr};
301 }
302 if ((S_IFMT & sb.st_mode) == S_IFDIR) /* skip '.' and '..' */
303 continue;
304 if (i == _cursorPos)
305 break;
306 i++;
307 }
308 /* Sanity check */
309 if (i > _cursorPos) {
310 const std::string errorStr{"Record cursor position out of "
311 "sync"};
312 if (dir != nullptr) {
313 if (closedir(dir)) {
314 throw Error::StrategyError("Could not close " +
315 this->_theFilesDir + " (" +
316 Error::errorStr() + ") while exiting with "
317 "error " + errorStr);
318 }
319 }
320
321 throw Error::StrategyError{errorStr};
322 }
323
324 BE::IO::RecordStore::Record record;
325 record.key = entry->d_name;
326 setCursor(BE_RECSTORE_SEQ_NEXT);
327 _cursorPos = i + 1;
328
329 if (dir != nullptr) {
330 if (closedir(dir)) {
331 throw Error::StrategyError("Could not close " +
332 _theFilesDir + " (" + Error::errorStr() +
333 ")");
334 }
335 }
336
337 if (returnData)
338 record.data = FileRecordStore::Impl::read(record.key);
339 return (record);
340}

357void
358BiometricEvaluation::IO::FileRecordStore::Impl::setCursorAtKey(
359 const std::string &key)
360{
361 if (!validateKeyString(key))
362 throw Error::StrategyError("Invalid key format");
363
364 DIR *dir;
365 dir = opendir(_theFilesDir.c_str());
366 if (dir == nullptr)
367 throw Error::StrategyError("Cannot open store directory");
368
369 struct dirent *entry;
370 struct stat sb;
371 int i = 1;
372 std::string cname;
373 while ((entry = readdir(dir)) != nullptr) {
374#ifndef _WIN32
375 if (entry->d_ino == 0)
376 continue;
377#endif
378 cname = _theFilesDir + "/" + entry->d_name;
379 if (stat(cname.c_str(), &sb) != 0)
380 throw Error::StrategyError("Cannot stat store file (" +
381 Error::errorStr() + ")");
382 if ((S_IFMT & sb.st_mode) == S_IFDIR) /* skip '.' and '..' */
383 continue;
384 if (key == entry->d_name) {
385 _cursorPos = i;
386 break;
387 }
388 i++;
389 }
390
391 /* Exited the loop by exhausting the directory */
392 if (entry == nullptr)
393 throw Error::ObjectDoesNotExist(key);
394
395 if (dir != nullptr) {
396 if (closedir(dir)) {
397 throw Error::StrategyError("Could not close " +
398 _theFilesDir + " (" + Error::errorStr() +
399 ")");
400 }
401 }
