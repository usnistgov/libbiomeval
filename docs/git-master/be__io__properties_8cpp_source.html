55void
56BiometricEvaluation::IO::Properties::initWithBuffer(
57 const uint8_t *const buffer,
58 size_t size,
59 const BiometricEvaluation::IO::Properties::PropertiesMap &defaults)
60{
61 /* Initialize the PropertiesMap */
62 _properties.clear();
63 this->registerDefaults(defaults);
64
65 bool eof = false;
66 size_t offset = 0;
67 std::string::size_type idx;
68 std::string oneline, property, value;
69 for (;;) {
70 oneline = "";
71 /* Read one line */
72 for (;;) {
73 if (offset >= size) {
74 eof = true;
75 break;
76 }
77
78 char c = buffer[offset++];
79 if (c == '\n') break;
80 oneline += c;
81 }
82 if (eof) break;
83
84 /* Each line must contain a '=' separator*/
85 idx = oneline.find("=");
86 if ((idx == std::string::npos) || (idx == 0)) {
87 /* Allow blank lines */
88 if (Text::trimWhitespace(oneline).empty())
89 continue;
90 throw Error::StrategyError("Properties file has "
91 "invalid line");
92 }
93
94 _properties[Text::trimWhitespace(oneline.substr(0, idx))] =
95 Text::trimWhitespace(oneline.substr(idx + 1,
96 oneline.length()));
97 }
98}

168int64_t
169BiometricEvaluation::IO::Properties::getPropertyAsInteger(
170 const std::string &property)
171 const
172{
173 std::string p{Text::trimWhitespace(property)};
174 PropertiesMap::const_iterator it = _properties.find(p);
175 if (it == _properties.end())
176 throw Error::ObjectDoesNotExist(property);
177
178 std::string value = it->second; /* Whitespace already removed */
179 /* Empty values are allowed, but not for integers */
180 if (value == "")
181 throw Error::ConversionError();
182
183 int base = 10;
184 /* Check for hexadecimal value */
185 if ((value.find("0x") == 0) || (value.find("0X") == 0))
186 base = 16;
187
188 int startIdx = 0;
189 if (base == 16)
190 startIdx = 2;
191 if (value[0] == '-')
192 startIdx = 1;
193 int id;
194 for (unsigned int idx = startIdx; idx < value.size(); idx++) {
195 if (base == 16)
196 id = isxdigit(value[idx]);
197 else
198 id = isdigit(value[idx]);
199 if (id == 0)
200 throw Error::ConversionError();
201 }
202
203 /* Convert the string value to integer */
204 errno = 0;
205 int64_t conVal = (int64_t)strtoll(value.c_str(), nullptr, base);
206 if (errno == ERANGE)
207 throw Error::ConversionError("Value out of range");
208
209 return (conVal);
210}

228bool
229BiometricEvaluation::IO::Properties::getPropertyAsBoolean(
230 const std::string &property)
231 const
232{
233 std::string value = getProperty(property);
234 if (BE::Text::caseInsensitiveCompare(value, "TRUE") ||
235 BE::Text::caseInsensitiveCompare(value, "YES") ||
236 BE::Text::caseInsensitiveCompare(value, "ENABLE") ||
237 BE::Text::caseInsensitiveCompare(value, "ENABLED") ||
238 BE::Text::caseInsensitiveCompare(value, "1"))
239 return (true);
240 else if (BE::Text::caseInsensitiveCompare(value, "FALSE") ||
241 BE::Text::caseInsensitiveCompare(value, "NO") ||
242 BE::Text::caseInsensitiveCompare(value, "DISABLE") ||
243 BE::Text::caseInsensitiveCompare(value, "DISABLED") ||
244 BE::Text::caseInsensitiveCompare(value, "0"))
245 return (false);
246
247 throw BE::Error::ConversionError();
248}
