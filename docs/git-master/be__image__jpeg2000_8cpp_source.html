76BiometricEvaluation::Image::JPEG2000::JPEG2000(
77 const uint8_t *data,
78 const uint64_t size,
79 const std::string &identifier,
80 const statusCallback_t &statusCallback,
81 const int8_t codecFormat) :
82 Image::Image(
83 data,
84 size,
85 CompressionAlgorithm::JP2,
86 identifier,
87 statusCallback),
88 _codecFormat(codecFormat)
89{
90 std::unique_ptr<opj_codec_t, OpenJPEG_CodecDeleter> codec(
91 static_cast<opj_codec_t*>(this->getDecompressionCodec()),
92 OpenJPEG_CodecDeleter{});
93 std::unique_ptr<opj_stream_t, OpenJPEG_StreamDeleter> stream(
94 static_cast<opj_stream_t*>(this->getDecompressionStream()),
95 OpenJPEG_StreamDeleter{});
96
97 opj_image_t *imagePtr = nullptr;
98 if (opj_read_header(stream.get(), codec.get(), &imagePtr) == OPJ_FALSE)
99 throw Error::Exception("Could not read header");
100 if (imagePtr == nullptr)
101 throw Error::Exception("Parsed header is empty");
102 std::unique_ptr<opj_image_t, OpenJPEG_ImageDeleter> image(
103 imagePtr, OpenJPEG_ImageDeleter{});
104
105 if (image->numcomps <= 0)
106 throw Error::NotImplemented("No components");
107
108 if ((image->color_space != OPJ_CLRSPC_SRGB) &&
109 (image->color_space != OPJ_CLRSPC_GRAY) &&
110 (image->color_space != OPJ_CLRSPC_UNSPECIFIED))
111 throw Error::NotImplemented("Colorspace " +
112 std::to_string(image->color_space));
113
114 /*
115 * Assign Image class instance variables.
116 */
117
118 this->setDimensions(Size(image->x1, image->y1));
119
120 /* Color depth */
121 const uint32_t prec = image->comps[0].prec;
122 for (uint32_t component = 1; component < image->numcomps; ++component)
123 if (image->comps[component].prec != prec)
124 throw Error::NotImplemented("Non-equivalent component "
125 "bit depths");
126 setColorDepth(image->numcomps * prec);
127 this->setBitDepth(prec);
128
129 /* Resolution */
130 static const uint8_t resc[4] = { 0x72, 0x65, 0x73, 0x63 };
131 static const uint8_t resc_box_size = 10;
132 /* The Capture Resolution Box is optional under some codecs */
133 try {
134 setResolution(parse_res(find_marker(resc, 4,
135 this->getDataPointer(), this->getDataSize(),
136 resc_box_size)));
137 } catch (const Error::ObjectDoesNotExist&) {
138 setResolution(Resolution(72, 72, Resolution::Units::PPI));
139 }
140
141 /*
142 * Component definition optional, but appears to only be present when
143 * not Grayscale or RGB (such as RGBA).
144 */
145 try {
146 this->setHasAlphaChannel(checkForAlphaInCDEF());
147 } catch (BE::Error::Exception &e) {
148 /* Take best guess on alpha channel presence */
149 this->setHasAlphaChannel((
150 (image->color_space == OPJ_CLRSPC_GRAY) &&
151 (image->numcomps == 2)) ||
152 ((image->color_space == OPJ_CLRSPC_SRGB) &&
153 (image->numcomps == 4)) ||
154 ((image->color_space == OPJ_CLRSPC_UNSPECIFIED) &&
155 (image->numcomps == 2)) ||
156 ((image->color_space == OPJ_CLRSPC_UNSPECIFIED) &&
157 (image->numcomps == 4)));
158 }
159}

174BiometricEvaluation::Memory::uint8Array
175BiometricEvaluation::Image::JPEG2000::getRawData()
176 const
177{
178 std::unique_ptr<opj_codec_t, OpenJPEG_CodecDeleter> codec(
179 static_cast<opj_codec_t*>(this->getDecompressionCodec()),
180 OpenJPEG_CodecDeleter{});
181 std::unique_ptr<opj_stream_t, OpenJPEG_StreamDeleter> stream(
182 static_cast<opj_stream_t*>(this->getDecompressionStream()),
183 OpenJPEG_StreamDeleter{});
184
185 opj_image_t *imagePtr = nullptr;
186 if (opj_read_header(stream.get(), codec.get(), &imagePtr) == OPJ_FALSE)
187 throw Error::Exception("Could not read header");
188 if (imagePtr == nullptr)
189 throw Error::Exception("Parsed header is empty");
190 std::unique_ptr<opj_image_t, OpenJPEG_ImageDeleter> image(
191 imagePtr, OpenJPEG_ImageDeleter{});
192
193 if (image->numcomps <= 0)
194 throw Error::NotImplemented("No components");
195 if (image->comps[0].sgnd == 1)
196 throw Error::NotImplemented("Signed buffers");
197
198 if (opj_decode(codec.get(), stream.get(), image.get()) == OPJ_FALSE)
199 throw Error::StrategyError("Could not initialize decoding");
200
201 const uint32_t w = this->getDimensions().xSize;
202 const uint32_t h = this->getDimensions().ySize;
203 const uint8_t bpc = image->comps[0].prec;
204
205 std::vector<int32_t*> ptr;
206 for (uint32_t i = 0; i < image->numcomps; ++i) {
207 ptr.push_back(image->comps[i].data);
208 if ((image->comps[i].w != w) || (image->comps[i].h != h) ||
209 (image->comps[i].prec != bpc))
210 throw Error::NotImplemented("Non-equal components");
211 }
212
213 Memory::uint8Array rawData(image->numcomps * (bpc / 8) * image->x1 *
214 image->y1);
215 Memory::MutableIndexedBuffer buffer(rawData);
216
217 const int32_t mask = (1 << image->comps[0].prec) - 1;
218 for (uint32_t row = 0; row < h; ++row) {
219 for (uint32_t col = 0; col < w; ++col) {
220 if (bpc <= 8) {
221 for (uint32_t i = 0; i < image->numcomps; ++i) {
222 buffer.pushU8Val(*ptr[i] & mask);
223 ptr[i]++;
224 }
225 } else if (bpc <= 16) {
226 for (uint32_t i = 0; i < image->numcomps; ++i) {
227 buffer.pushU16Val(*ptr[i] & mask);
228 ptr[i]++;
229 }
230 } else {
231 throw Error::NotImplemented(
232 std::to_string(bpc) +
233 "-bit-per-component images");
234 }
235 }
236 }
237
238 return (rawData);
239}

249bool
250BiometricEvaluation::Image::JPEG2000::isJPEG2000(
251 const uint8_t *data,
252 uint64_t size)
253{
254 static const uint8_t SOC_size = 12;
255 static const uint8_t SOC[SOC_size] = {
256 0x00, 0x00, 0x00, 0x0C, 0x6A, 0x50,
257 0x20, 0x20, 0x0D, 0x0A, 0x87, 0x0A
258 };
259 if (size < SOC_size)
260 return (false);
261
262 return (memcmp(data, SOC, SOC_size) == 0);
263}

306bool
307BiometricEvaluation::Image::JPEG2000::checkForAlphaInCDEF()
308{
309 static constexpr uint8_t cdef[4] = { 0x63, 0x64, 0x65, 0x66 };
310 static constexpr uint8_t cdefTagSize{4};
311
312 const uint64_t offset{find_marker_offset(cdef, cdefTagSize,
313 this->getDataPointer(), this->getDataSize())};
314
315 BE::Memory::IndexedBuffer ib(this->getDataPointer(),
316 this->getDataSize());
317 ib.scan(nullptr, offset + cdefTagSize);
318
319 volatile uint16_t typ{};
320 uint16_t numCDEF{ib.scanBeU16Val()};
321 for (uint16_t c{0}; c < numCDEF; ++c) {
322 typ = ib.scanBeU16Val(); /* cn */
323
324 typ = ib.scanBeU16Val();
325 if (typ == 1)
326 return (true);
327
328 typ = ib.scanBeU16Val(); /* asoc */
329 }
330
331 return (false);
332}

334uint64_t
335BiometricEvaluation::Image::JPEG2000::find_marker_offset(
336 const uint8_t *marker,
337 uint64_t marker_size,
338 const uint8_t *buffer,
339 uint64_t buffer_size)
340{
341 uint64_t step;
342 const uint64_t max{buffer_size - marker_size};
343 for (step = 0; step < max; ++step) {
344 if (memcmp(buffer, marker, marker_size))
345 ++buffer;
346 else
347 break;
348 }
349
350 if (step >= max)
351 throw BE::Error::ObjectDoesNotExist();
352
353 return (step);
354}

356BiometricEvaluation::Memory::AutoArray<uint8_t>
357BiometricEvaluation::Image::JPEG2000::find_marker(
358 const uint8_t *marker,
359 uint64_t marker_size,
360 const uint8_t *buffer,
361 uint64_t buffer_size,
362 uint64_t value_size)
363{
364 uint64_t step{find_marker_offset(marker, marker_size, buffer,
365 buffer_size)};
366 if ((step + marker_size + value_size) > buffer_size)
367 throw Error::ObjectDoesNotExist();
368
369 Memory::AutoArray<uint8_t> ret(value_size);
370 ret.copy(buffer + step + marker_size, value_size);
371 return (ret);
372}

374BiometricEvaluation::Image::Resolution
375BiometricEvaluation::Image::JPEG2000::parse_res(
376 const BiometricEvaluation::Memory::AutoArray<uint8_t> &res)
377{
378 /* Sanity check */
379 if (res.size() != 10)
380 throw Error::DataError("Invalid size for Resolution Box");
381
382 Memory::IndexedBuffer ib(res);
383
384 /* I.7.3.6.1: Capture resolution box */
385 uint16_t VR_N = ib.scanBeU16Val();
386 uint16_t VR_D = ib.scanBeU16Val();
387 uint16_t HR_N = ib.scanBeU16Val();
388 uint16_t HR_D = ib.scanBeU16Val();
389 int8_t VR_E = static_cast<int8_t>(ib.scanU8Val());
390 int8_t HR_E = static_cast<int8_t>(ib.scanU8Val());
391
392 return (Resolution(
393 ((static_cast<float>(VR_N) / VR_D) * pow(10.0, VR_E)) / 100.0,
394 ((static_cast<float>(HR_N) / HR_D) * pow(10.0, HR_E)) / 100.0,
395 Resolution::Units::PPCM));
396}

398void*
399BiometricEvaluation::Image::JPEG2000::getDecompressionCodec()
400 const
401{
402 opj_codec_t *codec = nullptr;
403 switch (this->_codecFormat) {
404 case OPJ_CODEC_J2K: /* JPEG-2000 codestream (.J2K) */
405 /* FALLTHROUGH */
406 case OPJ_CODEC_JP2: /* JPEG-2000 compressed image data (.JP2) */
407 /* FALLTHROUGH */
408 case OPJ_CODEC_JPT: /* JPEG 2000, JPIP (.JPT) */
409 codec = opj_create_decompress(static_cast<OPJ_CODEC_FORMAT>(
410 this->_codecFormat));
411 break;
412 case OPJ_CODEC_UNKNOWN:
413 /* FALLTHROUGH */
414 default:
415 throw Error::StrategyError("Unsupported decoding format: " +
416 std::to_string(this->_codecFormat));
417 break;
418 }
419
420 /* libopenjpg2 error callbacks */
421 opj_set_error_handler(codec, openjpeg_error, (void *)this);
422 opj_set_warning_handler(codec, openjpeg_warning, (void *)this);
423 opj_set_info_handler(codec, openjpeg_info, (void *)this);
424
425 /* Use default decoding parameters, except codec, which is "unknown" */
426 opj_dparameters parameters;
427 opj_set_default_decoder_parameters(&parameters);
428 parameters.decod_format = this->_codecFormat;
429 if (opj_setup_decoder(codec, &parameters) == OPJ_FALSE) {
430 opj_destroy_codec(codec);
431 throw Error::StrategyError("Could not initialize decoding");
432 }
433
434 return (codec);
435}
