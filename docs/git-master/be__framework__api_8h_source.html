328template<typename T>
329typename BiometricEvaluation::Framework::API<T>::Result
330BiometricEvaluation::Framework::API<T>::call(
331 const std::function<T(void)> &operation,
332 const std::function<void(const Framework::API<T>::Result&)> &success,
333 const std::function<void(const Framework::API<T>::Result&)> &failure,
334 const bool rethrowExceptions)
335{
336 Result ret;
337
338 BEGIN_SIGNAL_BLOCK(this->getSignalManager(), SM_BLOCK);
339 BEGIN_WATCHDOG_BLOCK(this->getWatchdog(), WD_BLOCK);
340 ret.currentState = APICurrentState::Running;
341 this->getTimer()->start();
342 try {
343 ret.status = operation();
344 } catch (...) {
345 this->getTimer()->stop();
346 ret.elapsed = this->getTimer()->elapsed();
347 ret.currentState = APICurrentState::ExceptionCaught;
348 ret.setException(std::current_exception());
349
350 if (failure)
351 failure(ret);
352
353 if (rethrowExceptions)
354 throw;
355
356 return (ret);
357 }
358 this->getTimer()->stop();
359 END_WATCHDOG_BLOCK(this->getWatchdog(), WD_BLOCK);
360 END_SIGNAL_BLOCK(this->getSignalManager(), SM_BLOCK);
361 if (this->getSignalManager()->sigHandled()) {
362 this->getTimer()->stop();
363 ret.elapsed = this->getTimer()->elapsed();
364 ret.currentState = APICurrentState::SignalCaught;
365
366 if (failure)
367 failure(ret);
368 } else if (this->getWatchdog()->expired()) {
369 this->getTimer()->stop();
370 ret.elapsed = this->getTimer()->elapsed();
371 ret.currentState = APICurrentState::WatchdogExpired;
372
373 if (failure)
374 failure(ret);
375 } else {
376 ret.currentState = APICurrentState::Completed;
377 ret.elapsed = this->getTimer()->elapsed();
378
379 if (success)
380 success(ret);
381 }
382
383 return (ret);
384}
