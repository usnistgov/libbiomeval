66void
67BiometricEvaluation::Process::Manager::reset()
68{
69 /* Make sure no workers are working before resetting any of them */
70 if (this->getNumActiveWorkers() != 0)
71 throw Error::ObjectExists();
72
73 for (auto &worker : this->_workers)
74 worker->reset();

76 _pendingExit.clear();

83bool
84BiometricEvaluation::Process::Manager::waitForMessage(
85 std::shared_ptr<WorkerController> &sender,
86 int *nextFD,
87 int numSeconds)
88 const
89{
90 bool result = false;
91 fd_set set;
92
93 /* Listen for all Worker receiving pipes */
94 FD_ZERO(&set);
95 int maxfd = 0, curfd;
96 std::map<std::shared_ptr<WorkerController>, int> fds;
97
98 struct timeval timeout, *timeoutptr = nullptr;
99 if (numSeconds >= 0) {
100 timeout.tv_sec = numSeconds;
101 timeout.tv_usec = 0;
102 timeoutptr = &timeout;
103 }
104
105 /* Round up all receiving pipes */
106 bool finished = false;
107 bool haveFD = false;
108 while (!finished) {
109 for (size_t i = 0; i < _workers.size(); i++) {
110 /*
111 * Add only active pipes to list: If the worker is
112 * asked to stop, it will be in the pending exit list;
113 * if it ended on its own, it won't be working anymore.
114 */
115 if (find(_pendingExit.begin(), _pendingExit.end(),
116 _workers[i]) != _pendingExit.end())
117 continue;
118 if (!_workers[i]->isWorking())
119 continue;
120
121 try {
122 curfd = _workers[i]->getWorker()->
123 getReceivingPipe();
124 FD_SET(curfd, &set);
125 haveFD = true;
126 if (curfd > maxfd)
127 maxfd = curfd;
128 fds[_workers[i]] = curfd;
129 } catch (const Error::ObjectDoesNotExist&) {
130 /* Don't add pipes for exiting Workers */
131 }
132 }
133
134 /* Don't hang in select if there are no file descriptors */
135 if (!haveFD) {
136 result = false;
137 finished = true;
138 break;
139 }
140
141 int ret = select(maxfd + 1, &set, nullptr, nullptr, timeoutptr);
142 /*
143 * We don't reset timeout value here (for Linux systems)
144 * because when something, or nothing, is available, we return,
145 * and for interrupted calls can just leave the remaining
146 * timeout in place.
147 */
148 if (ret == 0) {
149 /* Nothing available */
150 result = false;
151 finished = true;
152 } else if (ret < 0) {
153 /* Could have been interrupted while blocking */
154 if (errno != EINTR)
155 finished = true;
156 } else {
157 /* Something available -- check what */
158 for (std::map<std::shared_ptr<WorkerController>, int>::
159 const_iterator it = fds.begin(); it != fds.end();
160 it++) {
161 if (FD_ISSET(it->second, &set) != 0) {
162 if (nextFD != nullptr)
163 *nextFD = it->second;
164 result = true;
165 sender = it->first;
166 break;
167 }
168 }
169 finished = true;
170 }
171 }
172
173 return (result);
174}

176bool
177BiometricEvaluation::Process::Manager::getNextMessage(
178 std::shared_ptr<WorkerController> &sender,
179 Memory::uint8Array &message,
180 int timeout)
181 const
182{
183 int fd = 0;
184 if (this->waitForMessage(sender, &fd, timeout) == false)
185 return (false);
186
187 uint64_t length;
188 IO::Utility::readPipe(&length, sizeof(length), fd);
189 message.resize(length);
190 IO::Utility::readPipe(message, fd);
191
192 return (true);
193}
