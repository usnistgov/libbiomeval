30BiometricEvaluation::Device::Smartcard::Impl::Impl(
31 unsigned int cardNum,
32 const Memory::uint8Array &appID)
33{
34 if (appID.size() > APDU::MAX_NC_SIZE)
35 throw (BE::Error::ParameterError("Application ID too large"));
36
37 /* Exceptions float out */
38 this->connectToCard(cardNum);
39
40 BE::Device::Smartcard::APDU apdu{};
41 apdu.cla = 0x00;
42 apdu.ins = 0xA4; /* SELECT command */
43 apdu.p1 = 0x04; /* command data is DF name */
44 apdu.p2 = 0x00;
45 apdu.lc = appID.size();
46 std::memcpy(apdu.nc, appID, appID.size());
47 apdu.le = 0x00;
48
49 /*
50 * Set the Le field flag so any file control info block, if any,
51 * is returned.
52 */
53 apdu.field_mask = APDU::FIELD_LC | APDU::FIELD_LE;
54
55 try {
56 auto response = this->sendAPDU(apdu);
57 this->_lastResponseData = response.data;
58 } catch (BE::Device::Smartcard::APDUException &e) {
59 if ((e.response.sw1 == APDU::CHECK_ERR_WRONG_PARAM_QUAL) &&
60 (e.response.sw2 == APDU::FILE_OR_APP_NOT_FOUND)) {
61 throw BE::Error::ParameterError(
62 "Application not found");
63 } else {
64 throw;
65 }
66 }

70BiometricEvaluation::Memory::uint8Array
71BiometricEvaluation::Device::Smartcard::Impl::getDedicatedFileObject(
72 const Memory::uint8Array &objectID)
73{
74 if (objectID.size() > APDU::MAX_NC_SIZE)
75 throw (BE::Error::ParameterError("Data object ID too large"));
76
77 BE::Device::Smartcard::APDU apdu{};
78 apdu.cla = 0x00;
79 apdu.ins = 0xCB; /* P1-P2 contains file ID */
80 apdu.p1 = 0x3F; /* Use the current dedicated file */
81 apdu.p2 = 0xFF;
82 apdu.lc = objectID.size();
83 std::memcpy(apdu.nc, objectID, objectID.size());
84 apdu.le = 0x00;
85 apdu.field_mask = APDU::FIELD_LC | APDU::FIELD_LE;
86
87 /* We could be check sw1/sw2, but it's not clear what the status
88 * words mean. 0x6A80 (incorrect parameters) is often returned when
89 * a 'known' object tag is sent over, while while 0x6A82 is returned
90 * returned when an unknown tag is sent over. So let the APDUException
91 * float out of here, and the app can examine the status words.
92 */
93 auto response = this->sendAPDU(apdu);
94 return (response.data);
95}

119static void
120getAPDUResponse(
121 SCARDHANDLE hCard,
122 SCARD_IO_REQUEST pioSendPci,
123 uint8_t *recvBuf, DWORD recvLen,
124 BE::Memory::uint8Array &response, uint8_t &sw1, uint8_t &sw2)
125{
126 LONG rc;
127 uint8_t lsw1, lsw2;
128 uint8_t bGetRes[5] = {0x00, 0xC0, 0x00, 0x00, 0x00};
129 DWORD lRecvLen;
130
131 response.resize(BE::Device::Smartcard::APDU::MAX_LE_SIZE);
132 BE::Memory::MutableIndexedBuffer indexedResponse(response);
133
134 /* Get the response status words and check whether chaining was done */
135 lsw1 = recvBuf[recvLen - 2];
136 lsw2 = recvBuf[recvLen - 1];
137
138 /* Handle response chaining */
139 lRecvLen = recvLen;
140 while (lsw1 == BE::Device::Smartcard::APDU::NORMAL_CHAINING) {
141 indexedResponse.push(recvBuf, lRecvLen - 2);
142 bGetRes[4] = lsw2; /* The Le field */
143 lRecvLen = (0 == lsw2) ? 256 : lsw2;
144 lRecvLen += 2; /* Account for the SW */
145 rc = SCardTransmit(hCard, &pioSendPci, bGetRes, 5, nullptr,
146 recvBuf, &lRecvLen);
147
148 if (rc != SCARD_S_SUCCESS) {
149 throw BE::Error::StrategyError(
150 "Transmit of GET RESPONSE: "
151 + std::string(pcsc_stringify_error(rc)));
152 }
153 lsw1 = recvBuf[lRecvLen - 2];
154 lsw2 = recvBuf[lRecvLen - 1];
155 }
156 sw1 = lsw1;
157 sw2 = lsw2;
158 indexedResponse.push(recvBuf, lRecvLen - 2);
159 response.resize(indexedResponse.getIndex());
160}

173void
174BiometricEvaluation::Device::Smartcard::Impl::connectToCard(
175 unsigned int cardNum)
176{
177 LONG rc;
178
179 SCARDCONTEXT context;
180 if (SCardEstablishContext(
181 SCARD_SCOPE_SYSTEM, nullptr, nullptr, &context) !=
182 SCARD_S_SUCCESS) {
183 throw BE::Error::StrategyError(
184 "Could not establish PCSC context");
185 }
186
187 /* Retrieve the available readers list */
188 DWORD dwReaders;
189 rc = SCardListReaders(context, nullptr, nullptr, &dwReaders);
190 if (rc != SCARD_S_SUCCESS) {
191 throw BE::Error::StrategyError(
192 "Could not list readers: "
193 + std::string(pcsc_stringify_error(rc)));
194 }
195 LPTSTR mszReaders;
196 mszReaders = (LPTSTR)malloc(sizeof(char)*dwReaders);
197 if (mszReaders == nullptr) {
198 throw BE::Error::MemoryError("Reader string array");
199 }
200 std::unique_ptr<void, decltype(freeCStyle)>
201 pmszReaders(mszReaders, freeCStyle);
202 rc = SCardListReaders(context, nullptr, mszReaders, &dwReaders);
203 if (rc != SCARD_S_SUCCESS) {
204 throw BE::Error::StrategyError(
205 "Could not list readers: "
206 + std::string(pcsc_stringify_error(rc)));
207 }
208 /*
209 * Extract readers from the nul separated set of strings.
210 */
211 std::vector<std::string> readers;
212 char *ptr = mszReaders;
213 while (*ptr != '\0') {
214 readers.push_back(std::string(ptr));
215 ptr += std::strlen(ptr)+1;
216 }
217 if (readers.size() == 0) {
218 throw BE::Error::StrategyError("No readers found");
219 }
220 SCARDHANDLE hCard;
221 DWORD rdrprot;
222 unsigned int readerNum = 0;
223 bool found = false;
224 for (readerNum = 0; readerNum < readers.size(); readerNum++) {
225 rc = SCardConnect(context, readers[readerNum].c_str(),
226 SCARD_SHARE_EXCLUSIVE,
227 SCARD_PROTOCOL_T0 | SCARD_PROTOCOL_T1,
228 &hCard, &rdrprot);
229 if (rc == 0) {
230 if (readerNum == cardNum) {
231 this->_hCard = hCard;
232 this->_readerID = readers[readerNum];
233 found = true;
234 break;
235 } else {
236 SCardDisconnect(hCard, SCARD_RESET_CARD);
237 }
238 } else {
239 /*
240 * We could throw an exception here, letting the
241 * the caller know that the card is missing or
242 * otherwise inaccessible, but they probably
243 * don't care that a reader/card they didn't ask
244 * for has an issue.
245 */
246 }
247 }
248 if (!found) {
249 throw BE::Error::ParameterError(
250 "No card found for given number");
251 }
252}

256void
257BiometricEvaluation::Device::Smartcard::Impl::sendChained(
258 SCARD_IO_REQUEST pioSendPci,
259 BE::Device::Smartcard::APDU &apdu,
260 BE::Memory::uint8Array &response,
261 uint8_t &sw1, uint8_t &sw2)
262{
263 LONG rc;
264 int LcLen;
265 int maxLcLen;
266 int ncIndex;
267 uint8_t bSendBuffer[MAX_BUFFER_SIZE];
268 uint8_t bRecvBuffer[MAX_BUFFER_SIZE];
269 DWORD sendIndex;
270 DWORD recvLength;
271
272 recvLength = sizeof(bRecvBuffer);
273
274 bSendBuffer[0] = apdu.cla;
275 bSendBuffer[1] = apdu.ins;
276 bSendBuffer[2] = apdu.p1;
277 bSendBuffer[3] = apdu.p2;
278
279 LcLen = apdu.lc;
280 maxLcLen = APDU::MAX_SHORT_LC - APDU::HEADER_LEN - APDU::FLEN_TRAILER;
281 if ((apdu.field_mask & APDU::FIELD_LC) != 0) {
282 maxLcLen -= APDU::FLEN_LC_SHORT;
283 }
284
285 if ((apdu.field_mask & APDU::FIELD_LE) != 0) {
286 if (apdu.le > APDU::MAX_SHORT_LE) {
287 throw BE::Error::ParameterError(
288 "Invalid Le value");
289 }
290 maxLcLen -= APDU::FLEN_LE_SHORT;
291 }
292 ncIndex = 0;
293
294 unsigned int lastAPDUIndex = 0;
295 while (LcLen > 0) {
296 sendIndex = 4;
297 if ((apdu.field_mask & APDU::FIELD_LC) != 0) {
298 if (LcLen > maxLcLen) {
299 bSendBuffer[0] |= APDU::FLAG_CLA_CHAIN;
300 bSendBuffer[sendIndex] = (uint8_t)maxLcLen;
301 sendIndex += 1;
302 std::memcpy(&bSendBuffer[sendIndex],
303 &apdu.nc[ncIndex], maxLcLen);
304 ncIndex += maxLcLen;
305 sendIndex += maxLcLen;
306 LcLen -= maxLcLen;
307 } else { /* Note that LcLen cannot be 0 here */
308 bSendBuffer[0] &= ~APDU::FLAG_CLA_CHAIN;
309 bSendBuffer[sendIndex] = (uint8_t)LcLen;
310 sendIndex += 1;
311 std::memcpy(&bSendBuffer[sendIndex],
312 &apdu.nc[ncIndex], LcLen);
313 ncIndex += LcLen;
314 sendIndex += LcLen;
315 LcLen = 0;
316 }
317 }
318 if ((apdu.field_mask & APDU::FIELD_LE) != 0) {
319 bSendBuffer[sendIndex] = (uint8_t)apdu.le;
320 sendIndex += 1;
321 }
322
323 /*
324 * At this point, sendIndex is the location of where the next
325 * byte would be written into the send buffer, and that happens
326 * to be the current length of the send buffer as well.
327 * Append the send buffer to the saved last APDU.
328 */
329 this->_lastAPDU.resize(lastAPDUIndex + sendIndex);
330 std::memcpy(
331 &this->_lastAPDU[lastAPDUIndex], bSendBuffer, sendIndex);
332 lastAPDUIndex += sendIndex;
333 this->_lastResponseData.resize(0);
334 if (this->_dryrun == false) {
335 rc = SCardTransmit(this->_hCard,
336 &pioSendPci, bSendBuffer,
337 sendIndex, nullptr, bRecvBuffer, &recvLength);
338 if (rc != SCARD_S_SUCCESS) {
339 throw BE::Error::StrategyError(
340 "Transmit failed: "
341 + std::string(pcsc_stringify_error(rc)));
342 }
343 getAPDUResponse(this->_hCard, pioSendPci,
344 bRecvBuffer, recvLength, response,
345 sw1, sw2);
346 this->_lastResponseData = response;
347 /*
348 * If we fail in the middle of the chain, send no more
349 * data. The client can retrieve the response and
350 * status.
351 */
352 if (sw1 != APDU::NORMAL_COMPLETE) {
353 LcLen = 0; /* Send no more data */
354 }
355 } else {
356 sw1 = APDU::NORMAL_COMPLETE;
357 sw2 = 0;
358 }
359 }

364void
365BiometricEvaluation::Device::Smartcard::Impl::sendExtended(
366 SCARD_IO_REQUEST pioSendPci,
367 BE::Device::Smartcard::APDU &apdu,
368 BE::Memory::uint8Array &response,
369 uint8_t &sw1, uint8_t &sw2)
370{
371 LONG rc;
372 bool lcle_extended;
373 uint8_t bSendBuffer[MAX_BUFFER_SIZE_EXTENDED];
374 uint8_t bRecvBuffer[MAX_BUFFER_SIZE_EXTENDED];
375 DWORD sendIndex;
376 DWORD recvLength;
377
378 recvLength = sizeof(bRecvBuffer);
379
380 bSendBuffer[0] = apdu.cla;
381 bSendBuffer[1] = apdu.ins;
382 bSendBuffer[2] = apdu.p1;
383 bSendBuffer[3] = apdu.p2;
384 sendIndex = 4;
385 /*
386 * The Lc and Le fields are 0, 1, or 3 bytes. If the length is present
387 * and will fit in one byte, store it; else, store a '00' byte, then
388 * the value. Also, if either field is extended, then both must be.
389 * (Note that we are assuming the card can accept extended fields;
390 * we really should check the card capabilities third table, if that's
391 * even present. Of course, it may not be, but that is also where the
392 * command chaining indicator is located; 7816-4 is just wonderful.
393 * You're probably better off just shoving data to the card and
394 * hoping its tiny little head doesn't explode; that can be detected.)
395 */
396 if ((apdu.lc > APDU::MAX_SHORT_LC) ||
397 (apdu.le > APDU::MAX_SHORT_LE)) {
398 lcle_extended = true;
399 } else {
400 lcle_extended = false;
401 }
402 if ((apdu.field_mask & APDU::FIELD_LC) != 0) {
403 if (lcle_extended) {
404 bSendBuffer[sendIndex] = 0;
405 uint16_t val = (uint16_t)htons(apdu.lc);
406 std::memcpy(bSendBuffer + sendIndex+1, &val, 2);
407 sendIndex += 3;
408 } else {
409 bSendBuffer[sendIndex] = (uint8_t)apdu.lc;
410 sendIndex += 1;
411 }
412 std::memcpy(bSendBuffer + sendIndex, apdu.nc, apdu.lc);
413 sendIndex += apdu.lc;
414 }
415 if ((apdu.field_mask & APDU::FIELD_LE) != 0) {
416 if (lcle_extended) {
417 bSendBuffer[sendIndex] = 0;
418 uint16_t val = (uint16_t)htons(apdu.le);
419 std::memcpy(bSendBuffer + sendIndex+1, &val, 2);
420 sendIndex += 3;
421 } else {
422 bSendBuffer[sendIndex] = (uint8_t)apdu.le;
423 sendIndex += 1;
424 }
425 }
426 /*
427 * At this point, sendIndex is the location of where the next
428 * byte would be written into the send buffer, and that happens
429 * to be the current length of the send buffer as well.
430 * Copy the send buffer to the saved last APDU.
431 */
432 this->_lastAPDU.resize(sendIndex);
433 this->_lastAPDU.copy(bSendBuffer, sendIndex);
434 this->_lastResponseData.resize(0);
435 if (this->_dryrun == false) {
436 rc = SCardTransmit(this->_hCard, &pioSendPci,
437 bSendBuffer, sendIndex,
438 nullptr, bRecvBuffer, &recvLength);
439 if (rc != SCARD_S_SUCCESS) {
440 throw BE::Error::StrategyError(
441 "Transmit failed: "
442 + std::string(pcsc_stringify_error(rc)));
443 }
444 getAPDUResponse(this->_hCard, pioSendPci,
445 bRecvBuffer, recvLength, response, sw1, sw2);
446 this->_lastResponseData = response;
447 } else {
448 sw1 = APDU::NORMAL_COMPLETE;
449 sw2 = 0;
450 }

453BiometricEvaluation::Device::Smartcard::APDUResponse
454BiometricEvaluation::Device::Smartcard::Impl::sendAPDU(
455 Device::Smartcard::APDU &apdu)
456{
457 LONG rc;
458 SCARD_IO_REQUEST pioSendPci;
459 DWORD dwActiveProtocol;
460
461 if (this->_dryrun == true) {
462 dwActiveProtocol = SCARD_PROTOCOL_T0;
463 } else {
464 /* connect to a reader (even without a card) */
465 dwActiveProtocol = -1;
466 rc = SCardReconnect(this->_hCard, SCARD_SHARE_EXCLUSIVE,
467 SCARD_PROTOCOL_T0|SCARD_PROTOCOL_T1, SCARD_LEAVE_CARD,
468 &dwActiveProtocol);
469 if (rc != SCARD_S_SUCCESS)
470 throw BE::Error::StrategyError("SCardReconnect failed: "
471 + std::string(pcsc_stringify_error(rc)));
472 }
473 switch(dwActiveProtocol) {
474 case SCARD_PROTOCOL_T0:
475 pioSendPci = *SCARD_PCI_T0;
476 break;
477 case SCARD_PROTOCOL_T1:
478 pioSendPci = *SCARD_PCI_T1;
479 break;
480 default:
481 throw BE::Error::StrategyError("Unknown card protocol");
482 }
483
484 if (this->_dryrun == false) {
485 rc = SCardBeginTransaction(this->_hCard);
486 if (rc != SCARD_S_SUCCESS) {
487 (void)SCardEndTransaction(this->_hCard,
488 SCARD_LEAVE_CARD);
489 throw BE::Error::StrategyError(
490 "SCardBeginTransaction failed: "
491 + std::string(pcsc_stringify_error(rc)));
492 }
493 }
494 Memory::uint8Array response;
495 uint8_t sw1, sw2;
496 if (dwActiveProtocol == SCARD_PROTOCOL_T0)
497 this->sendChained(
498 pioSendPci, apdu, response, sw1, sw2);
499 else
500 this->sendExtended(
501 pioSendPci, apdu, response, sw1, sw2);
502
503 if (sw1 != APDU::NORMAL_COMPLETE) {
504 throw BE::Device::Smartcard::APDUException(
505 BE::Device::Smartcard::APDUResponse(response, sw1, sw2),
506 this->_lastAPDU);
507 }
508 if (this->_dryrun == false) {
509 rc = SCardEndTransaction(this->_hCard, SCARD_LEAVE_CARD);
510 if (rc != SCARD_S_SUCCESS) {
511 throw BE::Error::StrategyError("End Transaction failed:"
512 + std::string(pcsc_stringify_error(rc)));
513 }
514 }
515 return (BE::Device::Smartcard::APDUResponse(response, sw1, sw2));
516}
