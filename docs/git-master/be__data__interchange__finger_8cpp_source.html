18BiometricEvaluation::Memory::uint8Array
19BiometricEvaluation::DataInterchange::Finger::ANSI2004ToISOCard2011(
20 const BiometricEvaluation::DataInterchange::ANSI2004Record &ansi2004,
21 const uint32_t viewNumber,
22 const uint8_t maximumMinutia,
23 const uint8_t minimumMinutia,
24 const BiometricEvaluation::Feature::Sort::Kind &sortOrder)
25{
26 /* ISO on-card: 1 unit == 5.625 degrees */
27 static constexpr double ISOOCAngleUnit = (360.0 / 64.0);
28 /* Multiply by this to go from degrees to ISO on-card units */
29 static constexpr double ISOOCAngleUnitsPerDegree = (1 / ISOOCAngleUnit);
30 /* Multiply by this to go from ANSI units to degrees */
31 static const double DegreesPerANSIAngleUnit = 2;
32 /* Maximum ISO on-card angle unit */
33 static const double ISOOCMaxAngleUnit = 63;
34 /* Maximum ISO on-card coordinate unit */
35 static const double ISOOCMaxCoordinateUnit = 255;
36
37 if (minimumMinutia < maximumMinutia)
38 throw BE::Error::ParameterError("Minimum minutia is less than"
39 "maximum minutia");
40
41 auto minutia = ansi2004.getMinutia(viewNumber).getMinutiaPoints();
42 if (minutia.size() < minimumMinutia)
43 throw BE::Error::StrategyError("Too few minutiae");
44
45 const auto resolution = ansi2004.getView(viewNumber).
46 getImageResolution().toUnits(Image::Resolution::Units::PPCM);
47 const auto imageSize = ansi2004.getView(viewNumber).getImageSize();
48
49 /* If image size is larger than can be fully represented on card */
50 if ((std::lround(imageSize.xSize * (1.0 / (resolution.xRes / 100.0))) >
51 ISOOCMaxCoordinateUnit) ||
52 (std::lround(imageSize.ySize * (1.0 / (resolution.yRes / 100.0))) >
53 ISOOCMaxCoordinateUnit)) {
54 const unsigned int maxX = std::lround(ISOOCMaxCoordinateUnit /
55 (1.0 / (resolution.xRes / 100.0)));
56 const unsigned int maxY = std::lround(ISOOCMaxCoordinateUnit /
57 (1.0 / (resolution.yRes / 100.0)));
58
59 /* Remove minutia that cannot be represented in on-card */
60 std::remove_if(minutia.begin(), minutia.end(),
61 [&](BE::Feature::MinutiaPoint &m) {
62 return ((m.coordinate.x > maxX) ||
63 (m.coordinate.y > maxY));
64 });
65 }
66 if (minutia.size() < minimumMinutia)
67 throw BE::Error::StrategyError("Too few minutiae");
68
69 /*
70 * Per ISO/IEC 19794-2:2011 9.3.2, before pruning, minutia are first
71 * sorted by quality, and then by decreasing distance from the center
72 * of mass.
73 */
74 BE::Feature::Sort::stableSort(minutia,
75 BE::Feature::Sort::Kind::QualityDescending);
76 BE::Feature::Sort::stableSort(minutia,
77 BE::Feature::Sort::Kind::PolarCOMAscending);
78 /* Prune minutia over maximum */
79 if (minutia.size() > maximumMinutia)
80 minutia.erase(std::next(minutia.begin(), maximumMinutia),
81 minutia.end());
82
83 /* Convert X,Y,Theta to ISO on-card units */
84 for (auto &m : minutia) {
85 /* Convert arbitrary PPCM resolution to 100 PPCM */
86 m.coordinate.x = std::lround(
87 static_cast<double>(m.coordinate.x) *
88 (1.0 / (resolution.xRes / 100.0)));
89 m.coordinate.y = std::lround(
90 static_cast<double>(m.coordinate.y) *
91 (1.0 / (resolution.yRes / 100.0)));
92
93 /* ANSI units -> degrees -> ISO on-card units. */
94 m.theta = static_cast<unsigned int>(
95 std::fmin(std::round(ISOOCAngleUnitsPerDegree *
96 static_cast<double>(m.theta) * DegreesPerANSIAngleUnit),
97 ISOOCMaxAngleUnit));
98 }
99
100 /* Sort, per BIT requirements */
101 BE::Feature::Sort::sort(minutia, sortOrder);
102
103 /* Assemble */
104 uint8_t typeAndTheta;
105 BE::Memory::uint8Array isoCard(3 * minutia.size());
106 BE::Memory::MutableIndexedBuffer buf(isoCard);
107 for (const auto &m : minutia) {
108 buf.pushU8Val(m.coordinate.x);
109 buf.pushU8Val(m.coordinate.y);
110
111 switch (m.type) {
112 case BE::Feature::MinutiaeType::RidgeEnding:
113 typeAndTheta = ((1 << 6) | m.theta);
114 break;
115 case BE::Feature::MinutiaeType::Bifurcation:
116 typeAndTheta = ((2 << 6) | m.theta);
117 break;
118 default:
119 typeAndTheta = m.theta;
120 break;
121 }
122 buf.pushU8Val(typeAndTheta);
123 }
124
125 return (isoCard);
126}
