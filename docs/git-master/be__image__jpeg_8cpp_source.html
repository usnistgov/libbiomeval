83BiometricEvaluation::Memory::uint8Array
84BiometricEvaluation::Image::JPEG::getRawData()
85 const
86{
87 /* Initialize custom JPEG error manager to throw exceptions */
88 struct jpeg_error_mgr jpeg_error_mgr;
89 jpeg_std_error(&jpeg_error_mgr);
90 jpeg_error_mgr.error_exit = JPEG::error_exit;
91 jpeg_error_mgr.emit_message = JPEG::emit_message;
92 jpeg_error_mgr.output_message = JPEG::output_message;
93
94 struct jpeg_decompress_struct dinfo;
95 dinfo.err = &jpeg_error_mgr;
96 dinfo.client_data = (void *)this;
97 jpeg_create_decompress(&dinfo);
98
99#if JPEG_LIB_VERSION >= 80
100 ::jpeg_mem_src(&dinfo, (unsigned char *)this->getDataPointer(),
101 this->getDataSize());
102#else
103 JPEG::jpeg_mem_src(&dinfo, (unsigned char *)this->getDataPointer(),
104 this->getDataSize());
105#endif
106
107 if (jpeg_read_header(&dinfo, TRUE) != JPEG_HEADER_OK)
108 throw Error::StrategyError("jpeg_read_header()");
109 if (jpeg_start_decompress(&dinfo) != TRUE)
110 throw Error::StrategyError("jpeg_start_decompress()");
111
112 uint64_t row_stride = dinfo.output_width * dinfo.output_components;
113 Memory::uint8Array rawData(dinfo.output_height * row_stride);
114
115 JSAMPARRAY buffer = (*dinfo.mem->alloc_sarray)(
116 (j_common_ptr)&dinfo, JPOOL_IMAGE, row_stride, 1);
117
118 for (int n = 0; dinfo.output_scanline < dinfo.output_height; n++) {
119 jpeg_read_scanlines(&dinfo, buffer, 1);
120 memcpy(&rawData[n * row_stride], buffer[0], row_stride);
121 }
122
123 /* Clean up after libjpeg */
124 jpeg_finish_decompress(&dinfo);
125 jpeg_destroy_decompress(&dinfo);
126
127 return (rawData);
128}

130BiometricEvaluation::Memory::uint8Array
131BiometricEvaluation::Image::JPEG::getRawGrayscaleData(
132 uint8_t depth)
133 const
134{
135 if (depth != 8 && depth != 1)
136 throw Error::ParameterError("Invalid value for bit depth");
137
138 /* Initialize custom JPEG error manager to throw exceptions */
139 struct jpeg_error_mgr jpeg_error_mgr;
140 jpeg_std_error(&jpeg_error_mgr);
141 jpeg_error_mgr.error_exit = JPEG::error_exit;
142 jpeg_error_mgr.emit_message = JPEG::emit_message;
143 jpeg_error_mgr.output_message = JPEG::output_message;
144
145 struct jpeg_decompress_struct dinfo;
146 dinfo.err = &jpeg_error_mgr;
147 dinfo.client_data = (void *)this;
148 jpeg_create_decompress(&dinfo);
149
150#if JPEG_LIB_VERSION >= 80
151 ::jpeg_mem_src(&dinfo, (unsigned char *)this->getDataPointer(),
152 this->getDataSize());
153#else
154 JPEG::jpeg_mem_src(&dinfo, (unsigned char *)this->getDataPointer(),
155 this->getDataSize());
156#endif
157
158 if (jpeg_read_header(&dinfo, TRUE) != JPEG_HEADER_OK)
159 throw Error::StrategyError("jpeg_read_header()");
160
161 dinfo.out_color_space = JCS_GRAYSCALE;
162 dinfo.dither_mode = JDITHER_NONE;
163 switch (depth) {
164 case 8:
165 dinfo.quantize_colors = FALSE;
166 break;
167 case 1:
168 /* We have to quantize to change the output bit depth */
169 dinfo.quantize_colors = TRUE;
170 dinfo.desired_number_of_colors = 2;
171 break;
172 }
173
174 if (jpeg_start_decompress(&dinfo) != TRUE)
175 throw Error::StrategyError("jpeg_start_decompress()");
176
177 uint64_t row_stride = dinfo.output_width * dinfo.output_components;
178 Memory::uint8Array rawGray(dinfo.output_height * row_stride);
179
180 JSAMPARRAY buffer = (*dinfo.mem->alloc_sarray)(
181 (j_common_ptr)&dinfo, JPOOL_IMAGE, row_stride, 1);
182
183 for (int n = 0; dinfo.output_scanline < dinfo.output_height; n++) {
184 jpeg_read_scanlines(&dinfo, buffer, 1);
185
186 switch (depth) {
187 case 1:
188 /*
189 * Quantize 1 bit per pixel value into an 8 bit
190 * container by mapping 1 to 255.
191 *
192 * TODO: Use a colormap to support 2-7 bit depth.
193 */
194 for (uint64_t i = 0; i < row_stride; i++)
195 if (buffer[0][i] == 0x01)
196 buffer[0][i] = 0xFF;
197 break;
198 }
199 memcpy(&rawGray[n * row_stride], buffer[0], row_stride);
200 }
201
202 /* Clean up after libjpeg */
203 jpeg_finish_decompress(&dinfo);
204 jpeg_destroy_decompress(&dinfo);
205
206 return (rawGray);
207}

209bool
210BiometricEvaluation::Image::JPEG::isJPEG(
211 const uint8_t *data,
212 uint64_t size)
213{
214 uint8_t *markerBuf = (uint8_t *)data;
215 uint8_t *endPtr = (uint8_t *)data + size;
216
217 /*
218 * JPEG markers (ISO/IEC 10918-1:1993)
219 */
220 static const uint16_t startOfScan = 0xFFDA;
221 static const uint16_t startOfImage = 0xFFD8;
222
223 /* Start of frame, non-differential, Huffman coding */
224 static const uint16_t SOFBaselineDCT = 0xFFC0;
225 static const uint16_t SOFExtendedSequentialDCT = 0xFFC1;
226 static const uint16_t SOFProgressiveDCT = 0xFFC2;
227 static const uint16_t SOFLosslessSequential = 0xFFC3;
228 /* Start of frame, differential, Huffman coding */
229 static const uint16_t SOFDifferentialSequentialDCT = 0xFFC5;
230 static const uint16_t SOFDifferentialProgressiveDCT = 0xFFC6;
231 static const uint16_t SOFDifferentialLossless = 0xFFC7;
232 /* Start of frame, non-differential, arithmetic coding */
233 static const uint16_t SOFExtendedSequentialDCTArith = 0xFFC9;
234 static const uint16_t SOFProgressiveDCTArith = 0xFFCA;
235 static const uint16_t SOFLosslessArith = 0xFFCB;
236 /* Start of frame, differential, arithmetic coding */
237 static const uint16_t SOFDifferentialSequentialDCTArith = 0xFFCD;
238 static const uint16_t SOFDifferentialProgressiveDCTArith = 0xFFCE;
239 static const uint16_t SOFDifferentialLosslessArith = 0xFFCF;
240
241 /* First marker should be start of image */
242 uint16_t marker;
243 if (biomeval_nbis_getc_ushort(&marker, &markerBuf, endPtr) != 0)
244 return (false);
245 if (marker != startOfImage)
246 return (false);
247
248 /* Read markers until end of buffer or an identifying marker is found */
249 for (;;) {
250 /* Get next 16 bits */
251 if (biomeval_nbis_getc_ushort(&marker, &markerBuf, endPtr) != 0)
252 return (false);
253
254 /* 16-bit markers start with 0xFF but aren't 0xFF00 or 0xFFFF */
255 while (((marker >> 8) != 0xFF) &&
256 ((marker == 0xFF00) || (marker == 0xFFFF)))
257 if (biomeval_nbis_getc_ushort(&marker, &markerBuf, endPtr) != 0)
258 return (false);
259
260 switch (marker) {
261 /* Lossy start of frame markers */
262 case SOFBaselineDCT:
263 /* FALLTHROUGH */
264 case SOFExtendedSequentialDCT:
265 /* FALLTHROUGH */
266 case SOFProgressiveDCT:
267 /* FALLTHROUGH */
268 case SOFDifferentialSequentialDCT:
269 /* FALLTHROUGH */
270 case SOFDifferentialProgressiveDCT:
271 /* FALLTHROUGH */
272 case SOFExtendedSequentialDCTArith:
273 /* FALLTHROUGH */
274 case SOFProgressiveDCTArith:
275 /* FALLTHROUGH */
276 case SOFDifferentialSequentialDCTArith:
277 /* FALLTHROUGH */
278 case SOFDifferentialProgressiveDCTArith:
279 return (true);
280
281 /* Lossless start of frame markers */
282 case SOFLosslessSequential:
283 /* FALLTHROUGH */
284 case SOFDifferentialLossless:
285 /* FALLTHROUGH */
286 case SOFLosslessArith:
287 /* FALLTHROUGH */
288 case SOFDifferentialLosslessArith:
289 /* FALLTHROUGH */
290
291 /* Start of scan found before a start of frame */
292 case startOfScan:
293 return (false);
294 }
295
296 /* Reposition marker pointer after current marker segment */
297 if (JPEG::getc_skip_marker_segment(marker, &markerBuf, endPtr))
298 return (false);
299 }
300
301 return (false);
302}

344void
345BiometricEvaluation::Image::JPEG::emit_message(
346 j_common_ptr cinfo,
347 int msg_level)
348{
349 Framework::Status::Type level{};
350 if (msg_level < 0)
351 level = Framework::Status::Type::Warning;
352 else
353 level = Framework::Status::Type::Debug;
354
355 if (level == Framework::Status::Type::Warning) {
356 /* Only print first warning unless high trace level set */
357 if ((cinfo->err->num_warnings == 0) ||
358 (cinfo->err->trace_level >= 3))
359 BE::Image::JPEG::callStatusCallback(cinfo, level);
360 cinfo->err->num_warnings++;
361 } else {
362 /* Only print debugging info if high trace level set */
363 if (cinfo->err->trace_level >= msg_level)
364 BE::Image::JPEG::callStatusCallback(cinfo, level);
365 }
366}

368int
369BiometricEvaluation::Image::JPEG::getc_skip_marker_segment(
370 const unsigned short marker,
371 unsigned char **cbufptr,
372 unsigned char *ebufptr)
373{
374 int ret;
375 unsigned short length;
376
377 /* Get ushort Length. */
378 if((ret = biomeval_nbis_getc_ushort(&length, cbufptr, ebufptr)))
379 return(ret);
380
381 length -= 2;
382
383 /* Check for EOB ... */
384 if ((((*cbufptr) + length) >= ebufptr))
385 return (-2);
386
387 /* Bump buffer pointer. */
388 (*cbufptr) += length;
389
390 return (0);
391}

425boolean
426BiometricEvaluation::Image::JPEG::fill_input_buffer_mem(
427 j_decompress_ptr cinfo)
428{
429 static const JOCTET EOIBuffer[4] = {
430 (JOCTET) 0xFF, (JOCTET) JPEG_EOI, 0, 0};
431
432 /*
433 * The whole JPEG data is expected to reside in the supplied memory
434 * buffer, so any request for more data beyond the given buffer size
435 * is treated as an error.
436 */
437 WARNMS(cinfo, JWRN_JPEG_EOF);
438
439 /* Insert a fake EOI marker */
440 cinfo->src->next_input_byte = EOIBuffer;
441 cinfo->src->bytes_in_buffer = 2;
442
443 return (TRUE);
444}

446void
447BiometricEvaluation::Image::JPEG::skip_input_data_mem(
448 j_decompress_ptr cinfo,
449 long num_bytes)
450{
451 struct jpeg_source_mgr * src = cinfo->src;
452
453 if (num_bytes > 0) {
454 while (num_bytes > (long) src->bytes_in_buffer) {
455 num_bytes -= (long) src->bytes_in_buffer;
456 (void) (*src->fill_input_buffer) (cinfo);
457 }
458
459 src->next_input_byte += (size_t) num_bytes;
460 src->bytes_in_buffer -= (size_t) num_bytes;
461 }
462}
