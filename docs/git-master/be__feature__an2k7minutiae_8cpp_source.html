37BiometricEvaluation::Feature::AN2K7Minutiae::AN2K7Minutiae(
38 const std::string &filename,
39 int recordNumber)
40{
41
42 if (!IO::Utility::fileExists(filename))
43 throw (BE::Error::FileError("File not found."));
44

133static void
134convertXYT(unsigned char *value, BiometricEvaluation::Feature::MinutiaPoint *mp)
135{
136 char buf[5];
137 buf[4] = '\0';
138 BE::Image::Coordinate coord;
139 for (int i = 0; i < 4; i++)
140 buf[i] = value[i];
141 mp->coordinate.x = atoi(buf);
142 for (int i = 4; i < 8; i++)
143 buf[i - 4] = value[i];
144 mp->coordinate.y = atoi(buf);
145 buf[3] = '\0';
146 for (int i = 8; i < 11; i++)
147 buf[i - 8] = value[i];
148 mp->theta = atoi(buf);
149}

151static void
152extractRidge(
153 const unsigned char *str,
154 int *point,
155 int *distance)
156{
157 /* Copy the string */
158 int len = strlen((char*)str);
159 unsigned char* copy = new unsigned char[len + 1];
160 strncpy((char *)copy, (char *)str, len + 1);
161
162 /* Split the copy */
163 int i;
164 for (i = 1; i < len; i++) {
165 if (copy[i] == ',') {
166 copy[i] = '\0';
167 break;
168 }
169 }
170
171 /* Make sure it split */
172 if (i == (len - 1))
173 throw BE::Error::DataError("Invalid format for MRC ridge data");
174
175 /* Assign the two integers */
176 *point = atoi((char*)copy);
177 *distance = atoi((char*)copy + i + 1);
178 delete [] copy;
179}

181static void
182readMRC(
183 const RECORD *type9,
184 BiometricEvaluation::Feature::MinutiaPointSet &mps,
185 BiometricEvaluation::Feature::RidgeCountItemSet &rcs,
186 bool hasRDG)
187{
188 FIELD *field;
189 int idx;
190
191 /* Number of minutiae */
192 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, MIN_ID, type9) == FALSE)
193 throw BE::Error::DataError("Field MIN not found");
194 int count = atoi((char*)field->subfields[0]->items[0]->value);
195
196 /* Minutiae and Ridge Count data */
197 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, MRC_ID, type9) == FALSE)
198 throw BE::Error::DataError("Field MRC not found");
199 for (int i = 0; i < count; i++) {
200 BiometricEvaluation::Feature::MinutiaPoint mp;
201
202 int numItems = field->subfields[i]->num_items;
203
204 /* Get required fields */
205 mp.index =
206 atoi((char*)field->subfields[i]->items[0]->value);
207 convertXYT(field->subfields[i]->items[1]->value, &mp);
208
209 /* Quality field present */
210 if (numItems >= 3) {
211 mp.has_quality = true;
212 mp.quality = (unsigned int)atoi(
213 (char*)field->subfields[i]->items[2]->value);
214 } else {
215 mp.has_quality = false;
216 }
217
218 /* Minutiae type present */
219 if (numItems >= 4) {
220 mp.type = (BE::Feature::MinutiaeType)atoi((char*)
221 field->subfields[i]->items[3]->value);
222 mp.has_type = true;
223 } else {
224 mp.has_type = false;
225 }
226 mps.push_back(mp);
227
228 /* The remaining information is ridge count data */
229 if (hasRDG) {
230 if (numItems < 5)
231 throw BE::Error::DataError("Corrupt Type-9");
232 for (int j = 4; j < numItems; j++) {
233 BE::Feature::RidgeCountItem ridge;
234 ridge.index_one = mp.index;
235 extractRidge(
236 field->subfields[i]->items[j]->value,
237 &ridge.index_two,
238 &ridge.count);
239 rcs.push_back(ridge);
240 }
241 }
242 }

245BiometricEvaluation::Finger::PatternClassification
246BiometricEvaluation::Feature::AN2K7Minutiae::convertPatternClassification(
247 const char *fpc)
248{
249 std::string str(fpc);
250 if (str == "PA")
251 return (Finger::PatternClassification::PlainArch);
252 else if (str == "TA")
253 return (Finger::PatternClassification::TentedArch);
254 else if (str == "RL")
255 return (Finger::PatternClassification::RadialLoop);
256 else if (str == "UL")
257 return (Finger::PatternClassification::UlnarLoop);
258 else if (str == "PW")
259 return (Finger::PatternClassification::PlainWhorl);
260 else if (str == "CP")
261 return (Finger::PatternClassification::CentralPocketLoop);
262 else if (str == "DL")
263 return (Finger::PatternClassification::DoubleLoop);
264 else if (str == "AW")
265 return (Finger::PatternClassification::AccidentalWhorl);
266 else if (str == "WN")
267 return (Finger::PatternClassification::Whorl);
268 else if (str == "RS")
269 return (Finger::PatternClassification::RightSlantLoop);
270 else if (str == "LS")
271 return (Finger::PatternClassification::LeftSlantLoop);
272 else if (str == "SR")
273 return (Finger::PatternClassification::Scar);
274 else if (str == "XX")
275 return (Finger::PatternClassification::Amputation);
276 else if (str == "UN")
277 return (Finger::PatternClassification::Unknown);
278 else
279 throw BE::Error::DataError("Invalid value for "
280 "pattern classification");
281}

320BiometricEvaluation::Image::Coordinate
321BiometricEvaluation::Feature::AN2K7Minutiae::convertCoordinate(
322 const char *str,
323 bool calculateDistance)
324{
325 if (strlen(str) != 8)
326 throw BE::Error::DataError(
327 "Coordinate encoding of invalid size");
328
329 char x[5], y[5];
330 for (int i = 0; i < 4; i++) x[i] = str[i];
331 for (int i = 4; i < 8; i++) y[i - 4] = str[i];
332 x[4] = y[4] = '\0';
333
334 BE::Image::Coordinate ret(atoi(x), atoi(y));
335
336 if (calculateDistance) {
337 static BE::Image::Coordinate origin(0, 0, 0.0, 0.0);
338 ret.xDistance = BE::Image::distance(ret, origin);
339 ret.yDistance = BE::Image::distance(ret, origin);
340 }
341
342 return (ret);
343}

349void
350BiometricEvaluation::Feature::AN2K7Minutiae::readType9Record(
351 Memory::uint8Array &buf,
352 int recordNumber)
353{
354 Memory::AutoBuffer<ANSI_NIST> an2k =
355 Memory::AutoBuffer<ANSI_NIST>(&biomeval_nbis_alloc_ANSI_NIST,
356 &biomeval_nbis_free_ANSI_NIST, &biomeval_nbis_copy_ANSI_NIST);
357
358 AN2KBDB bdb;
359 INIT_AN2KBDB(&bdb, buf, buf.size());
360 if (biomeval_nbis_scan_ANSI_NIST(&bdb, an2k) != 0)
361 throw BE::Error::DataError(
362 "Could not read complete AN2K record");
363
364 /*
365 * Find the requested Type-9 in the file, throwing an exception
366 * if not present. The first record in an AN2K file is always
367 * the Type-1, so skip that one.
368 */
369 RECORD *type9 = nullptr;
370 for (int i = 1; i < an2k->num_records; i++) {
371 if (an2k->records[i]->type == TYPE_9_ID) {
372 if (i == recordNumber) {
373 type9 = an2k->records[i];
374 break;
375 }
376 }
377 }
378 if (type9 == nullptr)
379 throw (BE::Error::DataError(
380 "Could not find requested Type-9 in AN2K record"));
381
382 /*********************************************************************/
383 /* Required Fields. */
384 /*********************************************************************/
385 FIELD *field;
386 int idx;
387
388 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, FGP2_ID, type9) == FALSE)
389 throw BE::Error::DataError("Field FGP not found");
390 _fgp = Finger::AN2KView::populateFGP(field);
391
392 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, FPC_ID, type9) == FALSE)
393 throw BE::Error::DataError("Field FPC not found");
394 for (int i = 0; i < field->num_subfields; i++) {
395 if (field->subfields[i]->num_items == 2)
396 _fpc.push_back(PatternClassification::Entry(
397 (std::string((char*)field->subfields[i]->items[0]->
398 value) == "T"), (char *)field->subfields[i]->
399 items[1]->value));
400 else
401 throw BE::Error::DataError{"Invalid number of items "
402 "for field FPC, subfield " + std::to_string(i)};
403 }
404
405 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, RDG_ID, type9) == FALSE)
406 throw BE::Error::DataError("Field RDG not found");
407 bool hasRDG = atoi((char*)field->subfields[0]->items[0]->value);
408
409 readMRC(type9, _minutiaPointSet, _ridgeCountItemSet, hasRDG);
410
411 /*********************************************************************/
412 /* Optional Fields. */
413 /*********************************************************************/
414
415 /* Originating Fingerprint Reader System */
416 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, OFR_ID, type9) == TRUE) {
417 _ofr.name = (char*)field->subfields[0]->items[0]->value;
418 _ofr.method = convertEncodingMethod((char*)field->
419 subfields[0]->items[1]->value);
420
421 /* Equipment ID is optional */
422 if (field->subfields[0]->num_items == 3)
423 _ofr.equipment = (char*)field->subfields[0]->
424 items[2]->value;
425 }
426
427 /* Cores */
428 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, CRP_ID, type9) == TRUE)
429 for (int i = 0; i < field->num_subfields; i++)
430 _corePointSet.push_back(CorePoint(convertCoordinate(
431 (char *)field->subfields[i]->items[0]->value)));
432
433 /* Deltas */
434 if (biomeval_nbis_lookup_ANSI_NIST_field(&field, &idx, DLT_ID, type9) == TRUE)
435 for (int i = 0; i < field->num_subfields; i++)
436 _deltaPointSet.push_back(DeltaPoint(convertCoordinate(
437 (char *)field->subfields[i]->items[0]->value,
438 false)));
439
440}
