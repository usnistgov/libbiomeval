14std::ostream&
15BiometricEvaluation::Feature::AN2K11EFS::operator<< (
16 std::ostream& s,
17 const AN2K11EFS::Orientation& ort)
18{
19 switch (ort.encodingMethod) {
20 case Orientation::EncodingMethod::Default:
21 /* FALLTHROUGH */
22 case Orientation::EncodingMethod::UserDefined:
23 s << "EOD: " << ort.eod << "; ";
24 s << "EUC: "; if (ort.has_euc) s << ort.euc; else s << "N/A";
25 break;
26 case Orientation::EncodingMethod::Indeterminate:
27 s << "false; Indeterminate";
28 break;
29 }
30
31 return (s);
32}

203std::ostream&
204BiometricEvaluation::Feature::AN2K11EFS::operator<< (
205 std::ostream& s,
206 const AN2K11EFS::MinutiaeRidgeCountInfo& mrci)
207{
208 using BE::Framework::Enumeration::operator<<;
209
210 s << "MRA: ";
211 if (mrci.has_mra) s << mrci.mra << "\n"; else s << "N/A\n";
212 s << "MRC: ";
213 if (mrci.has_mrcs) {
214 s << mrci.mrcs.size() << "\n";
215 for (auto const &mrc: mrci.mrcs) {
216 s << mrc << "\n";
217 }
218 } else {
219 s << "N/A\n";
220 }
221 s << "RCC: ";
222 if (mrci.has_rccs) {
223 s << mrci.rccs.size() << "\n";
224 for (auto const &rcc: mrci.rccs) {
225 s << rcc;
226 }
227 } else {
228 s << "N/A\n";
229 }
230 return (s);
231}

344std::ostream&
345BiometricEvaluation::Feature::AN2K11EFS::operator<<(
346 std::ostream &s,
347 const ExaminerAnalysisAssessment &eaa)
348{
349 if (!eaa.present)
350 return (s << "<# NOT SET #>");
351
352 s << "Determination: " << BE::Framework::Enumeration::to_string(
353 eaa.aav) << ", as determined by " << eaa.afn << " " <<
354 eaa.aln << " of " << eaa.aaf << " on " << eaa.amt;
355
356 if ((eaa.has_cxf && eaa.cxf) || !eaa.acm.empty()) {
357 s << " (";
358 if (eaa.has_cxf && eaa.cxf) {
359 s << "[Complex analysis]";
360 if (!eaa.acm.empty())
361 s << ", ";
362 }
363 if (!eaa.acm.empty())
364 s << eaa.acm;
365
366 s << ')';
367 }
368
369 return (s);
370}
